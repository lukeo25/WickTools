  function rasterizeSvgToPng(svgText, scale) {
    return new Promise(function(resolve) {
      try {
        var doc = new DOMParser().parseFromString(svgText, "image/svg+xml");
        var svg = doc.documentElement;
        
        var vb = svg.getAttribute("viewBox");
        var fullW, fullH;
        if (vb) {
          var parts = vb.split(/[\s,]+/).map(parseFloat);
          fullW = parts[2] || 100;
          fullH = parts[3] || 100;
        } else {
          fullW = parseFloat(svg.getAttribute("width")) || 100;
          fullH = parseFloat(svg.getAttribute("height")) || 100;
        }
        
        var tempCanvas = document.createElement("canvas");
        tempCanvas.width = Math.ceil(fullW * scale);
        tempCanvas.height = Math.ceil(fullH * scale);
        var tempCtx = tempCanvas.getContext("2d");
        
        var blob = new Blob([svgText], { type: "image/svg+xml" });
        var url = URL.createObjectURL(blob);
        
        var img = new Image();
        img.onload = function() {
          tempCtx.drawImage(img, 0, 0, tempCanvas.width, tempCanvas.height);
          URL.revokeObjectURL(url);
          
          var imageData = tempCtx.getImageData(0, 0, tempCanvas.width, tempCanvas.height);
          var data = imageData.data;
          var minX = tempCanvas.width, minY = tempCanvas.height, maxX = 0, maxY = 0;
          var found = false;
          
          for (var y = 0; y < tempCanvas.height; y++) {
            for (var x = 0; x < tempCanvas.width; x++) {
              var alpha = data[(y * tempCanvas.width + x) * 4 + 3];
              if (alpha > 0) {
                found = true;
                if (x < minX) minX = x;
                if (x > maxX) maxX = x;
                if (y < minY) minY = y;
                if (y > maxY) maxY = y;
              }
            }
          }
          
          if (!found) { resolve(null); return; }
          
          minX = Math.max(0, minX - 1);
          minY = Math.max(0, minY - 1);
          maxX = Math.min(tempCanvas.width - 1, maxX + 1);
          maxY = Math.min(tempCanvas.height - 1, maxY + 1);
          
          var cropW = maxX - minX + 1;
          var cropH = maxY - minY + 1;
          
          var cropCanvas = document.createElement("canvas");
          cropCanvas.width = cropW;
          cropCanvas.height = cropH;
          var cropCtx = cropCanvas.getContext("2d");
          cropCtx.drawImage(tempCanvas, minX, minY, cropW, cropH, 0, 0, cropW, cropH);
          
          cropCanvas.toBlob(function(blob) {
            resolve({
              blob: blob,
              cropBox: {
                x: minX / scale,
                y: minY / scale,
                width: cropW / scale,
                height: cropH / scale
              },
              scale: scale
            });
          }, "image/png");
        };
        img.onerror = function() { URL.revokeObjectURL(url); resolve(null); };
        img.src = url;
      } catch (e) { resolve(null); }
    });
  }

  function groupLayers(layers, delim) {
    var groups = {}, order = [];
    layers.forEach(function(layer) {
      var p = parseLayerName(layer.name, delim);
      if (!groups[p.clipName]) { 
        groups[p.clipName] = { clipName: p.clipName, frames: [], selected: true, hasMulti: false }; 
        order.push(p.clipName);

      groups[p.clipName].frames.push({ frameName: p.frameName, layer: layer, selected: true });
    });
    order.forEach(function(k) { 
      groups[k].hasMulti = groups[k].frames.length > 1 || (groups[k].frames[0] && groups[k].frames[0].frameName !== null);
    });
    return order.map(function(k) { return groups[k]; });
  }

  function buildLayerSvg(svg, el) {
    var gs = '<svg xmlns="http://www.w3.org/2000/svg" xmlns:inkscape="http://www.inkscape.org/namespaces/inkscape" viewBox="0 0 ' + svgWidth + ' ' + svgHeight + '">';

  function extractLayers(svgDoc, topOnly) {
    var layers = [], svg = svgDoc.querySelector("svg");
    if (!svg) return layers;
    var vb = svg.getAttribute("viewBox");
    if (vb) { var p = vb.split(/[\s,]+/).map(parseFloat); svgWidth = p[2] || 100; svgHeight = p[3] || 100; }
    else { svgWidth = parseFloat(svg.getAttribute("width")) || 100; svgHeight = parseFloat(svg.getAttribute("height")) || 100; }

    var trueLayers = Array.from(svg.querySelectorAll('g')).filter(function(el) { return isInkscapeLayer(el); });
    var usingTrueLayers = trueLayers.length > 0;

    function pushLayer(el, name) {
      var pivotAttrs = findInkscapePivotAttrs(el);
      layers.push({
        name: name,
        svgText: buildLayerSvg(svg, el),
        hasPivot: pivotAttrs !== null,
        pivotAttrs: pivotAttrs
      });
    }

    function traverseTrueLayers(parent, depth) {
      Array.from(parent.children).forEach(function(el) {
        if ((el.tagName || '').toLowerCase() !== 'g') return;
        if (!isInkscapeLayer(el)) {
          if (!topOnly) traverseTrueLayers(el, depth + 1);
          return;
        }
        var name = getGroupName(el);
        if (name) pushLayer(el, name);
        if (!topOnly) traverseTrueLayers(el, depth + 1);
      });
    }

    function traverseLegacy(parent, depth) {
      Array.from(parent.children).forEach(function(el) {
        if ((el.tagName || '').toLowerCase() !== 'g') return;
        var name = getGroupName(el);
        if (name) pushLayer(el, name);
        if (!topOnly) traverseLegacy(el, depth + 1);
      });
    }

    if (usingTrueLayers) {
      log('Using real SVG layer traversal' + (topOnly ? ' at top level only.' : ' including sublayers.'));
      traverseTrueLayers(svg, 0);
    } else {
      log('No real SVG layers found, using legacy named group traversal' + (topOnly ? ' at top level only.' : ' including nested groups.'));
      traverseLegacy(svg, 0);
    }
    return layers;
  }

  // ===== Thumbnail =====
  function genThumb(svgText, size) {
    return new Promise(function(res) {
      try {
        var url = URL.createObjectURL(new Blob([svgText], { type: "image/svg+xml" }));
        var img = new Image();
        img.onload = function() {
          var s = Math.min(size / img.width, size / img.height, 1);
          var c = document.createElement("canvas"); c.width = img.width * s || size; c.height = img.height * s || size;
          c.getContext("2d").drawImage(img, 0, 0, c.width, c.height);
          URL.revokeObjectURL(url); res(c.toDataURL());
        };
        img.onerror = function() { URL.revokeObjectURL(url); res(""); };
        img.src = url;
      } catch (e) { res(""); }
    });
  }

  // ===== UI =====
  function renderList() {
    if (!clipGroups.length) { clipsList.innerHTML = '<div class="empty-msg">No groups found</div>'; return; }
    var multi = clipGroups.filter(function(g) { return g.hasMulti; });
    var single = clipGroups.filter(function(g) { return !g.hasMulti; });
    var html = "";

    multi.forEach(function(g) {
      var ri = clipGroups.indexOf(g);
      var hasPivot = g.frames.some(function(f) { return f.layer.hasPivot; });
      var pivotIcon = hasPivot ? '<span class="pivot-icon" title="Has Inkscape pivot">⊕</span>' : '';
      html += '<div class="clip-group"><div class="clip-header">';
      html += '<input type="checkbox" class="clip-check" data-g="' + ri + '"' + (g.selected ? ' checked' : '') + '>';
      html += '<span class="clip-name">' + g.clipName + pivotIcon + '</span><span class="clip-count">' + g.frames.length + ' frames</span></div>';
      html += '<div class="frame-list">';
      g.frames.forEach(function(f, fi) {
        var fpivot = f.layer.hasPivot ? '<span class="pivot-icon">⊕</span>' : '';
        html += '<div class="frame-item"><input type="checkbox" class="frame-check" data-g="' + ri + '" data-f="' + fi + '"' + (f.selected ? ' checked' : '') + '>';
        html += '<img class="frame-thumb" data-g="' + ri + '" data-f="' + fi + '" src="">';
        html += '<span class="frame-num">F' + (fi + 1) + '</span><span class="frame-name">' + (f.frameName || "(unnamed)") + fpivot + '</span></div>';
      });
      html += '</div></div>';
    });

    if (single.length) {
      html += '<div class="solo-clips"><div class="solo-label">Single-frame clips</div>';
      single.forEach(function(g) {
        var ri = clipGroups.indexOf(g);
        var hasPivot = g.frames[0] && g.frames[0].layer.hasPivot;
        var pivotIcon = hasPivot ? '<span class="pivot-icon">⊕</span>' : '';
        html += '<div class="clip-group"><div class="clip-header">';
        html += '<input type="checkbox" class="clip-check" data-g="' + ri + '"' + (g.selected ? ' checked' : '') + '>';
        html += '<span class="clip-name" style="color:#aaa">' + g.clipName + pivotIcon + '</span></div>';
        html += '<div class="frame-list"><div class="frame-item"><img class="frame-thumb" data-g="' + ri + '" data-f="0" src=""></div></div></div>';
      });
      html += '</div>';
    }

    clipsList.innerHTML = html;
    clipsList.querySelectorAll('.clip-check').forEach(function(cb) {
      cb.addEventListener('change', function() {
        var gi = +this.dataset.g; clipGroups[gi].selected = this.checked;
        clipGroups[gi].frames.forEach(function(f) { f.selected = cb.checked; });
        clipsList.querySelectorAll('.frame-check[data-g="' + gi + '"]').forEach(function(fc) { fc.checked = cb.checked; });
      });
    });
    clipsList.querySelectorAll('.frame-check').forEach(function(cb) {
      cb.addEventListener('change', function() { clipGroups[+this.dataset.g].frames[+this.dataset.f].selected = this.checked; });
    });
    clipGroups.forEach(function(g, gi) {
      g.frames.forEach(function(f, fi) {
        genThumb(f.layer.svgText, 36).then(function(t) {
          var img = clipsList.querySelector('.frame-thumb[data-g="' + gi + '"][data-f="' + fi + '"]');
          if (img && t) img.src = t;
        });
      });
    });
  }

  function regroup() {
    clipGroups = groupLayers(rawLayers, delimiterInput.value || "|");
    renderList();
    importBtn.disabled = !clipGroups.length;
    var m = clipGroups.filter(function(g) { return g.hasMulti; }).length;
    var withPivot = rawLayers.filter(function(l) { return l.hasPivot; }).length;
    log("Grouped: " + m + " multi-frame, " + (clipGroups.length - m) + " single. " + withPivot + " with pivots.");
  }

  // ===== Events =====
  dropZone.addEventListener("click", function() { svgInput.click(); });
  dropZone.addEventListener("dragover", function(e) { e.preventDefault(); dropZone.classList.add("dragover"); });
  dropZone.addEventListener("dragleave", function() { dropZone.classList.remove("dragover"); });
  dropZone.addEventListener("drop", function(e) { e.preventDefault(); dropZone.classList.remove("dragover"); if (e.dataTransfer.files[0]) loadFile(e.dataTransfer.files[0]); });
  svgInput.addEventListener("change", function() { if (svgInput.files[0]) loadFile(svgInput.files[0]); });

  function loadFile(file) {
    if (!file.name.toLowerCase().endsWith(".svg")) { log("ERROR: Not SVG"); return; }
    svgFileName = file.name.replace(/\.svg$/i, "");
    fileNameDisplay.textContent = "Main clip: " + svgFileName;
    fileNameDisplay.style.display = "block";
    log("Reading " + file.name + "...");
    var r = new FileReader();
    r.onload = function(e) {
      svgSourceDoc = new DOMParser().parseFromString(e.target.result, "image/svg+xml");
      if (svgSourceDoc.querySelector("parsererror")) { log("ERROR: Invalid SVG"); return; }
      rawLayers = extractLayers(svgSourceDoc, document.getElementById("topLevelOnly").checked);
      log("Found " + rawLayers.length + " layers");
      regroup();
    };
    r.readAsText(file);
  }

  regroupBtn.addEventListener("click", regroup);
  document.getElementById("topLevelOnly").addEventListener("change", function() { if (svgSourceDoc) { rawLayers = extractLayers(svgSourceDoc, this.checked); regroup(); } });
  selectAllBtn.addEventListener("click", function() { clipGroups.forEach(function(g) { g.selected = true; g.frames.forEach(function(f) { f.selected = true; }); }); clipsList.querySelectorAll('input[type="checkbox"]').forEach(function(c) { c.checked = true; }); });
  selectNoneBtn.addEventListener("click", function() { clipGroups.forEach(function(g) { g.selected = false; g.frames.forEach(function(f) { f.selected = false; }); }); clipsList.querySelectorAll('input[type="checkbox"]').forEach(function(c) { c.checked = false; }); });

  function delay(ms) { return new Promise(function(r) { setTimeout(r, ms); }); }

  // ===== Import =====
  importBtn.addEventListener("click", async function() {
    var bridge = getBridge(), ed = getEditor(bridge), project = getProject(ed), Wick = getWick();
    if (!ed || !project) { log("ERROR: No editor/project"); return; }
    if (!Wick) { log("ERROR: No Wick"); return; }
    if (typeof ed.createAssets !== "function") { log("ERROR: createAssets not available"); return; }

    var usePivot = document.getElementById("usePivot").checked;
    var scale = parseFloat(resolutionSelect.value) || 1;
    var selected = clipGroups.filter(function(g) { return g.selected; });
    if (!selected.length) { log("ERROR: Nothing selected"); return; }

    importBtn.disabled = true;
    log("Building rig: " + svgFileName + " (" + selected.length + " components) at " + scale + "x...");

    var componentClips = [];
    var svgCenterX = svgWidth / 2;
    var svgCenterY = svgHeight / 2;

    for (var gi = 0; gi < selected.length; gi++) {
      var group = selected[gi];
      var frames = group.frames.filter(function(f) { return f.selected; });
      if (!frames.length) continue;

      var hasMultipleFrames = frames.length > 1;
      log("[" + (gi + 1) + "/" + selected.length + "] " + group.clipName + " (" + frames.length + " frames)");

      focusRoot(project);
      await delay(50);

      // === STEP 1: Rasterize first frame ===
      var firstFrame = frames[0];
      var pngData = await rasterizeSvgToPng(firstFrame.layer.svgText, scale);
      if (!pngData || !pngData.blob) { log("  → WARN: Rasterize failed"); continue; }

      var pivotAttrs = usePivot ? firstFrame.layer.pivotAttrs : null;
      var firstPlacement = buildPlacementDataFromPngData(pngData, pivotAttrs, scale); // FIX
      if (!firstPlacement) { log("  → WARN: Placement failed"); continue; } // FIX
      var pivotInSvg = firstPlacement.pivotInSvg; // FIX
      
      // Component position = where pivot sits relative to SVG center (scaled)
      var componentX = (pivotInSvg.x - svgCenterX) * scale;
      var componentY = (pivotInSvg.y - svgCenterY) * scale;

      var filename = group.clipName + "_" + (firstFrame.frameName || "f1") + ".png";
      var file = blobToFile(pngData.blob, filename);

      var beforeUUIDs = listAssetUUIDs(project);
      ed.createAssets([file], []);

      var asset;
      try { asset = await waitForNewAsset(project, beforeUUIDs, 10000); } 
      catch (e) { log("  → WARN: Asset timeout"); continue; }

      var path;
      try { path = await placeImagePathFromAsset(project, asset.uuid, 0, 0); } 
      catch (e) { log("  → WARN: Place failed"); continue; }

      // === STEP 2: Create component clip ===
      selectionSelectSingle(project, ed, path);
      await delay(60);

      if (!createClipFromSelection(project, ed, group.clipName)) {
        log("  → WARN: Clip creation failed"); continue;
      }
      await delay(80);

      var componentClip = getSelectedClip(project);
      if (!componentClip) { log("  → WARN: No clip"); continue; }

      // === STEP 3: Enter clip and offset first frame for pivot ===
      if (!focusClip(project, componentClip)) {
        log("  → WARN: Can't enter clip");
        focusRoot(project); continue;
      }
      await delay(60);

      var clipFrame = getCurrentFrame(project);
      if (clipFrame) {
        if (firstPlacement.localOffset) { // FIX
          var objs = getObjectsInFrame(clipFrame);
          var firstOffset = offsetObjectsForPivot(objs, firstPlacement.localOffset); // FIX
          if (firstOffset) { // FIX
            log("  → F1 Offset: (" + firstOffset.x.toFixed(1) + ", " + firstOffset.y.toFixed(1) + ")"); // FIX
          } // FIX
        }
        if (firstFrame.frameName) {
          setFrameName(clipFrame, firstFrame.frameName);
          log("  → F1: " + firstFrame.frameName);
        }
      }

      // === STEP 4: Add additional frames ===
      if (hasMultipleFrames) {
        var layer = getActiveLayer(project);
        if (layer) {
          markDirty(ed);

          for (var fi = 1; fi < frames.length; fi++) {
            var fd = frames[fi];
            var frameNum = fi + 1;

            // Create new frame
            var newFrame = ensureFrameAt(layer, frameNum, Wick);
            if (!newFrame) { log("    → F" + frameNum + ": frame create failed"); continue; }

            // Name frame
            if (fd.frameName) {
              setFrameName(newFrame, fd.frameName);
            }

            // Move playhead BEFORE placing image
            setPlayhead(project, frameNum);
            await delay(50);

            // Rasterize this frame's content
            var fPngData = await rasterizeSvgToPng(fd.layer.svgText, scale);
            if (!fPngData || !fPngData.blob) { log("    → F" + frameNum + ": rasterize failed"); continue; }

            var fPivotAttrs = usePivot ? fd.layer.pivotAttrs : null; // FIX
            var fPlacement = buildPlacementDataFromPngData(fPngData, fPivotAttrs, scale); // FIX
            if (!fPlacement) { log("    → F" + frameNum + ": placement failed"); continue; } // FIX

            var fFilename = group.clipName + "_" + (fd.frameName || "f" + frameNum) + ".png";
            var fFile = blobToFile(fPngData.blob, fFilename);

            var fBeforeUUIDs = listAssetUUIDs(project);
            ed.createAssets([fFile], []);

            var fAsset;
            try { fAsset = await waitForNewAsset(project, fBeforeUUIDs, 10000); } 
            catch (e) { log("    → F" + frameNum + ": asset timeout"); continue; }

            // Place image at (0,0) - it goes on current frame (playhead position)
            var fPath;
            try { fPath = await placeImagePathFromAsset(project, fAsset.uuid, 0, 0); } 
            catch (e) { log("    → F" + frameNum + ": place failed"); continue; }

            // Offset for pivot
            if (fPlacement.localOffset && fPath) { // FIX
              try {
                fPath.x = (Number(fPath.x) || 0) + fPlacement.localOffset.x; // FIX
                fPath.y = (Number(fPath.y) || 0) + fPlacement.localOffset.y; // FIX
              } catch (e) {}
              log("    + F" + frameNum + " Offset: (" + fPlacement.localOffset.x.toFixed(1) + ", " + fPlacement.localOffset.y.toFixed(1) + ")"); // FIX
            }

            log("    + F" + frameNum + ": " + (fd.frameName || "(unnamed)"));
            await delay(30);
          }
        }
      }

      // Exit component clip
      focusRoot(project);
      await delay(50);

      // === STEP 5: Position component clip ===
      componentClip.x = componentX;
      componentClip.y = componentY;
      log("  → Pos: (" + componentX.toFixed(1) + ", " + componentY.toFixed(1) + ")");

      componentClips.push(componentClip);
      
      selectionClear(project);
      markDirty(ed);
      await delay(40);
    }

    // === STEP 6: Wrap all into main clip ===
    if (componentClips.length > 0) {
      log("Wrapping " + componentClips.length + " components into: " + svgFileName);

      focusRoot(project);
      await delay(50);

      selectionSelectMultiple(project, componentClips);
      await delay(60);

      if (createClipFromSelection(project, ed, svgFileName)) {
        await delay(80);
        var mainClip = getSelectedClip(project);
        if (mainClip) {
          var stageCenter = getStageCenter(project);
          mainClip.x = stageCenter.x;
          mainClip.y = stageCenter.y;
          log("✓ Created main clip: " + svgFileName);
          log("✓ Stage size: " + stageCenter.width + " x " + stageCenter.height);
          log("✓ Main clip centered at: (" + stageCenter.x + ", " + stageCenter.y + ")");
        } else {
          log("✓ Created main clip: " + svgFileName + " but could not reselect it for centering");
        }
      } else {
        log("→ WARN: Could not create main clip");
      }
      await delay(80);
    }

    selectionClear(project);
    markDirty(ed);
    log("Done!");
    importBtn.disabled = false;
  });

})();
