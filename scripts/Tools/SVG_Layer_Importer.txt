  // ===== SVG Sanitization =====
  function sanitizeSvg(svgText) {
    var parser = new DOMParser();
    var doc = parser.parseFromString(svgText, "image/svg+xml");
    var svg = doc.documentElement;
    if (!svg || svg.nodeName.toLowerCase() !== "svg") throw new Error("Not an SVG");

    var banned = ["script", "foreignobject", "iframe", "object", "embed"];
    banned.forEach(function (tag) {
      var nodes = doc.getElementsByTagName(tag);
      var list = [];
      for (var i = 0; i < nodes.length; i++) list.push(nodes[i]);
      list.forEach(function (n) { if (n.parentNode) n.parentNode.removeChild(n); });
    });

    if (!svg.getAttribute("viewBox")) {
      var w = parseFloat(svg.getAttribute("width")) || 100;
      var h = parseFloat(svg.getAttribute("height")) || 100;
      svg.setAttribute("viewBox", "0 0 " + w + " " + h);
    }

    return new XMLSerializer().serializeToString(doc);
  }

  function getAllPathsAndCompound(paper, item) {
    var out = [];
    if (!item) return out;
    try { var ps = item.getItems({ class: paper.Path }) || []; for (var i = 0; i < ps.length; i++) out.push(ps[i]); } catch (e) {}
    try { var cps = item.getItems({ class: paper.CompoundPath }) || []; for (var j = 0; j < cps.length; j++) out.push(cps[j]); } catch (e) {}
    return out;
  }

  function convertGroupToJsonList(paper, group, separate) {
    var list = [];
    if (!group) return list;

    if (separate) {
      var items = getAllPathsAndCompound(paper, group);
      for (var i = 0; i < items.length; i++) {
        try {
          var clone = items[i].clone({ insert: false });
          var json = clone.exportJSON({ asString: false });
          clone.remove();
          if (json) list.push(json);
        } catch (e) {}
      }
    } else {
      try {
        var compound = new paper.CompoundPath({ insert: false });
        var source = getAllPathsAndCompound(paper, group);
        for (var si = 0; si < source.length; si++) {
          var src = source[si];
          if (!src) continue;
          if (src.className === "CompoundPath" && src.children && src.children.length) {
            for (var ci = 0; ci < src.children.length; ci++) {
              try { compound.addChild(src.children[ci].clone({ insert: false })); } catch (e) {}
            }
          } else {
            try { compound.addChild(src.clone({ insert: false })); } catch (e) {}
          }
        }
        if (compound.children && compound.children.length) {
          if (source[0]) {
            try { compound.fillColor = source[0].fillColor; } catch (e) {}
            try { compound.strokeColor = source[0].strokeColor; } catch (e) {}
            try { compound.strokeWidth = source[0].strokeWidth; } catch (e) {}
          }
          list.push(compound.exportJSON({ asString: false }));
        }
        try { compound.remove(); } catch (e) {}
      } catch (e) {
        try { list.push(group.exportJSON({ asString: false })); } catch (e2) {}
      }
    }

    return list;
  }

  // ===== SVG Parsing =====
  function getGroupName(g) {
    return g.getAttribute("inkscape:label") || g.getAttribute("label") || g.id || null;
  }

  function extractLayers(svgDoc, topLevelOnly) {
    var layers = [];
    var svg = svgDoc.querySelector("svg");
    if (!svg) return layers;

    var vb = svg.getAttribute("viewBox");
    if (vb) {
      var parts = vb.split(/[\s,]+/).map(parseFloat);
      svgWidth = parts[2] || 100;
      svgHeight = parts[3] || 100;
    } else {
      svgWidth = parseFloat(svg.getAttribute("width")) || 100;
      svgHeight = parseFloat(svg.getAttribute("height")) || 100;
    }

    function traverse(parent, depth) {
      var children = parent.children;
      for (var i = 0; i < children.length; i++) {
        var el = children[i];
        if (el.tagName.toLowerCase() === "g") {
          var name = getGroupName(el);
          if (name) {
            var groupSvg = '<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 ' + svgWidth + ' ' + svgHeight + '">';
            var defs = svg.querySelector("defs");
            if (defs) groupSvg += defs.outerHTML;
            groupSvg += el.outerHTML + '</svg>';

            layers.push({
              name: name,
              svgText: groupSvg,
              depth: depth,
              selected: true,
              thumbnail: null
            });
          }
          if (!topLevelOnly) {
            traverse(el, depth + 1);
          }
        }
      }
    }

    traverse(svg, 0);
    return layers;
  }

  // ===== Thumbnail Generation =====
  function generateThumbnail(svgText, maxSize) {
    return new Promise(function (resolve) {
      try {
        var blob = new Blob([svgText], { type: "image/svg+xml;charset=utf-8" });
        var url = URL.createObjectURL(blob);
        var img = new Image();
        img.onload = function () {
          var scale = Math.min(maxSize / img.width, maxSize / img.height, 1);
          var w = Math.round(img.width * scale) || maxSize;
          var h = Math.round(img.height * scale) || maxSize;
          var canvas = document.createElement("canvas");
          canvas.width = w;
          canvas.height = h;
          var ctx = canvas.getContext("2d");
          ctx.drawImage(img, 0, 0, w, h);
          URL.revokeObjectURL(url);
          resolve(canvas.toDataURL("image/png"));
        };
        img.onerror = function () {
          URL.revokeObjectURL(url);
          resolve("");
        };
        img.src = url;
      } catch (e) {
        resolve("");
      }
    });
  }

  // ===== Load SVG =====
  function loadSVGFile(file) {
    if (!file.name.toLowerCase().endsWith(".svg")) {
      log("ERROR: Not an .svg file");
      return;
    }

    log("Reading: " + file.name + "...");
    parsedLayers = [];

    var reader = new FileReader();
    reader.onload = function (e) {
      try {
        var parser = new DOMParser();
        svgSourceDoc = parser.parseFromString(e.target.result, "image/svg+xml");

        var parseError = svgSourceDoc.querySelector("parsererror");
        if (parseError) {
          log("ERROR: Invalid SVG file");
          return;
        }

        var topLevelOnly = document.getElementById("topLevelOnly").checked;
        parsedLayers = extractLayers(svgSourceDoc, topLevelOnly);

        log("Parsed! Found " + parsedLayers.length + " groups. Canvas: " + svgWidth + "×" + svgHeight);

        if (parsedLayers.length === 0) {
          log("No named <g> groups found. Groups need id or inkscape:label.");
        }

        renderLayerList();
        importBtn.disabled = parsedLayers.length === 0;

        // Generate thumbnails async
        parsedLayers.forEach(function (L, idx) {
          generateThumbnail(L.svgText, 50).then(function (thumb) {
            L.thumbnail = thumb;
            var img = layersList.querySelector('.layer-thumb[data-idx="' + idx + '"]');
            if (img && thumb) img.src = thumb;
          });
        });

      } catch (err) {
        log("ERROR: " + err.message);
      }
    };
    reader.readAsText(file);
  }

      html += '<span class="layer-info">depth: ' + L.depth + '</span>';

      cb.addEventListener("change", function () {

  // ===== Import =====
  importBtn.addEventListener("click", async function () {
    var bridge = getBridge();
    var ed = getEditor(bridge);
    var project = getProject(ed);
    var Wick = getWick();
    var paper = getPaper();

    if (!ed) { log("ERROR: Editor not found"); return; }
    if (!project) { log("ERROR: Project not found"); return; }
    if (!Wick) { log("ERROR: Wick namespace not found"); return; }
    if (!Wick.Path) { log("ERROR: Wick.Path not found"); return; }
    if (!paper) { log("ERROR: paper.js not found"); return; }

    var tl = findActiveTimeline(project);
    if (!tl) { log("ERROR: No active timeline"); return; }

    var frameNum = getCurrentFrameNumber(project);
    var stageFrame = findFrameFromTimeline(tl, frameNum);
    if (!stageFrame) { log("ERROR: No current frame"); return; }

    var selected = parsedLayers.filter(function (L) { return L.selected; });
    if (!selected.length) { log("ERROR: No layers selected"); return; }

    var wrapAsClip = document.getElementById("wrapAsClip").checked;
    var separatePaths = document.getElementById("separatePaths").checked;

    importBtn.disabled = true;
    log("Importing " + selected.length + " groups as curves...");

    var totalInserted = 0;
    var totalFailed = 0;

    for (var idx = 0; idx < selected.length; idx++) {
      var L = selected[idx];
      log("[" + (idx + 1) + "/" + selected.length + "] " + L.name);

      try {
        var sanitized = sanitizeSvg(L.svgText);
        var group = importSvgAsGroup(paper, sanitized);
        if (!group) {
          log("  → WARN: Paper.js import failed");
          totalFailed++;
          continue;
        }

        var jsonList = convertGroupToJsonList(paper, group, separatePaths);
        try { group.remove(); } catch (e) {}

        if (!jsonList.length) {
          log("  → WARN: No paths found");
          totalFailed++;
          continue;
        }

        // Insert paths into the current frame
        var insertedPaths = [];
        for (var i = 0; i < jsonList.length; i++) {
          var wp = null;
          try { wp = new Wick.Path({ json: jsonList[i], project: project }); } catch (e) { wp = null; }
          if (!wp) { try { wp = new Wick.Path({ json: jsonList[i] }); } catch (e) { wp = null; } }
          if (wp && insertPathIntoFrame(stageFrame, wp)) {
            insertedPaths.push(wp);
          }
        }

        if (insertedPaths.length === 0) {
          log("  → WARN: Could not insert paths");
          totalFailed++;
          continue;
        }

        totalInserted += insertedPaths.length;

        if (wrapAsClip) {
          // Select all the paths we just inserted, then createClipFromSelection
          selectionSelectMultiple(project, ed, insertedPaths);
          await delay(50); // Give Wick time to process selection

          var r = createClipFromSelection(project, ed, L.name);
          if (r.ok) {
            log("  → Clip: " + L.name + " (" + insertedPaths.length + " paths)");
          } else {
            log("  → Paths added, clip wrap failed");
          }
        } else {
          log("  → Stage: " + insertedPaths.length + " paths");
        }

        markDirty(ed);
        await delay(30);

      } catch (err) {
        log("  → ERROR: " + err.message);
        totalFailed++;
      }
    }

    markDirty(ed);
    log("Done! Inserted " + totalInserted + " paths, " + totalFailed + " groups failed");
    importBtn.disabled = false;
  });

})();
