        if (list && typeof list.length === "number") {
          for (j = 0; j < list.length; j += 1) addUniqueObject(out, seen, list[j]);
        }
      }
    } catch(eChildren) {}
  }

  function getTimeline(obj){

  }

  function getTimelineLayers(timeline){

  function isStickRigInternalCarrier(clip){
    if (!clip || !looksLikeClip(clip)) return false;
    var identifier = s(safeGet(clip, "identifier") || safeGet(clip, "_identifier"));
    return safeGet(clip, "_stickRigInternalCarrier") === true ||
      identifier.indexOf(INTERNAL_CARRIER_PREFIX) === 0;
  }

  function rawTimelineChildClips(parentClip){
    var out = [];
    var seen = {};
    var timeline = getTimeline(parentClip);
    var layers;
    var frames;
    var i;
    var j;

    try {
      if (typeof parentClip.getChildren === "function") {
        var direct = parentClip.getChildren();
        if (direct && typeof direct.length === "number") {
          for (i = 0; i < direct.length; i += 1) if (looksLikeClip(direct[i])) addUniqueObject(out, seen, direct[i]);
        }
      }
    } catch(eDirect) {}

    if (!timeline) return out;

    try {
      var activeFrame = safeGet(timeline, "activeFrame");
      if (activeFrame) collectFrameObjects(activeFrame, out, seen);
    } catch(eActiveFrame) {}

    layers = getTimelineLayers(timeline);
    for (i = 0; i < layers.length; i += 1) {
      frames = getLayerFrames(layers[i]);
      for (j = 0; j < frames.length; j += 1) collectFrameObjects(frames[j], out, seen);
    }

    return out.filter(function(obj){ return looksLikeClip(obj); });
  }

  function getTimelineChildClips(parentClip){
    var out = [];
    var seen = {};

    function addLogical(clip, depth){
      if (!clip || depth > 20) return;
      if (isStickRigInternalCarrier(clip)) {
        rawTimelineChildClips(clip).forEach(function(child){ addLogical(child, depth + 1); });
        return;
      }
      addUniqueObject(out, seen, clip);
    }

    rawTimelineChildClips(parentClip).forEach(function(child){ addLogical(child, 0); });
    return out;
  }

  function readNumericProperty(obj, keys, fallback){
    var i;
    var value;
    for (i = 0; i < keys.length; i += 1) {
      value = safeGet(obj, keys[i]);
      if (value !== undefined && value !== null && isFinite(Number(value))) return Number(value);
    }
    return fallback;
  }

  function readTransform(obj){
    var transformation = safeGet(obj, "transformation");
    var x = readNumericProperty(obj, ["x", "_x"], NaN);
    var y = readNumericProperty(obj, ["y", "_y"], NaN);
    var rotation = readNumericProperty(obj, ["rotation", "_rotation"], NaN);
    var scaleX = readNumericProperty(obj, ["scaleX", "_scaleX"], NaN);
    var scaleY = readNumericProperty(obj, ["scaleY", "_scaleY"], NaN);

    if (transformation) {
      if (!isFinite(x)) x = n(safeGet(transformation, "x"), 0);
      if (!isFinite(y)) y = n(safeGet(transformation, "y"), 0);
      if (!isFinite(rotation)) rotation = n(safeGet(transformation, "rotation"), 0);
      if (!isFinite(scaleX)) scaleX = n(safeGet(transformation, "scaleX"), 1);
      if (!isFinite(scaleY)) scaleY = n(safeGet(transformation, "scaleY"), 1);
    }

    if (!isFinite(x)) x = 0;
    if (!isFinite(y)) y = 0;
    if (!isFinite(rotation)) rotation = 0;
    if (!isFinite(scaleX)) scaleX = 1;
    if (!isFinite(scaleY)) scaleY = 1;

    var absScale = (Math.abs(scaleX) + Math.abs(scaleY)) * 0.5;
    if (!isFinite(absScale) || absScale <= 0) absScale = 1;

    return {
      x: round3(x),
      y: round3(y),
      scale: round3(absScale),
      rotation: round3(rotation),
      signX: scaleX < 0 ? -1 : 1,
      signY: scaleY < 0 ? -1 : 1
    };
  }

  function cloneTransformation(current){
    var W = getWickNamespace();
    if (W && W.Transformation) {
      try { return new W.Transformation(current || {}); } catch(e1) {}
      try { return new W.Transformation(); } catch(e2) {}
    }
    var copy = {};
    var keys = ["x", "y", "scaleX", "scaleY", "rotation", "opacity"];
    var i;
    for (i = 0; i < keys.length; i += 1) copy[keys[i]] = n(current && current[keys[i]], keys[i].indexOf("scale") === 0 ? 1 : 0);
    return copy;
  }

  function setOneProperty(obj, key, value){
    var success = false;
    try { obj[key] = value; success = true; } catch(e1) {}
    try { if (obj.transformation && key in obj.transformation) { obj.transformation[key] = value; success = true; } } catch(e2) {}
    try { var under = "_" + key; if (under in obj) { obj[under] = value; success = true; } } catch(e3) {}
    return success;
  }

  function applyTransformToObject(node, pose, mark){
    if (!node || !node.obj || !pose) return false;
    var base = node.base || { signX: 1, signY: 1 };
    var signX = n(base.signX, 1) < 0 ? -1 : 1;
    var signY = n(base.signY, 1) < 0 ? -1 : 1;
    var uniformScale = Math.max(0.01, n(pose.scale, 1));
    var current = null;

    try { current = node.obj.transformation; } catch(eCurrent) {}

    if (current) {
      var tr = cloneTransformation(current);
      tr.x = n(pose.x, 0);
      tr.y = n(pose.y, 0);
      tr.rotation = n(pose.rotation, 0);
      tr.scaleX = uniformScale * signX;
      tr.scaleY = uniformScale * signY;
      try { node.obj.transformation = tr; } catch(eAssign) {}
    }

    setOneProperty(node.obj, "x", n(pose.x, 0));
    setOneProperty(node.obj, "y", n(pose.y, 0));
    setOneProperty(node.obj, "rotation", n(pose.rotation, 0));
    setOneProperty(node.obj, "scaleX", uniformScale * signX);
    setOneProperty(node.obj, "scaleY", uniformScale * signY);

    if (mark) markProjectChanged();
    return true;
  }

  function markProjectChanged(){

    try { if (editor && typeof editor.projectDidChange === "function") editor.projectDidChange(); } catch(e3) {}
    try { if (editor && typeof editor.forceUpdate === "function") editor.forceUpdate(); } catch(e4) {}
    try { if (project && project.view && typeof project.view.render === "function") project.view.render(); } catch(e5) {}
  }

  function getProjectDuration(){
    var timeline = getMainTimeline();
    var length = 1;
    var layers = getTimelineLayers(timeline);
    var hasNonProxyLayer = false;
    var i;
    var j;

    for (i = 0; i < layers.length; i += 1) {
      if (isStickRigProxyLayer(layers[i])) continue;
      hasNonProxyLayer = true;
      var frames = getLayerFrames(layers[i]);
      for (j = 0; j < frames.length; j += 1) length = Math.max(length, getFrameEnd(frames[j]));
    }

    if (!hasNonProxyLayer) {
      try {
        if (timeline && isFinite(Number(timeline.length))) length = Math.max(length, Math.round(Number(timeline.length)));
      } catch(e1) {}
    }

    return Math.max(1, length);
  }

  function makeNodeId(obj, parentId, index){
    var info = getObjectInfo(obj);
    if (info.uuid) return "uuid:" + info.uuid;

  function buildHierarchy(root){
    var nodes = [];
    var nodeMap = {};
    var visited = {};
    var maxDepth = 40;

    function walk(obj, parentNode, depth, path, siblingIndex){
      if (!obj || depth > maxDepth) return;
      var identity = objectIdentity(obj);
      if (visited[identity]) return;
      visited[identity] = true;

      var info = getObjectInfo(obj);
      var id = parentNode ? makeNodeId(obj, parentNode.id, siblingIndex) : makeNodeId(obj, "root", 0);
      var base = readTransform(obj);
      var node = {
        id: id,
        obj: obj,
        info: info,
        name: getObjectName(obj, parentNode ? "Child Clip" : "Root Clip"),
        depth: depth,
        parentId: parentNode ? parentNode.id : "",
        path: path.slice(),
        base: base,
        isRoot: !parentNode,
        childCount: 0,
        committedFrame: null
      };

      nodes.push(node);
      nodeMap[id] = node;

      var children = getTimelineChildClips(obj);
      node.childCount = children.length;
      var occurrenceByLookup = {};

      children.forEach(function(child, index){
        var childInfo = getObjectInfo(child);
        var lookup = childInfo.identifier || childInfo.name || childInfo.uuid || ("Child_" + (index + 1));
        var count = occurrenceByLookup[lookup] || 0;
        occurrenceByLookup[lookup] = count + 1;
        var segment = {
          uuid: childInfo.uuid,
          identifier: childInfo.identifier,
          name: childInfo.name,
          occurrence: count
        };
        walk(child, node, depth + 1, path.concat([segment]), index);
      });
    }

    walk(root, null, 0, [], 0);
    state.nodes = nodes;
    state.nodeMap = nodeMap;
  }

    if (!obj || !state.bridge) return "";

  function setUpdateScriptForObject(obj, source){
    if (!obj || !state.bridge) return { ok: false, reason: "object or bridge missing" };
    var info = getObjectInfo(obj);
    var result = null;

    try {
      if (typeof state.bridge.setUpdateScriptByInfo === "function") {
        result = state.bridge.setUpdateScriptByInfo(info, s(source));
        if (!(result && result.ok === false)) return result || { ok: true };
      }
    } catch(eByInfo) {
      result = { ok: false, reason: eByInfo && eByInfo.message ? eByInfo.message : String(eByInfo) };
    }

    /* Moving a clip to a dedicated animation layer can leave a bridge lookup
       cache stale for a moment. Re-select the same real clip and use the
       selection writer rather than treating that cache miss as animation
       failure. No clip is copied or replaced here. */
    try { refreshRefs(); } catch(eRefresh) {}
    try {
      if (state.bridge && typeof state.bridge.selectObjectByInfo === "function") {
        state.bridge.selectObjectByInfo(info);
      }
      if (state.bridge && typeof state.bridge.setUpdateScriptOnSelection === "function") {
        var selectionResult = state.bridge.setUpdateScriptOnSelection(s(source));
        if (!(selectionResult && selectionResult.ok === false)) return selectionResult || { ok: true };
        result = selectionResult;
      }
    } catch(eSelection) {
      result = { ok: false, reason: eSelection && eSelection.message ? eSelection.message : String(eSelection) };
    }

    return result || { ok: false, reason: "target not found" };
  }


  function stripLiveDriverBlock(source){
    var text = s(source);
    var start = text.indexOf(LIVE_DRIVER_BEGIN);
    while (start !== -1) {
      var end = text.indexOf(LIVE_DRIVER_END, start);
      if (end === -1) text = text.substring(0, start);
      else text = text.substring(0, start) + text.substring(end + LIVE_DRIVER_END.length);
      start = text.indexOf(LIVE_DRIVER_BEGIN);
    }
    return text.trim();
  }

  function buildLiveDriverData(record, duration){
    ensureChannelMaps(record);
    var rotationPath = rotationPathValues(record);
    var data = {
      version: 2,
      duration: Math.max(1, Math.round(n(duration, 1))),
      base: {
        x: round3(n(record && record.base && record.base.x, 0)),
        y: round3(n(record && record.base && record.base.y, 0)),
        scale: round3(Math.max(0.01, n(record && record.base && record.base.scale, 1))),
        rotation: round3(n(rotationPath && rotationPath.base, n(record && record.base && record.base.rotation, 0))),
        signX: n(record && record.base && record.base.signX, 1) < 0 ? -1 : 1,
        signY: n(record && record.base && record.base.signY, 1) < 0 ? -1 : 1
      },
      channels: {
        move: [],
        scale: [],
        rotate: []
      }
    };

    CHANNELS.forEach(function(channel){
      channelKeyFrames(record, channel).forEach(function(frameNumber){
        var pose = record.keys && record.keys[String(frameNumber)] ? record.keys[String(frameNumber)] : record.base;
        var entry = {
          frame: Math.max(1, Math.round(n(frameNumber, 1))),
          type: channelType(record, frameNumber, channel),
          easing: channelEasing(record, frameNumber, channel)
        };
        if (channel === "move") {
          entry.x = round3(n(pose && pose.x, data.base.x));
          entry.y = round3(n(pose && pose.y, data.base.y));
        } else if (channel === "scale") {
          entry.scale = round3(Math.max(0.01, n(pose && pose.scale, data.base.scale)));
        } else {
          entry.rotation = round3(
            rotationPath && rotationPath.keys[String(frameNumber)] !== undefined
              ? rotationPath.keys[String(frameNumber)]
              : n(pose && pose.rotation, data.base.rotation)
          );
        }
        data.channels[channel].push(entry);
      });
    });
    return data;
  }

  function buildLiveDriverBlock(record, duration){
    var data = buildLiveDriverData(record, duration);
    return [
      LIVE_DRIVER_BEGIN,
      "(function(){",
      "  try {",
      "    var D = " + JSON.stringify(data) + ";",
      "    var tl = this.timeline || this._timeline;",
      "    var frame = Number(tl && (tl.playheadPosition !== undefined ? tl.playheadPosition : tl._playheadPosition));",
      "    if (!isFinite(frame)) frame = 1;",
      "    function num(value, fallback){ value = Number(value); return isFinite(value) ? value : fallback; }",
      "    function clamp(value, min, max){ return Math.max(min, Math.min(max, value)); }",
      "    function ease(value, kind){",
      "      var t = clamp(num(value, 0), 0, 1);",
      "      if (kind === 'in') return t * t;",
      "      if (kind === 'out') return 1 - Math.pow(1 - t, 2);",
      "      if (kind === 'in-out') return t < 0.5 ? 2 * t * t : 1 - Math.pow(-2 * t + 2, 2) / 2;",
      "      return t;",
      "    }",
      "    function interpolate(list, base, properties){",
      "      if (!list || !list.length) return base;",
      "      var previous = null;",
      "      var next = null;",
      "      for (var i = 0; i < list.length; i += 1) {",
      "        if (list[i].frame === frame) return list[i];",
      "        if (list[i].frame < frame) previous = list[i];",
      "        if (list[i].frame > frame) { next = list[i]; break; }",
      "      }",
      "      if (!previous && next) {",
      "        previous = {};",
      "        for (var b = 0; b < properties.length; b += 1) previous[properties[b]] = base[properties[b]];",
      "        previous.frame = 1; previous.type = 'auto'; previous.easing = 'none';",
      "      }",
      "      if (previous && !next) return previous;",
      "      if (!previous || !next || next.frame <= previous.frame) return base;",
      "      var ratio = previous.type === 'hold' ? 0 : ease((frame - previous.frame) / (next.frame - previous.frame), previous.easing || 'none');",
      "      var output = {};",
      "      for (var p = 0; p < properties.length; p += 1) {",
      "        var property = properties[p];",
      "        var from = num(previous[property], num(base[property], 0));",
      "        var to = num(next[property], from);",
      "        output[property] = from + (to - from) * ratio;",
      "      }",
      "      return output;",
      "    }",
      "    var move = interpolate(D.channels.move, {x:D.base.x, y:D.base.y}, ['x','y']);",
      "    var scale = interpolate(D.channels.scale, {scale:D.base.scale}, ['scale']);",
      "    var rotate = interpolate(D.channels.rotate, {rotation:D.base.rotation}, ['rotation']);",
      "    var value;",
      "    value = num(move.x, D.base.x); if (isFinite(value)) this.x = value;",
      "    value = num(move.y, D.base.y); if (isFinite(value)) this.y = value;",
      "    value = Math.max(0.01, num(scale.scale, D.base.scale)); this.scaleX = value * (D.base.signX < 0 ? -1 : 1);",
      "    this.scaleY = value * (D.base.signY < 0 ? -1 : 1);",
      "    value = num(rotate.rotation, D.base.rotation); if (isFinite(value)) this.rotation = value;",
      "  } catch (error) {}",
      "}).call(this);",
      LIVE_DRIVER_END
    ].join("\n");
  }

  function writeLiveDriverToObject(obj, record, duration){
    if (!obj) return { ok: false, reason: "clip missing" };
    if (!record) return { ok: false, reason: "animation record missing" };
    var existing = getUpdateScriptForObject(obj);
    var clean = stripLiveDriverBlock(existing);
    var block = buildLiveDriverBlock(record, duration);
    var source = clean ? clean + "\n\n" + block : block;
    var result = setUpdateScriptForObject(obj, source);
    if (!(result && result.ok === false)) return result || { ok: true };

    try { markProjectChanged(); } catch(eMark) {}
    try { refreshRefs(); } catch(eRefresh) {}
    result = setUpdateScriptForObject(obj, source);
    if (!(result && result.ok === false)) return result || { ok: true };

    try {
      var info = getObjectInfo(obj);
      if (state.bridge && typeof state.bridge.selectObjectByInfo === "function") {
        state.bridge.selectObjectByInfo(info);
      }
      if (state.bridge && typeof state.bridge.setUpdateScriptOnSelection === "function") {
        var selectionResult = state.bridge.setUpdateScriptOnSelection(source);
        if (!(selectionResult && selectionResult.ok === false)) return selectionResult || { ok: true };
        result = selectionResult;
      }
    } catch(eSelectionWrite) {
      result = { ok: false, reason: eSelectionWrite && eSelectionWrite.message ? eSelectionWrite.message : String(eSelectionWrite) };
    }
    return result || { ok: false, reason: "target not found" };
  }

  function removeLiveDriverFromObject(obj){
    if (!obj) return { ok: true };
    var existing = getUpdateScriptForObject(obj);
    var clean = stripLiveDriverBlock(existing);
    if (clean === s(existing).trim()) return { ok: true };
    return setUpdateScriptForObject(obj, clean);
  }

  function parseCommittedData(source){
    var text = s(source);
    var start = text.indexOf(COMMITTED_DATA_BEGIN);
    if (start === -1) return null;
    var end = text.indexOf(COMMITTED_DATA_END, start);
    if (end === -1) return null;
    var body = text.substring(start + COMMITTED_DATA_BEGIN.length, end);
    var assign = body.indexOf("var LT_STICKRIG_COMMITTED_DATA =");
    if (assign === -1) return null;
    assign += "var LT_STICKRIG_COMMITTED_DATA =".length;
    var semicolon = body.lastIndexOf(";");
    if (semicolon === -1 || semicolon <= assign) return null;
    try { return JSON.parse(body.substring(assign, semicolon).trim()); } catch(e) { return null; }
  }

  function buildCommittedDataBlock(data){
    return [
      COMMITTED_DATA_BEGIN,
      "var LT_STICKRIG_COMMITTED_DATA = " + JSON.stringify(data) + ";",
      COMMITTED_DATA_END
    ].join("\n");
  }

  function writeCommittedDataToObject(obj, data){

    var block = buildCommittedDataBlock(data);
    return setUpdateScriptForObject(obj, clean ? clean + "\n\n" + block : block);
  }

  function committedWriteFailed(result){
    return !!(result && result.ok === false);
  }

  function writeCommittedDataToObjectRobust(obj, data){
    var result = writeCommittedDataToObject(obj, data);
    if (!committedWriteFailed(result)) return result || { ok: true };

    /* A newly inserted Wick clip can exist in the model before the LukeTools
       bridge has refreshed its lookup table. Force a model refresh and retry
       before falling back to selection-based script writing. */
    try { markProjectChanged(); } catch(eMark) {}
    try { refreshRefs(); } catch(eRefresh) {}
    result = writeCommittedDataToObject(obj, data);
    if (!committedWriteFailed(result)) return result || { ok: true };

    var info = getObjectInfo(obj);
    var existing = getUpdateScriptForObject(obj);
    var clean = stripCommittedDataBlock(existing);
    var source = clean ? clean + "\n\n" + buildCommittedDataBlock(data) : buildCommittedDataBlock(data);
    try {
      if (state.bridge && typeof state.bridge.selectObjectByInfo === "function") {
        state.bridge.selectObjectByInfo(info);
      }
      if (state.bridge && typeof state.bridge.setUpdateScriptOnSelection === "function") {
        var selectionResult = state.bridge.setUpdateScriptOnSelection(source);
        if (!committedWriteFailed(selectionResult)) return selectionResult || { ok: true };
        result = selectionResult;
      }
    } catch(eSelectionWrite) {
      result = { ok: false, reason: eSelectionWrite && eSelectionWrite.message ? eSelectionWrite.message : String(eSelectionWrite) };
    }

    return result || { ok: false, reason: "target not found" };
  }

  function setUpdateScript(source){
    if (!state.rootInfo) return { ok: false, reason: "root information missing" };
    try {
      if (state.bridge && typeof state.bridge.setUpdateScriptByInfo === "function") return state.bridge.setUpdateScriptByInfo(state.rootInfo, source);
    } catch(eByInfo) {}
    try {
      if (state.bridge && typeof state.bridge.selectObjectByInfo === "function") state.bridge.selectObjectByInfo(state.rootInfo);
    } catch(eSelect) {}
    try {
      if (state.bridge && typeof state.bridge.setUpdateScriptOnSelection === "function") return state.bridge.setUpdateScriptOnSelection(source);
    } catch(eSelection) {
      return { ok: false, reason: eSelection && eSelection.message ? eSelection.message : String(eSelection) };
    }
    return { ok: false, reason: "LukeToolsBridge has no update script writer" };
  }

  function stripExistingBlock(source){
    var text = s(source);
    var markerPairs = [
      { start: BLOCK_START, end: BLOCK_END },
      { start: "// LukeTools Native Clip Stack Commit GPT v3 BEGIN", end: "// LukeTools Native Clip Stack Commit GPT v3 END" },
      { start: "// LukeTools Native Clip Stack Commit GPT v4 BEGIN", end: "// LukeTools Native Clip Stack Commit GPT v4 END" }
    ];
    var changed = true;
    while (changed) {
      changed = false;
      markerPairs.forEach(function(pair){
        var start = text.indexOf(pair.start);
        if (start === -1) return;
        var end = text.indexOf(pair.end, start);
        if (end === -1) {
          text = text.substring(0, start);
        } else {
          text = text.substring(0, start) + text.substring(end + pair.end.length);
        }
        changed = true;
      });
    }
    return text.trim();
  }

  function parseSavedData(source){
    var text = s(source);
    var start = text.indexOf(DATA_BEGIN);
    var end;
    var body;
    var assignStart;
    var assignEnd;
    var jsonText;

    if (start === -1) return null;
    end = text.indexOf(DATA_END, start);
    if (end === -1) return null;
    body = text.substring(start + DATA_BEGIN.length, end);
    assignStart = body.indexOf("var LT_NCSA_DATA =");
    if (assignStart === -1) return null;
    assignStart += "var LT_NCSA_DATA =".length;
    assignEnd = body.lastIndexOf(";");
    if (assignEnd === -1 || assignEnd <= assignStart) return null;
    jsonText = body.substring(assignStart, assignEnd).trim();
    try {
      return JSON.parse(jsonText);
    } catch(e) {
      setStatus("The clip contains Native Commit staging data, but the saved JSON could not be read.", "warn");
      return null;
    }
  }

  function mergeSavedAnimation(saved){
    var fresh = {
      version: 1,
      root: state.rootInfo,
      nodes: {}
    };
    var savedNodes = saved && saved.nodes ? saved.nodes : {};
    var savedByPath = {};
    Object.keys(savedNodes).forEach(function(id){
      var record = savedNodes[id];
      if (record && record.pathKey) savedByPath[record.pathKey] = record;
    });

    state.nodes.forEach(function(node){
      var pathKey = JSON.stringify(node.path || []);
      var previous = savedNodes[node.id] || savedByPath[pathKey] || null;
      var base = {
        x: node.base.x,
        y: node.base.y,
        scale: node.base.scale,
        rotation: node.base.rotation,
        signX: node.base.signX,
        signY: node.base.signY
      };
      var previousKeyChannels = previous && previous.keyChannels ? previous.keyChannels : null;
      var hasAnyExplicitChannelMaps = !!(previousKeyChannels && Object.keys(previousKeyChannels).length);
      var record = {
        id: node.id,
        name: node.name,
        path: node.path,
        pathKey: pathKey,
        isRoot: node.isRoot,
        base: previous && previous.base ? previous.base : base,
        rotateMin: previous && isFinite(Number(previous.rotateMin)) ? Number(previous.rotateMin) : -359,
        rotateMax: previous && isFinite(Number(previous.rotateMax)) ? Number(previous.rotateMax) : 360,
        rotateLongestWay: !!(previous && previous.rotateLongestWay === true),
        keys: previous && previous.keys ? previous.keys : {},
        keyTypes: previous && previous.keyTypes ? previous.keyTypes : {},
        keyChannels: previousKeyChannels ? JSON.parse(JSON.stringify(previousKeyChannels)) : {},
        channelTypes: previous && previous.channelTypes ? JSON.parse(JSON.stringify(previous.channelTypes)) : {},
        channelEasing: previous && previous.channelEasing ? JSON.parse(JSON.stringify(previous.channelEasing)) : {},
        channelDefaults: previous && previous.channelDefaults ? JSON.parse(JSON.stringify(previous.channelDefaults)) : {
          move: "none",
          scale: "none",
          rotate: "none"
        }
      };

      if (record.rotateMin >= record.rotateMax) {
        record.rotateMin = -359;
        record.rotateMax = 360;
      }

      CHANNELS.forEach(function(channel){
        if (!TRANSITIONS[record.channelDefaults[channel]] || record.channelDefaults[channel] === "hold") {
          record.channelDefaults[channel] = "none";
        }
      });

      Object.keys(record.keys || {}).forEach(function(frame){
        var hadExplicitFrameMap = !!(
          hasAnyExplicitChannelMaps &&
          previousKeyChannels &&
          Object.prototype.hasOwnProperty.call(previousKeyChannels, frame)
        );

        if (!record.keyChannels[frame]) {
          /* Legacy StickRig data had no per-channel map, so its old combined
             transform key represented all three channels. New/Reanimate data
             supplies an explicit sparse map; missing channels must remain off. */
          record.keyChannels[frame] = hadExplicitFrameMap
            ? {}
            : (hasAnyExplicitChannelMaps ? {} : { move: true, scale: true, rotate: true });
        }
        if (!record.channelTypes[frame]) record.channelTypes[frame] = {};
        if (!record.channelEasing[frame]) record.channelEasing[frame] = {};

        CHANNELS.forEach(function(channel){
          if (record.keyChannels[frame][channel] === undefined) {
            record.keyChannels[frame][channel] = hasAnyExplicitChannelMaps ? false : true;
          }
          if (!record.keyChannels[frame][channel]) {
            delete record.channelTypes[frame][channel];
            delete record.channelEasing[frame][channel];
            return;
          }
          if (!record.channelTypes[frame][channel]) {
            record.channelTypes[frame][channel] = record.keyTypes[frame] === "hold" ? "hold" : "auto";
          }
          if (!TRANSITIONS[record.channelEasing[frame][channel]] || record.channelEasing[frame][channel] === "hold") {
            record.channelEasing[frame][channel] = record.channelDefaults[channel];
          }
        });

        if (!CHANNELS.some(function(channel){ return !!record.keyChannels[frame][channel]; })) {
          delete record.keys[frame];
          delete record.keyTypes[frame];
          delete record.keyChannels[frame];
          delete record.channelTypes[frame];
          delete record.channelEasing[frame];
        }
      });

      fresh.nodes[node.id] = record;
      node.base = record.base;
    });
    state.anim = fresh;
  }

  function getNodeRecord(id){
    return state.anim && state.anim.nodes ? state.anim.nodes[id] : null;
  }

  function ensureChannelMaps(record){
    if (!record) return;
    if (!record.keyChannels) record.keyChannels = {};
    if (!record.channelTypes) record.channelTypes = {};
    if (!record.channelEasing) record.channelEasing = {};
    if (!record.channelDefaults) record.channelDefaults = { move: "none", scale: "none", rotate: "none" };
  }

  function channelForProperty(property){
    if (property === "x" || property === "y") return "move";
    if (property === "scale") return "scale";
    return "rotate";
  }

  function channelKeyExists(record, frame, channel){
    ensureChannelMaps(record);
    var map = record && record.keyChannels ? record.keyChannels[String(frame)] : null;
    return !!(map && map[channel]);
  }

  function channelKeyFrames(record, channel){
    ensureChannelMaps(record);
    return Object.keys(record.keyChannels || {})
      .filter(function(frame){ return record.keyChannels[frame] && record.keyChannels[frame][channel] && record.keys && record.keys[frame]; })
      .map(function(frame){ return Number(frame); })
      .filter(function(frame){ return isFinite(frame); })
      .sort(function(a, b){ return a - b; });
  }

  function channelType(record, frame, channel){
    ensureChannelMaps(record);
    var types = record.channelTypes[String(frame)] || {};
    return types[channel] === "hold" ? "hold" : "auto";
  }

  function channelEasing(record, frame, channel){
    ensureChannelMaps(record);
    var easing = record.channelEasing[String(frame)] && record.channelEasing[String(frame)][channel];
    if (!TRANSITIONS[easing] || easing === "hold") easing = record.channelDefaults[channel] || "none";
    if (!TRANSITIONS[easing] || easing === "hold") easing = "none";
    return easing;
  }

  function transitionInfo(record, frame, channel){
    if (channelType(record, frame, channel) === "hold") return TRANSITIONS.hold;
    return TRANSITIONS[channelEasing(record, frame, channel)] || TRANSITIONS.none;
  }

  function setChannelKeyMeta(record, frame, channel, type){
    ensureChannelMaps(record);
    var key = String(frame);
    var frameNumber = Math.max(1, Math.round(n(frame, 1)));

    if (frameNumber > 1 && record.keys && record.keys["1"] && channelKeyFrames(record, channel).length === 0) {
      if (!record.keyChannels["1"]) record.keyChannels["1"] = {};
      if (!record.channelTypes["1"]) record.channelTypes["1"] = {};
      if (!record.channelEasing["1"]) record.channelEasing["1"] = {};
      record.keyChannels["1"][channel] = true;
      record.channelTypes["1"][channel] = "auto";
      record.channelEasing["1"][channel] = record.channelDefaults[channel] || "none";
      record.keyTypes["1"] = "auto";
    }

    if (!record.keyChannels[key]) record.keyChannels[key] = {};
    if (!record.channelTypes[key]) record.channelTypes[key] = {};
    if (!record.channelEasing[key]) record.channelEasing[key] = {};
    record.keyChannels[key][channel] = true;
    record.channelTypes[key][channel] = type === "hold" ? "hold" : "auto";
    record.channelEasing[key][channel] = record.channelDefaults[channel] || "none";
    var active = CHANNELS.filter(function(name){ return record.keyChannels[key] && record.keyChannels[key][name]; });
    record.keyTypes[key] = active.length && active.every(function(name){ return record.channelTypes[key][name] === "hold"; }) ? "hold" : "auto";
  }

  function removeChannelKeyMeta(record, frame, channel){
    ensureChannelMaps(record);
    var key = String(frame);
    if (record.keyChannels[key]) delete record.keyChannels[key][channel];
    if (record.channelTypes[key]) delete record.channelTypes[key][channel];
    if (record.channelEasing[key]) delete record.channelEasing[key][channel];
    var remains = CHANNELS.some(function(name){ return record.keyChannels[key] && record.keyChannels[key][name]; });
    if (!remains) {
      delete record.keyChannels[key];
      delete record.channelTypes[key];
      delete record.channelEasing[key];
      delete record.keys[key];
      if (record.keyTypes) delete record.keyTypes[key];
    } else {
      var active = CHANNELS.filter(function(name){ return record.keyChannels[key] && record.keyChannels[key][name]; });
      record.keyTypes[key] = active.every(function(name){ return record.channelTypes[key][name] === "hold"; }) ? "hold" : "auto";
    }
  }

  function previousChannelValues(record, frame, channel){
    var frames = channelKeyFrames(record, channel).filter(function(value){ return value < frame; });
    var pose = frames.length ? copyPose(record.keys[String(frames[frames.length - 1])]) : copyPose(record.base || {});
    if (channel === "move") return { x: pose.x, y: pose.y };
    if (channel === "scale") return { scale: pose.scale };
    return { rotation: pose.rotation };
  }

  function applyHeldChannelValues(pose, values, channel){
    if (channel === "move") {
      pose.x = values.x;
      pose.y = values.y;
    } else if (channel === "scale") {
      pose.scale = values.scale;
    } else {
      pose.rotation = values.rotation;
    }
    return pose;
  }

  function currentChannelStatus(record, frame, channel){
    if (!channelKeyExists(record, frame, channel)) return { label: "No key", keyed: false, info: TRANSITIONS.none };
    var type = channelType(record, frame, channel);
    var info = transitionInfo(record, frame, channel);
    return {
      label: type === "hold" ? ("Hold " + frame) : (info.label + " " + frame),
      keyed: true,
      info: info
    };
  }

  function sortedKeyFrames(record){
    if (!record || !record.keys) return [];
    return Object.keys(record.keys)
      .map(function(key){ return Number(key); })
      .filter(function(frame){ return isFinite(frame); })
      .sort(function(a, b){ return a - b; });
  }

  function copyPose(pose){
    return {
      x: round3(n(pose && pose.x, 0)),
      y: round3(n(pose && pose.y, 0)),
      scale: round3(Math.max(0.01, n(pose && pose.scale, 1))),
      rotation: round3(n(pose && pose.rotation, 0))
    };
  }

  function easeValue(t, easing){
    var value = clamp(n(t, 0), 0, 1);
    if (easing === "in") return value * value;
    if (easing === "out") return 1 - Math.pow(1 - value, 2);
    if (easing === "in-out") {
      return value < 0.5 ? 2 * value * value : 1 - Math.pow(-2 * value + 2, 2) / 2;
    }
    return value;
  }

  function canonicalRotationAngle(value){
    var angle = n(value, 0);
    angle = ((angle + 180) % 360 + 360) % 360 - 180;
    return round3(angle);
  }

  function shortestRotationDelta(fromValue, toValue){
    var from = canonicalRotationAngle(fromValue);
    var to = canonicalRotationAngle(toValue);
    var delta = ((to - from + 180) % 360 + 360) % 360 - 180;

    /* At exactly 180 degrees both directions are equally short. Preserve the
       direction implied by the user's entered values rather than arbitrarily
       reversing it. */
    if (Math.abs(Math.abs(delta) - 180) < 0.0001) {
      delta = n(toValue, 0) - n(fromValue, 0) < 0 ? -180 : 180;
    }
    return round3(delta);
  }

  function nextRotationPathValue(currentEffective, rawTarget, longestWay){
    var shortDelta = shortestRotationDelta(currentEffective, rawTarget);
    if (!longestWay) return round3(currentEffective + shortDelta);

    /* Longest Way deliberately takes the opposite route around the dial. A
       zero-degree visual change becomes one complete turn, with the direction
       taken from the user's entered target when possible. */
    var longDelta;
    if (Math.abs(shortDelta) < 0.0001) {
      longDelta = n(rawTarget, 0) - n(currentEffective, 0) < 0 ? -360 : 360;
    } else {
      longDelta = shortDelta > 0 ? shortDelta - 360 : shortDelta + 360;
    }
    return round3(currentEffective + longDelta);
  }

  function rotationPathValues(record){
    var frames = channelKeyFrames(record, "rotate");
    var longestWay = !!(record && record.rotateLongestWay === true);
    var baseRaw = n(record && record.base && record.base.rotation, 0);
    var effectiveBase = longestWay ? round3(baseRaw) : canonicalRotationAngle(baseRaw);
    var values = { base: effectiveBase, keys: {} };
    var current = effectiveBase;

    frames.forEach(function(frameNumber){
      var rawPose = record.keys && record.keys[String(frameNumber)] ? record.keys[String(frameNumber)] : null;
      var rawTarget = n(rawPose && rawPose.rotation, current);

      /* A key at frame 1 defines the starting visual orientation. In shortest
         mode 360 is therefore stored to Wick as 0. In longest mode its entered
         numeric value is retained so 360 to 15 becomes a deliberate -345 turn. */
      if (frameNumber === 1) {
        current = longestWay ? round3(rawTarget) : canonicalRotationAngle(rawTarget);
      } else {
        current = nextRotationPathValue(current, rawTarget, longestWay);
      }
      values.keys[String(frameNumber)] = current;
    });

    return values;
  }

  function channelPoseAtFrame(record, frame, channel){
    var base = copyPose(record.base || {});
    var rotationPath = channel === "rotate" ? rotationPathValues(record) : null;
    var frames = channelKeyFrames(record, channel);
    var previousFrame = null;
    var nextFrame = null;
    var exact = null;
    var i;

    for (i = 0; i < frames.length; i += 1) {
      if (frames[i] === frame) { exact = frames[i]; break; }
      if (frames[i] < frame) previousFrame = frames[i];
      if (frames[i] > frame) { nextFrame = frames[i]; break; }
    }

    function pick(pose, frameNumber, isBase){
      var source = pose || base;
      if (channel === "move") return { x: n(source.x, base.x), y: n(source.y, base.y) };
      if (channel === "scale") return { scale: n(source.scale, base.scale) };
      if (isBase) return { rotation: n(rotationPath && rotationPath.base, base.rotation) };
      if (rotationPath && frameNumber !== null && rotationPath.keys[String(frameNumber)] !== undefined) {
        return { rotation: n(rotationPath.keys[String(frameNumber)], source.rotation) };
      }
      return { rotation: n(source.rotation, base.rotation) };
    }

    if (exact !== null) return pick(record.keys[String(exact)], exact, false);
    if (!frames.length) return pick(base, null, true);
    if (previousFrame === null && nextFrame !== null) {
      previousFrame = 1;
      var startPose = pick(base, null, true);
      var endPose = pick(record.keys[String(nextFrame)], nextFrame, false);
      if (frame <= previousFrame || nextFrame <= previousFrame) return startPose;
      var earlyT = easeValue((frame - previousFrame) / (nextFrame - previousFrame), "none");
      if (channel === "move") return { x: round3(startPose.x + (endPose.x - startPose.x) * earlyT), y: round3(startPose.y + (endPose.y - startPose.y) * earlyT) };
      if (channel === "scale") return { scale: round3(startPose.scale + (endPose.scale - startPose.scale) * earlyT) };
      return { rotation: round3(startPose.rotation + (endPose.rotation - startPose.rotation) * earlyT) };
    }
    if (previousFrame !== null && nextFrame === null) return pick(record.keys[String(previousFrame)], previousFrame, false);
    if (previousFrame === null || nextFrame === null) return pick(base, null, true);

    var from = pick(record.keys[String(previousFrame)], previousFrame, false);
    var to = pick(record.keys[String(nextFrame)], nextFrame, false);
    var type = channelType(record, previousFrame, channel);
    var easing = type === "hold" ? "hold" : channelEasing(record, previousFrame, channel);
    var rawT = clamp((frame - previousFrame) / (nextFrame - previousFrame), 0, 1);
    /* Linear is a true continuous interpolation. Only an explicitly held
       key keeps its previous channel value until the following key. */
    var t = type === "hold" ? 0 : easeValue(rawT, easing === "none" ? "none" : easing);

    if (channel === "move") return { x: round3(from.x + (to.x - from.x) * t), y: round3(from.y + (to.y - from.y) * t) };
    if (channel === "scale") return { scale: round3(from.scale + (to.scale - from.scale) * t) };
    return { rotation: round3(from.rotation + (to.rotation - from.rotation) * t) };
  }

  function interpolatePose(record, frame){
    if (!record) return { x: 0, y: 0, scale: 1, rotation: 0 };
    var move = channelPoseAtFrame(record, frame, "move");
    var scale = channelPoseAtFrame(record, frame, "scale");
    var rotate = channelPoseAtFrame(record, frame, "rotate");
    return {
      x: round3(move.x),
      y: round3(move.y),
      scale: round3(scale.scale),
      rotation: round3(rotate.rotation)
    };
  }

  function previousHeldPose(record, frame){
    if (!record) return { x: 0, y: 0, scale: 1, rotation: 0 };
    var frames = sortedKeyFrames(record).filter(function(value){ return value < frame; });
    if (!frames.length) return copyPose(record.base || {});
    return copyPose(record.keys[String(frames[frames.length - 1])]);
  }

  function ensureKeyAtFrame(node, frame, keyType){
    var record = getNodeRecord(node.id);
    if (!record) return null;
    if (!record.keyTypes) record.keyTypes = {};
    var key = String(frame);
    var type = keyType === "hold" ? "hold" : "auto";

    if (!record.keys[key]) {
      if (!sortedKeyFrames(record).length && frame > 1) {
        record.keys["1"] = copyPose(record.base);
        record.keyTypes["1"] = "auto";
      }
      record.keys[key] = type === "hold" ? previousHeldPose(record, frame) : interpolatePose(record, frame);
    }

    if (keyType) record.keyTypes[key] = type;
    if (!record.keyTypes[key]) record.keyTypes[key] = "auto";
    return record.keys[key];
  }

  function keyTypeAtCurrentFrame(id){
    var record = getNodeRecord(id);
    if (!record || !record.keys || !record.keys[String(state.frame)]) return "";
    return record.keyTypes && record.keyTypes[String(state.frame)] === "hold" ? "hold" : "auto";
  }

  function keyExistsAtCurrentFrame(id){
    var record = getNodeRecord(id);
    return !!(record && record.keys && record.keys[String(state.frame)]);
  }

  function cardPose(id){
    var record = getNodeRecord(id);
    return interpolatePose(record, state.frame);
  }

  function rotationLimits(record){
    var min = record && isFinite(Number(record.rotateMin)) ? Number(record.rotateMin) : -359;
    var max = record && isFinite(Number(record.rotateMax)) ? Number(record.rotateMax) : 360;
    if (min >= max) {
      min = -359;
      max = 360;
    }
    return { min: round3(min), max: round3(max), step: 0.5 };
  }

  function rangeConfig(property){
    if (property === "x" || property === "y") return { min: -3000, max: 3000, step: 1 };
    if (property === "scale") return { min: 1, max: 400, step: 1 };
    return { min: -360, max: 360, step: 1 };
  }

  function displayValue(property, value){
    if (property === "scale") return round3(value * 100);
    return round3(value);
  }

  function poseValue(property, display){
    if (property === "scale") return Math.max(0.01, n(display, 100) / 100);
    return n(display, 0);
  }

  function renderAll(){

      $("cards").innerHTML = "";
      return;
    }

    $("empty").style.display = "none";
    $("cards").innerHTML = visibleNodes.map(renderCard).join("");
    bindCardEvents();
  }

  function renderChannelPanel(node, record, channel, pose, disabled){
    var status = currentChannelStatus(record, state.frame, channel);
    var heldAtFrame = status.keyed && channelType(record, state.frame, channel) === "hold";
    var easing = heldAtFrame ? "hold" : channelEasing(record, state.frame, channel);
    if (!status.keyed) easing = record.channelDefaults[channel] || "none";
    var info = status.keyed ? status.info : (TRANSITIONS[easing] || TRANSITIONS.none);
    var body = "";
    var isOpen = state.openChannels[node.id] === channel;

    if (channel === "move") {
      body = renderControl(node, "x", "X", pose.x, disabled) + renderControl(node, "y", "Y", pose.y, disabled);
    } else if (channel === "scale") {
      body = renderControl(node, "scale", "%", pose.scale, disabled);
    } else {
      body = renderRotateControl(node, pose.rotation, disabled);
    }

    return "" +

  function dialRadiansForValue(value){
    return (n(value, 0) - 90) * Math.PI / 180;
  }

  function dialPoint(radius, value){
    var rad = dialRadiansForValue(value);
    return {
      x: 90 + Math.cos(rad) * radius,
      y: 90 + Math.sin(rad) * radius
    };
  }

  function dialQuickValues(record){
    var limits = rotationLimits(record);
    var min = limits.min;
    var max = limits.max;
    var values = [];
    var canonical = [0, 45, 90, 135, 180, -135, -90, -45];
    canonical.forEach(function(value){
      if (value >= min && value <= max && values.indexOf(value) === -1) values.push(value);
    });
    [min, max, round3((min + max) / 2)].forEach(function(value){
      value = round3(value);
      if (value >= min && value <= max && values.indexOf(value) === -1) values.push(value);
    });
    if (values.length < 5) {
      var count = 5;
      if (max === min) values = [min];
      else {
        values = [];
        for (var i = 0; i < count; i += 1) {
          values.push(round3(min + ((max - min) * i / (count - 1))));
        }
      }
    }
    values.sort(function(a, b){ return dialRadiansForValue(a) - dialRadiansForValue(b); });
    return values;
  }

  function dialValueFromClientPosition(clientX, clientY, area, record){
    var rect = area.getBoundingClientRect();
    var cx = rect.left + rect.width / 2;
    var cy = rect.top + rect.height / 2;
    var dx = clientX - cx;
    var dy = clientY - cy;
    var angle = Math.atan2(dy, dx) * 180 / Math.PI + 90;
    if (angle > 180) angle -= 360;
    if (angle < -180) angle += 360;
    var limits = rotationLimits(record);
    return clamp(round3(angle), limits.min, limits.max);
  }

  function normaliseDialAngle(value){
    var angle = n(value, 0);
    angle = ((angle + 180) % 360 + 360) % 360 - 180;
    return round3(angle);
  }

  function nearestEquivalentDialAngle(target, reference, limits, preserveExplicitTurn){
    var rawTarget = n(target, 0);
    var current = n(reference, 0);
    var min = n(limits && limits.min, -359);
    var max = n(limits && limits.max, 360);

    /* A labelled endpoint such as 360 or -359 is an intentional absolute
       turn request. Normal quick-set points and donut clicks use the nearest
       equivalent angle so they cannot accidentally create a full spin. */
    if (preserveExplicitTurn && Math.abs(rawTarget) > 180) {
      return round3(clamp(rawTarget, min, max));
    }

    var base = normaliseDialAngle(rawTarget);
    var firstK = Math.floor((min - base) / 360) - 1;
    var lastK = Math.ceil((max - base) / 360) + 1;
    var best = null;
    var bestDistance = Infinity;
    var bestTargetDistance = Infinity;

    for (var k = firstK; k <= lastK; k += 1) {
      var candidate = base + k * 360;
      if (candidate < min - 0.0001 || candidate > max + 0.0001) continue;
      var distance = Math.abs(candidate - current);
      var targetDistance = Math.abs(candidate - rawTarget);
      if (distance < bestDistance - 0.0001 ||
          (Math.abs(distance - bestDistance) <= 0.0001 && targetDistance < bestTargetDistance)) {
        best = candidate;
        bestDistance = distance;
        bestTargetDistance = targetDistance;
      }
    }

    if (best === null) {
      var nearestUnclamped = base + Math.round((current - base) / 360) * 360;
      best = clamp(nearestUnclamped, min, max);
    }

    return round3(best);
  }

  function setRotationFromDial(id, rawValue, commit, area, sourceMode){
    var node = state.nodeMap[id];
    var record = getNodeRecord(id);
    if (!node || !record) return;
    state.activeId = id;
    state.frame = currentProjectFrame();

    var existingKey = record.keys && record.keys[String(state.frame)];
    var referencePose = existingKey ? copyPose(existingKey) : interpolatePose(record, state.frame);
    var limits = rotationLimits(record);
    var preserveExplicitTurn = sourceMode === 'quick' && Math.abs(n(rawValue, 0)) > 180;
    var nextValue = nearestEquivalentDialAngle(
      rawValue,
      n(referencePose.rotation, record.base && record.base.rotation),
      limits,
      preserveExplicitTurn
    );

    var key = ensureKeyAtFrame(node, state.frame, 'auto');
    key.rotation = nextValue;
    setChannelKeyMeta(record, state.frame, 'rotate', 'auto');
    applyTransformToObject(node, key, commit);
    queueNodeNativeTweenSync(node, record, commit);
    if (area) updateRotateDialVisual(area, record, nextValue);
    if (commit) {
      renderAll();
      saveStagingData(false);
      syncStickRigProxyLayers();
    }
  }

  function bindCardEvents(){
    var cards = $("cards");
    cards.onclick = function(event){
      var target = event.target;
      var action = target && target.getAttribute ? target.getAttribute("data-action") : "";
      var id = target && target.getAttribute ? target.getAttribute("data-id") : "";
      var channel = target && target.getAttribute ? target.getAttribute("data-channel") : "";
      if (!action || !id) return;
      if (action === "toggle") { setOpenCard(state.openIds[id] ? "" : id); renderAll(); return; }
      if (action === "toggleChannel") { setOpenChannel(id, channel); renderAll(); return; }
      if (action === "setChannelKey") { setHeldChannelKey(id, channel); return; }
      if (action === "animate") { armNode(id); return; }
      if (action === "deleteChannelKey") { deleteCurrentChannelKey(id, channel); return; }
      if (action === "resetBase") { resetCurrentKeyToBase(id); return; }
      if (action === "rotate-quick") {
        setRotationFromDial(id, n(target.getAttribute("data-angle"), 0), true, target.closest('.rotateDialArea'), 'quick');
        return;
      }
    };

    cards.onpointerdown = function(event){
      var target = event.target;
      if (!target || !target.closest) return;
      if (target.closest('[data-action="rotate-quick"]')) return;
      var area = target.closest('.rotateDialArea');
      if (!area) return;
      var id = area.getAttribute('data-id');
      if (!id) return;
      var card = area.closest('.partCard');
      if (!card || card.className.indexOf('open') === -1) return;
      event.preventDefault();
      event.stopPropagation();
      var record = getNodeRecord(id);
      if (!record) return;
      state.rotateDialDrag = { id: id, area: area, pointerId: event.pointerId };
      try { area.setPointerCapture(event.pointerId); } catch(eCapture) {}
      setRotationFromDial(id, dialValueFromClientPosition(event.clientX, event.clientY, area, record), false, area);
    };

    cards.onpointermove = function(event){
      var drag = state.rotateDialDrag;
      if (!drag || !drag.area) return;
      var record = getNodeRecord(drag.id);
      if (!record) return;
      setRotationFromDial(drag.id, dialValueFromClientPosition(event.clientX, event.clientY, drag.area, record), false, drag.area);
    };

    cards.onpointerup = function(event){
      var drag = state.rotateDialDrag;
      if (!drag || !drag.area) return;
      var record = getNodeRecord(drag.id);
      if (!record) { state.rotateDialDrag = null; return; }
      try { drag.area.releasePointerCapture(drag.pointerId); } catch(eRelease) {}
      setRotationFromDial(drag.id, dialValueFromClientPosition(event.clientX, event.clientY, drag.area, record), true, drag.area);
      state.rotateDialDrag = null;
    };

    cards.onpointercancel = function(){
      state.rotateDialDrag = null;
    };

    cards.oninput = function(event){
      var target = event.target;
      var action = target && target.getAttribute ? target.getAttribute("data-action") : "";
      var id = target && target.getAttribute ? target.getAttribute("data-id") : "";
      var property = target && target.getAttribute ? target.getAttribute("data-property") : "";
      var channel = target && target.getAttribute ? target.getAttribute("data-channel") : "";
      if (!id) return;
      if (action === "transition") {
        updateChannelTransition(id, channel, target.value, false);
        return;
      }
      if (action === "rotate-min" || action === "rotate-max") {
        updateRotationLimit(id, action === "rotate-min" ? "min" : "max", target.value, false);
        return;
      }
      if ((action !== "range" && action !== "number") || !property) return;
      syncPairedControl(target, action);
      updateKeyFromControl(id, property, target.value, false);
    };

    cards.onchange = function(event){
      var target = event.target;
      var action = target && target.getAttribute ? target.getAttribute("data-action") : "";
      var id = target && target.getAttribute ? target.getAttribute("data-id") : "";
      var property = target && target.getAttribute ? target.getAttribute("data-property") : "";
      var channel = target && target.getAttribute ? target.getAttribute("data-channel") : "";
      if (!id) return;
      if (action === "transition") {
        updateChannelTransition(id, channel, target.value, true);
        return;
      }
      if (action === "rotate-min" || action === "rotate-max") {
        updateRotationLimit(id, action === "rotate-min" ? "min" : "max", target.value, true);
        return;
      }
      if (action === "rotate-longest") {
        updateRotateLongestWay(id, !!target.checked);
        return;
      }
      if ((action !== "range" && action !== "number") || !property) return;
      syncPairedControl(target, action);
      updateKeyFromControl(id, property, target.value, true);
    };
  }

  function syncPairedControl(source, action){
    var id = source.getAttribute("data-id");
    var property = source.getAttribute("data-property");
    var otherAction = action === "range" ? "number" : "range";
    var selector = '[data-action="' + otherAction + '"][data-id="' + cssEscapeAttribute(id) + '"][data-property="' + cssEscapeAttribute(property) + '"]';
    var other = $("cards").querySelector(selector);
    if (other) other.value = source.value;
  }

  function refreshRotationControl(id, record, value){
    var card = $("cards").querySelector('.partCard[data-id="' + cssEscapeAttribute(id) + '"]');
    if (!card || !record) return;
    var limits = rotationLimits(record);
    var number = card.querySelector('.rotateCurrentValue');
    var minField = card.querySelector('[data-action="rotate-min"]');
    var maxField = card.querySelector('[data-action="rotate-max"]');
    var clamped = clamp(n(value, 0), limits.min, limits.max);
    if (number) {
      number.min = limits.min;
      number.max = limits.max;
      number.value = clamped;
    }
    if (minField) minField.value = limits.min;
    if (maxField) maxField.value = limits.max;
    var area = card.querySelector('.rotateDialArea');
    if (area) updateRotateDialVisual(area, record, clamped);
  }

  function updateRotationLimit(id, which, rawValue, commit){
    var record = getNodeRecord(id);
    var node = state.nodeMap[id];
    if (!record || !node) return;
    if (state.activeId !== id) armNode(id);

    var limits = rotationLimits(record);
    var value = n(rawValue, which === "min" ? limits.min : limits.max);

    if (which === "min") {
      record.rotateMin = round3(value);
      if (record.rotateMin >= limits.max) record.rotateMin = round3(limits.max - 0.5);
      record.rotateMax = limits.max;
    } else {
      record.rotateMax = round3(value);
      if (record.rotateMax <= limits.min) record.rotateMax = round3(limits.min + 0.5);
      record.rotateMin = limits.min;
    }

    var updatedLimits = rotationLimits(record);
    state.frame = currentProjectFrame();
    var currentPose = interpolatePose(record, state.frame);
    var clampedRotation = clamp(n(currentPose.rotation, 0), updatedLimits.min, updatedLimits.max);

    if (clampedRotation !== currentPose.rotation) {
      var key = ensureKeyAtFrame(node, state.frame, "auto");
      key.rotation = round3(clampedRotation);
      setChannelKeyMeta(record, state.frame, "rotate", "auto");
      applyTransformToObject(node, key, commit);
    }

    queueNodeNativeTweenSync(node, record, commit);
    renderAll();
    if (commit) {
      saveStagingData(false);
      syncStickRigProxyLayers();
    } else {
      queueSave();
      queueProxySync();
    }
  }

      " rotation path selected for " + node.name + ".",
      "good"
    );
  }

  function updateChannelTransition(id, channel, value, commit){
    var record = getNodeRecord(id);
    if (!record || CHANNELS.indexOf(channel) === -1) return;
    var selected = TRANSITIONS[value] ? value : "none";
    ensureChannelMaps(record);
    state.frame = currentProjectFrame();
    var frameKey = String(state.frame);
    var hasKey = channelKeyExists(record, state.frame, channel);

    if (selected === "hold") {
      if (hasKey) {
        if (!record.channelTypes[frameKey]) record.channelTypes[frameKey] = {};
        if (!record.channelEasing[frameKey]) record.channelEasing[frameKey] = {};
        record.channelTypes[frameKey][channel] = "hold";
        record.channelEasing[frameKey][channel] = "none";
      }
    } else {
      record.channelDefaults[channel] = selected;
      if (hasKey) {
        if (!record.channelTypes[frameKey]) record.channelTypes[frameKey] = {};
        if (!record.channelEasing[frameKey]) record.channelEasing[frameKey] = {};
        record.channelTypes[frameKey][channel] = "auto";
        record.channelEasing[frameKey][channel] = selected;
      }
    }

    if (hasKey) {
      var active = CHANNELS.filter(function(name){
        return record.keyChannels[frameKey] && record.keyChannels[frameKey][name];
      });
      record.keyTypes[frameKey] = active.length && active.every(function(name){
        return record.channelTypes[frameKey] && record.channelTypes[frameKey][name] === "hold";
      }) ? "hold" : "auto";
    }

    if (hasKey) queueNodeNativeTweenSync(state.nodeMap[id], record, commit);
    renderAll();
    if (commit) {
      saveStagingData(false);
      syncStickRigProxyLayers();
    } else {
      queueSave();
      queueProxySync();
    }
  }

    if (commit) {
      saveStagingData(false);
      syncStickRigProxyLayers();
    } else {
      queueSave();

    }
  }

  function deleteCurrentChannelKey(id, channel){
    var record = getNodeRecord(id);

    removeChannelKeyMeta(record, state.frame, channel);

    var nativeSync = syncNodeNativeTweensInstantly(node, record);

    renderAll();
    var stagingSaved = saveStagingData(false);
    syncStickRigProxyLayers();

    if (!nativeSync.ok) {
      setStatus(
        "The " + channel + " key was removed from StickRig, but the real clip timeline could not be refreshed: " +
        nativeSync.reason + ".",
        "warn"
      );
    } else if (stagingSaved === false) {
      setStatus(
        "The " + channel + " key and native keyframe were deleted, but editable Reanimate data could not be saved.",
        "warn"
      );
    } else {
      setStatus(
        "Deleted " + channel + " key for " + node.name + " at frame " + state.frame +
        (nativeSync.skipped ? "." : " and removed its native keyframe data on the real owning timeline immediately."),
        "good"
      );
    }
  }

  function resetCurrentKeyToBase(id){

    applyTransformToObject(node, record.keys[String(state.frame)], true);

  }

  function clearAllKeys(){
    if (!state.nodes.length) return;
    clearTimeout(state.saveTimer);
    clearTimeout(state.proxySyncTimer);
    state.saveTimer = null;
    state.proxySyncTimer = null;

    animatableNodes().forEach(function(node){
      var editableRecord = state.anim.nodes[node.id];
      if (!editableRecord) return;
      editableRecord.keys = {};
      editableRecord.keyTypes = {};
      editableRecord.keyChannels = {};
      editableRecord.channelTypes = {};
      editableRecord.channelEasing = {};
    });

    var nativeSyncFailures = [];
    animatableNodes().forEach(function(node){
      var record = getNodeRecord(node.id);
      if (!record) return;
      applyTransformToObject(node, record.base, false);
      var nativeSync = syncNodeNativeTweensInstantly(node, record);
      if (!nativeSync.ok) nativeSyncFailures.push(node.name + ": " + nativeSync.reason);
    });

    removeTimelineOverlay();
    removeStickRigProxyLayers();
    removeStaleStickRigProxyLayers(getMainTimeline());
    clearStagingBlock();
    state.timelineOverlay.selected = null;
    state.openChannels = {};
    markProjectChanged();
    renderAll();
    if (nativeSyncFailures.length) {
      setStatus(
        "Cleared all StickRig channel keys, but some stage clips could not be refreshed: " + nativeSyncFailures.join("; "),
        "warn"
      );
    } else {
      setStatus(
        "Cleared all StickRig keys and removed all corresponding real Wick keyframes from the clips immediately.",
        "good"
      );
    }
  }

  function serializableAnimation(){
    var output = {
      version: 1,
      toolVersion: VERSION,
      root: state.rootInfo,
      nodes: {}
    };
    Object.keys(state.anim.nodes).forEach(function(id){
      var record = state.anim.nodes[id];
      ensureChannelMaps(record);
      output.nodes[id] = {
        id: record.id,
        name: record.name,
        path: record.path,
        pathKey: record.pathKey,
        isRoot: !!record.isRoot,
        rotateMin: rotationLimits(record).min,
        rotateMax: rotationLimits(record).max,
        rotateLongestWay: !!record.rotateLongestWay,
        base: {
          x: round3(record.base.x),
          y: round3(record.base.y),
          scale: round3(record.base.scale),
          rotation: round3(record.base.rotation),
          signX: n(record.base.signX, 1) < 0 ? -1 : 1,
          signY: n(record.base.signY, 1) < 0 ? -1 : 1
        },
        keys: {},
        keyTypes: {},
        keyChannels: JSON.parse(JSON.stringify(record.keyChannels || {})),
        channelTypes: JSON.parse(JSON.stringify(record.channelTypes || {})),
        channelEasing: JSON.parse(JSON.stringify(record.channelEasing || {})),
        channelDefaults: JSON.parse(JSON.stringify(record.channelDefaults || { move: "none", scale: "none", rotate: "none" }))
      };
      Object.keys(record.keys || {}).forEach(function(frame){ output.nodes[id].keys[String(frame)] = copyPose(record.keys[frame]); });
      Object.keys(record.keyTypes || {}).forEach(function(frame){ output.nodes[id].keyTypes[String(frame)] = record.keyTypes[frame] === "hold" ? "hold" : "auto"; });
    });
    return output;
  }

  function committedAnimationAtPreservedOrigin(preservedBase, rootCopy){
    var animation = cloneAnimationData(serializableAnimation());
    animation.root = getObjectInfo(rootCopy);
    var rootRecord = null;
    Object.keys(animation.nodes || {}).some(function(id){
      if (animation.nodes[id] && animation.nodes[id].isRoot) {
        rootRecord = animation.nodes[id];
        return true;
      }
      return false;
    });
    if (!rootRecord) return animation;

    rootRecord.base = rootPoseRelativeToPreservedOrigin(rootRecord.base, preservedBase);
    rootRecord.base.signX = n(preservedBase.signX, 1) < 0 ? -1 : 1;
    rootRecord.base.signY = n(preservedBase.signY, 1) < 0 ? -1 : 1;
    Object.keys(rootRecord.keys || {}).forEach(function(frame){
      rootRecord.keys[frame] = rootPoseRelativeToPreservedOrigin(rootRecord.keys[frame], preservedBase);
    });
    return animation;
  }

  function safeFileName(value){

      .replace(/[\\/:*?"<>|]+/g, "_")

  }

  function triggerTextDownload(filename, text, mimeType){
    var bridgeResult = null;

    try {
      if (state.bridge && typeof state.bridge.downloadTextFile === "function") {
        bridgeResult = state.bridge.downloadTextFile(filename, text, mimeType || "application/json");
        if (bridgeResult !== false) return true;
      }
    } catch(eBridgeDownload) {}

    try {
      if (state.bridge && typeof state.bridge.saveTextFile === "function") {
        bridgeResult = state.bridge.saveTextFile(filename, text, mimeType || "application/json");
        if (bridgeResult !== false) return true;
      }
    } catch(eBridgeSave) {}

    try {
      var blob = new Blob([text], { type: mimeType || "application/json" });
      var url = URL.createObjectURL(blob);
      var anchor = document.createElement("a");
      anchor.href = url;
      anchor.download = filename;
      anchor.style.display = "none";
      document.body.appendChild(anchor);
      anchor.click();
      setTimeout(function(){
        try { document.body.removeChild(anchor); } catch(eRemove) {}
        try { URL.revokeObjectURL(url); } catch(eRevoke) {}
      }, 1000);
      return true;
    } catch(eDownload) {
      console.error(eDownload);
      return false;
    }
  }

  function saveRigJson(){
    if (!state.root || !state.nodes.length) {
      setStatus("Load a rig first.", "warn");
      return false;
    }

    var data = buildRigJsonData();
    var filename = safeFileName(state.rootName) + "_StickRig.json";
    var json = JSON.stringify(data, null, 2);

    if (!ok) {
      setStatus("Rig JSON could not be downloaded.", "bad");
      return false;
    }

    setStatus("Saved " + filename + ".", "good");
    return true;
  }

  function saveStagingData(showSuccess){
    if (!state.root || !state.nodes.length) {
      if (showSuccess !== false) setStatus("Read a selected clip before saving staging data.", "warn");
      return false;
    }
    clearTimeout(state.saveTimer);
    state.saveTimer = null;

    /* Native Wick tweens are now the live source of truth. Automatic metadata
       saves must never interrupt key editing or report a false animation
       failure when the bridge cache is refreshing after a layer move. */
    if (showSuccess === false) return true;

    var existing = getUpdateScriptForObject(state.root);
    var clean = stripExistingBlock(existing);
    var block = buildStagingBlock();
    var finalSource = clean ? (clean + "\n\n" + block) : block;
    var result = setUpdateScriptForObject(state.root, finalSource);
    if (result && result.ok === false) {
      setStatus("Save failed: " + (result.reason || "unknown bridge error") + ". The native Wick animation was not affected.", "bad");
      return false;
    }
    state.rootInfo = getObjectInfo(state.root);
    markProjectChanged();
    setStatus("Saved editable StickRig channel data into " + state.rootName + ".", "good");
    return true;
  }

  function clearStagingBlock(){
    if (!state.root) return true;
    var existing = getExistingUpdateScript();
    var clean = stripExistingBlock(existing);
    var result = setUpdateScript(clean);
    if (result && result.ok === false) return false;
    return true;
  }

  function clearStagingBlockOnObject(obj){
    if (!obj || !state.bridge) return true;
    var info = getObjectInfo(obj);
    try {
      if (typeof state.bridge.getUpdateScriptByInfo === "function" && typeof state.bridge.setUpdateScriptByInfo === "function") {
        var existing = s(state.bridge.getUpdateScriptByInfo(info));
        var clean = stripExistingBlock(existing);
        var result = state.bridge.setUpdateScriptByInfo(info, clean);
        if (result && result.ok === false) return false;
      }
    } catch(e) {
      return false;
    }
    return true;
  }

  function parentLayerOfFrame(frame){
    return safeGet(frame, "parentLayer") || safeGet(frame, "_parentLayer") || safeGet(frame, "parent") || safeGet(frame, "_parent") || null;
  }

  function parentTimelineOfLayer(layer){
    return safeGet(layer, "parentTimeline") || safeGet(layer, "_parentTimeline") || safeGet(layer, "parent") || safeGet(layer, "_parent") || null;
  }

  function parentClipOfTimeline(timeline){
    var candidates = [
      safeGet(timeline, "parentClip"), safeGet(timeline, "_parentClip"),
      safeGet(timeline, "clip"), safeGet(timeline, "_clip"),
      safeGet(timeline, "parent"), safeGet(timeline, "_parent")
    ];
    for (var i = 0; i < candidates.length; i += 1) if (looksLikeClip(candidates[i])) return candidates[i];
    return null;
  }

  function parentClipOfObject(obj){
    var frame = getParentFrame(obj);
    if (!frame) return null;
    var layer = parentLayerOfFrame(frame);
    if (!layer) return null;
    return parentClipOfTimeline(parentTimelineOfLayer(layer));
  }

  function clipHierarchyHasNativeTweens(clip, depth){
    if (!clip || !looksLikeClip(clip) || depth > 24) return false;
    var timeline = getTimeline(clip);
    if (timeline) {
      var layers = getTimelineLayers(timeline);
      for (var i = 0; i < layers.length; i += 1) {
        var frames = getLayerFrames(layers[i]);
        for (var j = 0; j < frames.length; j += 1) {
          if (getFrameTweens(frames[j]).length) return true;
        }
      }
    }
    var children = getTimelineChildClips(clip);
    for (var k = 0; k < children.length; k += 1) {
      if (clipHierarchyHasNativeTweens(children[k], depth + 1)) return true;
    }
    return false;
  }

  function clipHasOwnNativeTweens(clip){
    if (!clip || !looksLikeClip(clip)) return false;
    var internalFrame = findStickRigMotionFrame(clip);
    if (internalFrame && getFrameTweens(internalFrame).length > 0) return true;
    return getFrameTweens(getParentFrame(clip)).length > 0;
  }

  function isCommittedWrapperCandidate(clip){
    if (!clip || !looksLikeClip(clip)) return false;
    if (parseCommittedData(getUpdateScriptForObject(clip))) return true;

    /* v54 direct-root commits and v55 internal-timeline commits are both
       readable. v55 exact commits are primarily identified by their saved
       committed-data marker. */
    if (clipHasOwnNativeTweens(clip)) return true;

    var info = getObjectInfo(clip);
    var label = (info.identifier + " " + info.name + " " + getObjectName(clip, "")).toLowerCase();
    if (label.indexOf("committed_") !== -1 && label.indexOf("wrapper") !== -1) return true;

    var children = getTimelineChildClips(clip);
    return children.length === 1 && clipHierarchyHasNativeTweens(clip, 0);
  }

  function findCommittedWrapper(selected){
    var current = selected;
    var guard = 0;
    var outermostDirectAnimatedClip = null;
    var outermostStructuralMatch = null;
    var explicitLegacyWrapper = null;

    while (current && guard < 12) {
      var manifest = parseCommittedData(getUpdateScriptForObject(current));
      if (manifest && manifest.directRoot === true) outermostDirectAnimatedClip = current;

      if (clipHasOwnNativeTweens(current)) outermostDirectAnimatedClip = current;

      var info = getObjectInfo(current);
      var label = (info.identifier + " " + info.name + " " + getObjectName(current, "")).toLowerCase();
      if (label.indexOf("committed_") !== -1 && label.indexOf("wrapper") !== -1) explicitLegacyWrapper = current;

      if (isCommittedWrapperCandidate(current)) outermostStructuralMatch = current;
      current = parentClipOfObject(current);
      guard += 1;
    }

    /* Prefer the outermost clip whose own containing frame has native keyframes.
       That is the actual committed root in v54 and prevents a same-named
       legacy wrapper from being ingested as an extra TopLeg level. */
    return outermostDirectAnimatedClip || explicitLegacyWrapper || outermostStructuralMatch;
  }

  function findCommittedRoot(container, manifest){
    /* Backward compatibility for v54 direct-root commits. */
    if (clipHasOwnNativeTweens(container) || (manifest && manifest.directRoot === true)) return container;

    /* Backward compatibility for v43-v53 wrapper commits. */
    var children = getTimelineChildClips(container);
    if (!children.length) return null;
    var target = manifest && manifest.committedRoot ? manifest.committedRoot : null;
    if (target) {
      for (var i = 0; i < children.length; i += 1) {
        var info = getObjectInfo(children[i]);
        if (target.uuid && info.uuid === target.uuid) return children[i];
        if (target.identifier && info.identifier === target.identifier) return children[i];
        if (target.name && info.name === target.name) return children[i];
      }
    }
    return children[0];
  }

  function readTweenPose(tween, fallback){
    var base = fallback || { x: 0, y: 0, scale: 1, rotation: 0, signX: 1, signY: 1 };
    var tr = safeGet(tween, "transformation") || safeGet(tween, "_transformation") || {};
    var scaleX = n(safeGet(tr, "scaleX"), n(base.scale, 1) * (n(base.signX, 1) < 0 ? -1 : 1));
    var scaleY = n(safeGet(tr, "scaleY"), n(base.scale, 1) * (n(base.signY, 1) < 0 ? -1 : 1));
    var scale = (Math.abs(scaleX) + Math.abs(scaleY)) * 0.5;
    return {
      x: round3(n(safeGet(tr, "x"), base.x)),
      y: round3(n(safeGet(tr, "y"), base.y)),
      scale: round3(scale > 0 ? scale : n(base.scale, 1)),
      rotation: round3(n(safeGet(tr, "rotation"), base.rotation)),
      signX: scaleX < 0 ? -1 : 1,
      signY: scaleY < 0 ? -1 : 1
    };
  }

  function channelPoseDiffers(a, b, channel){
    var left = a || {};
    var right = b || {};
    var epsilon = 0.0005;
    if (channel === "move") {
      return Math.abs(n(left.x, 0) - n(right.x, 0)) > epsilon ||
        Math.abs(n(left.y, 0) - n(right.y, 0)) > epsilon;
    }
    if (channel === "scale") return Math.abs(n(left.scale, 1) - n(right.scale, 1)) > epsilon;
    return Math.abs(n(left.rotation, 0) - n(right.rotation, 0)) > epsilon;
  }

  function nativeAnimationFallback(){
    var output = { version: 1, toolVersion: VERSION, root: state.rootInfo, nodes: {} };
    animatableNodes().forEach(function(node){
      var frame = findStickRigMotionFrame(node.obj) || getParentFrame(node.obj);
      var tweens = getFrameTweens(frame).slice().sort(function(a, b){
        return n(getTweenFrameNumber(a), 1) - n(getTweenFrameNumber(b), 1);
      });
      var base = readTransform(node.obj);
      var entries = [];

      tweens.forEach(function(tween){
        var frameNumber = getTweenFrameNumber(tween);
        if (frameNumber === null) return;
        var easing = s(safeGet(tween, "easingType") || "none");
        if (!TRANSITIONS[easing] || easing === "hold") easing = "none";
        entries.push({
          frame: frameNumber,
          pose: readTweenPose(tween, base),
          easing: easing
        });
      });

      /* Keep only actual Wick tween positions. A channel receives a key only
         when that property changes into or out of this tween. Constant X/Y,
         scale or rotation values are not converted into false channel keys. */
      var record = {
        id: node.id,
        name: node.name,
        path: node.path,
        pathKey: JSON.stringify(node.path || []),
        isRoot: !!node.isRoot,
        rotateMin: -359,
        rotateMax: 360,
        rotateLongestWay: false,
        base: base,
        keys: {},
        keyTypes: {},
        keyChannels: {},
        channelTypes: {},
        channelEasing: {},
        channelDefaults: { move: "none", scale: "none", rotate: "none" }
      };

      var previousRotationForPath = n(base.rotation, 0);
      entries.forEach(function(entry){
        if (Math.abs(n(entry.pose && entry.pose.rotation, previousRotationForPath) - previousRotationForPath) > 180.0001) {
          record.rotateLongestWay = true;
        }
        previousRotationForPath = n(entry.pose && entry.pose.rotation, previousRotationForPath);
      });

      entries.forEach(function(entry, index){
        var previousPose = index > 0 ? entries[index - 1].pose : base;
        var nextPose = index + 1 < entries.length ? entries[index + 1].pose : entry.pose;
        var activeChannels = {};

        CHANNELS.forEach(function(channel){
          var changesFromPrevious = channelPoseDiffers(entry.pose, previousPose, channel);
          var changesToNext = channelPoseDiffers(entry.pose, nextPose, channel);
          if (changesFromPrevious || changesToNext) activeChannels[channel] = true;
        });

        if (!CHANNELS.some(function(channel){ return !!activeChannels[channel]; })) return;
        var key = String(entry.frame);
        record.keys[key] = copyPose(entry.pose);
        record.keyTypes[key] = "auto";
        record.keyChannels[key] = activeChannels;
        record.channelTypes[key] = {};
        record.channelEasing[key] = {};
        CHANNELS.forEach(function(channel){
          if (!activeChannels[channel]) return;
          record.channelTypes[key][channel] = "auto";
          record.channelEasing[key][channel] = entry.easing;
        });
      });

      output.nodes[node.id] = record;
    });
    return output;
  }

  function alignExactAnimationToNativeTweens(exactAnimation, nativeAnimation){
    if (!exactAnimation || !exactAnimation.nodes) return nativeAnimation;
    var result = cloneAnimationData(nativeAnimation);
    var exactNodes = exactAnimation.nodes || {};
    var exactByPath = {};
    Object.keys(exactNodes).forEach(function(id){
      var item = exactNodes[id];
      if (item && item.pathKey) exactByPath[item.pathKey] = item;
    });

    Object.keys(result.nodes || {}).forEach(function(id){
      var nativeRecord = result.nodes[id];
      var exactRecord = exactNodes[id] || exactByPath[nativeRecord.pathKey] || null;
      if (!exactRecord) return;

      nativeRecord.rotateMin = isFinite(Number(exactRecord.rotateMin)) ? Number(exactRecord.rotateMin) : nativeRecord.rotateMin;
      nativeRecord.rotateMax = isFinite(Number(exactRecord.rotateMax)) ? Number(exactRecord.rotateMax) : nativeRecord.rotateMax;
      nativeRecord.rotateLongestWay = exactRecord.rotateLongestWay === true;
      nativeRecord.base = exactRecord.base ? copyPose(exactRecord.base) : nativeRecord.base;
      nativeRecord.base.signX = exactRecord.base && n(exactRecord.base.signX, 1) < 0 ? -1 : 1;
      nativeRecord.base.signY = exactRecord.base && n(exactRecord.base.signY, 1) < 0 ? -1 : 1;
      nativeRecord.channelDefaults = exactRecord.channelDefaults
        ? JSON.parse(JSON.stringify(exactRecord.channelDefaults))
        : nativeRecord.channelDefaults;

      /* Native tween frames are authoritative. Exact v43+ metadata is used
         only to restore the separate channel identity, Hold state and easing
         at those real tween positions. Metadata-only frames are not invented. */
      Object.keys(nativeRecord.keys || {}).forEach(function(frame){
        if (exactRecord.keys && exactRecord.keys[frame]) nativeRecord.keys[frame] = copyPose(exactRecord.keys[frame]);
        if (!(exactRecord.keyChannels && exactRecord.keyChannels[frame])) return;

        var explicitMap = exactRecord.keyChannels[frame];
        var sparseMap = {};
        CHANNELS.forEach(function(channel){
          if (explicitMap[channel]) sparseMap[channel] = true;
        });
        if (!CHANNELS.some(function(channel){ return !!sparseMap[channel]; })) return;

        nativeRecord.keyChannels[frame] = sparseMap;
        nativeRecord.keyTypes[frame] = exactRecord.keyTypes && exactRecord.keyTypes[frame] === "hold" ? "hold" : "auto";
        nativeRecord.channelTypes[frame] = {};
        nativeRecord.channelEasing[frame] = {};
        CHANNELS.forEach(function(channel){
          if (!sparseMap[channel]) return;
          var exactType = exactRecord.channelTypes && exactRecord.channelTypes[frame]
            ? exactRecord.channelTypes[frame][channel]
            : null;
          var exactEasing = exactRecord.channelEasing && exactRecord.channelEasing[frame]
            ? exactRecord.channelEasing[frame][channel]
            : null;
          nativeRecord.channelTypes[frame][channel] = exactType === "hold" ? "hold" : "auto";
          nativeRecord.channelEasing[frame][channel] = TRANSITIONS[exactEasing] && exactEasing !== "hold"
            ? exactEasing
            : "none";
        });
      });
    });

    result.root = exactAnimation.root || result.root;
    return result;
  }

  function committedRigDuration(container, manifest){
    if (manifest && isFinite(Number(manifest.duration))) return Math.max(1, Math.round(Number(manifest.duration)));
    if (clipHasOwnNativeTweens(container)) {
      var ownFrame = findStickRigMotionFrame(container) || getParentFrame(container);
      if (ownFrame) return Math.max(1, getFrameEnd(ownFrame));
    }
    var children = getTimelineChildClips(container);
    var frame = children.length ? getParentFrame(children[0]) : null;
    if (frame) return Math.max(1, getFrameEnd(frame));
    return getProjectDuration();
  }

  function readSelectedClip(){
    if (state.reading) return;
    state.reading = true;
    setStatus("Loading…", "");
    try {
      removeStickRigProxyLayers();
      resetReanimateState();
      resetLiveNativeState();
      if (!refreshRefs()) { setStatus("LukeToolsBridge or Wick project could not be found.", "bad"); return; }
      var root = getSelectedObject();
      if (!root) { setStatus("Select a clip first.", "warn"); return; }
      if (!looksLikeClip(root)) { setStatus("Selection is not a clip.", "warn"); return; }

      /* A committed parent contains its root animation on an internal frame.
         Loading that outer parent as an ordinary hierarchy would display the
         parent name twice and the hierarchy could be interpreted twice.
         Route exact StickRig committed parents through the same ingest used by
         Reanimate through the Reanimate ingest path. */
      var committedMarker = parseCommittedData(getUpdateScriptForObject(root));
      if (committedMarker && committedMarker.format === "LukeTools.StickRig.Committed") {
        state.reading = false;
        reanimateSelectedRig();
        return;
      }

      state.root = root;
      state.rootInfo = getObjectInfo(root);
      state.rootName = getObjectName(root, "Selected Root Clip");
      state.frame = currentProjectFrame();
      state.lastPolledFrame = state.frame;
      state.activeId = "";
      state.openIds = {};
      state.openChannels = {};
      buildHierarchy(root);
      if (!animatableNodes().length) { setStatus("The selected rig container has no nested clips to animate.", "warn"); return; }
      animatableNodes().forEach(function(node){ consolidateLiveLayersForNode(node, getProjectDuration()); });
      var stagedAnimation = parseSavedData(getUpdateScriptForObject(root));
      mergeSavedAnimation(stagedAnimation);
      state.liveNative.active = !!stagedAnimation;
      var loadMigration = stagedAnimation
        ? migrateLoadedKeysIntoRealClipTimelines()
        : { ok: true, written: 0, failures: [] };
      var firstEditable = firstAnimatableNode();
      if (firstEditable) state.openIds[firstEditable.id] = true;
      renderAll();
      $("saveBtn").disabled = false;
      $("jsonBtn").disabled = false;
      $("clearAllBtn").disabled = false;
      setupStickRigProxyLayers();
      setStatus(
        !loadMigration.ok
          ? "The rig loaded, but some real clip timelines could not be updated: " + loadMigration.failures.join(" | ")
          : ("Read " + animatableNodes().length + " nested clip" + (animatableNodes().length === 1 ? "" : "s") + " from " + state.rootName + ". The selected outer container is not animatable. " +
            (stagedAnimation
              ? "Existing StickRig keys are now native Wick tweens on the real frames that own the clip instances."
              : "Every new key is written immediately as a native Wick tween on the one real frame that owns that clip instance; Every change is written automatically.")),
        !loadMigration.ok ? "bad" : "good"
      );
    } catch(error) {
      console.error(error);
      setStatus("Read failed: " + (error && error.message ? error.message : String(error)), "bad");
    } finally {
      state.reading = false;
    }
  }

  function createTween(frameNumber, pose, base, easingType, colour){
    var W = getWickNamespace();
    if (!W || !W.Tween || !W.Transformation) throw new Error("Wick.Tween or Wick.Transformation is not available");
    var signX = n(base && base.signX, 1) < 0 ? -1 : 1;
    var signY = n(base && base.signY, 1) < 0 ? -1 : 1;
    var easing = TRANSITIONS[easingType] && easingType !== "hold" ? easingType : "none";
    var transformation = new W.Transformation({
      x: n(pose && pose.x, 0),
      y: n(pose && pose.y, 0),
      scaleX: Math.max(0.01, n(pose && pose.scale, 1)) * signX,
      scaleY: Math.max(0.01, n(pose && pose.scale, 1)) * signY,
      rotation: n(pose && pose.rotation, 0),
      opacity: 1
    });
    var tween = new W.Tween({
      playheadPosition: Math.max(1, Math.round(n(frameNumber, 1))),
      transformation: transformation,
      easingType: easing
    });
    applyTimelineColour(tween, colour || (TRANSITIONS[easing] || TRANSITIONS.none).colour);
    return tween;
  }

  function getAccessibleEditorDocuments(){
    var result = [];
    var seen = [];

    function addDocument(doc){
      if (!doc || !doc.body) return;
      for (var i = 0; i < seen.length; i += 1) {
        if (seen[i] === doc) return;
      }
      seen.push(doc);
      result.push(doc);
    }

    var currentWindow = window;
    var guard = 0;
    while (currentWindow && guard < 12) {
      try { addDocument(currentWindow.document); } catch(eDoc) {}
      var parentWindow = null;
      try { parentWindow = currentWindow.parent; } catch(eParent) { parentWindow = null; }
      if (!parentWindow || parentWindow === currentWindow) break;
      currentWindow = parentWindow;
      guard += 1;
    }

    try { if (window.top) addDocument(window.top.document); } catch(eTop) {}
    return result;
  }

  function scoreTimelineDocument(doc){
    if (!doc || !doc.body) return -Infinity;
    var score = 0;
    var viewportWidth = 0;
    var viewportHeight = 0;
    try {
      viewportWidth = Math.max(doc.documentElement ? doc.documentElement.clientWidth : 0, doc.defaultView ? doc.defaultView.innerWidth : 0);
      viewportHeight = Math.max(doc.documentElement ? doc.documentElement.clientHeight : 0, doc.defaultView ? doc.defaultView.innerHeight : 0);
    } catch(eViewport) {}
    score += Math.min(40, (viewportWidth * viewportHeight) / 100000);

    try {
      var timelineMatches = doc.querySelectorAll('[class*="timeline" i],[id*="timeline" i]');
      score += Math.min(60, timelineMatches.length * 3);
    } catch(eTimelineQuery) {}

    try {
      var bodyText = cleanInvisibleText(doc.body.innerText || doc.body.textContent || "");
      if (/\bLayer\b/.test(bodyText)) score += 12;
      if (/\bFrame\b/.test(bodyText)) score += 4;
    } catch(eBodyText) {}

    if (state.proxy && state.proxy.layers && state.proxy.layers.length) {
      state.proxy.layers.forEach(function(info){
        try {
          if (findLayerLabelElement(doc, info.displayName, info.cleanName)) score += 250;
        } catch(eLabel) {}
      });
    }

    try { if (doc === window.document) score -= 20; } catch(eCurrent) {}
    return score;
  }

  function getParentEditorDocument(){
    var docs = getAccessibleEditorDocuments();
    var best = null;
    var bestScore = -Infinity;
    docs.forEach(function(doc){
      if (!doc || !doc.body) return;
      try {
        if (doc === window.document) return;
        if (doc.getElementById("app") && doc.getElementById("cards")) return;
        if (/StickRig/i.test(doc.title || "")) return;
      } catch(eExclude) {}
      var score = scoreTimelineDocument(doc);
      try {
        if (doc.defaultView === window.top) score += 500;
      } catch(eTop) {}
      if (score > bestScore) {
        best = doc;
        bestScore = score;
      }
    });
    if (best) return best;
    try {
      if (window.parent && window.parent !== window && window.parent.document && window.parent.document.body) return window.parent.document;
    } catch(eParent) {}
    return null;
  }

  function forceStickRigRowHeight(row, labelElement){
    if (!row) return;
    var targetHeight = 72;
    var baseRect = null;
    try { baseRect = row.getBoundingClientRect(); } catch(eBase) {}
    var baseTop = baseRect ? baseRect.top : 0;

    function stretchChain(start, maxDepth){
      var current = start;
      var guard = 0;
      while (current && guard < maxDepth) {
        var rect = null;
        try { rect = current.getBoundingClientRect(); } catch(eRect) {}
        if (!rect) break;

          current.style.setProperty("height", targetHeight + "px", "important");

          current.style.setProperty("max-height", targetHeight + "px", "important");

          current.style.setProperty("align-items", "stretch", "important");
          current.style.setProperty("box-sizing", "border-box", "important");
        }
        current = current.parentElement;
        guard += 1;
      }
    }

    stretchChain(row, 6);
    if (labelElement) stretchChain(labelElement, 7);
  }

    } catch(ePanelZ) {}

    if (!overlay.keyboardBound) {
      doc.addEventListener("keydown", handleTimelineOverlayKeydown, true);
      overlay.keyboardBound = true;
    }
    return overlay.root;
  }

  function cleanInvisibleText(value){
    return s(value).replace(/[\u200B-\u200D\u2060\u2063\uFEFF]/g, "").trim();
  }


  function parseCssColour(value){
    var match = s(value).match(/rgba?\s*\(\s*([\d.]+)\s*,\s*([\d.]+)\s*,\s*([\d.]+)(?:\s*,\s*([\d.]+))?/i);
    if (!match) return null;
    return {
      r: Number(match[1]),
      g: Number(match[2]),
      b: Number(match[3]),
      a: match[4] === undefined ? 1 : Number(match[4])
    };
  }

  function isLikelyNativeFrameBar(doc, element, rect){
    if (!doc || !element || !rect) return false;
    var view = doc.defaultView;
    var viewportWidth = Math.max(doc.documentElement ? doc.documentElement.clientWidth : 0, view ? view.innerWidth : 0);
    var viewportHeight = Math.max(doc.documentElement ? doc.documentElement.clientHeight : 0, view ? view.innerHeight : 0);
    if (rect.width < 180 || rect.height < 24 || rect.height > 62) return false;
    if (rect.top < viewportHeight * 0.42 || rect.bottom > viewportHeight + 4) return false;
    if (rect.left < 70 || rect.left > viewportWidth * 0.62) return false;
    if (rect.right < rect.left + 180) return false;
    try {
      if (element.closest && element.closest('#luke_tools_fullscreen_overlay_271,#luke_tools_panel_271,#stickrig-compact-timeline-root')) return false;
    } catch(eClosest) {}

    var style = null;
    try { style = view.getComputedStyle(element); } catch(eStyle) {}
    if (!style || style.display === 'none' || style.visibility === 'hidden' || Number(style.opacity) === 0) return false;
    var colour = parseCssColour(style.backgroundColor);
    var brightness = colour ? (colour.r + colour.g + colour.b) / 3 : 0;
    var signature = (s(element.className) + ' ' + s(element.id)).toLowerCase();
    var frameSignature = signature.indexOf('frame') !== -1 || signature.indexOf('timeline') !== -1 || signature.indexOf('cell') !== -1;
    if (!(colour && colour.a > 0.45 && brightness > 175) && !frameSignature) return false;
    var text = cleanInvisibleText(element.textContent || '');
    if (text.length > 8) return false;
    return true;
  }

  function sampleCanvasPixel(data, width, height, x, y){
    var px = Math.max(0, Math.min(width - 1, Math.round(x)));
    var py = Math.max(0, Math.min(height - 1, Math.round(y)));
    var index = (py * width + px) * 4;
    return {
      r: data[index] || 0,
      g: data[index + 1] || 0,
      b: data[index + 2] || 0,
      a: data[index + 3] || 0
    };
  }

  function isBrightTimelinePixel(pixel){
    if (!pixel || pixel.a < 90) return false;
    var max = Math.max(pixel.r, pixel.g, pixel.b);
    var min = Math.min(pixel.r, pixel.g, pixel.b);
    var brightness = (pixel.r + pixel.g + pixel.b) / 3;
    return brightness >= 145 && max - min <= 105;
  }

  /*
   * The vertical row finder can be permissive because it only needs to locate
   * timeline-height bands. The horizontal calibration cannot be permissive:
   * Wick's grey layer-name column is also bright and was being joined to the
   * white frame strip. That made the compact tracks begin at the layer panel,
   * use the wrong width and place every key in the wrong frame. This stricter
   * test isolates the near-white native frame interior only.
   */
  function isNativeFrameInteriorPixel(pixel){
    if (!pixel || pixel.a < 170) return false;
    var max = Math.max(pixel.r, pixel.g, pixel.b);
    var min = Math.min(pixel.r, pixel.g, pixel.b);
    return min >= 222 && max - min <= 34;
  }

  function scanCanvasForFrameBars(doc, canvas){
    var result = [];
    if (!doc || !canvas) return result;
    var rect = rectOf(canvas);
    if (!rect || rect.width < 240 || rect.height < 90) return result;

    var view = doc.defaultView;
    var viewportHeight = Math.max(doc.documentElement ? doc.documentElement.clientHeight : 0, view ? view.innerHeight : 0);
    if (rect.bottom < viewportHeight * 0.50 || rect.top > viewportHeight + 4) return result;

    try {
      if (canvas.closest && canvas.closest('#luke_tools_fullscreen_overlay_271,#luke_tools_panel_271,#stickrig-compact-timeline-root')) return result;
    } catch(eClosest) {}

    var context = null;
    try { context = canvas.getContext('2d', { willReadFrequently: true }); } catch(eContext1) {}
    try { if (!context) context = canvas.getContext('2d'); } catch(eContext2) {}
    if (!context || !canvas.width || !canvas.height) return result;

    var image = null;
    try { image = context.getImageData(0, 0, canvas.width, canvas.height); } catch(eRead) { return result; }
    if (!image || !image.data) return result;

    var scaleX = canvas.width / Math.max(1, rect.width);
    var scaleY = canvas.height / Math.max(1, rect.height);
    var sampleCssXs = [
      Math.min(rect.width - 2, 8),
      Math.min(rect.width - 2, 18),
      Math.min(rect.width - 2, 34),
      Math.min(rect.width - 2, 62),
      Math.min(rect.width - 2, Math.max(84, rect.width * 0.10)),
      Math.min(rect.width - 2, Math.max(120, rect.width * 0.16))
    ].filter(function(value, index, array){ return value > 1 && array.indexOf(value) === index; });

    var brightByY = [];
    var cssHeight = Math.floor(rect.height);
    for (var cssY = 0; cssY < cssHeight; cssY += 1) {
      var votes = 0;
      for (var xIndex = 0; xIndex < sampleCssXs.length; xIndex += 1) {
        var pixel = sampleCanvasPixel(image.data, canvas.width, canvas.height, sampleCssXs[xIndex] * scaleX, cssY * scaleY);
        if (isBrightTimelinePixel(pixel)) votes += 1;
      }
      brightByY[cssY] = votes >= 2;
    }

    var runs = [];
    var runStart = -1;
    for (var yIndex = 0; yIndex <= brightByY.length; yIndex += 1) {
      var isBright = yIndex < brightByY.length && brightByY[yIndex];
      if (isBright && runStart < 0) runStart = yIndex;
      if ((!isBright || yIndex === brightByY.length) && runStart >= 0) {
        var runEnd = yIndex - 1;
        var runHeight = runEnd - runStart + 1;
        if (runHeight >= 18 && runHeight <= 78) runs.push({ top: runStart, bottom: runEnd, height: runHeight });
        runStart = -1;
      }
    }

    runs.forEach(function(run){
      /*
       * Do not sample only the vertical centre of the row. Wick draws the
       * playhead, tween line and tween diamonds through that centre. Those
       * dark/coloured pixels split the white frame strip into separate runs,
       * which previously made StickRig choose only the section ahead of the
       * playhead. Sample the whole row instead, then bridge small gaps so the
       * detected bounds remain fixed while the scrubber moves.
       */
      var step = 2;
      var sampleYs = [];
      var inset = Math.max(2, Math.min(6, Math.floor(run.height * 0.16)));
      var usableTop = run.top + inset;
      var usableBottom = run.bottom - inset;
      var sampleCount = Math.max(5, Math.min(11, Math.floor(run.height / 4)));
      for (var syIndex = 0; syIndex < sampleCount; syIndex += 1) {
        var syT = sampleCount <= 1 ? 0.5 : syIndex / (sampleCount - 1);
        sampleYs.push(usableTop + (usableBottom - usableTop) * syT);
      }

      var horizontalBright = [];
      for (var cssX = 0; cssX < rect.width; cssX += step) {
        var votes = 0;
        for (var sampleIndex = 0; sampleIndex < sampleYs.length; sampleIndex += 1) {
          var pixel = sampleCanvasPixel(
            image.data,
            canvas.width,
            canvas.height,
            cssX * scaleX,
            sampleYs[sampleIndex] * scaleY
          );
          if (isNativeFrameInteriorPixel(pixel)) votes += 1;
        }
        horizontalBright.push({
          x: cssX,
          bright: votes >= Math.max(2, Math.ceil(sampleYs.length * 0.30))
        });
      }

      /* Bridge only narrow interruptions made by the native playhead and
         tween diamonds. A wider bridge can incorrectly join the layer-name
         column to the actual frame strip. */
      var maxGapCss = Math.max(6, Math.min(12, Math.round(run.height * 0.28)));
      var maxGapSamples = Math.max(1, Math.ceil(maxGapCss / step));
      var bridged = horizontalBright.map(function(item){ return item.bright; });
      var gapStart = -1;
      for (var bridgeIndex = 0; bridgeIndex <= bridged.length; bridgeIndex += 1) {
        var isOn = bridgeIndex < bridged.length && bridged[bridgeIndex];
        if (!isOn && gapStart < 0) gapStart = bridgeIndex;
        if ((isOn || bridgeIndex === bridged.length) && gapStart >= 0) {
          var gapEnd = bridgeIndex - 1;
          var beforeOn = gapStart > 0 && bridged[gapStart - 1];
          var afterOn = bridgeIndex < bridged.length && bridged[bridgeIndex];
          var gapLength = gapEnd - gapStart + 1;
          if (beforeOn && afterOn && gapLength <= maxGapSamples) {
            for (var fillIndex = gapStart; fillIndex <= gapEnd; fillIndex += 1) bridged[fillIndex] = true;
          }
          gapStart = -1;
        }
      }

      var horizontalRuns = [];
      var horizontalStart = -1;
      for (var hIndex = 0; hIndex <= bridged.length; hIndex += 1) {
        var itemBright = hIndex < bridged.length && bridged[hIndex];
        if (itemBright && horizontalStart < 0) horizontalStart = horizontalBright[hIndex].x;
        if ((!itemBright || hIndex === bridged.length) && horizontalStart >= 0) {
          var horizontalEnd = hIndex < horizontalBright.length ? horizontalBright[hIndex].x : rect.width;
          if (horizontalEnd - horizontalStart >= 80) {
            horizontalRuns.push({
              left: horizontalStart,
              right: horizontalEnd,
              width: horizontalEnd - horizontalStart
            });
          }
          horizontalStart = -1;
        }
      }

      horizontalRuns.sort(function(a, b){ return b.width - a.width; });
      var bestHorizontal = horizontalRuns[0] || { left: 0, right: rect.width, width: rect.width };

      /* Never snap to the canvas edge. The Wick canvas also contains the
         layer controls on the left; snapping there is what moved the S/M/R
         tracks away from frame 1. */
      bestHorizontal.width = Math.max(80, bestHorizontal.right - bestHorizontal.left);

      result.push({
        element: canvas,
        source: 'canvas',
        rect: {
          left: rect.left + bestHorizontal.left,
          right: rect.left + bestHorizontal.right,
          top: rect.top + run.top,
          bottom: rect.top + run.bottom + 1,
          width: bestHorizontal.width,
          height: run.height
        }
      });
    });

    return result;
  }

  function collectCanvasFrameBars(doc){
    if (!doc || !doc.body) return [];
    var canvases = [];
    try { canvases = Array.prototype.slice.call(doc.querySelectorAll('canvas')); } catch(eQuery) {}
    var rows = [];
    canvases.forEach(function(canvas){
      scanCanvasForFrameBars(doc, canvas).forEach(function(row){ rows.push(row); });
    });

    rows.sort(function(a, b){
      if (Math.abs(a.rect.top - b.rect.top) > 2) return a.rect.top - b.rect.top;
      return b.rect.width - a.rect.width;
    });

    var collapsed = [];
    rows.forEach(function(candidate){
      var existing = null;
      for (var i = 0; i < collapsed.length; i += 1) {
        if (Math.abs(collapsed[i].rect.top - candidate.rect.top) <= 3 && Math.abs(collapsed[i].rect.left - candidate.rect.left) <= 16) {
          existing = collapsed[i];
          break;
        }
      }
      if (!existing) collapsed.push(candidate);
      else if (candidate.rect.width > existing.rect.width) {
        existing.element = candidate.element;
        existing.rect = candidate.rect;
        existing.source = candidate.source;
      }
    });
    return collapsed;
  }

  function collectNativeFrameBars(doc){
    if (!doc || !doc.body) return [];
    var view = doc.defaultView;
    var viewportWidth = Math.max(doc.documentElement ? doc.documentElement.clientWidth : 0, view ? view.innerWidth : 0);
    var viewportHeight = Math.max(doc.documentElement ? doc.documentElement.clientHeight : 0, view ? view.innerHeight : 0);
    var startY = Math.max(0, Math.floor(viewportHeight * 0.46));
    var probeXs = [
      Math.max(90, Math.min(viewportWidth - 20, 180)),
      Math.max(110, Math.min(viewportWidth - 20, 230)),
      Math.max(140, Math.min(viewportWidth - 20, 310)),
      Math.max(180, Math.min(viewportWidth - 20, 430)),
      Math.max(220, Math.min(viewportWidth - 20, Math.round(viewportWidth * 0.42)))
    ];
    var candidates = [];
    var seen = [];

    function alreadySeen(element){
      for (var i = 0; i < seen.length; i += 1) if (seen[i] === element) return true;
      seen.push(element);
      return false;
    }

    for (var y = startY; y < viewportHeight - 4; y += 7) {
      for (var xIndex = 0; xIndex < probeXs.length; xIndex += 1) {
        var x = probeXs[xIndex];
        var stack = [];
        try { stack = doc.elementsFromPoint(x, y) || []; } catch(ePoint) {}
        for (var stackIndex = 0; stackIndex < stack.length; stackIndex += 1) {
          var element = stack[stackIndex];
          if (!element || alreadySeen(element)) continue;
          var rect = rectOf(element);
          if (!isLikelyNativeFrameBar(doc, element, rect)) continue;
          candidates.push({ element: element, rect: rect });
        }
      }
    }

    candidates.sort(function(a, b){
      if (Math.abs(a.rect.top - b.rect.top) > 2) return a.rect.top - b.rect.top;
      return b.rect.width - a.rect.width;
    });

    var collapsed = [];
    candidates.forEach(function(candidate){
      var existing = null;
      for (var i = 0; i < collapsed.length; i += 1) {
        if (Math.abs(collapsed[i].rect.top - candidate.rect.top) <= 3 && Math.abs(collapsed[i].rect.left - candidate.rect.left) <= 12) {
          existing = collapsed[i];
          break;
        }
      }
      if (!existing) {
        collapsed.push(candidate);
      } else if (candidate.rect.width > existing.rect.width) {
        existing.element = candidate.element;
        existing.rect = candidate.rect;
      }
    });

    collapsed.sort(function(a, b){ return a.rect.top - b.rect.top; });
    return collapsed;
  }

  function medianNumber(values){
    var clean = (values || []).filter(function(value){ return isFinite(Number(value)); }).map(Number).sort(function(a, b){ return a - b; });
    if (!clean.length) return NaN;
    var middle = Math.floor(clean.length / 2);
    return clean.length % 2 ? clean[middle] : (clean[middle - 1] + clean[middle]) * 0.5;
  }

  /*
   * Calibrate the horizontal frame grid from Wick's numbered timeline ruler.
   * The ruler is much more reliable than reading the white frame strip: the
   * frame strip contains the playhead, tween line and key diamonds, while the
   * ruler labels retain a stable frame number and x position.
   */
  function findTimelineRulerGrid(doc, selectedRows){
    if (!doc || !doc.body) return null;

    var view = doc.defaultView;
    var viewportWidth = Math.max(doc.documentElement ? doc.documentElement.clientWidth : 0, view ? view.innerWidth : 0);
    var viewportHeight = Math.max(doc.documentElement ? doc.documentElement.clientHeight : 0, view ? view.innerHeight : 0);
    var firstRowTop = Infinity;

    (selectedRows || []).forEach(function(row){
      if (row && row.rect && isFinite(Number(row.rect.top))) firstRowTop = Math.min(firstRowTop, Number(row.rect.top));
    });

    var candidates = [];
    var walker = null;
    try { walker = doc.createTreeWalker(doc.body, 4, null); } catch(eWalker) {}

    if (walker) {
      var textNode = null;
      while ((textNode = walker.nextNode())) {
        var raw = cleanInvisibleText(textNode.nodeValue || '');
        if (!/^\d{1,4}$/.test(raw)) continue;

        var value = Number(raw);
        if (!isFinite(value) || value < 1 || value > 9999) continue;

        var element = textNode.parentElement;
        if (!element) continue;

        try {
          if (element.closest && element.closest('#luke_tools_fullscreen_overlay_271,#luke_tools_panel_271,#stickrig-compact-timeline-root,#app')) continue;
        } catch(eClosest) {}

        var rect = rectOf(element);
        if (!rect) continue;
        if (rect.width < 2 || rect.width > 58 || rect.height < 7 || rect.height > 32) continue;
        if (rect.left < 70 || rect.right > viewportWidth + 4) continue;
        if (rect.top < viewportHeight * 0.40 || rect.bottom > viewportHeight + 4) continue;

        if (isFinite(firstRowTop)) {
          var centreY = rect.top + rect.height * 0.5;
          if (centreY > firstRowTop + 8 || centreY < firstRowTop - 90) continue;
        }

        var style = null;
        try { style = view.getComputedStyle(element); } catch(eStyle) {}
        if (style && (style.display === 'none' || style.visibility === 'hidden' || Number(style.opacity) === 0)) continue;

        candidates.push({
          value: value,
          x: rect.left + rect.width * 0.5,
          y: rect.top + rect.height * 0.5,
          rect: rect,
          element: element
        });
      }
    }

    if (candidates.length < 4) return null;

    candidates.sort(function(a, b){
      if (Math.abs(a.y - b.y) > 3) return a.y - b.y;
      return a.x - b.x;
    });

    var groups = [];
    candidates.forEach(function(candidate){
      var group = null;
      for (var i = 0; i < groups.length; i += 1) {
        if (Math.abs(groups[i].y - candidate.y) <= 3) {
          group = groups[i];
          break;
        }
      }
      if (!group) {
        group = { y: candidate.y, items: [] };
        groups.push(group);
      }
      group.items.push(candidate);
      group.y = medianNumber(group.items.map(function(item){ return item.y; }));
    });

    var best = null;
    var bestScore = -Infinity;

    groups.forEach(function(group){
      var items = group.items.slice().sort(function(a, b){ return a.x - b.x; });
      var deduped = [];

      items.forEach(function(item){
        var duplicate = false;
        for (var i = 0; i < deduped.length; i += 1) {
          if (deduped[i].value === item.value && Math.abs(deduped[i].x - item.x) <= 5) {
            duplicate = true;
            break;
          }
        }
        if (!duplicate) deduped.push(item);
      });

      if (deduped.length < 4) return;

      var cellWidths = [];
      for (var pairIndex = 1; pairIndex < deduped.length; pairIndex += 1) {
        var previous = deduped[pairIndex - 1];
        var current = deduped[pairIndex];
        var frameDelta = current.value - previous.value;
        var xDelta = current.x - previous.x;
        if (frameDelta <= 0 || frameDelta > 20 || xDelta <= 0) continue;
        var cellWidth = xDelta / frameDelta;
        if (cellWidth >= 12 && cellWidth <= 100) cellWidths.push(cellWidth);
      }

      if (cellWidths.length < 3) return;
      var canonicalCellWidth = medianNumber(cellWidths);
      if (!isFinite(canonicalCellWidth) || canonicalCellWidth < 12 || canonicalCellWidth > 100) return;

      var origins = [];
      var residuals = [];
      deduped.forEach(function(item){
        origins.push(item.x - (item.value - 0.5) * canonicalCellWidth);
      });
      var origin = medianNumber(origins);
      if (!isFinite(origin)) return;

      deduped.forEach(function(item){
        var predicted = origin + (item.value - 0.5) * canonicalCellWidth;
        residuals.push(Math.abs(predicted - item.x));
      });
      var residual = medianNumber(residuals);
      if (!isFinite(residual) || residual > 5.5) return;

      var rowDistance = isFinite(firstRowTop) ? Math.abs(firstRowTop - group.y) : 0;
      var score = deduped.length * 100 - residual * 25 - rowDistance;

      if (score > bestScore) {
        bestScore = score;
        best = {
          origin: origin,
          cellWidth: canonicalCellWidth,
          rulerY: group.y,
          firstVisibleFrame: deduped[0].value,
          lastVisibleFrame: deduped[deduped.length - 1].value,
          firstVisibleLeft: deduped[0].x - canonicalCellWidth * 0.5,
          lastVisibleRight: deduped[deduped.length - 1].x + canonicalCellWidth * 0.5,
          itemCount: deduped.length,
          residual: residual
        };
      }
    });

    return best;
  }

  function chooseProxyFrameRows(doc, count){
    /* Prefer canvas-derived frame strips. DOM timeline containers often span
       the entire editor width, which is exactly what produced the full-width
       S/M/R channels in v20. */
    var canvasRows = collectCanvasFrameBars(doc);
    var domRows = collectNativeFrameBars(doc);
    var rows = canvasRows.length >= count ? canvasRows.slice() : domRows.slice();

    if (rows.length < count) {
      canvasRows.concat(domRows).forEach(function(candidate){
        var duplicate = false;
        for (var existingIndex = 0; existingIndex < rows.length; existingIndex += 1) {
          if (Math.abs(rows[existingIndex].rect.top - candidate.rect.top) <= 3) {
            duplicate = true;
            if (candidate.source === 'canvas' || candidate.rect.width < rows[existingIndex].rect.width) rows[existingIndex] = candidate;
            break;
          }
        }
        if (!duplicate) rows.push(candidate);
      });
      rows.sort(function(a, b){ return a.rect.top - b.rect.top; });
    }
    if (!rows.length || count <= 0) return [];
    var groups = [];
    var current = [];

    rows.forEach(function(row){
      if (!current.length) {
        current.push(row);
        return;
      }
      var previous = current[current.length - 1];
      var gap = row.rect.top - previous.rect.top;
      var similarLeft = Math.abs(row.rect.left - previous.rect.left) <= 24;
      var similarWidth = Math.abs(row.rect.width - previous.rect.width) <= 120;
      if (gap >= 20 && gap <= 82 && similarLeft && similarWidth) {
        current.push(row);
      } else {
        groups.push(current);
        current = [row];
      }
    });
    if (current.length) groups.push(current);

    groups.sort(function(a, b){
      if (a.length !== b.length) return b.length - a.length;
      var aBottom = a.length ? a[a.length - 1].rect.bottom : 0;
      var bBottom = b.length ? b[b.length - 1].rect.bottom : 0;
      return bBottom - aBottom;
    });

    var best = groups[0] || [];
    if (best.length < count) return [];

    var selected = best.slice(best.length - count);

    /*
     * Use Wick's numbered ruler as the single source of truth for horizontal
     * alignment. The overlay begins at the exact left edge of frame 1 and its
     * width is the frame-cell width multiplied by the project duration.
     */
    var rulerGrid = findTimelineRulerGrid(doc, selected);
    var duration = Math.max(1, getProjectDuration());

    if (rulerGrid && isFinite(rulerGrid.origin) && isFinite(rulerGrid.cellWidth)) {
      var canonicalLeft = rulerGrid.origin;
      var canonicalWidth = rulerGrid.cellWidth * duration;
      var canonicalRight = canonicalLeft + canonicalWidth;

      selected.forEach(function(row){
        row.rect.left = canonicalLeft;
        row.rect.right = canonicalRight;
        row.rect.width = canonicalWidth;
        row.rulerGrid = rulerGrid;
      });
      state.timelineOverlay.rulerGrid = rulerGrid;
      return selected;
    }

    /* Fallback only when the ruler is unavailable. Keep all selected rows on
       one shared horizontal measurement so key positions remain consistent. */
    var validLefts = selected.map(function(row){ return row.rect.left; }).filter(function(value){ return isFinite(value); });
    var validRights = selected.map(function(row){ return row.rect.right; }).filter(function(value){ return isFinite(value); });
    var fallbackLeft = medianNumber(validLefts);
    var fallbackRight = medianNumber(validRights);

    if (isFinite(fallbackLeft) && isFinite(fallbackRight) && fallbackRight - fallbackLeft >= 80) {
      selected.forEach(function(row){
        row.rect.left = fallbackLeft;
        row.rect.right = fallbackRight;
        row.rect.width = fallbackRight - fallbackLeft;
      });
    }

    return selected;
  }

  function requestNativeTimelineDraw(clearCanvas){
    var nativeState = state.nativeTimeline;
    var gui = nativeState.lastTimelineGUI;

    if (gui && gui.canvas && gui.ctx && clearCanvas) {
      try {
        gui.ctx.save();
        if (typeof gui.ctx.setTransform === "function") gui.ctx.setTransform(1, 0, 0, 1, 0, 0);
        gui.ctx.clearRect(0, 0, gui.canvas.width, gui.canvas.height);
        gui.ctx.restore();
      } catch(eClear) {}
    }

    try {
      if (gui && typeof gui.draw === "function") {
        gui.draw();
        return true;
      }
    } catch(eGuiDraw) {}

    try {
      if (state.project && state.project.guiElement && typeof state.project.guiElement.draw === "function") {
        state.project.guiElement.draw();
        return true;
      }
    } catch(eProjectDraw) {}

    try {
      if (state.editor && typeof state.editor.forceUpdate === "function") {
        state.editor.forceUpdate();
        return true;
      }
    } catch(eEditorDraw) {}
    return false;
  }

  function forceNativeTimelineRestore(gui, originalDraw){
    var targets = [];
    function addTarget(target){
      if (!target || targets.indexOf(target) !== -1) return;
      targets.push(target);
    }
    addTarget(gui);
    addTarget(state.nativeTimeline.lastTimelineGUI);
    try {
      var W = getWickNamespace();
      if (W && W.GUIElement && W.GUIElement.Timeline) {
        addTarget(W.GUIElement.Timeline.__stickRigLastTimelineGUI);
      }
    } catch(eLast) {}

    function redraw(){
      targets.forEach(function(target){
        try {
          if (target.canvas && target.ctx) {
            target.ctx.save();
            if (typeof target.ctx.setTransform === "function") target.ctx.setTransform(1, 0, 0, 1, 0, 0);
            target.ctx.clearRect(0, 0, target.canvas.width, target.canvas.height);
            target.ctx.restore();
          }
        } catch(eClear) {}
        try {
          if (typeof originalDraw === "function") originalDraw.call(target);
          else if (typeof target.draw === "function") target.draw();
        } catch(eDraw) {
          try { if (typeof target.draw === "function") target.draw(); } catch(eFallback) {}
        }
      });
    }

    redraw();
    try {
      var ownerWindow = targets[0] && targets[0].canvas && targets[0].canvas.ownerDocument
        ? targets[0].canvas.ownerDocument.defaultView
        : null;
      if (ownerWindow && typeof ownerWindow.requestAnimationFrame === "function") {
        ownerWindow.requestAnimationFrame(function(){
          redraw();
          ownerWindow.requestAnimationFrame(redraw);
        });
      } else if (ownerWindow && typeof ownerWindow.setTimeout === "function") {
        ownerWindow.setTimeout(redraw, 0);
        ownerWindow.setTimeout(redraw, 80);
      }
    } catch(eSchedule) {}
  }

  function nativeCanvasPoint(canvas, event){
    var rect = canvas.getBoundingClientRect();
    var scaleX = rect.width ? canvas.width / rect.width : 1;
    var scaleY = rect.height ? canvas.height / rect.height : 1;
    return {
      x: (event.clientX - rect.left) * scaleX,
      y: (event.clientY - rect.top) * scaleY
    };
  }

  function nativeFrameFromX(x, geometry){
    if (!geometry) return 1;
    return clamp(Math.floor((x - geometry.baseX) / geometry.cellWidth) + 1, 1, geometry.duration);
  }

  function findNativeChannelAtPoint(x, y){
    var geometry = state.nativeTimeline.geometry || [];
    for (var i = geometry.length - 1; i >= 0; i -= 1) {
      var item = geometry[i];
      if (x < item.visibleLeft || x > item.visibleRight || y < item.trackTop || y > item.trackBottom) continue;
      return item;
    }
    return null;
  }

  function findNativeMarkerAtPoint(x, y){
    var geometry = state.nativeTimeline.geometry || [];
    var best = null;
    var bestDistance = Infinity;
    geometry.forEach(function(item){
      (item.markers || []).forEach(function(marker){
        var dx = x - marker.x;
        var dy = y - marker.y;
        var distance = Math.sqrt(dx * dx + dy * dy);
        if (distance <= 11 && distance < bestDistance) {
          bestDistance = distance;
          best = marker;
        }
      });
    });
    return best;
  }

  function stopTimelineMouseEvent(event){
    if (!event) return;
    try { event.preventDefault(); } catch(e1) {}
    try { event.stopPropagation(); } catch(e2) {}
    try { if (typeof event.stopImmediatePropagation === 'function') event.stopImmediatePropagation(); } catch(e3) {}
  }

  function copyTimelineKey(marker){
    if (!marker) return false;
    var record = getNodeRecord(marker.nodeId);
    var node = state.nodeMap[marker.nodeId];
    if (!record || !channelKeyExists(record, marker.frame, marker.channel)) return false;
    var sourcePose = copyPose((record.keys && record.keys[String(marker.frame)]) || interpolatePose(record, marker.frame));
    state.keyClipboard = {
      channel: marker.channel,
      pose: sourcePose,
      type: channelType(record, marker.frame, marker.channel),
      easing: channelEasing(record, marker.frame, marker.channel),
      sourceNodeId: marker.nodeId,
      sourceFrame: marker.frame
    };
    setStatus('Copied ' + marker.channel + ' key ' + marker.frame + ' from ' + (node ? node.name : 'part') + '.', 'good');
    return true;
  }

  function pasteTimelineKey(nodeId, channel, frameNumber){
    var clipboard = state.keyClipboard;
    var record = getNodeRecord(nodeId);
    var node = state.nodeMap[nodeId];
    if (!clipboard || !record || !node) return false;
    if (clipboard.channel !== channel) {
      setStatus('A ' + clipboard.channel + ' key can only be pasted into the ' + clipboard.channel + ' channel.', 'warn');
      return false;
    }

    var frame = clamp(Math.round(n(frameNumber, currentProjectFrame())), 1, Math.max(1, getProjectDuration()));
    var key = String(frame);
    ensureChannelMaps(record);
    if (!record.keys) record.keys = {};
    var targetPose = copyPose(record.keys[key] || interpolatePose(record, frame));
    copyChannelToPose(targetPose, clipboard.pose, channel);
    record.keys[key] = targetPose;
    if (!record.keyChannels[key]) record.keyChannels[key] = {};
    if (!record.channelTypes[key]) record.channelTypes[key] = {};
    if (!record.channelEasing[key]) record.channelEasing[key] = {};
    record.keyChannels[key][channel] = true;
    record.channelTypes[key][channel] = clipboard.type === 'hold' ? 'hold' : 'auto';
    record.channelEasing[key][channel] = clipboard.easing || 'none';
    var active = CHANNELS.filter(function(name){ return record.keyChannels[key] && record.keyChannels[key][name]; });
    record.keyTypes[key] = active.length && active.every(function(name){ return record.channelTypes[key][name] === 'hold'; }) ? 'hold' : 'auto';

    state.timelineOverlay.selected = { nodeId: nodeId, channel: channel, frame: frame };
    setMainTimelineFrame(frame);
    applyTransformToObject(node, interpolatePose(record, frame), false);
    queueNodeNativeTweenSync(node, record, true);
    markProjectChanged();
    renderAll();
    saveStagingData(false);
    queueProxySync();
    requestNativeTimelineDraw();
    setStatus('Pasted ' + channel + ' key at frame ' + frame + ' for ' + node.name + '.', 'good');
    return true;
  }

  function cutTimelineKey(marker){
    if (!copyTimelineKey(marker)) return false;
    deleteTimelineMarker(marker);
    return true;
  }

  function hideTimelineContextMenu(){
    var nativeState = state.nativeTimeline;
    var menu = nativeState.contextMenuElement;
    var doc = nativeState.document || (menu ? menu.ownerDocument : null);
    try { if (menu && menu.parentNode) menu.parentNode.removeChild(menu); } catch(eRemove) {}
    if (doc && nativeState.contextMenuDismissHandler) {
      try { doc.removeEventListener('pointerdown', nativeState.contextMenuDismissHandler, true); } catch(e1) {}
      try { doc.removeEventListener('contextmenu', nativeState.contextMenuDismissHandler, true); } catch(e2) {}
      try { doc.removeEventListener('keydown', nativeState.contextMenuDismissHandler, true); } catch(e3) {}
      try { doc.defaultView.removeEventListener('blur', nativeState.contextMenuDismissHandler, true); } catch(e4) {}
    }
    nativeState.contextMenuElement = null;
    nativeState.contextMenuDismissHandler = null;
  }

  function showTimelineContextMenu(event, marker, channelGeometry, frameNumber){
    stopTimelineMouseEvent(event);
    hideTimelineContextMenu();
    var nativeState = state.nativeTimeline;
    var doc = nativeState.document || (event && event.target ? event.target.ownerDocument : null) || getParentEditorDocument();
    if (!doc || !doc.body) return;
    nativeState.document = doc;

    var root = ensureTimelineOverlay();
    if (!root) return;
    var menu = doc.createElement('div');
    menu.className = 'sr-timeline-context-menu';
    menu.setAttribute('role', 'menu');

    function addItem(label, className, action){
      var button = doc.createElement('button');
      button.type = 'button';
      button.textContent = label;
      if (className) button.className = className;
      button.addEventListener('pointerdown', stopTimelineMouseEvent, true);
      button.addEventListener('mousedown', stopTimelineMouseEvent, true);
      button.addEventListener('mouseup', stopTimelineMouseEvent, true);
      button.addEventListener('click', function(clickEvent){
        stopTimelineMouseEvent(clickEvent);
        hideTimelineContextMenu();
        action();
      }, true);
      menu.appendChild(button);
    }

    if (marker) {
      state.timelineOverlay.selected = { nodeId: marker.nodeId, channel: marker.channel, frame: marker.frame };
      setMainTimelineFrame(marker.frame);
      addItem('Copy', '', function(){ copyTimelineKey(marker); });
      addItem('Cut', '', function(){ cutTimelineKey(marker); });
      addItem('Delete', 'sr-danger', function(){ deleteTimelineMarker(marker); });
    } else if (channelGeometry && state.keyClipboard && state.keyClipboard.channel === channelGeometry.channel) {
      var pasteFrame = clamp(Math.round(n(frameNumber, currentProjectFrame())), 1, Math.max(1, channelGeometry.duration || getProjectDuration()));
      addItem('Paste', '', function(){ pasteTimelineKey(channelGeometry.nodeId, channelGeometry.channel, pasteFrame); });
    } else {
      return;
    }

    root.appendChild(menu);
    nativeState.contextMenuElement = menu;
    var viewportWidth = Math.max(doc.documentElement ? doc.documentElement.clientWidth : 0, doc.defaultView ? doc.defaultView.innerWidth : 0);
    var viewportHeight = Math.max(doc.documentElement ? doc.documentElement.clientHeight : 0, doc.defaultView ? doc.defaultView.innerHeight : 0);
    var left = n(event.clientX, 0);
    var top = n(event.clientY, 0);
    try {
      var rect = menu.getBoundingClientRect();
      if (left + rect.width > viewportWidth - 6) left = Math.max(6, viewportWidth - rect.width - 6);
      if (top + rect.height > viewportHeight - 6) top = Math.max(6, viewportHeight - rect.height - 6);
    } catch(eRect) {}
    menu.style.left = Math.round(left) + 'px';
    menu.style.top = Math.round(top) + 'px';

    nativeState.contextMenuDismissHandler = function(dismissEvent){
      if (dismissEvent && dismissEvent.type === 'keydown' && dismissEvent.key !== 'Escape') return;
      if (dismissEvent && menu.contains && menu.contains(dismissEvent.target)) return;
      hideTimelineContextMenu();
    };
    setTimeout(function(){
      if (!nativeState.contextMenuElement) return;
      doc.addEventListener('pointerdown', nativeState.contextMenuDismissHandler, true);
      doc.addEventListener('contextmenu', nativeState.contextMenuDismissHandler, true);
      doc.addEventListener('keydown', nativeState.contextMenuDismissHandler, true);
      try { doc.defaultView.addEventListener('blur', nativeState.contextMenuDismissHandler, true); } catch(eBlur) {}
    }, 0);
  }

  function beginNativeMarkerDrag(event, canvas, marker){
    if (!marker) return;
    stopTimelineMouseEvent(event);
    hideTimelineContextMenu();
    var nativeState = state.nativeTimeline;
    var doc = canvas.ownerDocument || nativeState.document;
    state.timelineOverlay.selected = { nodeId: marker.nodeId, channel: marker.channel, frame: marker.frame };
    setMainTimelineFrame(marker.frame);
    var startPoint = nativeCanvasPoint(canvas, event);
    var drag = {
      active: true,
      pointerId: event.pointerId,
      nodeId: marker.nodeId,
      channel: marker.channel,
      oldFrame: marker.frame,
      previewFrame: marker.frame,
      geometry: marker.geometry,
      moved: false,
      startX: startPoint.x,
      cleanup: null
    };
    nativeState.drag = drag;
    state.timelineOverlay.drag = drag;

    function moveDrag(moveEvent){
      if (!drag.active || moveEvent.pointerId !== drag.pointerId) return;
      stopTimelineMouseEvent(moveEvent);
      var point = nativeCanvasPoint(canvas, moveEvent);
      if (Math.abs(point.x - drag.startX) > 2) drag.moved = true;
      drag.previewFrame = nativeFrameFromX(point.x, drag.geometry);
      requestNativeTimelineDraw();
    }

    function finishDrag(upEvent, cancelled){
      if (!drag.active) return;
      if (upEvent && upEvent.pointerId !== drag.pointerId) return;
      if (upEvent) stopTimelineMouseEvent(upEvent);
      drag.active = false;
      try { doc.removeEventListener('pointermove', moveDrag, true); } catch(e1) {}
      try { doc.removeEventListener('pointerup', finishPointer, true); } catch(e2) {}
      try { doc.removeEventListener('pointercancel', cancelPointer, true); } catch(e3) {}
      nativeState.drag = null;
      state.timelineOverlay.drag = null;
      if (!cancelled && drag.moved && drag.previewFrame !== drag.oldFrame) {
        moveTimelineMarker(drag.nodeId, drag.channel, drag.oldFrame, drag.previewFrame);
      } else {
        requestNativeTimelineDraw();
      }
    }

    function finishPointer(upEvent){ finishDrag(upEvent, false); }
    function cancelPointer(cancelEvent){ finishDrag(cancelEvent, true); }
    drag.cleanup = function(cancelled){ finishDrag(null, cancelled !== false); };
    doc.addEventListener('pointermove', moveDrag, true);
    doc.addEventListener('pointerup', finishPointer, true);
    doc.addEventListener('pointercancel', cancelPointer, true);
  }

  function renderNativeInteractionBarriers(canvas, geometry){
    if (!canvas || !geometry || !geometry.length) return;
    if (state.nativeTimeline.drag && state.nativeTimeline.drag.active) return;
    var root = ensureTimelineOverlay();
    var doc = canvas.ownerDocument || state.nativeTimeline.document;
    if (!root || !doc) return;
    state.nativeTimeline.document = doc;

    try {
      var old = root.querySelectorAll('.sr-native-barrier');
      for (var i = 0; i < old.length; i += 1) if (old[i].parentNode) old[i].parentNode.removeChild(old[i]);
    } catch(eClear) {}

    var canvasRect = canvas.getBoundingClientRect();
    var scaleX = canvasRect.width ? canvas.width / canvasRect.width : 1;
    var scaleY = canvasRect.height ? canvas.height / canvasRect.height : 1;

    geometry.forEach(function(item){
      var leftCanvas = Math.max(item.visibleLeft, item.baseX);
      var rightCanvas = Math.min(item.visibleRight, item.baseX + item.duration * item.cellWidth);
      if (rightCanvas <= leftCanvas || item.trackBottom <= item.trackTop) return;
      var barrier = doc.createElement('div');
      barrier.className = 'sr-native-barrier';
      barrier.dataset.nodeId = item.nodeId;
      barrier.dataset.channel = item.channel;
      barrier.style.left = Math.round(canvasRect.left + leftCanvas / scaleX) + 'px';
      barrier.style.top = Math.round(canvasRect.top + item.trackTop / scaleY) + 'px';
      barrier.style.width = Math.max(1, Math.round((rightCanvas - leftCanvas) / scaleX)) + 'px';
      barrier.style.height = Math.max(1, Math.round((item.trackBottom - item.trackTop) / scaleY)) + 'px';

      ['mousedown','mouseup','click','dblclick','auxclick','dragstart'].forEach(function(type){
        barrier.addEventListener(type, stopTimelineMouseEvent, true);
      });

      barrier.addEventListener('pointerdown', function(pointerEvent){
        if (pointerEvent.target && pointerEvent.target.closest && pointerEvent.target.closest('.sr-native-marker-hit')) return;
        stopTimelineMouseEvent(pointerEvent);
        hideTimelineContextMenu();
        var point = nativeCanvasPoint(canvas, pointerEvent);
        setMainTimelineFrame(nativeFrameFromX(point.x, item));
        requestNativeTimelineDraw();
      }, true);
      barrier.addEventListener('pointerup', stopTimelineMouseEvent, true);
      barrier.addEventListener('pointercancel', stopTimelineMouseEvent, true);
      barrier.addEventListener('contextmenu', function(menuEvent){
        if (menuEvent.target && menuEvent.target.closest && menuEvent.target.closest('.sr-native-marker-hit')) return;
        stopTimelineMouseEvent(menuEvent);
        var point = nativeCanvasPoint(canvas, menuEvent);
        showTimelineContextMenu(menuEvent, null, item, nativeFrameFromX(point.x, item));
      }, true);

      (item.markers || []).forEach(function(marker){
        if (marker.x < leftCanvas - 11 || marker.x > rightCanvas + 11) return;
        var hit = doc.createElement('div');
        hit.className = 'sr-native-marker-hit';
        hit.dataset.nodeId = marker.nodeId;
        hit.dataset.channel = marker.channel;
        hit.dataset.frame = String(marker.frame);
        hit.style.left = ((marker.x - leftCanvas) / scaleX) + 'px';
        hit.style.top = ((marker.y - item.trackTop) / scaleY) + 'px';
        ['mousedown','mouseup','click','dblclick','auxclick','dragstart'].forEach(function(type){
          hit.addEventListener(type, stopTimelineMouseEvent, true);
        });
        hit.addEventListener('pointerdown', function(pointerEvent){
          beginNativeMarkerDrag(pointerEvent, canvas, marker);
        }, true);
        hit.addEventListener('pointerup', stopTimelineMouseEvent, true);
        hit.addEventListener('contextmenu', function(menuEvent){
          showTimelineContextMenu(menuEvent, marker, item, marker.frame);
        }, true);
        barrier.appendChild(hit);
      });

      root.appendChild(barrier);
    });
  }

  function attachNativeTimelineEvents(canvas){
    if (!canvas || !canvas.addEventListener) return;
    var nativeState = state.nativeTimeline;
    if (nativeState.canvas === canvas && nativeState.pointerDownHandler) return;
    detachNativeTimelineEvents();

    nativeState.canvas = canvas;
    nativeState.document = canvas.ownerDocument || null;

    function channelAtEvent(event){
      var point = nativeCanvasPoint(canvas, event);
      var marker = findNativeMarkerAtPoint(point.x, point.y);
      return {
        point: point,
        marker: marker,
        channel: marker ? marker.geometry : findNativeChannelAtPoint(point.x, point.y)
      };
    }

    nativeState.pointerDownHandler = function(event){
      if (!state.proxy.active || event.button !== 0) return;
      var hit = channelAtEvent(event);
      if (!hit.channel) return;
      stopTimelineMouseEvent(event);
      nativeState.barrierPointerId = event.pointerId;
      hideTimelineContextMenu();
      if (hit.marker) {
        beginNativeMarkerDrag(event, canvas, hit.marker);
      } else {
        setMainTimelineFrame(nativeFrameFromX(hit.point.x, hit.channel));
        requestNativeTimelineDraw();
      }
    };

    nativeState.pointerMoveHandler = function(event){
      if (!state.proxy.active) return;
      if (nativeState.barrierPointerId === event.pointerId) stopTimelineMouseEvent(event);
    };

    nativeState.pointerUpHandler = function(event){
      if (nativeState.barrierPointerId !== event.pointerId) return;
      stopTimelineMouseEvent(event);
      nativeState.barrierPointerId = null;
    };

    nativeState.pointerCancelHandler = function(event){
      if (nativeState.barrierPointerId !== event.pointerId) return;
      stopTimelineMouseEvent(event);
      nativeState.barrierPointerId = null;
    };

    nativeState.contextMenuHandler = function(event){
      if (!state.proxy.active) return;
      var hit = channelAtEvent(event);
      if (!hit.channel) return;
      showTimelineContextMenu(event, hit.marker, hit.channel, nativeFrameFromX(hit.point.x, hit.channel));
    };

    function blockMouseEvent(event){
      if (!state.proxy.active) return;
      var hit = channelAtEvent(event);
      if (hit.channel) stopTimelineMouseEvent(event);
    }
    nativeState.mouseDownHandler = blockMouseEvent;
    nativeState.mouseUpHandler = blockMouseEvent;
    nativeState.clickHandler = blockMouseEvent;
    nativeState.doubleClickHandler = blockMouseEvent;
    nativeState.auxClickHandler = blockMouseEvent;

    nativeState.keyDownHandler = function(event){
      if (!state.proxy.active) return;
      var target = event.target;
      var tag = target && target.tagName ? target.tagName.toLowerCase() : '';
      if (tag === 'input' || tag === 'textarea' || tag === 'select') return;
      var selected = state.timelineOverlay.selected;
      var command = !!(event.ctrlKey || event.metaKey);
      var key = s(event.key).toLowerCase();

      if (command && key === 'c' && selected) {
        stopTimelineMouseEvent(event);
        copyTimelineKey(selected);
        return;
      }
      if (command && key === 'x' && selected) {
        stopTimelineMouseEvent(event);
        cutTimelineKey(selected);
        return;
      }
      if (command && key === 'v' && selected && state.keyClipboard) {
        stopTimelineMouseEvent(event);
        pasteTimelineKey(selected.nodeId, selected.channel, currentProjectFrame());
        return;
      }
      if ((event.key === 'Delete' || event.key === 'Backspace') && selected) {
        stopTimelineMouseEvent(event);
        deleteTimelineMarker(selected);
      }
    };

    canvas.addEventListener('pointerdown', nativeState.pointerDownHandler, true);
    canvas.addEventListener('pointermove', nativeState.pointerMoveHandler, true);
    canvas.addEventListener('pointerup', nativeState.pointerUpHandler, true);
    canvas.addEventListener('pointercancel', nativeState.pointerCancelHandler, true);
    canvas.addEventListener('contextmenu', nativeState.contextMenuHandler, true);
    canvas.addEventListener('mousedown', nativeState.mouseDownHandler, true);
    canvas.addEventListener('mouseup', nativeState.mouseUpHandler, true);
    canvas.addEventListener('click', nativeState.clickHandler, true);
    canvas.addEventListener('dblclick', nativeState.doubleClickHandler, true);
    canvas.addEventListener('auxclick', nativeState.auxClickHandler, true);
    if (nativeState.document) {
      try {
        if (nativeState.document.defaultView) nativeState.document.defaultView.addEventListener('keydown', nativeState.keyDownHandler, true);
        else nativeState.document.addEventListener('keydown', nativeState.keyDownHandler, true);
      } catch(eKeyBind) {
        try { nativeState.document.addEventListener('keydown', nativeState.keyDownHandler, true); } catch(eKeyFallback) {}
      }
    }
  }

  function drawNativeKeyShape(ctx, x, y, colour, transitionColour, shape, selected){
    ctx.save();
    ctx.fillStyle = colour;
    ctx.strokeStyle = selected ? '#ffffff' : transitionColour;
    ctx.lineWidth = selected ? 2 : 1;
    ctx.beginPath();
    if (shape === 'square') {
      ctx.rect(x - 4, y - 4, 8, 8);
    } else if (shape === 'circle') {
      ctx.arc(x, y, 4.5, 0, Math.PI * 2);
    } else if (shape === 'triangle-left') {
      ctx.moveTo(x - 5, y);
      ctx.lineTo(x + 4, y - 5);
      ctx.lineTo(x + 4, y + 5);
      ctx.closePath();
    } else if (shape === 'triangle-right') {
      ctx.moveTo(x + 5, y);
      ctx.lineTo(x - 4, y - 5);
      ctx.lineTo(x - 4, y + 5);
      ctx.closePath();
    } else {
      ctx.moveTo(x, y - 5);
      ctx.lineTo(x + 5, y);
      ctx.lineTo(x, y + 5);
      ctx.lineTo(x - 5, y);
      ctx.closePath();
    }
    ctx.fill();
    ctx.stroke();
    ctx.restore();
  }

  function drawNativeStickRigTimeline(timelineGUI){
    if (timelineGUI) state.nativeTimeline.lastTimelineGUI = timelineGUI;
    if (!state.proxy.active || !state.proxy.layers.length || !timelineGUI) return;
    var W = getWickNamespace();
    if (!W || !W.GUIElement) return;
    var ctx = timelineGUI.ctx;
    var canvas = timelineGUI.canvas;
    if (!ctx || !canvas) return;

    attachNativeTimelineEvents(canvas);

    var G = W.GUIElement;
    var layerWidth = n(G.LAYERS_CONTAINER_WIDTH, 160);
    var breadcrumbHeight = n(G.BREADCRUMBS_HEIGHT, 30);
    var numberHeight = n(G.NUMBER_LINE_HEIGHT, 35);
    var rowHeight = n(timelineGUI.gridCellHeight, n(G.GRID_DEFAULT_CELL_HEIGHT, 42));
    var cellWidth = n(timelineGUI.gridCellWidth, n(G.GRID_DEFAULT_CELL_WIDTH, 38));
    var scrollbarSize = n(G.SCROLLBAR_SIZE, 14);
    var scrollX = n(timelineGUI.project && timelineGUI.project.scrollX, 0);
    var scrollY = n(timelineGUI.project && timelineGUI.project.scrollY, 0);
    var duration = Math.max(1, getProjectDuration());
    var layers = getTimelineLayers(timelineGUI.model);
    var baseX = layerWidth - scrollX;
    var clipLeft = layerWidth;
    var clipTop = breadcrumbHeight + numberHeight;
    var clipRight = Math.max(clipLeft, canvas.width - scrollbarSize);
    var clipBottom = Math.max(clipTop, canvas.height - scrollbarSize);
    var order = ['scale', 'move', 'rotate'];
    var selected = state.timelineOverlay.selected;
    var drag = state.nativeTimeline.drag;
    var geometry = [];

    ctx.save();
    ctx.beginPath();
    ctx.rect(clipLeft, clipTop, Math.max(0, clipRight - clipLeft), Math.max(0, clipBottom - clipTop));
    ctx.clip();
    ctx.font = 'bold 10px Arial, sans-serif';
    ctx.textBaseline = 'middle';

    state.proxy.layers.forEach(function(info){
      var layerIndex = layers.indexOf(info.layer);
      if (layerIndex < 0) return;
      var rowTop = clipTop + layerIndex * rowHeight - scrollY;
      var rowBottom = rowTop + rowHeight;
      if (rowBottom < clipTop || rowTop > clipBottom) return;

      var record = getNodeRecord(info.nodeId);
      if (!record) return;
      var trackHeight = rowHeight / 3;
      var totalWidth = duration * cellWidth;
      var visibleLeft = Math.max(clipLeft, baseX);
      var visibleRight = Math.min(clipRight, baseX + totalWidth);

      order.forEach(function(channel, channelIndex){
        var visual = CHANNEL_VISUALS[channel];
        var trackTop = rowTop + channelIndex * trackHeight;
        var trackBottom = channelIndex === 2 ? rowBottom : rowTop + (channelIndex + 1) * trackHeight;
        var trackHeightActual = Math.max(1, trackBottom - trackTop);

        ctx.fillStyle = 'rgba(31,112,82,0.66)';
        ctx.fillRect(baseX, trackTop, totalWidth, trackHeightActual);
        ctx.strokeStyle = visual.colour;
        ctx.lineWidth = 2;
        ctx.strokeRect(baseX + 0.5, trackTop + 0.5, Math.max(0, totalWidth - 1), Math.max(0, trackHeightActual - 1));
        ctx.fillStyle = visual.colour;
        ctx.fillText(visual.label, baseX + 4, trackTop + trackHeightActual * 0.5);

        var item = {
          nodeId: info.nodeId,
          channel: channel,
          duration: duration,
          baseX: baseX,
          cellWidth: cellWidth,
          trackTop: trackTop,
          trackBottom: trackBottom,
          visibleLeft: visibleLeft,
          visibleRight: visibleRight,
          markers: []
        };

        channelKeyFrames(record, channel).forEach(function(frameNumber){
          if (frameNumber < 1 || frameNumber > duration) return;
          var drawFrame = frameNumber;
          if (drag && drag.active && drag.nodeId === info.nodeId && drag.channel === channel && drag.oldFrame === frameNumber) {
            drawFrame = drag.previewFrame;
          }
          var keyX = baseX + (drawFrame - 0.5) * cellWidth;
          var keyY = trackTop + trackHeightActual * 0.5;
          var transition = transitionInfo(record, frameNumber, channel);
          var markerSelected = !!(selected && selected.nodeId === info.nodeId && selected.channel === channel && selected.frame === frameNumber);
          drawNativeKeyShape(ctx, keyX, keyY, visual.colour, transition.colour, transitionShape(record, frameNumber, channel), markerSelected);
          var marker = {
            nodeId: info.nodeId,
            channel: channel,
            frame: frameNumber,
            x: keyX,
            y: keyY,
            geometry: item
          };
          item.markers.push(marker);
        });

        geometry.push(item);
      });
    });

    /* Redraw Wick's current playhead through the StickRig rows at the exact
       native frame-centre position because the channels are drawn after the
       native timeline. This is the same playhead position, not a second one. */
    var currentFrame = currentProjectFrame();
    var playheadX = baseX + (currentFrame - 0.5) * cellWidth;
    ctx.strokeStyle = '#ff5b63';
    ctx.lineWidth = 2;
    ctx.beginPath();
    ctx.moveTo(playheadX, clipTop);
    ctx.lineTo(playheadX, clipBottom);
    ctx.stroke();

    ctx.restore();
    state.nativeTimeline.geometry = geometry;
    state.timelineOverlay.attachedCount = state.proxy.layers.length;
    renderNativeInteractionBarriers(canvas, geometry);
  }

  function installNativeTimelineRenderer(){
    var W = getWickNamespace();
    if (!W || !W.GUIElement || !W.GUIElement.Timeline || !W.GUIElement.Timeline.prototype) return false;

    var TimelineClass = W.GUIElement.Timeline;
    var proto = TimelineClass.prototype;
    var nativeState = state.nativeTimeline;

    if (nativeState.installed && nativeState.TimelineClass === TimelineClass && TimelineClass.__stickRigNativeRenderer === drawNativeStickRigTimeline) {
      requestNativeTimelineDraw(false);
      return true;
    }

    /* Remove any wrapper left by an earlier StickRig iframe before installing
       this instance. This prevents old closures from surviving after a tool
       panel has been closed or a newer script version has been loaded. */
    try {
      if (proto.__stickRigNativeOriginalDraw && typeof proto.__stickRigNativeOriginalDraw === "function") {
        proto.draw = proto.__stickRigNativeOriginalDraw;
        delete proto.__stickRigNativeOriginalDraw;
      }
    } catch(eRestoreOld) {}
    try { TimelineClass.__stickRigNativeRenderer = null; } catch(eClearOldRenderer) {}

    var originalDraw = proto.draw;
    if (typeof originalDraw !== "function") return false;

    proto.__stickRigNativeOriginalDraw = originalDraw;
    proto.draw = function(){
      TimelineClass.__stickRigLastTimelineGUI = this;
      var result = originalDraw.apply(this, arguments);
      try {
        var renderer = TimelineClass.__stickRigNativeRenderer;
        if (typeof renderer === "function") renderer(this);
      } catch(error) {
        try { console.error('[StickRig] Native timeline renderer failed.', error); } catch(eLog) {}
      }
      return result;
    };

    TimelineClass.__stickRigNativeRenderer = drawNativeStickRigTimeline;
    nativeState.installed = true;
    nativeState.TimelineClass = TimelineClass;
    nativeState.prototype = proto;
    nativeState.originalDraw = originalDraw;
    try {
      if (TimelineClass.__stickRigLastTimelineGUI) nativeState.lastTimelineGUI = TimelineClass.__stickRigLastTimelineGUI;
    } catch(eLastGui) {}
    requestNativeTimelineDraw(false);
    return true;
  }

  function uninstallNativeTimelineRenderer(suppressRedraw){
    var nativeState = state.nativeTimeline;
    var W = getWickNamespace();
    var TimelineClass = nativeState.TimelineClass || (W && W.GUIElement && W.GUIElement.Timeline ? W.GUIElement.Timeline : null);
    var proto = nativeState.prototype || (TimelineClass && TimelineClass.prototype ? TimelineClass.prototype : null);

    try {
      if (TimelineClass) TimelineClass.__stickRigNativeRenderer = null;
    } catch(eRenderer) {}

    try {
      if (proto && proto.__stickRigNativeOriginalDraw && typeof proto.__stickRigNativeOriginalDraw === "function") {
        proto.draw = proto.__stickRigNativeOriginalDraw;
        delete proto.__stickRigNativeOriginalDraw;
      } else if (proto && nativeState.originalDraw && typeof nativeState.originalDraw === "function") {
        proto.draw = nativeState.originalDraw;
      }
    } catch(eRestoreDraw) {}

    nativeState.installed = false;
    nativeState.geometry = [];
    detachNativeTimelineEvents();

    if (!suppressRedraw) requestNativeTimelineDraw(true);
  }

  function scheduleTimelineOverlayRetries(){
    clearTimeout(state.timelineOverlay.retryTimer);
    state.timelineOverlay.retryTimer = null;
    state.timelineOverlay.attachAttempts = 0;
    state.timelineOverlay.warningShown = false;
    if (!installNativeTimelineRenderer()) {
      setStatus("StickRig could not access Wick's native timeline renderer.", "bad");
      return;
    }
    state.timelineOverlay.attachedCount = state.proxy.layers.length;
    requestNativeTimelineDraw();
  }

  function findLayerLabelElement(doc, rawName, cleanName){
    if (!doc || !doc.body) return null;
    var exactMatches = [];
    var cleanMatches = [];
    var walker;
    try { walker = doc.createTreeWalker(doc.body, 4, null); } catch(eWalker) { walker = null; }

    function allowed(element){
      if (!element) return false;
      try {
        if (element.closest && element.closest("#stickrig-compact-timeline-root,#cards,.partCard,.channelPanel,[id*='luke_tools_panel'],[id*='luke_tools_fullscreen']")) return false;
      } catch(eClosest) {}
      return true;
    }

    if (walker) {
      var node;
      while ((node = walker.nextNode())) {
        var raw = s(node.nodeValue).trim();
        if (!raw || !node.parentElement || !allowed(node.parentElement)) continue;
        if (raw === rawName) exactMatches.push(node.parentElement);
        else if (cleanInvisibleText(raw) === cleanName) cleanMatches.push(node.parentElement);
      }
    }

    var matches = exactMatches.length ? exactMatches : cleanMatches;
    var best = null;
    var bestScore = -Infinity;
    matches.forEach(function(element){
      var rect = rectOf(element);
      if (!rect || rect.width < 4 || rect.height < 8) return;
      var score = exactMatches.indexOf(element) !== -1 ? 1000 : 0;
      if (rect.left < 260) score += 180;
      if (rect.height >= 18 && rect.height <= 76) score += 60;
      if (rect.width >= 20 && rect.width <= 260) score += 35;
      try {
        var ancestors = element;
        var depth = 0;
        while (ancestors && depth < 8) {
          var signature = (s(ancestors.className) + " " + s(ancestors.id)).toLowerCase();
          if (signature.indexOf("timeline") !== -1 || signature.indexOf("layer") !== -1) score += 45;
          ancestors = ancestors.parentElement;
          depth += 1;
        }
      } catch(eAncestors) {}
      if (score > bestScore) { best = element; bestScore = score; }
    });
    return best;
  }

  function rectOf(element){
    if (!element) return null;
    try { return element.getBoundingClientRect(); } catch(e) { return null; }
  }

  function findLayerRow(labelElement){
    if (!labelElement) return null;
    var labelRect = rectOf(labelElement);
    if (!labelRect) return null;
    var current = labelElement;
    var best = labelElement;
    var bestScore = -Infinity;
    var depth = 0;

    while (current && depth < 14) {
      var rect = rectOf(current);
      if (!rect) break;
      var sameLine = Math.abs((rect.top + rect.height * 0.5) - (labelRect.top + labelRect.height * 0.5)) < 12;
      var usefulHeight = rect.height >= 22 && rect.height <= 72;
      var usefulWidth = rect.width >= Math.max(90, labelRect.width) && rect.width <= 360;
      if (sameLine && usefulHeight && usefulWidth) {
        var score = rect.width + rect.height * 4;
        if (score > bestScore) {
          best = current;
          bestScore = score;
        }
      }
      if (rect.height > 110 || rect.width > 500) break;
      current = current.parentElement;
      depth += 1;
    }
    return best;
  }

  function findTimelineRowAtY(doc, labelRow, labelElement){
    if (!doc) return null;
    var labelRect = rectOf(labelRow) || rectOf(labelElement);
    if (!labelRect) return null;
    var viewportWidth = Math.max(
      doc.documentElement ? doc.documentElement.clientWidth : 0,
      doc.defaultView ? doc.defaultView.innerWidth : 0
    );
    var y = Math.round(labelRect.top + Math.max(2, Math.min(labelRect.height - 2, labelRect.height * 0.5)));
    var probeXs = [
      labelRect.right + 12,
      labelRect.right + 60,
      labelRect.right + 150,
      labelRect.right + 300,
      viewportWidth - 40
    ];
    var candidates = [];

    probeXs.forEach(function(x){
      if (!isFinite(x) || x <= labelRect.right || x >= viewportWidth) return;
      var stack = [];
      try { stack = doc.elementsFromPoint(x, y) || []; } catch(ePoint) {}
      for (var i = 0; i < stack.length; i += 1) {
        var current = stack[i];
        var guard = 0;
        while (current && guard < 7) {
          var rect = rectOf(current);
          if (!rect) break;
          var containsY = y >= rect.top - 2 && y <= rect.bottom + 2;
          var rightSide = rect.left >= labelRect.right - 8 || rect.right > labelRect.right + 100;
          var rowHeight = rect.height >= 22 && rect.height <= 78;
          var rowWidth = rect.width >= 100;
          if (containsY && rightSide && rowHeight && rowWidth) {
            candidates.push({ element: current, rect: rect });
          }
          if (rect.height > 100) break;
          current = current.parentElement;
          guard += 1;
        }
      }
    });

    if (!candidates.length) return null;
    candidates.sort(function(a, b){
      var aLeft = Math.max(a.rect.left, labelRect.right);
      var bLeft = Math.max(b.rect.left, labelRect.right);
      var aVisibleWidth = a.rect.right - aLeft;
      var bVisibleWidth = b.rect.right - bLeft;
      var aTopPenalty = Math.abs(a.rect.top - labelRect.top) * 25;
      var bTopPenalty = Math.abs(b.rect.top - labelRect.top) * 25;
      return (bVisibleWidth - bTopPenalty) - (aVisibleWidth - aTopPenalty);
    });
    return candidates[0].element;
  }

  function findTimelineRect(row, labelElement, doc){
    var labelRow = row || findLayerRow(labelElement);
    var labelRect = rectOf(labelRow) || rectOf(labelElement);
    if (!labelRect) return null;
    var timelineRow = findTimelineRowAtY(doc, labelRow, labelElement);
    var timelineRect = rectOf(timelineRow);
    var viewportWidth = Math.max(
      doc && doc.documentElement ? doc.documentElement.clientWidth : 0,
      doc && doc.defaultView ? doc.defaultView.innerWidth : 0
    );

    var left = labelRect.right + 4;
    var right = viewportWidth - 4;
    var top = labelRect.top;

    if (timelineRect) {
      left = Math.max(labelRect.right + 2, timelineRect.left);
      right = Math.max(left + 80, timelineRect.right);
      top = Math.min(labelRect.top, timelineRect.top);
    }

    return {
      left: left,
      top: top,
      width: Math.max(80, right - left),
      height: 72,
      rowRect: labelRect,
      labelRow: labelRow,
      timelineRow: timelineRow
    };
  }

  function transitionShape(record, frame, channel){
    if (channelType(record, frame, channel) === "hold") return "hold";
    var easing = channelEasing(record, frame, channel);
    if (easing === "in") return "ease-in";
    if (easing === "out") return "ease-out";
    if (easing === "in-out") return "ease-in-out";
    return "linear";
  }

  function frameToOverlayX(frameNumber, duration, width){
    var safeDuration = Math.max(1, Math.round(n(duration, 1)));
    var safeFrame = clamp(Math.round(n(frameNumber, 1)), 1, safeDuration);
    var cellWidth = width / safeDuration;
    /* Wick positions a key at the centre of its frame cell. */
    return (safeFrame - 0.5) * cellWidth;
  }

  function overlayXToFrame(clientX, groupRect, duration){
    var safeDuration = Math.max(1, Math.round(n(duration, 1)));
    var local = clientX - groupRect.left;
    var cellWidth = groupRect.width / safeDuration;
    return clamp(Math.floor(local / Math.max(0.0001, cellWidth)) + 1, 1, safeDuration);
  }

  function setMainTimelineFrame(frameNumber){
    var timeline = getMainTimeline();
    if (!timeline) return;
    var frame = Math.max(1, Math.round(n(frameNumber, 1)));
    try { timeline.playheadPosition = frame - 1; } catch(e1) {}
    try { state.project.currentFrameNumber = frame; } catch(e2) {}
    state.frame = frame;
    state.lastPolledFrame = frame;
    try { renderAll(); } catch(e3) {}
  }

  function selectTimelineMarker(marker, element){
    var overlay = state.timelineOverlay;
    var doc = overlay.parentDocument;
    if (doc) {
      var selected = doc.querySelectorAll(".sr-compact-key.sr-selected");
      for (var i = 0; i < selected.length; i += 1) selected[i].classList.remove("sr-selected");
    }
    overlay.selected = marker;
    if (element) element.classList.add("sr-selected");
    setMainTimelineFrame(marker.frame);
  }

  function deleteTimelineMarker(marker){
    if (!marker) return;
    var record = getNodeRecord(marker.nodeId);
    var node = state.nodeMap[marker.nodeId];
    if (!record || !channelKeyExists(record, marker.frame, marker.channel)) return;
    removeChannelKeyMeta(record, marker.frame, marker.channel);
    if (node) applyTransformToObject(node, interpolatePose(record, currentProjectFrame()), false);

    /* Right-click Delete, keyboard Delete and Cut all pass through this
       function, so every deletion path updates the real clip immediately. */
    var nativeSync = node
      ? syncNodeNativeTweensInstantly(node, record)
      : { ok: false, skipped: false, reason: "the animated clip could not be found" };

    state.timelineOverlay.selected = null;
    markProjectChanged();
    renderAll();
    var stagingSaved = saveStagingData(false);
    queueProxySync();
    requestNativeTimelineDraw();

    if (!nativeSync.ok) {
      setStatus(
        "Deleted the " + marker.channel + " channel key in StickRig, but the real clip timeline could not be refreshed: " +
        nativeSync.reason + ".",
        "warn"
      );
    } else if (stagingSaved === false) {
      setStatus(
        "Deleted the " + marker.channel + " key and its native keyframe data, but editable Reanimate data could not be saved.",
        "warn"
      );
    } else {
      setStatus(
        "Deleted " + marker.channel + " key " + marker.frame + " for " + (node ? node.name : "part") +
        (nativeSync.skipped ? "." : " and removed its native keyframe data on the real owning timeline immediately."),
        "good"
      );
    }
  }

  function handleTimelineOverlayKeydown(event){
    var overlay = state.timelineOverlay;
    if (!overlay.selected) return;
    if (event.key !== "Delete" && event.key !== "Backspace") return;
    var target = event.target;
    var tag = target && target.tagName ? target.tagName.toLowerCase() : "";
    if (tag === "input" || tag === "textarea" || tag === "select") return;
    event.preventDefault();
    event.stopPropagation();
    deleteTimelineMarker(overlay.selected);
  }

  function copyChannelToPose(targetPose, sourcePose, channel){
    if (channel === "move") {
      targetPose.x = n(sourcePose.x, targetPose.x);
      targetPose.y = n(sourcePose.y, targetPose.y);
    } else if (channel === "scale") {
      targetPose.scale = n(sourcePose.scale, targetPose.scale);
    } else {
      targetPose.rotation = n(sourcePose.rotation, targetPose.rotation);
    }
  }

  function moveTimelineMarker(nodeId, channel, oldFrame, newFrame){
    if (oldFrame === newFrame) return;
    var record = getNodeRecord(nodeId);
    var node = state.nodeMap[nodeId];
    if (!record || !channelKeyExists(record, oldFrame, channel)) return;

    ensureChannelMaps(record);
    var oldKey = String(oldFrame);
    var newKey = String(newFrame);
    var oldPose = copyPose(record.keys[oldKey] || interpolatePose(record, oldFrame));
    var oldType = channelType(record, oldFrame, channel);
    var oldEasing = channelEasing(record, oldFrame, channel);
    var targetPose = copyPose(record.keys[newKey] || interpolatePose(record, newFrame));
    copyChannelToPose(targetPose, oldPose, channel);

    removeChannelKeyMeta(record, oldFrame, channel);
    record.keys[newKey] = targetPose;
    if (!record.keyChannels[newKey]) record.keyChannels[newKey] = {};
    if (!record.channelTypes[newKey]) record.channelTypes[newKey] = {};
    if (!record.channelEasing[newKey]) record.channelEasing[newKey] = {};
    record.keyChannels[newKey][channel] = true;
    record.channelTypes[newKey][channel] = oldType;
    record.channelEasing[newKey][channel] = oldEasing;
    var active = CHANNELS.filter(function(name){ return record.keyChannels[newKey] && record.keyChannels[newKey][name]; });
    record.keyTypes[newKey] = active.length && active.every(function(name){ return record.channelTypes[newKey][name] === "hold"; }) ? "hold" : "auto";

    if (node) applyTransformToObject(node, interpolatePose(record, currentProjectFrame()), false);
    var nativeSync = node
      ? syncNodeNativeTweensInstantly(node, record)
      : { ok: false, skipped: false, reason: "the animated clip could not be found" };
    markProjectChanged();
    renderAll();
    var stagingSaved = saveStagingData(false);
    queueProxySync();
    requestNativeTimelineDraw();

    if (!nativeSync.ok) {
      setStatus(
        "Moved the " + channel + " key in StickRig, but the real clip timeline could not be refreshed: " + nativeSync.reason + ".",
        "warn"
      );
    } else if (stagingSaved === false) {
      setStatus(
        "Moved the " + channel + " key and updated the native keyframe, but editable Reanimate data could not be saved.",
        "warn"
      );
    } else {
      setStatus(
        "Moved " + channel + " key from " + oldFrame + " to " + newFrame +
        (nativeSync.skipped ? "." : " and updated the native keyframe on the real owning timeline immediately."),
        "good"
      );
    }
  }

  function bindTimelineMarkerEvents(element, marker, groupRect, duration){
    var doc = state.timelineOverlay.parentDocument;
    if (!doc) return;

    element.addEventListener("pointerdown", function(event){
      if (event.button !== 0) return;
      event.preventDefault();
      event.stopPropagation();

      var overlay = state.timelineOverlay;
      if (overlay.drag && typeof overlay.drag.cleanup === "function") {
        try { overlay.drag.cleanup(true); } catch(ePreviousDrag) {}
      }

      selectTimelineMarker(marker, element);
      var pointerId = event.pointerId;
      var drag = {
        marker: marker,
        element: element,
        groupRect: {
          left: groupRect.left,
          top: groupRect.top,
          width: groupRect.width,
          height: groupRect.height
        },
        duration: duration,
        pointerId: pointerId,
        startX: event.clientX,
        currentFrame: marker.frame,
        moved: false,
        active: true,
        cleanup: null
      };

      function moveDrag(moveEvent){
        if (!drag.active || moveEvent.pointerId !== pointerId) return;
        moveEvent.preventDefault();
        moveEvent.stopPropagation();
        if (Math.abs(moveEvent.clientX - drag.startX) > 2) drag.moved = true;
        if (!drag.moved) return;
        drag.currentFrame = overlayXToFrame(moveEvent.clientX, drag.groupRect, drag.duration);
        element.style.left = frameToOverlayX(drag.currentFrame, drag.duration, drag.groupRect.width) + "px";
        element.title = marker.channel.toUpperCase() + " frame " + drag.currentFrame;
      }

      function finishDrag(upEvent, cancelled){
        if (!drag.active) return;
        if (upEvent && upEvent.pointerId !== pointerId) return;
        drag.active = false;
        try { doc.removeEventListener("pointermove", moveDrag, true); } catch(eMoveOff) {}
        try { doc.removeEventListener("pointerup", finishPointer, true); } catch(eUpOff) {}
        try { doc.removeEventListener("pointercancel", cancelPointer, true); } catch(eCancelOff) {}
        try { element.releasePointerCapture(pointerId); } catch(eRelease) {}
        element.classList.remove("sr-dragging");
        if (state.timelineOverlay.drag === drag) state.timelineOverlay.drag = null;

        if (cancelled) {
          renderTimelineOverlays();
          return;
        }
        if (!drag.moved || drag.currentFrame === marker.frame) {
          renderTimelineOverlays();
          return;
        }
        moveTimelineMarker(marker.nodeId, marker.channel, marker.frame, drag.currentFrame);
      }

      function finishPointer(upEvent){
        upEvent.preventDefault();
        upEvent.stopPropagation();
        finishDrag(upEvent, false);
      }

      function cancelPointer(cancelEvent){
        finishDrag(cancelEvent, true);
      }

      drag.cleanup = function(cancelled){ finishDrag(null, cancelled !== false); };
      overlay.drag = drag;
      element.classList.add("sr-dragging");
      try { element.setPointerCapture(pointerId); } catch(eCapture) {}
      doc.addEventListener("pointermove", moveDrag, true);
      doc.addEventListener("pointerup", finishPointer, true);
      doc.addEventListener("pointercancel", cancelPointer, true);
    });

    element.addEventListener("contextmenu", function(event){
      stopTimelineMouseEvent(event);
      selectTimelineMarker(marker, element);
      showTimelineContextMenu(event, marker, {
        nodeId: marker.nodeId,
        channel: marker.channel,
        duration: duration,
        baseX: groupRect.left,
        cellWidth: groupRect.width / Math.max(1, duration)
      }, marker.frame);
    });
  }

  function resolveStickRigTimelineSpan(doc, resolvedRows, duration){
    var firstPartTop = Infinity;
    (resolvedRows || []).forEach(function(item){
      if (item && item.rowRect && isFinite(Number(item.rowRect.top))) {
        firstPartTop = Math.min(firstPartTop, Number(item.rowRect.top));
      }
    });

    var bars = collectCanvasFrameBars(doc);
    if (!bars.length) bars = collectNativeFrameBars(doc);

    var candidatesAbove = bars.filter(function(item){
      if (!item || !item.rect) return false;
      var rect = item.rect;
      if (!isFinite(Number(rect.left)) || !isFinite(Number(rect.width)) || Number(rect.width) < 80) return false;
      if (!isFinite(firstPartTop)) return true;
      var distance = firstPartTop - Number(rect.bottom);
      return distance >= -4 && distance <= 110;
    });

    candidatesAbove.sort(function(a, b){
      var aDistance = isFinite(firstPartTop) ? Math.abs(firstPartTop - Number(a.rect.bottom)) : 0;
      var bDistance = isFinite(firstPartTop) ? Math.abs(firstPartTop - Number(b.rect.bottom)) : 0;
      if (Math.abs(aDistance - bDistance) > 2) return aDistance - bDistance;
      return Number(b.rect.width) - Number(a.rect.width);
    });

    var selectedBar = candidatesAbove[0] || null;
    if (!selectedBar && bars.length) {
      selectedBar = bars.slice().sort(function(a, b){
        return Number(b.rect.width) - Number(a.rect.width);
      })[0];
    }

    if (selectedBar && selectedBar.rect) {
      return {
        left: Number(selectedBar.rect.left),
        width: Number(selectedBar.rect.width),
        source: 'wick-frame-bar',
        rulerGrid: null
      };
    }

    var rulerRows = bars.slice();
    var rulerGrid = findTimelineRulerGrid(doc, rulerRows);
    if (rulerGrid && isFinite(Number(rulerGrid.origin)) && isFinite(Number(rulerGrid.cellWidth))) {
      return {
        left: Number(rulerGrid.origin),
        width: Math.max(Number(rulerGrid.cellWidth), Number(rulerGrid.cellWidth) * Math.max(1, duration)),
        source: 'wick-ruler',
        rulerGrid: rulerGrid
      };
    }

    if (resolvedRows && resolvedRows.length) {
      var sample = resolvedRows[0];
      var fallbackRect = findTimelineRect(sample.labelRow, sample.labelElement, doc);
      if (fallbackRect && isFinite(Number(fallbackRect.left)) && isFinite(Number(fallbackRect.width)) && Number(fallbackRect.width) >= 80) {
        return {
          left: Number(fallbackRect.left),
          width: Number(fallbackRect.width),
          source: 'timeline-pane-fallback',
          rulerGrid: null
        };
      }

      /* Guaranteed fallback: the timeline begins immediately to the right of
         Wick's native layer-name column. Wick's default frame cell is about
         43 CSS pixels at normal zoom. This fallback is used only when the
         ruler and native frame bar are canvas-rendered and cannot be read. */
      var rowRights = resolvedRows.map(function(item){
        return item && item.rowRect ? Number(item.rowRect.right) : NaN;
      }).filter(function(value){ return isFinite(value); });
      var layerRight = medianNumber(rowRights);
      if (isFinite(layerRight)) {
        var viewportWidth = Math.max(
          doc.documentElement ? doc.documentElement.clientWidth : 0,
          doc.defaultView ? doc.defaultView.innerWidth : 0
        );
        var fallbackLeft = layerRight + 4;
        var fallbackCellWidth = 43;
        var requestedWidth = fallbackCellWidth * Math.max(1, duration);
        var visibleWidth = Math.max(100, viewportWidth - fallbackLeft - 8);
        return {
          left: fallbackLeft,
          width: Math.min(requestedWidth, visibleWidth),
          source: 'native-layer-edge-fallback',
          rulerGrid: {
            origin: fallbackLeft,
            cellWidth: fallbackCellWidth,
            fallback: true
          }
        };
      }
    }

    return null;
  }

  function indexOfLayerReference(layers, target){
    if (!layers || !target) return -1;
    for (var i = 0; i < layers.length; i += 1) {
      if (layers[i] === target) return i;
    }
    var targetUuid = s(safeGet(target, "uuid") || safeGet(target, "_uuid"));
    if (targetUuid) {
      for (var j = 0; j < layers.length; j += 1) {
        var uuid = s(safeGet(layers[j], "uuid") || safeGet(layers[j], "_uuid"));
        if (uuid && uuid === targetUuid) return j;
      }
    }
    return -1;
  }

  function firstLayerWithFrameIndex(layers){
    for (var i = 0; i < layers.length; i += 1) {
      if (isStickRigProxyLayer(layers[i])) continue;
      var frames = getLayerFrames(layers[i]);
      if (frames && frames.length) return i;
    }
    return -1;
  }

  /*
   * Empty StickRig layers have no native frame bars, and their names can be
   * rendered by Wick's canvas rather than HTML. Therefore the overlay must
   * not search for the text TopLeg/LowerLeg/Foot in the DOM. Instead:
   * 1. locate a real native frame bar on an existing Wick layer;
   * 2. use that bar as the horizontal frame span;
   * 3. use Wick's actual timeline layer array to calculate each empty
   *    StickRig layer's row position.
   */
  function resolveStickRigRowsFromMainFrame(doc){
    if (!doc || !state.proxy.active || !state.proxy.layers.length) return null;

    var view = doc.defaultView;
    var viewportWidth = Math.max(
      doc.documentElement ? doc.documentElement.clientWidth : 0,
      view ? view.innerWidth : 0
    );
    var viewportHeight = Math.max(
      doc.documentElement ? doc.documentElement.clientHeight : 0,
      view ? view.innerHeight : 0
    );

    var bars = collectCanvasFrameBars(doc);
    if (!bars.length) bars = collectNativeFrameBars(doc);

    bars = (bars || []).filter(function(item){
      if (!item || !item.rect) return false;
      var rect = item.rect;
      if (!isFinite(Number(rect.left)) || !isFinite(Number(rect.top))) return false;
      if (!isFinite(Number(rect.width)) || !isFinite(Number(rect.height))) return false;
      if (Number(rect.left) < 90) return false;
      if (Number(rect.left) > viewportWidth * 0.55) return false;
      if (Number(rect.top) < viewportHeight * 0.45) return false;
      if (Number(rect.height) < 20 || Number(rect.height) > 72) return false;
      if (Number(rect.width) < 80) return false;
      return true;
    });

    if (!bars.length) return null;

    /* The only native white frame bar that remains is the selected clip's
       real stage-layer span. It is the correct horizontal start, duration and
       row-height reference. Prefer the uppermost valid bar; if several share
       that row, prefer the narrower one rather than a full canvas container. */
    bars.sort(function(a, b){
      var topDiff = Number(a.rect.top) - Number(b.rect.top);
      if (Math.abs(topDiff) > 3) return topDiff;
      return Number(a.rect.width) - Number(b.rect.width);
    });

    var anchor = bars[0].rect;
    var rowStride = Math.max(20, Number(anchor.height));
    var timelineLeft = Number(anchor.left);
    var channelWidth = Number(anchor.width);

    if (!isFinite(timelineLeft) || !isFinite(channelWidth) || channelWidth < 80) return null;

    var rows = state.proxy.layers.map(function(info, index){
      var top = Number(anchor.top) + rowStride * (index + 1);
      return {
        info: info,
        rowRect: {
          left: timelineLeft,
          right: timelineLeft + channelWidth,
          top: top,
          bottom: top + rowStride,
          width: channelWidth,
          height: rowStride
        }
      };
    });

    return {
      rows: rows,
      left: timelineLeft,
      width: channelWidth,
      rowStride: rowStride,
      source: 'main-native-frame-span'
    };
  }

  function renderTimelineOverlays(){
    if (!state.proxy.active || !state.proxy.layers.length) {
      state.timelineOverlay.attachedCount = 0;
      requestNativeTimelineDraw();
      return;
    }
    installNativeTimelineRenderer();
    state.timelineOverlay.attachedCount = state.proxy.layers.length;
    requestNativeTimelineDraw();
  }

  function removeStickRigProxyLayers(){
    state.proxy.syncing = true;
    state.proxy.ignoreChangesUntil = Date.now() + 500;
    clearTimeout(state.proxySyncTimer);
    state.proxySyncTimer = null;
    var timeline = state.proxy.timeline || getMainTimeline();
    (state.proxy.layers || []).slice().forEach(function(info){
      if (info && info.layer) removeLayerFromTimeline(timeline, info.layer);
    });
    removeStaleStickRigProxyLayers(timeline);
    state.proxy.active = false;
    state.proxy.timeline = null;
    state.proxy.layers = [];
    state.proxy.previousActiveLayer = null;
    state.proxy.syncing = false;
    removeTimelineOverlay();
    markProjectChanged();
    requestNativeTimelineDraw();
  }

  function setupStickRigProxyLayers(){
    var visibleNodes = animatableNodes();
    if (!visibleNodes.length) return false;
    var timeline = getMainTimeline();
    if (!timeline) return false;

    removeStickRigProxyLayers();
    removeStaleStickRigProxyLayers(timeline);

    state.proxy.timeline = timeline;
    state.proxy.layers = [];
    try {
      state.proxy.previousActiveLayer = timeline.activeLayer;
    } catch(eActiveRead) {
      state.proxy.previousActiveLayer = null;
    }

    visibleNodes.forEach(function(node, nodeIndex){
      /* Invisible separators keep duplicate part names unique in the DOM while
         remaining visually identical in Wick's native layer-name column. */
      var displayName = node.name + new Array(nodeIndex + 2).join('\u2063');
      var layer = createLayer(displayName);

      try { layer.identifier = STICKRIG_PROXY_PREFIX + String(nodeIndex + 1); } catch(eIdentifier) {}
      try { layer.name = displayName; } catch(eName) {}
      try { layer._stickRigChannelLayer = true; } catch(eFlag) {}

      /* Deliberately add only the layer. No Wick frame and no Wick clip are
         created, therefore no native white frame oblong can appear. */
      addLayerToTimeline(timeline, layer);

      state.proxy.layers.push({
        nodeId: node.id,
        layer: layer,
        frame: null,
        clip: null,
        displayName: displayName,
        cleanName: node.name,
        virtualIndex: nodeIndex,
        lastUnionFrames: [],
        lastFramesByChannel: {
          move: [],
          scale: [],
          rotate: []
        }
      });
    });

    state.proxy.active = true;

    try {
      if (state.proxy.previousActiveLayer !== null && state.proxy.previousActiveLayer !== undefined) {
        timeline.activeLayer = state.proxy.previousActiveLayer;
      }
    } catch(eActiveRestore) {}

    installNativeTimelineRenderer();
    markProjectChanged();
    syncStickRigProxyLayers();
    renderTimelineOverlays();
    scheduleTimelineOverlayRetries();
    return true;
  }

  function getFrameTweens(frame){
    if (!frame) return [];
    var tweens = safeGet(frame, "tweens") || safeGet(frame, "_tweens") || null;
    if (tweens && typeof tweens.length === "number") return Array.prototype.slice.call(tweens);

    try {
      if (typeof frame.getChildren === "function") {
        return Array.prototype.slice.call(frame.getChildren()).filter(function(child){
          var className = s(safeGet(child, "className")).toLowerCase();
          var type = s(safeGet(child, "type")).toLowerCase();
          return className.indexOf("tween") !== -1 || type.indexOf("tween") !== -1 || safeGet(child, "playheadPosition") !== undefined;
        });
      }
    } catch(eChildren) {}

    return [];
  }

  function getTweenFrameNumber(tween){
    var value = safeGet(tween, "playheadPosition");
    if (!isFinite(Number(value))) value = safeGet(tween, "frameNumber");
    if (!isFinite(Number(value))) value = safeGet(tween, "position");
    return isFinite(Number(value)) ? Math.max(1, Math.round(Number(value))) : null;
  }

  function readProxyFrameNumbers(info){
    if (!info || !info.frame) return [];
    var seen = {};
    var frames = [];
    getFrameTweens(info.frame).forEach(function(tween){
      var frameNumber = getTweenFrameNumber(tween);
      if (frameNumber === null || seen[frameNumber]) return;
      seen[frameNumber] = true;
      frames.push(frameNumber);
    });
    frames.sort(function(a, b){ return a - b; });
    return frames;
  }

  function sameFrameNumberList(a, b){
    var left = a || [];
    var right = b || [];
    if (left.length !== right.length) return false;
    for (var i = 0; i < left.length; i += 1) {
      if (Number(left[i]) !== Number(right[i])) return false;
    }
    return true;
  }

  function unionChannelFrames(record){
    var seen = {};
    var result = [];
    CHANNELS.forEach(function(channel){
      channelKeyFrames(record, channel).forEach(function(frame){
        if (seen[frame]) return;
        seen[frame] = true;
        result.push(frame);
      });
    });
    result.sort(function(a, b){ return a - b; });
    return result;
  }

  function syncStickRigProxyLayers(){
    if (state.timelineOverlay.drag && state.timelineOverlay.drag.active) return;
    if (!state.proxy.active || !state.proxy.layers.length) return;
    state.proxy.syncing = true;
    state.proxy.ignoreChangesUntil = Date.now() + 120;
    try {
      state.proxy.layers.forEach(function(info){
        var record = getNodeRecord(info.nodeId);
        if (!record) return;
        info.lastUnionFrames = unionChannelFrames(record);
        info.lastFramesByChannel = {
          move: channelKeyFrames(record, 'move').slice(),
          scale: channelKeyFrames(record, 'scale').slice(),
          rotate: channelKeyFrames(record, 'rotate').slice()
        };
      });
    } finally {
      state.proxy.syncing = false;
      renderTimelineOverlays();
    }
  }

  function reconcileDeletedTimelineKeys(){
    if (!state.proxy.active || !state.proxy.layers.length) return;
    if (state.timelineOverlay.drag && state.timelineOverlay.drag.active) return;
    renderTimelineOverlays();
  }

  function queueProxySync(){
    clearTimeout(state.proxySyncTimer);
    state.proxySyncTimer = setTimeout(function(){
      if (!state.proxy.active && state.nodes.length) setupStickRigProxyLayers();
      else syncStickRigProxyLayers();
    }, 90);
  }

  function getParentFrame(obj){
    var direct = safeGet(obj, "parentFrame") || safeGet(obj, "_parentFrame") || null;
    if (looksLikeFrame(direct)) return direct;
    direct = safeGet(obj, "parent") || safeGet(obj, "_parent") || null;
    if (looksLikeFrame(direct)) return direct;
    return null;
  }

  function ensureLayerHasDuration(layer, duration){
    var frames = getLayerFrames(layer).slice().sort(function(a, b){ return getFrameStart(a) - getFrameStart(b); });
    if (!frames.length) {
      var newFrame = createFrame(1, duration);
      addFrameToLayer(layer, newFrame);
      return newFrame;
    }
    var last = frames[frames.length - 1];
    if (getFrameEnd(last) < duration) last.end = duration;
    return frames[0];
  }

  function ensureClipTimelineDuration(clip, duration){
    var timeline = getTimeline(clip);
    if (!timeline) throw new Error("Timeline missing from " + getObjectName(clip, "clip"));
    var layers = getTimelineLayers(timeline);
    if (!layers.length) {
      var layer = createLayer(getObjectName(clip, "Clip") + "_Layer1");
      addLayerToTimeline(timeline, layer);
      ensureLayerHasDuration(layer, duration);
      layers = getTimelineLayers(timeline);
    }
    layers.forEach(function(layer){ ensureLayerHasDuration(layer, duration); });
  }

  function createDedicatedLayerFrame(parentClip, childClip, duration, label){
    var parentTimeline = getTimeline(parentClip);
    if (!parentTimeline) throw new Error("Parent clip timeline missing for " + getObjectName(parentClip));
    var oldFrame = getParentFrame(childClip);
    if (!oldFrame) throw new Error("Could not find the existing parent frame for " + getObjectName(childClip));
    removeChildFromFrame(oldFrame, childClip);
    var layer = createLayer(label || (getObjectName(childClip, "Clip") + "_Animated"));
    addLayerToTimeline(parentTimeline, layer);
    var frame = createFrame(1, duration);
    addFrameToLayer(layer, frame);
    addChildToFrame(frame, childClip);
    return frame;
  }

  function resolveCopiedPath(rootCopy, path){
    var current = rootCopy;
    var i;
    for (i = 0; i < path.length; i += 1) {
      var segment = path[i] || {};
      var children = getTimelineChildClips(current);
      var matches = children.filter(function(child){
        var info = getObjectInfo(child);
        if (segment.identifier) return info.identifier === segment.identifier;
        if (segment.name) return info.name === segment.name;
        return true;
      });
      var occurrence = Math.max(0, Math.round(n(segment.occurrence, 0)));
      current = matches[occurrence] || matches[0] || null;
      if (!current) return null;
    }
    return current;
  }

  function copyRootHierarchy(root){
    if (!root || typeof root.copy !== "function") {
      throw new Error("The selected clip does not expose Wick's copy() method");
    }
    var copy = root.copy();
    if (!copy || !looksLikeClip(copy)) throw new Error("Wick failed to copy the selected clip hierarchy");
    return copy;
  }


  function preserveIngestedClipIdentity(sourceClip, copiedClip){
    if (!sourceClip || !copiedClip) return;
    var sourceInfo = getObjectInfo(sourceClip);
    var exactName = safeGet(sourceClip, "name");
    if (exactName === undefined || exactName === null || exactName === "") exactName = safeGet(sourceClip, "_name");

    /* Preserve the exact identifier and visible name from ingest. StickRig
       must never prepend Committed_, append _Wrapper, or otherwise rename the
       selected parent or any copied child clip. */
    if (sourceInfo.identifier) {
      try { copiedClip.identifier = sourceInfo.identifier; } catch(eIdentifier) {}
    }
    if (exactName !== undefined && exactName !== null && s(exactName) !== "") {
      try { copiedClip.name = exactName; } catch(eName) {}
      try { copiedClip._name = exactName; } catch(ePrivateName) {}
    } else if (sourceInfo.name) {
      try { copiedClip.name = sourceInfo.name; } catch(eFallbackName) {}
      try { copiedClip._name = sourceInfo.name; } catch(eFallbackPrivateName) {}
    }
  }

  /*
   * Parent-origin rule:
   * The surviving committed parent inherits the original root clip's loaded
   * X/Y origin. The copied root hierarchy is placed at local 0,0 inside that
   * parent so its native root tweens live in the parent's own timeline. Child
   * positions and artwork bounds never calculate the parent origin.
   */
  function setExactClipTransform(clip, values){
    if (!clip || !values) return;
    var current = safeGet(clip, "transformation");
    var transformation = cloneTransformation(current || {});
    transformation.x = n(values.x, 0);
    transformation.y = n(values.y, 0);
    transformation.rotation = n(values.rotation, 0);
    transformation.scaleX = n(values.scaleX, 1);
    transformation.scaleY = n(values.scaleY, 1);
    try { clip.transformation = transformation; } catch(eAssign) {}
    setOneProperty(clip, "x", transformation.x);
    setOneProperty(clip, "y", transformation.y);
    setOneProperty(clip, "rotation", transformation.rotation);
    setOneProperty(clip, "scaleX", transformation.scaleX);
    setOneProperty(clip, "scaleY", transformation.scaleY);
  }

  function rootBaseForPreservedOrigin(sourceRoot, rootRecord){
    var base = rootRecord && rootRecord.base ? rootRecord.base : readTransform(sourceRoot);
    return {
      x: round3(n(base.x, 0)),
      y: round3(n(base.y, 0)),
      scale: round3(Math.max(0.01, n(base.scale, 1))),
      rotation: round3(n(base.rotation, 0)),
      signX: n(base.signX, 1) < 0 ? -1 : 1,
      signY: n(base.signY, 1) < 0 ? -1 : 1
    };
  }

  function rootPoseRelativeToPreservedOrigin(pose, preservedBase){
    return {
      x: round3(n(pose && pose.x, preservedBase.x) - preservedBase.x),
      y: round3(n(pose && pose.y, preservedBase.y) - preservedBase.y),
      scale: round3(Math.max(0.01, n(pose && pose.scale, preservedBase.scale))),
      rotation: round3(n(pose && pose.rotation, preservedBase.rotation))
    };
  }

  function createCommittedWrapper(sourceRoot, rootCopy, duration, preservedBase){
    var sourceParentFrame = getParentFrame(sourceRoot);
    if (!sourceParentFrame) throw new Error("Could not find the selected root clip's containing stage frame");

    var origin = preservedBase || rootBaseForPreservedOrigin(sourceRoot, null);
    var sourceInfo = getObjectInfo(sourceRoot);
    var rootName = getObjectName(sourceRoot, "RootClip");
    var wrapper = createClip(sourceInfo.identifier || sourceInfo.name || rootName);
    preserveIngestedClipIdentity(sourceRoot, wrapper);
    preserveIngestedClipIdentity(sourceRoot, rootCopy);

    var wrapperTimeline = getTimeline(wrapper);
    var wrapperLayer = getTimelineLayers(wrapperTimeline)[0];
    var wrapperFrame = wrapperLayer && getLayerFrames(wrapperLayer)[0];
    if (!wrapperLayer || !wrapperFrame) throw new Error("The new parent clip did not contain its default layer and frame");
    wrapperFrame.end = duration;

    /* The replacement parent owns the original parent origin. It remains
       otherwise unrotated and unscaled, so root X/Y keys can be represented as
       offsets from the preserved origin without changing their stage result. */
    setExactClipTransform(wrapper, {
      x: origin.x,
      y: origin.y,
      rotation: 0,
      scaleX: 1,
      scaleY: 1
    });

    /* The copied ingested parent is centred on that preserved origin. Its
       original rotation, scale, signs, exact name and exact identifier are
       retained. Every copied child also keeps its ingested identity. */
    setExactClipTransform(rootCopy, {
      x: 0,
      y: 0,
      rotation: origin.rotation,
      scaleX: origin.scale * origin.signX,
      scaleY: origin.scale * origin.signY
    });

    sourceParentFrame.addClip(wrapper);
    wrapperFrame.addClip(rootCopy);

    return {
      wrapper: wrapper,
      frame: wrapperFrame,
      sourceParentFrame: sourceParentFrame,
      preservedBase: origin
    };
  }

  function directFrameObjectList(frame){
    var objects = [];
    var seen = {};
    collectFrameObjects(frame, objects, seen);
    return objects;
  }

  function directFrameClipList(frame){
    return directFrameObjectList(frame).filter(function(object){ return looksLikeClip(object); });
  }

  function prepareDirectRootCommitFrame(sourceRoot, rootCopy, duration){
    var sourceParentFrame = getParentFrame(sourceRoot);
    if (!sourceParentFrame) throw new Error("Could not find the selected root clip's containing frame");

    var sourceLayer = parentLayerOfFrame(sourceParentFrame);
    var sourceTimeline = sourceLayer ? parentTimelineOfLayer(sourceLayer) : null;
    if (!sourceTimeline) throw new Error("Could not find the selected root clip's containing timeline");

    var directObjects = directFrameObjectList(sourceParentFrame);
    var canReuseSourceFrame = directObjects.length === 1 && directObjects[0] === sourceRoot;
    var result = {
      frame: null,
      sourceParentFrame: sourceParentFrame,
      sourceLayer: sourceLayer,
      sourceTimeline: sourceTimeline,
      createdLayer: null,
      createdFrame: null,
      reusedSourceFrame: canReuseSourceFrame,
      originalFrameStart: getFrameStart(sourceParentFrame),
      originalFrameEnd: getFrameEnd(sourceParentFrame),
      originalTweens: getFrameTweens(sourceParentFrame).slice(),
      rootCopyAttached: false
    };

    if (canReuseSourceFrame) {
      /* Remove the old root immediately before inserting its replacement in
         the same frame. This keeps one TopLeg clip, one identity and one
         timeline location. The caller restores the old clip on any failure. */
      removeChildFromFrame(sourceParentFrame, sourceRoot);
      setFrameSpan(sourceParentFrame, 1, duration);
      clearTweensOnFrame(sourceParentFrame);
      addChildToFrame(sourceParentFrame, rootCopy);
      result.frame = sourceParentFrame;
      result.rootCopyAttached = true;
      result.sourceRemovedDuringPrepare = true;
      return result;
    }

    /* If the source frame contains other clips, use a dedicated sibling layer
       so root tweens cannot modify those unrelated objects. This is a layer,
       not another wrapper clip, and the source root is deleted after success. */
    var layer = createLayer(getObjectName(sourceRoot, "RootClip"));
    addLayerToTimeline(sourceTimeline, layer);
    var frame = createFrame(1, duration);
    addFrameToLayer(layer, frame);

    /* Remove the source before attaching its same-named replacement. This
       prevents Wick from ever containing two clips with the ingested identity,
       even briefly during Commit. */
    removeChildFromFrame(sourceParentFrame, sourceRoot);
    addChildToFrame(frame, rootCopy);
    result.createdLayer = layer;
    result.createdFrame = frame;
    result.frame = frame;
    result.rootCopyAttached = true;
    result.sourceRemovedDuringPrepare = true;
    return result;
  }

  function rollbackDirectRootCommit(prepared, sourceRoot, rootCopy){
    if (!prepared) return;
    try {
      if (prepared.rootCopyAttached && prepared.frame && rootCopy) {
        removeChildFromFrame(prepared.frame, rootCopy);
      }
    } catch(eRemoveCopy) {}

    try {
      if (prepared.createdLayer && prepared.sourceTimeline) {
        removeLayerFromTimeline(prepared.sourceTimeline, prepared.createdLayer);
      }
    } catch(eRemoveLayer) {}

    try {
      if (prepared.reusedSourceFrame && prepared.sourceParentFrame) {
        clearTweensOnFrame(prepared.sourceParentFrame);
        (prepared.originalTweens || []).forEach(function(tween){
          try { addTweenToFrame(prepared.sourceParentFrame, tween); } catch(eTween) {}
        });
        setFrameSpan(prepared.sourceParentFrame, prepared.originalFrameStart, prepared.originalFrameEnd);
      }
    } catch(eRestoreFrame) {}

    try {
      if (prepared.sourceParentFrame && sourceRoot) {
        var restoredClips = directFrameClipList(prepared.sourceParentFrame);
        if (restoredClips.indexOf(sourceRoot) === -1) addChildToFrame(prepared.sourceParentFrame, sourceRoot);
      }
    } catch(eRestoreSource) {}
  }

  function preflightNativeCommit(){
    var W = getWickNamespace();
    if (!W) throw new Error("Wick namespace is unavailable");
    ["Clip", "Layer", "Frame", "Tween", "Transformation"].forEach(function(className){
      if (!W[className]) throw new Error("Wick." + className + " is unavailable");
    });
    if (!state.root || typeof state.root.copy !== "function") throw new Error("Selected root clip cannot be copied safely");
    var parentFrame = getParentFrame(state.root);
    if (!parentFrame || typeof parentFrame.addClip !== "function") throw new Error("Selected root clip is not inside a writable Wick frame");
  }

  function combinedEasingForFrame(record, frameNumber){
    var active = CHANNELS.filter(function(channel){ return channelKeyExists(record, frameNumber, channel); });
    if (!active.length) return "none";
    var values = active.map(function(channel){ return channelEasing(record, frameNumber, channel); });
    if (values.indexOf("in-out") !== -1) return "in-out";
    if (values.indexOf("in") !== -1 && values.indexOf("out") !== -1) return "in-out";
    if (values.indexOf("in") !== -1) return "in";
    if (values.indexOf("out") !== -1) return "out";
    return "none";
  }

  function addNativeTweensForNode(frame, record){
    clearTweensOnFrame(frame);
    var keyedFrames = sortedKeyFrames(record);
    var framesToWrite = [];
    if (!keyedFrames.length) return 0;
    if (keyedFrames[0] !== 1) framesToWrite.push(1);
    keyedFrames.forEach(function(f){ if (framesToWrite.indexOf(f) === -1) framesToWrite.push(f); });
    framesToWrite.sort(function(a, b){ return a - b; });
    var count = 0;
    framesToWrite.forEach(function(frameNumber){
      var pose = frameNumber === 1 && !record.keys["1"] ? copyPose(record.base) : interpolatePose(record, frameNumber);
      var easing = combinedEasingForFrame(record, frameNumber);
      var tween = createTween(frameNumber, pose, record.base, easing, (TRANSITIONS[easing] || TRANSITIONS.none).colour);
      addTweenToFrame(frame, tween);
      count += 1;
    });
    return count;
  }

  function addNativeTweensForRootAtPreservedOrigin(frame, record, preservedBase){
    clearTweensOnFrame(frame);
    var keyedFrames = sortedKeyFrames(record);
    var framesToWrite = [];
    if (!keyedFrames.length) return 0;
    if (keyedFrames[0] !== 1) framesToWrite.push(1);
    keyedFrames.forEach(function(frameNumber){
      if (framesToWrite.indexOf(frameNumber) === -1) framesToWrite.push(frameNumber);
    });
    framesToWrite.sort(function(a, b){ return a - b; });

    var localBase = {
      x: 0,
      y: 0,
      scale: preservedBase.scale,
      rotation: preservedBase.rotation,
      signX: preservedBase.signX,
      signY: preservedBase.signY
    };

    var count = 0;
    framesToWrite.forEach(function(frameNumber){
      var stagePose = frameNumber === 1 && !record.keys["1"]
        ? copyPose(record.base)
        : interpolatePose(record, frameNumber);
      var localPose = rootPoseRelativeToPreservedOrigin(stagePose, preservedBase);
      var easing = combinedEasingForFrame(record, frameNumber);
      var tween = createTween(
        frameNumber,
        localPose,
        localBase,
        easing,
        (TRANSITIONS[easing] || TRANSITIONS.none).colour
      );
      addTweenToFrame(frame, tween);
      count += 1;
    });
    return count;
  }

  function setClipSynced(clip, synced){
    try { clip.isSynced = !!synced; } catch(e1) {}
    try { clip._isSynced = !!synced; } catch(e2) {}
    if (synced) {
      try { clip.animationType = "sync"; } catch(e3) {}
      try { clip._animationType = "sync"; } catch(e4) {}
    }
  }

  function isStickRigMotionLayer(layer){
    if (!layer) return false;
    var identifier = s(safeGet(layer, "identifier") || safeGet(layer, "_identifier"));
    var name = s(safeGet(layer, "name") || safeGet(layer, "_name"));
    return safeGet(layer, "_stickRigMotionLayer") === true ||
      identifier === MOTION_LAYER_ID || name === MOTION_LAYER_NAME;
  }

  function findStickRigMotionLayer(clip){
    var timeline = getTimeline(clip);
    if (!timeline) return null;
    var layers = getTimelineLayers(timeline);
    for (var i = 0; i < layers.length; i += 1) {
      if (isStickRigMotionLayer(layers[i])) return layers[i];
    }
    return null;
  }

  function findStickRigMotionFrame(clip){
    var layer = findStickRigMotionLayer(clip);
    if (!layer) return null;
    var frames = getLayerFrames(layer);
    for (var i = 0; i < frames.length; i += 1) {
      var identifier = s(safeGet(frames[i], "identifier") || safeGet(frames[i], "_identifier"));
      if (safeGet(frames[i], "_stickRigMotionFrame") === true || identifier === MOTION_FRAME_ID) return frames[i];
    }
    return frames.length ? frames[0] : null;
  }

  function ensureStickRigMotionFrame(node, duration){
    if (!node || !node.obj) throw new Error("the animated clip is missing");
    var clip = node.obj;
    var timeline = getTimeline(clip);
    if (!timeline) throw new Error("the clip's own timeline could not be found");

    var layer = findStickRigMotionLayer(clip);
    if (!layer) {
      layer = createLayer(MOTION_LAYER_NAME);
      try { layer.name = MOTION_LAYER_NAME; } catch(eName) {}
      try { layer.identifier = MOTION_LAYER_ID; } catch(eIdentifier) {}
      try { layer._stickRigMotionLayer = true; } catch(eFlag) {}
      addLayerToTimeline(timeline, layer);
    }

    var frame = findStickRigMotionFrame(clip);
    if (!frame) {
      frame = createFrame(1, duration);
      try { frame.identifier = MOTION_FRAME_ID; } catch(eFrameIdentifier) {}
      try { frame._stickRigMotionFrame = true; } catch(eFrameFlag) {}
      addFrameToLayer(layer, frame);
    }

    setFrameSpan(frame, 1, duration);
    state.liveNative.active = true;
    state.liveNative.frames[node.id] = frame;
    node.committedFrame = frame;
    return frame;
  }

  function removeStickRigMotionLayer(node){
    if (!node || !node.obj) return false;
    var clip = node.obj;
    var timeline = getTimeline(clip);
    var layer = findStickRigMotionLayer(clip);
    if (layer && timeline) removeLayerFromTimeline(timeline, layer);
    /* This layer lived inside the clip and is unrelated to the real owning
       frame used by live native tweens. Do not discard that owning-frame map. */
    removeLiveDriverFromObject(clip);
    return !!layer;
  }

  function mappedLiveNativeFrame(node){
    if (!node || !state.liveNative || !state.liveNative.frames) return null;
    var frame = state.liveNative.frames[node.id] || null;
    if (!frame) return null;
    var actual = findStickRigMotionFrame(node.obj);
    if (actual === frame) return frame;
    delete state.liveNative.frames[node.id];
    return actual;
  }

  function clearLegacyExternalTweensForNode(node){
    if (!node || !node.obj) return false;
    var parentFrame = getParentFrame(node.obj);
    if (!parentFrame || !getFrameTweens(parentFrame).length) return false;

    /* Only remove the old StickRig external tween when this clip is the sole
       object on that frame. This avoids touching a user's unrelated artwork. */
    var directObjects = directFrameObjectList(parentFrame);
    if (directObjects.length !== 1 || directObjects[0] !== node.obj) return false;
    clearTweensOnFrame(parentFrame);
    return true;
  }

  function extendClipForLiveMotion(node, duration){
    var parentFrame = getParentFrame(node.obj);
    if (parentFrame && getFrameEnd(parentFrame) < duration) {
      setFrameSpan(parentFrame, getFrameStart(parentFrame), duration);
    }

    var timeline = getTimeline(node.obj);
    if (!timeline) return;
    getTimelineLayers(timeline).forEach(function(layer){
      if (isStickRigMotionLayer(layer) || isStickRigKeyLayer(layer)) return;
      getLayerFrames(layer).forEach(function(frame){
        if (getFrameEnd(frame) < duration) setFrameSpan(frame, getFrameStart(frame), duration);
      });
    });
  }

  function isStickRigKeyLayer(layer){
    if (!layer) return false;
    var identifier = s(safeGet(layer, "identifier") || safeGet(layer, "_identifier"));
    var name = s(safeGet(layer, "name") || safeGet(layer, "_name"));
    return safeGet(layer, "_stickRigKeyLayer") === true ||
      identifier === KEY_LAYER_ID || name === KEY_LAYER_NAME;
  }

  function findStickRigKeyLayer(clip){
    var timeline = getTimeline(clip);
    if (!timeline) return null;
    var layers = getTimelineLayers(timeline);
    for (var i = 0; i < layers.length; i += 1) {
      if (isStickRigKeyLayer(layers[i])) return layers[i];
    }
    return null;
  }

  function clearFramesOnLayer(layer){
    if (!layer) return;
    getLayerFrames(layer).slice().forEach(function(frame){
      try {
        /* Remove directly so Wick does not resolve gaps by stretching one of
           the remaining keyframes across the layer while it is being rebuilt. */
        if (typeof layer.removeChild === "function") layer.removeChild(frame);
        else if (frame && typeof frame.remove === "function") frame.remove();
        else if (typeof layer.removeFrame === "function") layer.removeFrame(frame);
      } catch(eRemoveFrame) {}
    });
  }

  function removeStickRigKeyLayer(node){
    if (!node || !node.obj) return false;
    var timeline = getTimeline(node.obj);
    var layer = findStickRigKeyLayer(node.obj);
    if (layer && timeline) removeLayerFromTimeline(timeline, layer);
    /* Removing the failed internal key layer must not invalidate the real
       external owning frame where Wick stores this clip instance's tweens. */
    return !!layer;
  }

  function addFrameDirectlyToLayer(layer, frame){
    if (!layer || !frame) throw new Error("Wick keyframe layer or frame is missing");
    if (typeof layer.addChild === "function") {
      layer.addChild(frame);
      return true;
    }
    return addFrameToLayer(layer, frame);
  }

  function liveTargetEntryFrame(entry){
    if (!entry) return null;
    if (looksLikeFrame(entry)) return entry;
    return entry.frame && looksLikeFrame(entry.frame) ? entry.frame : null;
  }

  function frameDirectlyContainsObject(frame, object){
    if (!frame || !object) return false;
    return directFrameObjectList(frame).indexOf(object) !== -1;
  }

  var LIVE_LAYER_PREFIX = "__LT_STICKRIG_LIVE_LAYER__";

  function liveLayerIdentifierForNode(node){
    return LIVE_LAYER_PREFIX + validIdentifier(node && node.id ? node.id : "Clip", "Clip");
  }

  function isStickRigLiveLayer(layer){
    if (!layer) return false;
    var identifier = s(safeGet(layer, "identifier") || safeGet(layer, "_identifier"));
    return safeGet(layer, "_stickRigLiveLayer") === true || identifier.indexOf(LIVE_LAYER_PREFIX) === 0;
  }

  function liveLayerMatchesNode(layer, node){
    if (!layer || !node) return false;
    var identifier = s(safeGet(layer, "identifier") || safeGet(layer, "_identifier"));
    var storedNodeId = s(safeGet(layer, "_stickRigNodeId"));
    return storedNodeId === node.id || identifier === liveLayerIdentifierForNode(node);
  }

  function owningTimelineForNode(node){
    if (!node) return null;
    if (!node.isRoot && node.parentId && state.nodeMap[node.parentId]) {
      var parentTimeline = getTimeline(state.nodeMap[node.parentId].obj);
      if (parentTimeline) return parentTimeline;
    }
    var frame = getParentFrame(node.obj);
    var layer = frame ? parentLayerOfFrame(frame) : null;
    return layer ? parentTimelineOfLayer(layer) : getMainTimeline();
  }

  function layerFrameContainingObject(layer, object){
    if (!layer || !object) return null;
    var frames = getLayerFrames(layer);
    for (var i = 0; i < frames.length; i += 1) {
      if (frameDirectlyContainsObject(frames[i], object)) return frames[i];
    }
    return null;
  }

  function removeGeneratedLiveLayer(timeline, layer, preservedObject){
    if (!timeline || !layer) return;
    getLayerFrames(layer).slice().forEach(function(frame){
      if (preservedObject && frameDirectlyContainsObject(frame, preservedObject)) {
        try { removeChildFromFrame(frame, preservedObject); } catch(eDetach) {}
      }
    });
    removeLayerFromTimeline(timeline, layer);
  }

  function consolidateLiveLayersForNode(node, duration){
    if (!node || !node.obj) return null;
    var timeline = owningTimelineForNode(node);
    if (!timeline) return null;
    var layers = getTimelineLayers(timeline).slice();
    var generated = layers.filter(function(layer){ return liveLayerMatchesNode(layer, node); });
    if (!generated.length) return null;

    var currentFrame = getParentFrame(node.obj);
    var currentLayer = currentFrame ? parentLayerOfFrame(currentFrame) : null;
    var keepLayer = currentLayer && liveLayerMatchesNode(currentLayer, node) ? currentLayer : null;
    var keepFrame = keepLayer && frameDirectlyContainsObject(currentFrame, node.obj) ? currentFrame : null;

    if (!keepFrame) {
      for (var i = generated.length - 1; i >= 0; i -= 1) {
        var candidateFrame = layerFrameContainingObject(generated[i], node.obj);
        if (candidateFrame) {
          keepLayer = generated[i];
          keepFrame = candidateFrame;
          break;
        }
      }
    }

    /* v65 could leave many same-named generated layers. Detach the one real
       ingested clip from every generated frame, remove the duplicate layers,
       then attach that same object exactly once to the retained frame. */
    if (keepLayer && keepFrame) {
      generated.forEach(function(layer){
        getLayerFrames(layer).forEach(function(frame){
          if (frameDirectlyContainsObject(frame, node.obj)) {
            try { removeChildFromFrame(frame, node.obj); } catch(eDetachDuplicate) {}
          }
        });
      });
    }
    generated.forEach(function(layer){
      if (layer === keepLayer) return;
      removeLayerFromTimeline(timeline, layer);
    });

    if (!keepLayer || !keepFrame) return null;
    if (!frameDirectlyContainsObject(keepFrame, node.obj)) addChildToFrame(keepFrame, node.obj);
    try { keepLayer.name = node.name || getObjectName(node.obj, "Clip"); } catch(eName) {}
    try { keepLayer.identifier = liveLayerIdentifierForNode(node); } catch(eIdentifier) {}
    try { keepLayer._stickRigLiveLayer = true; } catch(eFlag) {}
    try { keepLayer._stickRigNodeId = node.id; } catch(eNodeFlag) {}
    setFrameSpan(keepFrame, 1, Math.max(1, Math.round(n(duration, 1))));

    var entry = {
      frame: keepFrame,
      layer: keepLayer,
      timeline: timeline,
      createdLayer: true
    };
    state.liveNative.active = true;
    state.liveNative.frames[node.id] = entry;
    node.committedFrame = keepFrame;
    return entry;
  }

  function ensureLiveTweenTargetFrame(node, duration){
    if (!node || !node.obj) throw new Error("the animated clip is missing");
    var safeDuration = Math.max(1, Math.round(n(duration, 1)));

    var existing = state.liveNative && state.liveNative.frames ? state.liveNative.frames[node.id] : null;
    var existingFrame = liveTargetEntryFrame(existing);
    if (existingFrame && frameDirectlyContainsObject(existingFrame, node.obj)) {
      setFrameSpan(existingFrame, 1, safeDuration);
      return existing;
    }

    var consolidated = consolidateLiveLayersForNode(node, safeDuration);
    if (consolidated) return consolidated;

    var currentFrame = getParentFrame(node.obj);
    if (!currentFrame) throw new Error("the clip is not inside a writable Wick frame");
    var currentLayer = parentLayerOfFrame(currentFrame);
    var currentTimeline = currentLayer ? parentTimelineOfLayer(currentLayer) : null;
    if (!currentLayer || !currentTimeline) throw new Error("the clip's containing timeline could not be found");

    /* A previously generated live layer is always reused, even when Wick's
       frame child getters also expose paths or other implementation objects. */
    if (liveLayerMatchesNode(currentLayer, node) && frameDirectlyContainsObject(currentFrame, node.obj)) {
      setFrameSpan(currentFrame, 1, safeDuration);
      var reusedGenerated = {
        frame: currentFrame,
        layer: currentLayer,
        timeline: currentTimeline,
        createdLayer: true
      };
      state.liveNative.active = true;
      state.liveNative.frames[node.id] = reusedGenerated;
      node.committedFrame = currentFrame;
      return reusedGenerated;
    }

    var directObjects = directFrameObjectList(currentFrame);
    var canReuseOriginal = directObjects.length === 1 && directObjects[0] === node.obj && getFrameStart(currentFrame) <= 1;
    var entry;

    if (canReuseOriginal) {
      setFrameSpan(currentFrame, 1, safeDuration);
      entry = {
        frame: currentFrame,
        layer: currentLayer,
        timeline: currentTimeline,
        createdLayer: false
      };
    } else {
      /* A Wick tween affects every object in its frame. Move this exact clip
         once to one dedicated sibling layer. Later key edits update this same
         frame in place; they never create another TopLeg/LowerLeg/Foot row. */
      removeChildFromFrame(currentFrame, node.obj);
      var layer = createLayer(node.name || getObjectName(node.obj, "Clip"));
      try { layer.name = node.name || getObjectName(node.obj, "Clip"); } catch(eLayerName) {}
      try { layer.identifier = liveLayerIdentifierForNode(node); } catch(eLayerIdentifier) {}
      try { layer._stickRigLiveLayer = true; } catch(eLayerFlag) {}
      try { layer._stickRigNodeId = node.id; } catch(eLayerNodeFlag) {}
      addLayerToTimeline(currentTimeline, layer);

      var frame = createFrame(1, safeDuration);
      addFrameToLayer(layer, frame);
      addChildToFrame(frame, node.obj);
      entry = {
        frame: frame,
        layer: layer,
        timeline: currentTimeline,
        createdLayer: true,
        previousFrame: currentFrame,
        previousLayer: currentLayer
      };
    }

    state.liveNative.active = true;
    state.liveNative.frames[node.id] = entry;
    node.committedFrame = entry.frame;
    return entry;
  }

  function clearLiveTweenDataForNode(node){
    if (!node || !node.obj) return;

    /* Native StickRig animation belongs on the frame that owns this exact
       clip instance. Clearing a channel key therefore clears and rebuilds that
       owning frame, never an overlay and never an invented internal carrier. */
    var entry = state.liveNative && state.liveNative.frames ? state.liveNative.frames[node.id] : null;
    var frame = liveTargetEntryFrame(entry) || getParentFrame(node.obj);
    if (frame && frameDirectlyContainsObject(frame, node.obj)) {
      clearTweensOnFrame(frame);
      try { if (frame.view && typeof frame.view.render === "function") frame.view.render(); } catch(eRender) {}
    }

    /* Remove only the failed empty layers created by v60-v62. These are not
       the user's animation layers and are no longer used by StickRig. */
    removeStickRigKeyLayer(node);
    removeStickRigMotionLayer(node);
    removeLiveDriverFromObject(node.obj);

    node.committedFrame = frame || null;
    if (state.liveNative && state.liveNative.frames && entry) {
      state.liveNative.frames[node.id] = entry;
    }
  }

  function rebuildActualKeyFrames(node, record, duration){
    if (!node || !node.obj) throw new Error("the animated clip is missing");

    /* This is the original working Commit arrangement, applied to the actual
       ingested hierarchy instead of to a copied wrapper hierarchy:

       - the root is tweened by the stage/project frame that owns the root;
       - every child is tweened by the frame in its real parent clip that owns
         that child;
       - Wick.Tween stores the real x, y, scaleX, scaleY and rotation values;
       - no carrier, wrapper, duplicate clip or fictional key layer exists. */
    if (!node._stickRigLegacyCleaned) {
      removeStickRigKeyLayer(node);
      removeStickRigMotionLayer(node);
      try { removeLiveDriverFromObject(node.obj); } catch(eLegacyDriver) {}
      node._stickRigLegacyCleaned = true;
    }

    var keyFrames = unionChannelFrames(record);
    if (!keyFrames.length) {
      clearLiveTweenDataForNode(node);
      var emptyFrame = getParentFrame(node.obj);
      return {
        layer: emptyFrame ? parentLayerOfFrame(emptyFrame) : null,
        frame: emptyFrame || null,
        frames: emptyFrame ? [emptyFrame] : [],
        tweens: [],
        count: 0
      };
    }

    var target = ensureLiveTweenTargetFrame(node, duration);
    if (!target || !target.frame) throw new Error("the real containing frame could not be prepared");
    if (!frameDirectlyContainsObject(target.frame, node.obj)) {
      throw new Error("the native tween target does not contain the real clip");
    }

    setFrameSpan(target.frame, 1, Math.max(1, Math.round(n(duration, 1))));

    /* Do not recreate this logic. This is the same native transform writer
       used by the original working Commit. */
    var count = addNativeTweensForNode(target.frame, record);
    var tweens = getFrameTweens(target.frame).slice();

    if (count !== tweens.length) {
      throw new Error("Wick retained " + tweens.length + " of " + count + " native tween keys");
    }

    tweens.forEach(function(tween){
      var frameNumber = getTweenFrameNumber(tween);
      var transformation = safeGet(tween, "transformation") || safeGet(tween, "_transformation");
      if (!transformation) {
        throw new Error("a native tween was created without transform data");
      }
      ["x", "y", "scaleX", "scaleY", "rotation"].forEach(function(property){
        if (!isFinite(Number(safeGet(transformation, property)))) {
          throw new Error("native tween " + frameNumber + " is missing " + property);
        }
      });
      var activeChannels = CHANNELS.filter(function(channel){
        return frameNumber !== null && channelKeyExists(record, frameNumber, channel);
      });
      try { tween._stickRigTween = true; } catch(eTweenFlag) {}
      try { tween._stickRigNodeId = node.id; } catch(eTweenNode) {}
      try { tween._stickRigKeyPosition = frameNumber; } catch(eTweenPosition) {}
      try { tween._stickRigChannels = activeChannels.slice(); } catch(eTweenChannels) {}
    });

    state.liveNative.active = true;
    state.liveNative.frames[node.id] = target;
    node.committedFrame = target.frame;

    try { if (target.frame.view && typeof target.frame.view.render === "function") target.frame.view.render(); } catch(eFrameRender) {}
    try { if (target.layer && target.layer.view && typeof target.layer.view.render === "function") target.layer.view.render(); } catch(eLayerRender) {}

    return {
      layer: target.layer,
      frame: target.frame,
      frames: [target.frame],
      tweens: tweens,
      count: count
    };
  }

  function syncNodeNativeTweensInstantly(node, record){
    if (!node || !node.obj || !record) {
      return { ok: false, skipped: false, reason: "animated node or animation record is missing", count: 0 };
    }

    try {
      var actualKeyFrames = unionChannelFrames(record);
      var lastKeyFrame = actualKeyFrames.length ? actualKeyFrames[actualKeyFrames.length - 1] : 1;
      var duration = Math.max(
        1,
        Math.round(n(getProjectDuration(), state.reanimate && state.reanimate.active ? state.reanimate.duration || 1 : 1)),
        Math.round(n(lastKeyFrame, 1))
      );

      if (!actualKeyFrames.length) {
        clearLiveTweenDataForNode(node);
        setClipSynced(node.obj, false);
        markProjectChanged();
        requestNativeTimelineDraw();
        return { ok: true, skipped: true, count: 0, frame: getParentFrame(node.obj) || null };
      }

      var rebuilt = rebuildActualKeyFrames(node, record, duration);
      setClipSynced(node.obj, true);

      /* Keep the real parent timelines long enough for Wick's synced child
         instances to play. This changes frame duration only; it does not make
         another clip or another animation store. */
      extendClipForLiveMotion(node, duration);
      markProjectChanged();
      requestNativeTimelineDraw();

      return {
        ok: true,
        skipped: false,
        count: rebuilt.count,
        frame: rebuilt.frame,
        frames: rebuilt.frames,
        layer: rebuilt.layer
      };
    } catch(error) {
      return {
        ok: false,
        skipped: false,
        reason: error && error.message ? error.message : String(error),
        count: 0
      };
    }
  }


  function queueNodeNativeTweenSync(node, record, immediate){
    if (!state.liveNative) resetLiveNativeState();
    var id = node && node.id ? node.id : "";
    if (id && state.liveNative.syncTimers[id]) {
      clearTimeout(state.liveNative.syncTimers[id]);
      delete state.liveNative.syncTimers[id];
    }
    if (immediate !== false) return syncNodeNativeTweensInstantly(node, record);

    if (!id) return { ok: false, skipped: false, reason: "animated node id is missing", count: 0 };
    state.liveNative.syncTimers[id] = setTimeout(function(){
      delete state.liveNative.syncTimers[id];
      var currentNode = state.nodeMap[id];
      var currentRecord = getNodeRecord(id);
      if (!currentNode || !currentRecord) return;
      var result = syncNodeNativeTweensInstantly(currentNode, currentRecord);
      if (!result.ok) {
        setStatus("The live native keyframe could not be refreshed: " + result.reason + ".", "warn");
      }
      requestNativeTimelineDraw();
    }, 35);
    return { ok: true, skipped: false, pending: true, count: 0 };
  }


  function migrateLoadedKeysIntoRealClipTimelines(){
    var failures = [];
    var written = 0;
    animatableNodes().forEach(function(node){
      var record = getNodeRecord(node.id);
      if (!record) return;
      var result = syncNodeNativeTweensInstantly(node, record);
      if (!result.ok) failures.push(node.name + ": " + result.reason);
      else written += Math.max(0, Math.round(n(result.count, 0)));
    });
    requestNativeTimelineDraw();
    return { ok: failures.length === 0, written: written, failures: failures };
  }

  function cleanupStickRigRuntime(){
    if (state.cleanupDone) return;
    state.cleanupDone = true;

    var nativeGui = state.nativeTimeline.lastTimelineGUI;
    var nativeOriginalDraw = state.nativeTimeline.originalDraw;

    try { clearTimeout(state.saveTimer); } catch(eSaveTimer) {}
    try { clearTimeout(state.proxySyncTimer); } catch(eProxyTimer) {}
    try { clearTimeout(state.timelineOverlay.retryTimer); } catch(eRetryTimer) {}
    try {
      Object.keys((state.liveNative && state.liveNative.syncTimers) || {}).forEach(function(id){
        clearTimeout(state.liveNative.syncTimers[id]);
      });
    } catch(eLiveTimers) {}
    try { if (state.framePollTimer) clearInterval(state.framePollTimer); } catch(eFrameTimer) {}
    try { if (state.proxyWatchTimer) clearInterval(state.proxyWatchTimer); } catch(eWatchTimer) {}
    try { if (state.lifecycleWatchTimer) clearInterval(state.lifecycleWatchTimer); } catch(eLifeTimer) {}
    state.saveTimer = null;
    state.proxySyncTimer = null;
    state.timelineOverlay.retryTimer = null;
    state.framePollTimer = null;
    state.proxyWatchTimer = null;
    state.lifecycleWatchTimer = null;

    /* Disable drawing and hit interception before removing the temporary Wick
       layers. This prevents one final StickRig renderer pass during close. */
    state.proxy.active = false;
    state.nativeTimeline.geometry = [];
    try { uninstallNativeTimelineRenderer(true); } catch(eRendererCleanup) {}
    try { removeStickRigProxyLayers(); } catch(eLayerCleanup) {}
    try { removeStaleStickRigProxyLayers(getMainTimeline()); } catch(eStaleCleanup) {}
    try { removeTimelineOverlay(); } catch(eOverlayCleanup) {}
    try { removeAllStickRigTimelineDom(); } catch(eDomCleanup) {}
    try { restoreTimelineRowStyles(); } catch(eRowCleanup) {}

    try {
      if (state.cleanupObserver) state.cleanupObserver.disconnect();
    } catch(eObserver) {}
    state.cleanupObserver = null;

    try {
      var parentWindow = window.parent && window.parent !== window ? window.parent : null;
      if (parentWindow && parentWindow[state.parentCleanupKey] === cleanupStickRigRuntime) {
        delete parentWindow[state.parentCleanupKey];
      }
    } catch(eParentKey) {}

    /* StickRig draws the coloured S/M/R strips directly into Wick's canvas.
       Clear that bitmap and invoke the original native renderer immediately,
       then again on the next two parent animation frames. */
    try { markProjectChanged(); } catch(eMark) {}
    try { forceNativeTimelineRestore(nativeGui, nativeOriginalDraw); } catch(eRestoreCanvas) {}
  }

  function stickRigFrameIsVisible(frame){
    if (!frame || !frame.isConnected) return false;
    var doc = frame.ownerDocument;
    var view = doc ? doc.defaultView : null;
    var current = frame;
    while (current && current.nodeType === 1) {
      try {
        var style = view && typeof view.getComputedStyle === "function" ? view.getComputedStyle(current) : null;
        if (style && (style.display === "none" || style.visibility === "hidden" || style.visibility === "collapse")) return false;
      } catch(eStyle) {}
      current = current.parentElement;
    }
    try {
      var rect = frame.getBoundingClientRect();
      return rect.width > 1 && rect.height > 1 && frame.getClientRects().length > 0;
    } catch(eRect) {
      return true;
    }
  }

  function loadedRigStillExists(){
    if (!state.root) return true;
    try {
      if (safeGet(state.root, "isRoot") === true) return true;
      return !!getParentFrame(state.root);
    } catch(error) {
      return false;
    }
  }

  function releaseDeletedLoadedRig(){
    if (!state.root || loadedRigStillExists()) return false;

    /* The root clip has been deleted from the stage. Remove the native tween
       frame that contained it as well, so no StickRig diamonds or arrows are
       left behind on the stage timeline. */
    try {
      var rootNode = state.nodes && state.nodes.length ? state.nodes[0] : null;
      var rootEntry = rootNode && state.liveNative && state.liveNative.frames
        ? state.liveNative.frames[rootNode.id]
        : null;
      var rootFrame = liveTargetEntryFrame(rootEntry);
      if (rootEntry && rootEntry.createdLayer && rootEntry.timeline && rootEntry.layer) {
        removeLayerFromTimeline(rootEntry.timeline, rootEntry.layer);
      } else if (rootFrame) {
        clearTweensOnFrame(rootFrame);
      }
    } catch(eRootCleanup) {}

    removeStickRigProxyLayers();
    removeTimelineOverlay();
    state.root = null;
    state.rootInfo = null;
    state.rootName = "";
    state.nodes = [];
    state.nodeMap = {};
    state.anim = { version: 1, root: null, nodes: {} };
    state.activeId = "";
    state.openIds = {};
    state.openChannels = {};
    resetReanimateState();
    resetLiveNativeState();
    $("saveBtn").disabled = true;
    $("jsonBtn").disabled = true;
    $("clearAllBtn").disabled = true;
    renderAll();
    setStatus("The loaded clip was deleted. Its StickRig keyframes were deleted with the clip.", "good");
    return true;
  }

  function installStickRigLifecycleGuard(){
    try {
      var parentWindow = window.parent && window.parent !== window ? window.parent : null;
      if (parentWindow) {
        var previousCleanup = parentWindow[state.parentCleanupKey];
        if (typeof previousCleanup === "function" && previousCleanup !== cleanupStickRigRuntime) {
          try { previousCleanup(); } catch(ePreviousCleanup) {}
        }
        parentWindow[state.parentCleanupKey] = cleanupStickRigRuntime;
      }
    } catch(eParent) {}

    try {
      var frame = window.frameElement;
      var parentDocument = frame && frame.ownerDocument ? frame.ownerDocument : null;
      if (frame && parentDocument && parentDocument.documentElement && typeof MutationObserver !== "undefined") {
        state.cleanupObserver = new MutationObserver(function(){
          try {
            if (!frame.isConnected || !parentDocument.documentElement.contains(frame)) cleanupStickRigRuntime();
          } catch(eCheck) {}
        });
        state.cleanupObserver.observe(parentDocument.documentElement, { childList: true, subtree: true, attributes: true, attributeFilter: ["style", "class", "hidden"] });
      }

      /* Luke Tools can close a tool by hiding its iframe rather than removing
         it, so unload and MutationObserver alone are not sufficient. Once the
         panel has been visibly open, two consecutive hidden checks mean Close
         was used and cleanup must run immediately. */
      if (frame) {
        state.lifecycleWatchTimer = setInterval(function(){
          if (state.cleanupDone) return;
          var visible = stickRigFrameIsVisible(frame);
          if (visible) {
            state.lifecycleSeenVisible = true;
            state.lifecycleHiddenChecks = 0;
            return;
          }
          if (!frame.isConnected) {
            cleanupStickRigRuntime();
            return;
          }
          if (state.lifecycleSeenVisible) {
            state.lifecycleHiddenChecks += 1;
            if (state.lifecycleHiddenChecks >= 2) cleanupStickRigRuntime();
          }
        }, 180);
      }
    } catch(eMutation) {}
  }

  $("readBtn").addEventListener("click", readSelectedClip);
  $("reanimateBtn").addEventListener("click", reanimateSelectedRig);
  $("saveBtn").addEventListener("click", function(){ saveStagingData(true); });
  $("jsonBtn").addEventListener("click", saveRigJson);
  $("refreshFrameBtn").addEventListener("click", function(){ refreshCurrentFrame(true); setStatus("Current Wick frame refreshed: " + state.frame + ".", "good"); });
  $("clearAllBtn").addEventListener("click", clearAllKeys);

  window.addEventListener("beforeunload", cleanupStickRigRuntime);
  window.addEventListener("pagehide", cleanupStickRigRuntime);
  window.addEventListener("unload", cleanupStickRigRuntime);

  state.framePollTimer = setInterval(function(){
    try { if (releaseDeletedLoadedRig()) return; } catch(eDeletedRig) {}
    try { refreshCurrentFrame(false); } catch(ePoll) {}
  }, 300);

  state.proxyWatchTimer = setInterval(function(){
    try { reconcileDeletedTimelineKeys(); } catch(eProxyWatch) {}
    try {
      if (state.proxy.active && !(state.timelineOverlay.drag && state.timelineOverlay.drag.active) && state.timelineOverlay.attachedCount === state.proxy.layers.length) renderTimelineOverlays();
    } catch(eOverlayWatch) {}
  }, 420);

  installStickRigLifecycleGuard();
  refreshRefs();
  try {
    /* Clean up a native draw wrapper or temporary layers left by an older
       StickRig instance before this tool becomes active. */
    var WStartup = getWickNamespace();
    if (WStartup && WStartup.GUIElement && WStartup.GUIElement.Timeline && WStartup.GUIElement.Timeline.prototype) {
      var startupProto = WStartup.GUIElement.Timeline.prototype;
      if (startupProto.__stickRigNativeOriginalDraw && typeof startupProto.__stickRigNativeOriginalDraw === "function") {
        startupProto.draw = startupProto.__stickRigNativeOriginalDraw;
        delete startupProto.__stickRigNativeOriginalDraw;
      }
      WStartup.GUIElement.Timeline.__stickRigNativeRenderer = null;
    }
  } catch(eStartupRendererCleanup) {}
  try {
    removeTimelineOverlay();
    removeStaleStickRigProxyLayers(getMainTimeline());
    markProjectChanged();
  } catch(eStartupCleanup) {}
  refreshCurrentFrame(true);
  setStatus("Ready.", "");
})();
