 function visit(obj, path, depth){

 uuid: objUUID,
 identifier: getIdentifier(obj)

 var pb = b.path.join("/").toLowerCase();

 function spliceNodeBlock(existing, blockText){

 var a = s.indexOf(start);
 var b = s.indexOf(end);
 if (blockText && String(blockText).trim()){
 var wrapped = start + "\n" + String(blockText).trim() + "\n" + end;
 if (a !== -1 && b !== -1 && b > a) return s.slice(0,a) + wrapped + s.slice(b + end.length);
 if (!s.trim()) return wrapped;
 return s.replace(/\s*$/,"") + "\n\n" + wrapped + "\n";
 }
 if (a !== -1 && b !== -1 && b > a) {
 var before = s.slice(0,a);
 var after = s.slice(b + end.length);
 return (before + after).replace(/\n{3,}/g,"\n\n").trim();
 }
 return s;
 }

 function calcNodeHeight(controls, extraPx) {
 // Per-control height based on type
 var h = 52; // header
 for (var ci=0; ci<controls.length; ci++) {
 var c = controls[ci];
 var t = c.type || "number";
 if (t === "number" || t === "slider") h += 52;
 else if (t === "select") h += 56;
 else if (t === "text") h += 58;
 else if (t === "toggle" || t === "checkbox") h += 44;
 else h += 52;
 }
 h += (extraPx || 0);
 h += 32; // port strip
 return h;
 }

 function staggerPos(x, y) {
 var attempts = 0;
 while (attempts < 8) {
 var overlap = false;
 for (var oi=0; oi<state.nodes.length; oi++) {
 var other = state.nodes[oi];
 if (Math.abs(other.x - x) < 200 && Math.abs(other.y - y) < 120) {
 overlap = true; break;
 }
 }
 if (!overlap) break;
 y += 60;
 attempts++;
 }
 return { x:x, y:y };
 }

function addScriptNodeFromDef(x, y, def){
  var sp4 = staggerPos(x, y); 
  x = sp4.x; 
  y = sp4.y;

  var n = { 
    id: uuid(), 
    kind: "ScriptNode", 
    title: String(def.label || def.id || "Tool"), 
    x: x, y: y, w: 320, ports: [], data: {}, defId: String(def.id || def.label || "") 
  };

  // Default ports: trigger in + code out
  n.ports = [
    { name: "trigger", dir: "in",  x: 10,       y: 0 },
    { name: "code",    dir: "out", x: n.w - 16, y: 0 }
  ];

  var ctrlCount = 0;
  if (def && def.ui && Array.isArray(def.ui.controls)) {
    ctrlCount = def.ui.controls.length;
  }

  // Mouse and Keyboard nodes get an extra "body" input port (wrapper-like)
  var isEventWrapper = (def && (def.id === "mouse" || def.id === "keyboard"));
  var cardH, portY;

  if (isEventWrapper) {
    // Add body port for inner code
    n.ports.push({ name: "body", dir: "in", x: n.w / 2, y: 0 }); // y repositioned later

    var clampExtra = 0; // no clamping on event nodes
    cardH = calcNodeHeight(def.ui?.controls || [], clampExtra) + 80; // extra height
    portY = cardH - 24;

    // Reposition all ports
    n.ports[0].y = portY;           // trigger
    n.ports[1].y = portY;           // code out
    n.ports[2].y = portY - 60;      // body (visually centered higher)
  } else {
    // Normal nodes
    var CLAMP_IDS = { "move":1, "setPosition":1, "rotate":1, "scale":1 };
    var clampExtra = CLAMP_IDS[def?.id] ? ctrlCount * 34 : 0;
    cardH = calcNodeHeight(def.ui?.controls || [], clampExtra);
    portY = cardH - 24;

    n.ports[0].y = portY;
    n.ports[1].y = portY;
  }

  n.h = cardH;

  // Load default values from the node's UI controls
  if (def && def.ui && Array.isArray(def.ui.controls)){
    for (var i = 0; i < def.ui.controls.length; i++){
      var c = def.ui.controls[i];
      if (!c || typeof c !== "object" || !c.key) continue;

      var key = String(c.key).trim();
      if (typeof c.default !== "undefined") {
        if (c.type === "number") {
          n.data[key] = (typeof c.default === "number") 
            ? c.default 
            : parseFloat(c.default) || 0;
        } else {
          n.data[key] = String(c.default);
        }
      }
    }
  }

  state.nodes.push(n);
  selectNode(n.id);
  tryInsertNodeIntoLink(n, n.x, n.y);
  scheduleInject();
  return n;
}

 function addWrapperNode(x, y, wrapType) {
 var sp3=staggerPos(x,y); x=sp3.x; y=sp3.y;
 var wt = WRAPPER_TYPES[wrapType];
 if (!wt) return null;
 var ctrlCount = wt.controls.length;
 // height: header(64) + controls(ctrlCount*52) + hint(32) + strip(48)
 var cardH = 64 + ctrlCount * 52 + 32 + 48;
 var portY = cardH - 24;
 var data = {};
 for (var dk in wt.defaultData) data[dk] = wt.defaultData[dk];
 data._wrapType = wrapType;
 // v2: Wrappers are initiators - only code out, no body input
 var ports = [
 { name:"code", dir:"out", x:240, y:portY }
 ];
 // If...Else still needs else port for the else branch output
 if (wt.hasElsePort) {
 ports.push({ name:"else", dir:"out", x:80, y:portY });
 }
 var n = {

 title: wt.label,
 x: x, y: y, w: 320, h: cardH,

 data: data
 };
 state.nodes.push(n);
 return n;
 }

 function tryCompleteLink(toNodeId, toPort){
  if (!state.linking || !state.linking.active) return;
  if (state.linking._completing) return;
  state.linking._completing = true;

  var fromNodeId = state.linking.fromNodeId;
  var fromPort   = state.linking.fromPort;

  if (fromNodeId === toNodeId) { cancelLink(); return; }

  var a = nodeById(fromNodeId);
  var b = nodeById(toNodeId);
  if (!a || !b) { cancelLink(); return; }

  var ok = false;

  // Normal code → clip/event slot
  if ((a.kind === "ScriptNode" || a.kind === "Wrapper") && fromPort === "code") {
    if (b.kind === "Clip" && ["default","load","update","unload","mouse","keyboard"].includes(toPort)) ok = true;
    if (b.kind === "ProxyClip" && ["default","load","update","unload","mouse","keyboard"].includes(toPort)) ok = true;
    if (b.kind === "ScriptNode" && toPort === "trigger") ok = true;
  }

  // v2: Wrapper else port → ScriptNode trigger (for If...Else)
  if (a.kind === "Wrapper" && fromPort === "else") {
    if (b.kind === "ScriptNode" && toPort === "trigger") ok = true;
    if (b.kind === "Wrapper") ok = true;
  }

  // New: code → body port on wrappers and Mouse/Keyboard nodes
  if ((a.kind === "ScriptNode" || a.kind === "Wrapper") && fromPort === "code") {
    if (b.kind === "Wrapper" && toPort === "body") {
      ok = true;
    }
    if (b.kind === "ScriptNode" && (b.defId === "mouse" || b.defId === "keyboard") && toPort === "body") {
      ok = true;
    }
  }

  // Context clip out → clip slots
  if (a.kind === "Context" && fromPort === "clip") {
    if (b.kind === "Clip" && ["default","load","update","unload","mouse","keyboard"].includes(toPort)) ok = true;
  }

  // trigger → trigger chaining
  if (fromPort === "trigger" && toPort === "trigger") ok = true;

  if (!ok) { 
    cancelLink(); 
    return; 
  }

  state.links.push({ 
    id: uuid(), 
    fromNodeId: fromNodeId, 
    fromPort: fromPort, 
    toNodeId: toNodeId, 
    toPort: toPort 
  });

  cancelLink();
  drawNodes();
  scheduleInject();
}

 // ===== Generic node line renderer =====
 // Works for ANY node def — reads all data keys and substitutes into template
 function renderNodeLine(def, data){
 // Smart code node — regenerate from template + current values
 if (def && (def.id === "smartCode" || def.id === "customCode") && data && data._template) {
 // RANDO mode: rebuild the RANDO_CONFIG array with updated values and regenerate full code
 if (data._cfgMode && data._randoMode && data._randoConfig) {
 var updatedRando = [];
 var rngCount = data._randoConfig.length;
 for (var rri = 0; rri < rngCount; rri++) {
 var origRng = data._randoConfig[rri];
 var prefix = rngCount > 1 ? "rng" + (rri+1) + "_" : "";
 var newRng = {
 varName: data[prefix+"varName"] !== undefined ? data[prefix+"varName"] : (origRng.varName || "rng"),
 min: data[prefix+"min"] !== undefined ? String(data[prefix+"min"]) : String(origRng.min || "0"),
 max: data[prefix+"max"] !== undefined ? String(data[prefix+"max"]) : String(origRng.max || "10"),
 isInt: data[prefix+"isInt"] === "true" || data[prefix+"isInt"] === true,
 onlyOnce: data[prefix+"onlyOnce"] === "true" || data[prefix+"onlyOnce"] === true,
 storeVar: data[prefix+"storeVar"] === "true" || data[prefix+"storeVar"] === true,
 rules: origRng.rules || []
 };
 updatedRando.push(newRng);
 }
 // Regenerate the Rando script from the updated config
 return buildRandoScript(updatedRando);
 }
 // CFG mode: rebuild the JSON object with updated slider values
 if (data._cfgMode && data._cfgObj && data._cfgJsonStr) {
 var updatedCfg = {};
 for (var ck in data._cfgObj) {
 if (!data._cfgObj.hasOwnProperty(ck)) continue;
 updatedCfg[ck] = (typeof data[ck] !== "undefined") ? data[ck] : data._cfgObj[ck];
 }
 var newJson = JSON.stringify(updatedCfg);
 return data._template.replace(data._cfgJsonStr, newJson);
 }
 // Template mode: substitute {{key}} placeholders
 var tpl2 = data._template;
 var controls2 = data._controls || [];
 var out2 = tpl2;
 for (var ci2 = 0; ci2 < controls2.length; ci2++) {
 var c2 = controls2[ci2];
 var val2 = (typeof data[c2.key] !== "undefined") ? data[c2.key] : c2.default;
 out2 = out2.replace(new RegExp("\\{\\{" + c2.key + "\\}\\}", "g"), String(val2));
 }
 return out2;
 }
 // customCode node: try to auto-upgrade to smart controls on first render
 if (def && def.id === "customCode" && data && data.code && !data._template) {
 var smartParsed = parseCodeToControls(data.code);
 if (smartParsed) {
 data._template = smartParsed.template;
 data._controls = smartParsed.controls;
 data._rawCode = data.code;
 data._cfgMode = smartParsed._cfgMode || false;
 data._randoMode = smartParsed._randoMode || false;
 data._randoConfig = smartParsed._randoConfig || null;
 data._cfgObj = smartParsed._cfgObj || null;
 data._cfgJsonStr = smartParsed._cfgJsonStr || null;
 data._cfgVarName = smartParsed._cfgVarName || null;
 for (var uci = 0; uci < smartParsed.controls.length; uci++) {
 var uc = smartParsed.controls[uci];
 if (typeof data[uc.key] === "undefined") data[uc.key] = uc.default;
 }
 }
 }
 if (!def || !def.script) return "";
 data = data || {};

 // Build vars map from ALL data keys on this node
 var vars = {};
 for (var k in data) vars[k] = String(data[k]);

 var sc = def.script;

 // templateByAction — key field is sc.actionKey, or "statement" for codeStatement, else "action"
 if (sc.templateByAction) {
 var actionKey = sc.actionKey
 || (def.id === "codeStatement" ? "statement" : "action");
 var actionVal = String(data[actionKey] || "");
 var tpl = sc.templateByAction[actionVal] || "";
 if (!tpl && sc.template) tpl = sc.template;
 return tpl ? renderTemplate(tpl, vars) : "";
 }

 // templateByEvent — keyed on data.event (keyboard node)
 if (sc.templateByEvent) {
 var evVal = String(data.event || "");
 var keyVal = String(data.key || "");
 // If key is blank or a catch-all, use the Any template if available
 var useAny = (!keyVal || keyVal === "ANY" || keyVal === "any");
 if (useAny && sc.templateByEventAny) {
 var tplAny = sc.templateByEventAny[evVal] || "";
 if (tplAny) return renderTemplate(tplAny, vars);
 }
 var tpl2 = sc.templateByEvent[evVal] || "";
 return tpl2 ? renderTemplate(tpl2, vars) : "";
 }

 // plain template — substitute all data vars directly
 if (sc.template) {
 return renderTemplate(sc.template, vars);
 }

 return "";
 }

 // ===== Injection mapping =====
 function buildOpsByClip(){
  var ops = {};
  function key(path){ return path.join(">>"); }
  function ensure(path, slot){
    var k = key(path);
    if (!ops[k]) ops[k] = {};
    if (!ops[k][slot]) ops[k][slot] = { path:path, lines:[] };
    return ops[k][slot];
  }
  function findIncoming(nodeId, port){
    var out = [];
    for (var i=0;i<state.links.length;i++){
      var L = state.links[i];
      if (L.toNodeId === nodeId && L.toPort === port) out.push(L);
    }
    return out;
  }
  function findOutgoing(nodeId, port){
    var out = [];
    for (var i=0;i<state.links.length;i++){
      var L = state.links[i];
      if (L.fromNodeId === nodeId && L.fromPort === port) out.push(L);
    }
    return out;
  }

  // ── 1. Pre-render ALL wrappers (top-level and nested) ──
  var consumedByWrapper = {};
  var wrapperOutput = {}; // nodeId → rendered block string

  function renderWrapperBlock(wnId, visited) {
    visited = visited || {};
    if (visited[wnId]) return wrapperOutput[wnId] || "";
    visited[wnId] = true;

    var wn = nodeById(wnId);
    if (!wn || wn.kind !== "Wrapper") return "";

    var wt = WRAPPER_TYPES[wn.data?._wrapType] || null;
    if (!wt) return "";

    var bodyLines = [];

    // v2: Wrappers are initiators - find what's connected TO our "code" output
    function collectBody(startId, vis) {
      vis = vis || {};
      if (vis[startId]) return;
      vis[startId] = true;
      var sn = nodeById(startId);
      if (!sn) return;

      if (sn.kind === "ScriptNode") {
        var bdef = NodeConfig.byId[sn.defId];
        if (bdef && bdef.script) {
          var bline = renderNodeLine(bdef, sn.data);
          if (bline && String(bline).trim()) {
            bodyLines.push("  " + String(bline).trim());
            consumedByWrapper[sn.id] = true;
          }
        }
        // Chain forward through code-out ports to find more connected nodes
        findOutgoing(startId, "code").forEach(l => {
          var nextNode = nodeById(l.toNodeId);
          if (nextNode && nextNode.kind === "ScriptNode" && l.toPort === "trigger") {
            collectBody(l.toNodeId, vis);
          }
        });
      } else if (sn.kind === "Wrapper") {
        var inner = renderWrapperBlock(sn.id, visited);
        if (inner) {
          inner.split("\n").forEach(l => bodyLines.push("  " + l));
          consumedByWrapper[sn.id] = true;
        }
      }
    }

    // v2: Find nodes connected to our "code" output port
    findOutgoing(wnId, "code").forEach(l => {
      var targetNode = nodeById(l.toNodeId);
      if (targetNode && targetNode.kind === "ScriptNode" && l.toPort === "trigger") {
        collectBody(l.toNodeId, {});
      } else if (targetNode && targetNode.kind === "Wrapper") {
        collectBody(l.toNodeId, {});
      }
    });

    var elseLines = [];
    if (wt.hasElsePort) {
      var elseBodyLines = [];
      function collectElse(startId, vis) {
        vis = vis || {};
        if (vis[startId]) return;
        vis[startId] = true;
        var en = nodeById(startId);
        if (!en) return;

        if (en.kind === "ScriptNode") {
          var edef = NodeConfig.byId[en.defId];
          if (edef && edef.script) {
            var eline = renderNodeLine(edef, en.data);
            if (eline && String(eline).trim()) {
              elseBodyLines.push("  " + String(eline).trim());
              consumedByWrapper[en.id] = true;
            }
          }
          findOutgoing(startId, "code").forEach(l => {
            var nextNode = nodeById(l.toNodeId);
            if (nextNode && nextNode.kind === "ScriptNode" && l.toPort === "trigger") {
              collectElse(l.toNodeId, vis);
            }
          });
        } else if (en.kind === "Wrapper") {
          var eInner = renderWrapperBlock(en.id, visited);
          if (eInner) {
            eInner.split("\n").forEach(l => elseBodyLines.push("  " + l));
            consumedByWrapper[en.id] = true;
          }
        }
      }
      // v2: Find nodes connected to "else" output port
      findOutgoing(wnId, "else").forEach(l => {
        var targetNode = nodeById(l.toNodeId);
        if (targetNode && targetNode.kind === "ScriptNode" && l.toPort === "trigger") {
          collectElse(l.toNodeId, {});
        } else if (targetNode && targetNode.kind === "Wrapper") {
          collectElse(l.toNodeId, {});
        }
      });
      elseLines = elseBodyLines;
    }

    var bodyStr = bodyLines.join("\n");
    var elseStr = elseLines.length ? elseLines.join("\n") : "  // no else code";

    var vars = {};
    for (var wvk in wn.data) vars[wvk] = String(wn.data[wvk]);

    var block = wt.wrapTemplate;
    for (var wvk in vars) block = block.split("{{" + wvk + "}}").join(vars[wvk]);
    block = block.split("{{body}}").join(bodyStr);
    block = block.split("{{elseBody}}").join(elseStr);

    wrapperOutput[wnId] = block;
    return block;
  }

  // Render every wrapper first (nested ones included)
  for (var wi = 0; wi < state.nodes.length; wi++) {
    var wn = state.nodes[wi];
    if (wn?.kind === "Wrapper") renderWrapperBlock(wn.id, {});
  }

  // ── 2. Push top-level (non-nested) wrappers to clips ──
  for (var wi = 0; wi < state.nodes.length; wi++) {
    var wn = state.nodes[wi];
    if (!wn || wn.kind !== "Wrapper") continue;
    if (consumedByWrapper[wn.id]) continue; // nested

    var wrapped = wrapperOutput[wn.id] || "";
    if (!wrapped.trim()) continue;

    var targets = getDownstreamClipSlots(wn.id, {});
    for (var ti = 0; ti < targets.length; ti++) {
      var tgt = targets[ti];
      var path = tgt.clipNode.data?.path || [];
      if (!path.length && tgt.clipNode.kind === "ProxyClip" && tgt.clipNode.data?.proxyName) {
        path = ["proxy::", tgt.clipNode.data.proxyName];
      }
      if (!path.length) continue;

      var slot = tgt.slotPort;
      ensure(path, slot).lines.push(wrapped);
    }
  }

  // ── 3. Regular ScriptNodes (skip if consumed by wrapper) ──
  for (var ni = 0; ni < state.nodes.length; ni++) {
    var sn = state.nodes[ni];
    if (!sn || sn.kind !== "ScriptNode") continue;
    if (consumedByWrapper[sn.id]) continue;

    var def = NodeConfig.byId[sn.defId];
    if (!def || !def.script) continue;

    var line = renderNodeLine(def, sn.data);
    if (!line || !line.trim()) continue;

    var targets = getDownstreamClipSlots(sn.id, {});
    for (var ti = 0; ti < targets.length; ti++) {
      var tgt = targets[ti];
      var path = tgt.clipNode.data?.path || [];
      if (!path.length && tgt.clipNode.kind === "ProxyClip" && tgt.clipNode.data?.proxyName) {
        path = ["proxy::", tgt.clipNode.data.proxyName];
      }
      if (!path.length) continue;

      var slot = tgt.slotPort;

      // Smart slot override for mouse/keyboard nodes
      if (def.script.slotMode === "mouseEvent") {
        slot = sn.data?.event || "mouseclick";
      } else if (def.script.slotMode === "keyEvent") {
        slot = sn.data?.event || "keypressed";
      } else if (tgt.slotPort === "mouse") {
        slot = sn.data?.event || "mouseclick";
      } else if (tgt.slotPort === "keyboard") {
        slot = sn.data?.event || "keypressed";
      } else if (def.script.slotMode === "byTargetPort") {
        slot = tgt.slotPort;
      } else if (def.script.slotDefault) {
        slot = def.script.slotDefault;
      }

      ensure(path, slot).lines.push(line.trim());
    }
  }

  // ── 4. Mouse & Keyboard as wrappers ──
  for (var ni = 0; ni < state.nodes.length; ni++) {
    var en = state.nodes[ni];
    if (!en || en.kind !== "ScriptNode") continue;
    if (en.defId !== "mouse" && en.defId !== "keyboard") continue;

    var bodyLinks = findIncoming(en.id, "body");
    if (!bodyLinks.length) continue;

    var innerLines = [];

    function collectInner(startId, vis) {
      vis = vis || new Set();
      if (vis.has(startId)) return;
      vis.add(startId);

      var inn = nodeById(startId);
      if (!inn) return;

      if (inn.kind === "ScriptNode") {
        var idef = NodeConfig.byId[inn.defId];
        if (idef?.script) {
          var iline = renderNodeLine(idef, inn.data);
          if (iline?.trim()) innerLines.push("  " + iline.trim());
        }
        findIncoming(startId, "trigger").forEach(l => collectInner(l.fromNodeId, vis));
      }
      // Could recurse into nested wrappers here if desired
    }

    bodyLinks.forEach(l => collectInner(l.fromNodeId, new Set()));

    if (!innerLines.length) continue;

    var eventName = en.data?.event || (en.defId === "mouse" ? "mouseclick" : "keypressed");
    var wrapped = `onEvent('${eventName}', function() {\n${innerLines.join("\n")}\n});`;

    var targets = getDownstreamClipSlots(en.id, {});
    for (var ti = 0; ti < targets.length; ti++) {
      var tgt = targets[ti];
      var path = tgt.clipNode.data?.path || [];
      if (!path.length && tgt.clipNode.kind === "ProxyClip" && tgt.clipNode.data?.proxyName) {
        path = ["proxy::", tgt.clipNode.data.proxyName];
      }
      if (!path.length) continue;

      var slot = en.defId === "mouse" ? "mouse" : "keyboard";
      ensure(path, slot).lines.push(wrapped);
    }
  }

  // ── 5. Context nodes ──
  for (var n = 0; n < state.nodes.length; n++) {
    var ctx = state.nodes[n];
    if (!ctx || ctx.kind !== "Context") continue;

    var slotC = String(ctx.data?.slot || "update");
    var clipLinks = findOutgoing(ctx.id, "clip");
    if (!clipLinks.length) continue;

    var clipNode = nodeById(clipLinks[0].toNodeId);
    if (!clipNode || (clipNode.kind !== "Clip" && clipNode.kind !== "ProxyClip")) continue;

    var clipPathC = clipNode.data?.path || [];
    if (!clipPathC.length && clipNode.kind === "ProxyClip" && clipNode.data?.proxyName) {
      clipPathC = ["proxy::", clipNode.data.proxyName];
    }
    if (!clipPathC.length) continue;

    var codeLinks = findIncoming(ctx.id, "code");
    if (!codeLinks.length) continue;

    var bucketC = ensure(clipPathC, slotC);

    codeLinks.forEach(LC => {
      var fromC = nodeById(LC.fromNodeId);
      if (!fromC || fromC.kind !== "ScriptNode") return;
      var defC = NodeConfig.byId[fromC.defId];
      if (!defC || !defC.script) return;
      var genLineC = renderNodeLine(defC, fromC.data);
      if (genLineC?.trim()) bucketC.lines.push(genLineC.trim());
    });
  }

  return ops;
}

 // ===== Insert node into an existing link on drop =====
 function pointNearCubicBezier(px, py, x0, y0, x1, y1, x2, y2, x3, y3, threshold) {
 // Sample the cubic bezier at N points and find min distance
 var minD = Infinity;
 var STEPS = 40;
 for (var t = 0; t <= STEPS; t++) {
 var u = t / STEPS;
 var mt = 1 - u;
 var bx = mt*mt*mt*x0 + 3*mt*mt*u*x1 + 3*mt*u*u*x2 + u*u*u*x3;
 var by = mt*mt*mt*y0 + 3*mt*mt*u*y1 + 3*mt*u*u*y2 + u*u*u*y3;
 var dx = px - bx, dy = py - by;
 var d = Math.sqrt(dx*dx + dy*dy);
 if (d < minD) minD = d;
 }
 return minD < threshold;
 }

 function tryInsertNodeIntoLink(newNode, wx, wy) {
 // Find a link whose bezier curve passes close to the drop point
 var THRESHOLD = 60; // world-space pixels
 var hitLink = null;
 for (var li = 0; li < state.links.length; li++) {
 var lk = state.links[li];
 var fromN = nodeById(lk.fromNodeId);
 var toN = nodeById(lk.toNodeId);
 if (!fromN || !toN) continue;
 // Get port world positions
 var fp = null, tp = null;
 for (var pi = 0; pi < fromN.ports.length; pi++) if (fromN.ports[pi].name === lk.fromPort) fp = fromN.ports[pi];
 for (var pi2 = 0; pi2 < toN.ports.length; pi2++) if (toN.ports[pi2].name === lk.toPort) tp = toN.ports[pi2];
 if (!fp || !tp) continue;
 var x0 = fromN.x + fp.x, y0 = fromN.y + fp.y;
 var x3 = toN.x + tp.x, y3 = toN.y + tp.y;
 var dx = Math.abs(x3 - x0) * 0.5;
 var x1 = x0 + dx, y1 = y0;
 var x2 = x3 - dx, y2 = y3;
 if (pointNearCubicBezier(wx, wy, x0, y0, x1, y1, x2, y2, x3, y3, THRESHOLD)) {
 hitLink = lk; break;
 }
 }
 if (!hitLink) return;

 // Find code-in and code-out ports on the new node
 var newIn = null, newOut = null;
 for (var npi = 0; npi < newNode.ports.length; npi++) {
 var np = newNode.ports[npi];
 if (np.dir === "in" && !newIn) newIn = np;
 if (np.dir === "out" && !newOut) newOut = np;
 }
 if (!newIn || !newOut) return;

 // Remove the original link
 state.links = state.links.filter(function(l){ return l !== hitLink; });

 // Add: fromNode.code -> newNode.codeIn
 state.links.push({ id:uuid(), fromNodeId:hitLink.fromNodeId, fromPort:hitLink.fromPort, toNodeId:newNode.id, toPort:newIn.name });
 // Add: newNode.codeOut -> toNode.slot
 state.links.push({ id:uuid(), fromNodeId:newNode.id, fromPort:newOut.name, toNodeId:hitLink.toNodeId, toPort:hitLink.toPort });

 scheduleInject();
 }

 // ===== Fit =====
 function fitToNodes(){
 if (!state.nodes.length) return;
 var minX=Infinity, minY=Infinity, maxX=-Infinity, maxY=-Infinity;
 for (var i=0;i<state.nodes.length;i++){
 var n = state.nodes[i];
 minX = Math.min(minX, n.x);
 minY = Math.min(minY, n.y);
 maxX = Math.max(maxX, n.x + n.w);
 maxY = Math.max(maxY, n.y + 160);
 }
 if (!isFinite(minX)) return;
 var pad = 120;
 var w = viewport.clientWidth || 1;
 var h = viewport.clientHeight || 1;
 var graphW = (maxX - minX) + pad*2;
 var graphH = (maxY - minY) + pad*2;
 var s = Math.min(w / graphW, h / graphH);
 s = clamp(s, 0.20, 1.40);
 state.scale = s;
 var centerX = minX + graphW/2 - pad;
 var centerY = minY + graphH/2 - pad;
 state.panX = (w/2) - (centerX * s);
 state.panY = (h/2) - (centerY * s);
 applyTransform();
 drawLinks();
 }

 // ===== Smart code parser: extract editable values from arbitrary code =====
 // Returns { controls, template } or null if nothing useful found.
 // Controls follow the same schema as nodeconfig ui.controls.
 // Template uses {{key}} placeholders matching control keys.
 function escRx(s) { return String(s).replace(/[.*+?^${}()|[\]\\]/g, "\\$&"); }

 function parseCodeToControls(code) {
 if (!code || !code.trim()) return null;
 var src = code.trim();

 // Priority 0: RANDO_CONFIG comment format (from Rando tool)
 // Matches: // RANDO_CONFIG: [{...}] or // RANDO_CONFIG: {...}
 var randoRx = /\/\/\s*RANDO_CONFIG:\s*(\[[\s\S]*?\]|\{[\s\S]*?\})/;
 var randoMatch = src.match(randoRx);
 if (randoMatch) {
 var randoJsonStr = randoMatch[1];
 var randoObj = null;
 try { randoObj = JSON.parse(randoJsonStr); } catch(eRJ) {}
 if (randoObj) {
 var randoArr = Array.isArray(randoObj) ? randoObj : [randoObj];
 var controls = [];
 // Create controls for each RNG in the array
 for (var ri = 0; ri < randoArr.length; ri++) {
 var rng = randoArr[ri];
 var prefix = randoArr.length > 1 ? "rng" + (ri+1) + "_" : "";
 // varName control
 controls.push({ type:"text", key:prefix+"varName", label:(prefix?"RNG"+(ri+1)+" ":"")+"Variable", default:rng.varName||"rng" });
 // min/max controls
 var minVal = parseFloat(rng.min) || 0;
 var maxVal = parseFloat(rng.max) || 10;
 controls.push({ type:"number", key:prefix+"min", label:(prefix?"RNG"+(ri+1)+" ":"")+"Min", default:minVal, min:-1000, max:1000, step:1 });
 controls.push({ type:"number", key:prefix+"max", label:(prefix?"RNG"+(ri+1)+" ":"")+"Max", default:maxVal, min:-1000, max:1000, step:1 });
 // isInt toggle
 controls.push({ type:"select", key:prefix+"isInt", label:(prefix?"RNG"+(ri+1)+" ":"")+"Integer", default:rng.isInt?"true":"false",
 options:[{value:"true",label:"yes"},{value:"false",label:"no"}] });
 // onlyOnce toggle

 // ===== Rando script generator (mirrors Rando.txt buildScript) =====
 function buildRandoScript(rngList) {
 if (!rngList || !rngList.length) return "";
 var L = [];
 L.push("// ── Rando BEGIN ──────────────────────────────────────────");
 L.push("// RANDO_CONFIG: " + JSON.stringify(rngList));
 for (var ri = 0; ri < rngList.length; ri++) {
 var r = rngList[ri];
 var vn = (r.varName || "rng").replace(/[^a-zA-Z0-9_]/g, "_") || ("rng" + ri);
 var mn = parseFloat(r.min) || 0;
 var mx = parseFloat(r.max) || 10;
 L.push("");
 L.push("// RNG: " + vn);
 if (r.onlyOnce) L.push("if(this._rando_" + vn + "_done===undefined){");
 var ind = r.onlyOnce ? "  " : "";
 // Roll
 if (r.isInt) {
 L.push(ind + "var " + vn + " = Math.floor(Math.random()*(" + (mx - mn + 1) + ")+" + mn + ");");
 } else {
 L.push(ind + "var " + vn + " = +(Math.random()*(" + mx + "-" + mn + ")+" + mn + ").toFixed(4);");
 }
 if (r.storeVar) L.push(ind + "this." + vn + " = " + vn + ";");
 // Rules (simplified - full rule support would need more UI)
 if (r.rules && r.rules.length) {
 for (var rui = 0; rui < r.rules.length; rui++) {
 var rule = r.rules[rui];
 if (rule.action === "none") continue;
 var hasIf = (String(rule.val || "").trim() !== "");
 var ci = ind + (hasIf ? "  " : "");
 var tgt = rule.clipName || "self";
 var tRef = (tgt === "self" || tgt === "") ? "this" : "this." + tgt;
 if (hasIf) L.push(ind + "if(" + vn + " " + rule.op + " " + String(rule.val).trim() + "){");
 var fn = rule.action, fv = String(rule.frame || "1").trim();
 if (fn === "gotoAndStop" || fn === "gotoAndPlay") {
 var fvStr = isNaN(Number(fv)) ? "\"" + fv + "\"" : fv;
 L.push(ci + "try{ " + tRef + "." + fn + "(" + fvStr + "); }catch(e){}");
 } else if (fn === "stop" || fn === "play" || fn === "gotoNextFrame" || fn === "gotoPrevFrame") {
 L.push(ci + "try{ " + tRef + "." + fn + "(); }catch(e){}");
 } else if (fn === "setOpacity") {
 L.push(ci + "try{ " + tRef + ".opacity=" + vn + "; }catch(e){}");
 } else if (fn === "setX") {
 L.push(ci + "try{ " + tRef + ".x=" + vn + "; }catch(e){}");
 } else if (fn === "setY") {
 L.push(ci + "try{ " + tRef + ".y=" + vn + "; }catch(e){}");
 }
 if (hasIf) L.push(ind + "}");
 }
 }
 if (r.onlyOnce) { L.push("  this._rando_" + vn + "_done=true;"); L.push("}"); }
 }
 L.push("");
 L.push("// ── Rando END ────────────────────────────────────────────");
 return L.join("\n");
 }


 // For a "smartCode" node, regenerate code from template + current data
 function renderSmartCode(n) {
 if (!n || !n.data || !n.data._template) return n && n.data && n.data._rawCode ? n.data._rawCode : "";
 var tpl = n.data._template;
 var result = tpl;
 var controls = n.data._controls || [];
 for (var ci = 0; ci < controls.length; ci++) {
 var c = controls[ci];
 var val = (typeof n.data[c.key] !== "undefined") ? n.data[c.key] : c.default;
 result = result.replace(new RegExp("\{\{" + c.key + "\}\}", "g"), String(val));
 }
 return result;
 }

 // ===== Try to match a script line against known node definitions =====
 // Returns { defId, data } if matched, or null
 function tryMatchNodeDef(scriptLine) {
 var line = String(scriptLine || "").trim();
 if (!line) return null;
 var defs = NodeConfig.list || [];
 for (var di = 0; di < defs.length; di++) {
 var def = defs[di];
 if (!def || !def.script || def.id === "smartCode" || def.id === "customCode") continue;
 var sc = def.script;
 // Collect all template variants for this def
 var templates = [];
 if (sc.templateByAction) {
 for (var ak in sc.templateByAction) templates.push({ actionKey: sc.actionKey || "action", actionVal: ak, tpl: sc.templateByAction[ak] });
 } else if (sc.templateByEvent) {
 for (var ek in sc.templateByEvent) templates.push({ eventKey: "event", eventVal: ek, tpl: sc.templateByEvent[ek] });
 } else if (sc.template) {
 templates.push({ tpl: sc.template });
 }
 // Build default data from ui.controls
 var defaultData = {};
 var ui = def.ui || {};
 var controls = (ui.controls && Array.isArray(ui.controls)) ? ui.controls : [];
 for (var ci3 = 0; ci3 < controls.length; ci3++) {
 var ctrl = controls[ci3];
 if (ctrl.key) defaultData[ctrl.key] = String(ctrl.default !== undefined ? ctrl.default : "");
 }
 // For each template variant, check if rendered output is a substring of the script line
 // OR if the script line matches a pattern derived from the template
 for (var ti2 = 0; ti2 < templates.length; ti2++) {
 var tv = templates[ti2];
 var testData = {};
 for (var dk in defaultData) testData[dk] = defaultData[dk];
 if (tv.actionKey) testData[tv.actionKey] = tv.actionVal;
 if (tv.eventKey) testData[tv.eventKey] = tv.actionVal;
 // Render the template with default values
 var rendered = renderNodeLine(def, testData);
 if (!rendered || !rendered.trim()) continue;
 // Build a regex from the template by replacing {{key}} with a capture pattern
 // First check: does the rendered output (with defaults) appear in the script line?
 var renderedClean = rendered.trim().replace(/\s+/g, " ");
 var lineClean = line.replace(/\s+/g, " ");
 if (renderedClean === lineClean) {
 // Exact match with defaults — use them
 var matchData = {};
 for (var dk2 in defaultData) matchData[dk2] = defaultData[dk2];
 if (tv.actionKey) matchData[tv.actionKey] = tv.actionVal;
 if (tv.eventKey) matchData[tv.eventKey] = tv.actionVal;
 return { defId: def.id, data: matchData, slotDefault: sc.slotDefault };
 }
 // Second check: build regex from template, extract actual values
 var tplStr = String(tv.tpl || "");
 if (!tplStr) continue;
 // Find all {{key}} placeholders
 var placeholderRx = /\{\{([^}]+)\}\}/g;
 var placeholders = [];
 var pm;
 while ((pm = placeholderRx.exec(tplStr)) !== null) placeholders.push(pm[1]);
 if (!placeholders.length) continue;
 // Build regex: escape the template, replace placeholders with capture groups
 var regexStr = tplStr .replace(/[.*+?^${}()|[\\\]\\\\]/g, "\\$&");
        for (var pi2 = 0; pi2 < placeholders.length; pi2++) {
          regexStr = regexStr.replace("\\{\\{" + placeholders[pi2] + "\\}\\}", "([\\w.\\-+]+)");
        }
        try {
          var rx = new RegExp("^\\s*" + regexStr + "\\s*$");
          var rxMatch = rx.exec(lineClean);
          if (rxMatch) {
            var extractedData = {};
            for (var dk3 in defaultData) extractedData[dk3] = defaultData[dk3];
            if (tv.actionKey) extractedData[tv.actionKey] = tv.actionVal;
            if (tv.eventKey) extractedData[tv.eventKey] = tv.eventVal;
            for (var pi3 = 0; pi3 < placeholders.length; pi3++) {
              extractedData[placeholders[pi3]] = rxMatch[pi3 + 1] || defaultData[placeholders[pi3]] || "";
            }
            return { defId: def.id, data: extractedData, slotDefault: sc.slotDefault };
          }
        } catch(eRx) {}
      }
    }
    return null;
  }

  // ===== Import nodes from existing clip scripts =====
  function importClipNodes(clipNode, dropX, dropY) {
    if (clipNode.data && clipNode.data.isRoot) return;
    var SLOTS = ["default", "update", "load", "unload"];
    var nodeX = dropX + 320;
    var nodeY = dropY - 80;

    // Remove existing ScriptNodes already linked to this clip
    var toRemove = [];
    for (var ni = 0; ni < state.nodes.length; ni++) {
      var nd = state.nodes[ni];
      if (nd.kind !== "ScriptNode") continue;
      for (var li = 0; li < state.links.length; li++) {
        if (state.links[li].fromNodeId === nd.id && state.links[li].toNodeId === clipNode.id) {
          toRemove.push(nd.id); break;
        }
      }
    }
    for (var ri = 0; ri < toRemove.length; ri++) {
      state.nodes = state.nodes.filter(function(n){ return n.id !== toRemove[ri]; });
      state.links = state.links.filter(function(l){ return l.fromNodeId !== toRemove[ri] && l.toNodeId !== toRemove[ri]; });
    }

    var yOff = 0;
    var createdAny = false;
    var CLAMP_IDS_IMP = { "move":1, "setPosition":1, "rotate":1, "scale":1 };

    for (var si = 0; si < SLOTS.length; si++) {
      var slot = SLOTS[si];
      var raw = "";
      try { raw = readSlotViaBridge(clipNode.data, slot) || ""; } catch(e){}
      if (!raw.trim()) continue;

      // Strip LT_NODEPANEL wrapper markers if present
      var src = raw;
      var startMark = "/* LT_NODEPANEL_START */";
      var endMark   = "/* LT_NODEPANEL_END */";
      var sa = src.indexOf(startMark);
      var sb = src.indexOf(endMark);
      if (sa !== -1 && sb !== -1 && sb > sa) src = src.slice(sa + startMark.length, sb).trim();
      if (!src.trim()) continue;

      // ── Pass 1: try to match each line to a known node def (json-config node) ──
      var lines = src.split("\n")
        .map(function(l){ return l.trim(); })
        .filter(function(l){ return l && l.indexOf("//") !== 0; });

      var matchedDefIds = {};
      var unmatchedLines = [];

      for (var lmi = 0; lmi < lines.length; lmi++) {
        var lm = tryMatchNodeDef(lines[lmi]);
        if (lm && NodeConfig.byId[lm.defId]) {
          // Don't create duplicate def nodes for same defId
          if (matchedDefIds[lm.defId]) { unmatchedLines.push(lines[lmi]); continue; }
          matchedDefIds[lm.defId] = true;
          var mDef = NodeConfig.byId[lm.defId];
          var mControls = (mDef.ui && mDef.ui.controls) ? mDef.ui.controls : [];
          var mCtrl = mControls.length;
          var mClamp = CLAMP_IDS_IMP[lm.defId] ? mCtrl * 34 : 0;
          var mH = 60 + mCtrl * 44 + mClamp + 42;
          var mSn = { id:uuid(), kind:"ScriptNode", title:mDef.label || lm.defId,
            x:nodeX, y:nodeY+yOff, w:320, h:mH,
            ports:[
              { name:"trigger", dir:"in",  x:0,   y:mH-21 },
              { name:"code",    dir:"out", x:320, y:mH-21 }
            ],
            defId:lm.defId, data:lm.data };
          state.nodes.push(mSn);
          state.links.push({ id:uuid(), fromNodeId:mSn.id, fromPort:"code", toNodeId:clipNode.id, toPort:slot });
          yOff += mH + 16;
          createdAny = true;
        } else {
          unmatchedLines.push(lines[lmi]);
        }
      }

      // ── Pass 2: remaining lines → smartCode / CFG card ──
      // Use full src for CFG parsing (LT_CFG block needs the whole script)
      var fallbackSrc = Object.keys(matchedDefIds).length > 0 ? unmatchedLines.join("\n") : src;
      if (!fallbackSrc.trim()) continue;

      var parsed = parseCodeToControls(fallbackSrc);
      if (!parsed || !parsed.controls || !parsed.controls.length) continue;

      var nodeData = {
        _template: parsed.template, _rawCode: fallbackSrc, _controls: parsed.controls,
        _cfgMode: parsed._cfgMode || false, _cfgObj: parsed._cfgObj || null,
        _cfgJsonStr: parsed._cfgJsonStr || null, _cfgVarName: parsed._cfgVarName || null
      };
      for (var ci = 0; ci < parsed.controls.length; ci++) {
        var c = parsed.controls[ci];
        nodeData[c.key] = (typeof c.default !== "undefined") ? c.default : "";
      }
      var rawTitle = parsed._cfgVarName || "";
      if (rawTitle) rawTitle = rawTitle.replace(/^LT_/,"").replace(/_/g," ").toLowerCase().replace(/\b(\w)/g,function(x){ return x.toUpperCase(); });
      var title = rawTitle || (slot.charAt(0).toUpperCase() + slot.slice(1));

      // Card height matches drawNodes smartCode formula: 60 + ctrlCount*44 + 42
      var cardH = calcNodeHeight(parsed.controls, 0);
      var sn = { id:uuid(), kind:"ScriptNode", title:title,
        x:nodeX, y:nodeY+yOff, w:320, h:cardH,
        ports:[
          { name:"trigger", dir:"in",  x:0,   y:cardH-21 },
          { name:"code",    dir:"out", x:320, y:cardH-21 }
        ],
        defId:"smartCode", data:nodeData };
      state.nodes.push(sn);
      state.links.push({ id:uuid(), fromNodeId:sn.id, fromPort:"code", toNodeId:clipNode.id, toPort:slot });
      yOff += cardH + 16;
      createdAny = true;
    }
    return createdAny;
  }

      try {

          if (pipNode) state.hoverIn._fromElementFromPoint = true;
        }
      } catch(eEFP) {}

      // Proximity snap preview: find nearest valid in-port within snap radius
      var SNAP_RADIUS = 48;
      var bestDist = SNAP_RADIUS;
      var bestNodeId = null;
      var bestPort = null;
      for (var ni=0; ni<state.nodes.length; ni++){
        var nd = state.nodes[ni];

        for (var npi=0; npi<nd.ports.length; npi++){

          if (np.dir !== "in") continue;
          var wx = nd.x + np.x;
          var wy = nd.y + np.y;
          var dx0 = w.x - wx;
          var dy0 = w.y - wy;
          var dist = Math.sqrt(dx0*dx0 + dy0*dy0);
          var effectiveDist = (np.name === "else") ? dist * 0.55 : dist;
          if (effectiveDist < bestDist){ bestDist = effectiveDist; bestNodeId = nd.id; bestPort = np.name; }
        }
      }
      // Check else ports via world coords for snap preview
      var elseSnapHit = null;
      for (var esw=0; esw<state.nodes.length; esw++) {
        var esn = state.nodes[esw];
        if (!esn || esn.kind !== "Wrapper") continue;
        var eswdef = WRAPPER_TYPES[esn.data && esn.data._wrapType];
        if (!eswdef || !eswdef.hasElsePort) continue;
        for (var esp=0; esp<esn.ports.length; esp++) {

            var espx = esn.x + esn.ports[esp].x;
            var espy = esn.y + esn.ports[esp].y;
            var esdx = w.x - espx; var esdy = w.y - espy;
            if (Math.sqrt(esdx*esdx + esdy*esdy) < 60) {
              elseSnapHit = { nodeId: esn.id, port: "else", px: espx, py: espy };
            }
            break;
          }
        }
        if (elseSnapHit) break;
      }
      var snapNodeId, snapPort;
      if (elseSnapHit) {
        snapNodeId = elseSnapHit.nodeId; snapPort = "else";
        state.linking.mouseWorldX = elseSnapHit.px;
        state.linking.mouseWorldY = elseSnapHit.py;
      } else {
        snapNodeId = state.hoverIn ? state.hoverIn.nodeId : bestNodeId;
        snapPort   = state.hoverIn ? state.hoverIn.port   : bestPort;
      }

          if (snapPip) {
            snapPip.style.transform = "scale(1.8)";
            snapPip.style.boxShadow = "0 0 12px " + (snapPort === "else" ? "rgba(255,160,80,0.9)" : "rgba(100,220,180,0.9)");
          }
        }
      } catch(eVF){}
      drawLinks();
    }
    if (!state.draggingPan) return;
    var dx = ev.clientX - state.dragStartX;
    var dy = ev.clientY - state.dragStartY;
    state.panX = state.panStartX + dx;
    state.panY = state.panStartY + dy;
    applyTransform();
    drawLinks();
  });

        var wen = state.nodes[wei];
        if (!wen || wen.kind !== "Wrapper") continue;
        var wdef = WRAPPER_TYPES[wen.data && wen.data._wrapType];
        if (!wdef || !wdef.hasElsePort) continue;
        var elsePort = null;
        for (var esp=0; esp<wen.ports.length; esp++) {
          if (wen.ports[esp].name === "else") { elsePort = wen.ports[esp]; break; }
        }
        if (!elsePort) continue;
        var elseX = wen.x + elsePort.x;
        var elseY_w = wen.y + elsePort.y;
        var distToElse = Math.sqrt(Math.pow(mw.x - elseX,2) + Math.pow(mw.y - elseY_w,2));
        var centerHalfLeft  = wen.x + wen.w * 0.25;
        var centerHalfRight = wen.x + wen.w * 0.75;
        var inCenterX = mw.x >= centerHalfLeft && mw.x <= centerHalfRight;
        var nearBottom = Math.abs(mw.y - (wen.y + wen.h - 24)) < 35;
        if (distToElse < 45 && inCenterX && nearBottom) {
          elseHit = { nodeId: wen.id, port: "else" };
          state.linking.mouseWorldX = elseX;
          state.linking.mouseWorldY = elseY_w;
        }
        if (elseHit) break;
      }
      if (elseHit) {
        tryCompleteLink(elseHit.nodeId, elseHit.port);
        return;
      }

      // 2. pipRegistry screen rect check for other special pips
      var pipHit = null;
      for (var pri=0; pri<state.pipRegistry.length; pri++) {
        var pr = state.pipRegistry[pri];
        if (!pr || !pr.el || pr.port === "else") continue; // else handled above
        try {
          var rect = pr.el.getBoundingClientRect();
          var pad = 10;
          if (ev.clientX >= rect.left - pad && ev.clientX <= rect.right + pad &&
              ev.clientY >= rect.top - pad && ev.clientY <= rect.bottom + pad) {
            pipHit = pr; break;
          }
        } catch(eR){}
      }
      if (pipHit) {
        tryCompleteLink(pipHit.nodeId, pipHit.port);
      // 3. hoverIn fallback
      } else if (state.hoverIn) {
        try { tryCompleteLink(state.hoverIn.nodeId, state.hoverIn.port); } catch(eH){ cancelLink(); }
      } else {
        // Proximity snap fallback: find nearest valid in-port within SNAP_RADIUS world-units
        var SNAP_RADIUS = 48;
        var mw = screenToWorld(ev.clientX, ev.clientY);
        var bestDist = SNAP_RADIUS;
        var bestNodeId = null;
        var bestPort = null;

        for (var ni=0; ni<state.nodes.length; ni++){
          var nd = state.nodes[ni];
          if (!nd || nd.id === state.linking.fromNodeId) continue;
          for (var npi=0; npi<nd.ports.length; npi++){
            var np = nd.ports[npi];
            if (np.dir !== "in") continue;
            var wx = nd.x + np.x;
            var wy = nd.y + np.y;
            var dx = mw.x - wx;
            var dy = mw.y - wy;
            var dist = Math.sqrt(dx*dx + dy*dy);
            if (dist < bestDist){
              bestDist = dist;
              bestNodeId = nd.id;
              bestPort = np.name;
            }
          }
        }

        if (bestNodeId) {
          tryCompleteLink(bestNodeId, bestPort);
        } else {
          cancelLink();
        }
      }
    }
    if (state.draggingPan){
      state.draggingPan = false;
      viewport.style.cursor = "grab";
      try { saveLayout(); } catch(eSP){}
    }
  });

  function loadFFAndAdd(filename) {
    if (FF_CACHE[filename]) { addFFNodeFromCode(filename, FF_CACHE[filename]); return; }
    fetch(FF_RAW_BASE + filename)
      .then(function(r){ return r.text(); })
      .then(function(code) {
        FF_CACHE[filename] = code;
        addFFNodeFromCode(filename, code);
      })
      .catch(function(e){ console.warn("[LT FF] Failed to load", filename, e); });
  }

  function addFFNodeFromCode(filename, code) {
    var parsed = parseCodeToControls(code);
    if (!parsed) { console.warn("[LT FF] No LT_CFG or RANDO_CONFIG found in", filename); return; }
    var label = filename.replace(/\.txt$/i, "");
    // Detect slot from CFG type or default to update
    var slot = "update";
    if (parsed._cfgObj && parsed._cfgObj.slot) slot = parsed._cfgObj.slot;
    else if (/solid/i.test(filename)) slot = "default";
    else if (code.indexOf("onEvent(") !== -1) slot = "default";
    var ctrlCount = parsed.controls.length;
    var nodeData = {
      _template: code, _rawCode: code, _controls: parsed.controls,
      _cfgMode: true, _cfgVarName: parsed._cfgVarName,
      _cfgJsonStr: parsed._cfgJsonStr, _cfgObj: parsed._cfgObj,
      _randoMode: parsed._randoMode || false,
      _randoConfig: parsed._randoConfig || null,
      slot: slot
    };
    for (var ci=0; ci<parsed.controls.length; ci++) {
      var ctrl = parsed.controls[ci];
      nodeData[ctrl.key] = ctrl.default !== undefined ? ctrl.default : 0;
    }
    var nH = calcNodeHeight(parsed.controls, 0);
    var portY2 = nH - 24;
    var _fp = worldCenter(); var _fsp = staggerPos(_fp.x, _fp.y);
    var nn = {
      id: uuid(), kind: "ScriptNode", defId: "smartCode",
      title: label,
      x: _fsp.x, y: _fsp.y, w: 320,
      ports: [
        { name:"trigger", dir:"in",  x:10,  y:portY2 },
        { name:"code",    dir:"out", x:310, y:portY2 }
      ],
      data: nodeData
    };
    state.nodes.push(nn);

    // Find existing clip card on canvas, or auto-create one from the active Wick selection
    var selClip2 = null;

    // 1. Check if selected node on canvas is a clip card
    if (state.selectedNodeId) {
      var sc2 = nodeById(state.selectedNodeId);
      if (sc2 && (sc2.kind === "Clip" || sc2.kind === "ProxyClip")) selClip2 = sc2;
    }
    // 2. Check for any clip card already on canvas
    if (!selClip2) {
      for (var fci=0; fci<state.nodes.length; fci++) {
        var fcn2 = state.nodes[fci];
        if (fcn2 && fcn2.id !== nn.id && (fcn2.kind === "Clip" || fcn2.kind === "ProxyClip")) { selClip2 = fcn2; break; }
      }
    }
    // 3. No clip card — resolve from the Wick selection and auto-create one
    if (!selClip2) {
      try {
        var _abr = window.parent.LukeToolsBridge || window.top.LukeToolsBridge || null;
        if (_abr) {
          // Try to get the selected clip from the editor
          var _aed = _abr.getEditor ? _abr.getEditor() : null;
          var _aobj = null;
          if (_aed) {
            // Try selection
            try {
              var _asel = (_aed.project && _aed.project.selection) ? _aed.project.selection : _aed.selection || null;
              if (_asel) {
                var _aobjs = typeof _asel.getSelectedObjects === "function" ? _asel.getSelectedObjects() : [];
                if (_aobjs && _aobjs.length) _aobj = _aobjs[0];
              }
            } catch(e0){}
            // Try focus / activeClip
            if (!_aobj) try { _aobj = _aed.focusedClip || _aed.activeClip || null; } catch(e1){}
            if (!_aobj) try { _aobj = _aed.project && (_aed.project.focus || _aed.project.activeClip); } catch(e2){}
          }
          if (_aobj) {
            var _aname  = _aobj.identifier || _aobj.name || _aobj.uuid || "clip";
            var _auuid  = String(_aobj.uuid || _aobj.UUID || _aobj.id || "");
            var _aident = String(_aobj.identifier || _aobj._identifier || "");
            var _aclipData = { path:["Project", _aname], name:_aname, uuid:_auuid, identifier:_aident, isRoot:false };
            var _acx = _fsp.x + 360;
            var _acy = _fsp.y;
            selClip2 = addClipNode(_acx, _acy, _aclipData);
          }
        }
      } catch(eAuto){}
    }

    if (selClip2) {
      state.links.push({ id: uuid(), fromNodeId: nn.id, fromPort: "code", toNodeId: selClip2.id, toPort: slot });
    }
    selectNode(nn.id);
    drawNodes();
    scheduleInject();
  }
