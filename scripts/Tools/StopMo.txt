function tryCall(obj, names, args){
  if(!obj) return { ok:false };
  for(var i=0;i<names.length;i++){
    try {
      if(typeof obj[names[i]] === 'function'){
        return { ok:true, name:names[i], out: obj[names[i]].apply(obj, args||[]) };
      }
    } catch(e){ return { ok:false, err:String(e) }; }
  }
  return { ok:false };
}

function getOrCreateLayer(tl, name){
  var W = getWick();
  // Wick timelines expose layers via .layers array (NOT ._children which holds other things)
  var layers = null;
  try { layers = tl.layers; } catch(e){}
  // fallback: _children but only if they look like layers (have .frames or .identifier)
  if(!layers || !layers.length){
    try {
      var ch = tl._children;
      if(ch && ch.length && (ch[0].frames || ch[0].identifier !== undefined)) layers = ch;
    } catch(e){}
  }
  // search existing by name
  if(layers){
    for(var i=0;i<layers.length;i++){
      var l = layers[i]; if(!l) continue;
      var n = String(l.name || l.identifier || l._name || '');
      if(n === name) return l;
    }
  }
  // create one new layer
  try {
    if(W && W.Layer){
      var nl = new W.Layer();
      try { nl.name = name; } catch(e){}
      try { nl.identifier = name; } catch(e){}
      // Try the official API first — if it works, done
      try {
        if(typeof tl.addLayer === 'function'){ tl.addLayer(nl); return nl; }
      } catch(e){}
      // Fallback: push into whichever array holds layers (only one, not both)
      try {
        if(tl.layers && Array.isArray(tl.layers)){ tl.layers.unshift(nl); try{nl.parent=tl;}catch(e){} return nl; }
      } catch(e){}
      try {
        if(tl._children && Array.isArray(tl._children)){ tl._children.unshift(nl); try{nl.parent=tl;}catch(e){} return nl; }
      } catch(e){}
    }
  } catch(e){}
  // absolute fallback: use first existing layer
  if(layers && layers.length) return layers[0];
  return null;
}
function ensureFrameAt(layer, frameNum){
  // frameNum is 1-based; Wick frame.start is 0-based
  var ph = Math.max(0, frameNum - 1);
  var W = getWick();
  // check existing
  try {
    var frames = layer.frames || layer._children;
    if(frames){
      for(var i=0;i<frames.length;i++){
        var f = frames[i]; if(!f) continue;
        var st = Number(f.start !== undefined ? f.start : (f.startFrame !== undefined ? f.startFrame : 0));
        var le = Number(f.length !== undefined ? f.length : (f.duration !== undefined ? f.duration : 1));
        if(ph >= st && ph < st+le) return f;
      }
    }
  } catch(e){}
  // create
  try {
    if(W && W.Frame){
      var nf = new W.Frame({ start:ph, length:1 });
      var r = tryCall(layer, ['addFrame'], [nf]);
      if(r.ok) return nf;
      var fa = layer.frames || layer._children;
      if(fa && Array.isArray(fa)){ fa.push(nf); try{nf.parent=layer;}catch(e){} return nf; }
    }
  } catch(e){}
  try { if(typeof layer.addBlankFrame === 'function') return layer.addBlankFrame(ph); } catch(e){}
  return null;
}
function addToFrame(frame, obj){

  try {
    var ch = frame._children || frame.children;
    if(ch && Array.isArray(ch)){ ch.push(obj); try{obj._parent=frame;}catch(e){} return true; }
  } catch(e){}
  return false;
}

// ─── Wait for asset in Wick ObjectCache ─────────────────────────────────────
function sleep(ms){ return new Promise(function(res){ setTimeout(res, ms); }); }
async function waitForAssetInCache(uuid, msMax){
  var deadline = Date.now() + (msMax || 10000);
  while(Date.now() < deadline){
    try {
      var W = getWick();
      if(W && W.ObjectCache && typeof W.ObjectCache.getObjectByUUID === 'function'){
        if(W.ObjectCache.getObjectByUUID(String(uuid))) return true;
      }
    } catch(e){}
    await sleep(60);
  }
  return false;
}

// ─── Import PNG dataURL as Wick asset ────────────────────────────────────────
async function importPNGAsset(dataUrl, name, project){
  var W = getWick();

  // Method 1: project helper methods (try all known names)
  var importMethods = [
    'addAssetFromDataURL','addAssetFromURL',
    'importAssetFromDataURL','importAssetFromURL',
    'importAsset','addImageAssetFromDataURL','addImageAssetFromURL'
  ];
  for(var i=0;i<importMethods.length;i++){
    try {
      if(typeof project[importMethods[i]] !== 'function') continue;
      var result = project[importMethods[i]](dataUrl, name);
      if(result && typeof result.then === 'function') result = await result;
      if(result && (result.uuid || result.id)){
        try { if(!result.name) result.name = name; } catch(e){}
        return result;
      }
    } catch(e){}
  }

  // Method 2: construct asset manually then add to project
  if(W){
    var ctors = ['ImageAsset','BitmapAsset','RasterAsset','Asset'];
    for(var j=0;j<ctors.length;j++){
      try {
        if(!W[ctors[j]]) continue;
        var a = new W[ctors[j]]();
        try { a.name = name; } catch(e){}
        try { a.filename = name + '.png'; } catch(e){}
        // try to assign source
        var srcProps = ['src','source','data','url','dataURL','dataUrl'];
        for(var k=0;k<srcProps.length;k++){
          try { if(srcProps[k] in a){ a[srcProps[k]] = dataUrl; break; } } catch(e){}
        }
        try { if(typeof a.loadFromURL === 'function') { a.loadFromURL(dataUrl); await sleep(200); } } catch(e){}
        var addMethods = ['addAsset','addAssets','addAssetToLibrary','libraryAddAsset'];
        for(var m=0;m<addMethods.length;m++){
          try {
            if(typeof project[addMethods[m]] === 'function'){
              project[addMethods[m]](a);
              await sleep(100);
              return a;
            }
          } catch(e){}
        }
      } catch(e){}
    }
  }

  return null;
}

// ─── Create image object from asset (via createImagePathFromAsset) ────────────
async function createImageFromAsset(asset, project, x, y){
  var W = getWick();
  if(!asset) return null;

  var uuid = String(asset.uuid || asset.id || '');

  // Wait for asset to appear in ObjectCache (needed for createImagePathFromAsset)
  if(uuid) await waitForAssetInCache(uuid, 8000);

  // Method 1: project.createImagePathFromAsset (the real Wick way)
  if(typeof project.createImagePathFromAsset === 'function' && uuid && W && W.ObjectCache){
    try {
      var assetObj = null;
      try { assetObj = W.ObjectCache.getObjectByUUID(uuid); } catch(e){}
      if(assetObj){
        var path = await new Promise(function(resolve){
          try {
            project.createImagePathFromAsset(assetObj, Number(x||0), Number(y||0), function(p){ resolve(p||null); });
          } catch(e){ resolve(null); }
        });
        if(path) return path;
      }
    } catch(e){}
  }

  // Method 2: construct image object with asset ref
  if(W){
    var imgCtors = ['Image','ImagePath','Bitmap','Raster','AssetImage','BitmapObject','ImageObject'];
    for(var i=0;i<imgCtors.length;i++){
      try {
        if(!W[imgCtors[i]]) continue;
        var o = null;
        try { o = new W[imgCtors[i]](asset); } catch(e){}
        try { if(!o) o = new W[imgCtors[i]]({asset:asset}); } catch(e){}
        try { if(!o) o = new W[imgCtors[i]]({assetUUID:uuid}); } catch(e){}
        if(o){ return o; }
      } catch(e){}
    }
  }

  // Method 3: direct dataURL path (last resort)
  return null;
}

// ─── Main capture + place function ───────────────────────────────────────────
async function captureAndPlace(){
  if(!s.stream || !vid.srcObject || vid.readyState < 2){
    setStatus('Camera not ready — press Start Camera first', 'err'); return;
  }

  var project = getProject();
  if(!project){ setStatus('Wick editor not found', 'err'); return; }

  setStatus('Capturing…');
  showFlash();

  // ── Draw webcam frame to offscreen canvas at project resolution ──
  var pw = Number(project.width) || 480;
  var ph = Number(project.height) || 270;
  var cap = document.createElement('canvas');
  cap.width = pw; cap.height = ph;
  var capCtx = cap.getContext('2d');

  var vw = vid.videoWidth  || pw;
  var vh = vid.videoHeight || ph;
  var scale = Math.max(pw/vw, ph/vh);
  var dw = vw*scale, dh = vh*scale;
  var dx = (pw-dw)*0.5, dy = (ph-dh)*0.5;

  // Apply brightness/contrast filter
  capCtx.filter = 'brightness(' + s.brightness + ') contrast(' + s.contrast + ')';
  if(s.flipH){
    capCtx.save(); capCtx.translate(pw,0); capCtx.scale(-1,1);
    capCtx.drawImage(vid, dx, dy, dw, dh);
    capCtx.restore();
  } else {
    capCtx.drawImage(vid, dx, dy, dw, dh);
  }
  capCtx.filter = 'none';

  // Save as onion skin (copy pixels)
  s.onionImageData = capCtx.getImageData(0,0,pw,ph);

  var dataUrl = cap.toDataURL('image/png');
  var frameNum = s.frameTarget;
  var imgName = 'webcam_f' + String(frameNum).padStart(3,'0') + '_' + (Date.now()%100000);

  setStatus('Importing image asset…');

  try {
    // 1. Import as asset
    var asset = await importPNGAsset(dataUrl, imgName, project);
    if(!asset){ setStatus('Asset import failed — check console', 'err'); return; }

    // 2. Use whichever layer is active in Wick right now
    var layer = getActiveLayer(project);
    if(!layer){ setStatus('No active layer — click a layer in Wick first', 'err'); return; }

    var frame = ensureFrameAt(layer, frameNum);
    if(!frame){ setStatus('Could not create frame ' + frameNum, 'err'); return; }

    // 3. Set active layer + playhead so createImagePathFromAsset lands in the right place
    setActiveLayer(project, layer);
    setPlayhead(project, frameNum);

    // 4. Create image path — Wick places it on the now-active layer/frame
    var cx = pw * 0.5, cy = ph * 0.5;
    var imgObj = await createImageFromAsset(asset, project, cx, cy);

    if(!imgObj){
      // Last resort: manually construct and add to frame
      setStatus('createImagePathFromAsset failed, trying manual insert…');
      var W = getWick();
      var ok2 = false;
      if(W){
        var imgCtors = ['ImagePath','Image','Bitmap','Raster'];
        for(var ii=0;ii<imgCtors.length;ii++){
          try {
            if(!W[imgCtors[ii]]) continue;
            var o = new W[imgCtors[ii]](asset);
            try { o.x = cx; o.y = cy; } catch(e){}
            ok2 = addToFrame(frame, o);
            if(ok2){ imgObj = o; break; }
          } catch(e){}
        }
      }
      if(!ok2){ setStatus('Could not place image', 'err'); return; }
    }

    // Position at center (createImagePathFromAsset uses x/y we passed, but reinforce)
    try { imgObj.x = cx; imgObj.y = cy; } catch(e){}
    try { imgObj.identifier = imgName; } catch(e){}
    try { imgObj.name = imgName; } catch(e){}

    markDirty();
    var ok = true;

    if(ok){
      s.captureCount++;
      updateCountUI();
      setStatus('✓ Captured to frame ' + frameNum, 'ok');

      // Advance frame
      if(s.autoAdvance){
        s.frameTarget++;
        updateFrameUI();
        setPlayhead(project, s.frameTarget);
      }
    } else {
      setStatus('Image created but could not add to frame', 'err');
    }

  } catch(e){
    setStatus('Error: ' + String(e), 'err');
  }
}

// ─── Preview loop ─────────────────────────────────────────────────────────────
function resizeCanvas(){
  var rect = canvas.parentElement.getBoundingClientRect();
  // Use 16:9 ratio based on actual width
  var w = Math.max(1, Math.round(rect.width));
  var h = Math.max(1, Math.round(rect.height));
  if(canvas.width !== w) canvas.width = w;
  if(canvas.height !== h) canvas.height = h;
}

function previewTick(){
  var w = canvas.width, h = canvas.height;

  ctx.clearRect(0,0,w,h);

  // Onion skin
  if(s.onionImageData && s.onionOpacity > 0){
    var tmp = document.createElement('canvas');
    tmp.width = s.onionImageData.width; tmp.height = s.onionImageData.height;
    tmp.getContext('2d').putImageData(s.onionImageData, 0, 0);
    ctx.globalAlpha = s.onionOpacity;
    // letterbox onion
    var ow = tmp.width, oh = tmp.height;
    var osc = Math.min(w/ow, h/oh);
    var odw = ow*osc, odh = oh*osc, odx = (w-odw)*0.5, ody = (h-odh)*0.5;
    ctx.drawImage(tmp, odx, ody, odw, odh);
    ctx.globalAlpha = 1;
  }

  // Live feed
  if(s.stream && vid.readyState >= 2){
    var vw = vid.videoWidth||w, vh = vid.videoHeight||h;
    var sc = Math.max(w/vw, h/vh);
    var dw = vw*sc, dh = vh*sc, ddx = (w-dw)*0.5, ddy = (h-dh)*0.5;
    ctx.filter = 'brightness(' + s.brightness + ') contrast(' + s.contrast + ')';
    if(s.flipH){
      ctx.save(); ctx.translate(w,0); ctx.scale(-1,1);
      ctx.drawImage(vid, ddx, ddy, dw, dh);
      ctx.restore();
    } else {
      ctx.drawImage(vid, ddx, ddy, dw, dh);
    }
    ctx.filter = 'none';
    overlay.style.display = 'none';
  } else {
    overlay.style.display = 'flex';
  }

  s.previewRAF = requestAnimationFrame(previewTick);
}

// ─── Camera start/stop ────────────────────────────────────────────────────────
async function startCamera(){
  if(s.stream) stopCamera();
  try {
    var constraints = {
      video: { deviceId: s.deviceId ? {exact:s.deviceId} : undefined, width:{ideal:1280}, height:{ideal:720} },
      audio: false
    };
    var stream = await navigator.mediaDevices.getUserMedia(constraints);
    s.stream = stream;
    vid.srcObject = stream;
    await vid.play();
    setStatus('Camera ready — press Capture Frame');
    populateDevices();
  } catch(e){
    setStatus('Camera error: ' + String(e), 'err');
  }
}
function stopCamera(){
  if(s.stream){ s.stream.getTracks().forEach(function(t){ try{t.stop();}catch(e){} }); s.stream = null; }
  vid.srcObject = null;
  setStatus('Camera stopped');
}
async function populateDevices(){
  try {
    var devs = await navigator.mediaDevices.enumerateDevices();
    var vids = devs.filter(function(d){ return d.kind==='videoinput'; });
    deviceSel.innerHTML = '<option value="">Default camera</option>';
    vids.forEach(function(d,i){
      var o = document.createElement('option');
      o.value = d.deviceId;
      o.textContent = d.label || ('Camera '+(i+1));
      if(d.deviceId === s.deviceId) o.selected = true;
      deviceSel.appendChild(o);
    });
  } catch(e){}
}
