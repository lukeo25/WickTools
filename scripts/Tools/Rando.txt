function getBridge(){
  try{ if(window.LukeToolsBridge) return window.LukeToolsBridge; }catch(e){}
  try{ if(window.parent&&window.parent.LukeToolsBridge) return window.parent.LukeToolsBridge; }catch(e){}
  try{ if(window.top&&window.top.LukeToolsBridge) return window.top.LukeToolsBridge; }catch(e){}
  return null;
}

  try{ if(window.parent.wickEditor) return window.parent.wickEditor; }catch(e){}

  return null;
}

  var ed=getEditor();
  try{ if(ed&&ed.project) return ed.project; }catch(e){}
  return null;
}

function tryLoadExisting(){
  var bridge=getBridge(); if(!bridge) return;
  if(!currentInfo||!currentInfo.uuid){ setStatus("Select a clip then click Create."); return; }
  var script="";
  try{
    if(typeof bridge.getUpdateScriptByInfo==="function"){
      script=bridge.getUpdateScriptByInfo(currentInfo)||"";
    }
  }catch(e){}
  if(!script){
    try{ if(typeof bridge.getUpdateScriptOnSelection==="function") script=bridge.getUpdateScriptOnSelection()||""; }catch(e){}
  }
  if(!script){ setStatus("Clip loaded. No existing script found. Configure and click Create."); return; }
  var parsed=parseConfig(script);
  if(parsed&&parsed.length){
    rngs=parsed;
    renderAll();
    setStatus("Loaded existing Rando config ("+parsed.length+" RNG"+(parsed.length>1?"s":"")+") . Edit and click Create to update.", "warn");
  } else {
    setStatus("Clip loaded. No existing Rando config. Configure and click Create.");
  }
}

  var bridge=getBridge();
  if(!bridge){ document.getElementById("selLabel").textContent="No bridge"; return; }

    if(typeof bridge.getSelectionInfo==="function"){

        currentInfo=info;

        rebuildAllClipDropdowns();
        tryLoadExisting();
        return;
      }
    }
  }catch(e){}
  currentInfo=null;
  document.getElementById("selLabel").textContent="No Clip Chosen";
  childNames=[];
  allTargetPaths=[];
  renderChildPills();
}

function getObjectName(obj){
  try{
    var nm=String(obj&& (obj._identifier||obj.identifier||obj.name||obj._name) || "").trim();
    return nm;
  }catch(e){ return ""; }
}

function getObjectUUID(obj){
  try{ return obj&&(obj.uuid||obj._uuid||obj.id||null); }catch(e){ return null; }
}

function getFrameChildren(frame){
  if(!frame) return [];
  return frame.clips||frame.children||frame._children||frame.objects||[];
}

function getTimelineChildrenFromClip(clip){
  var out=[];
  try{
    var tl=clip&&(clip.timeline||(clip.clipAsset&&clip.clipAsset.timeline))||null;
    if(!tl) return out;
    var layers=tl.layers||[];
    for(var li=0;li<layers.length;li++){
      var layer=layers[li]; if(!layer) continue;
      var frames=layer.frames||[];
      if(!frames.length){
        var af=layer.activeFrame||layer.currentFrame;
        if(af) frames=[af];
      }
      for(var fi=0;fi<frames.length;fi++){
        var frame=frames[fi]; if(!frame) continue;
        var kids=getFrameChildren(frame);
        for(var ci=0;ci<kids.length;ci++) out.push(kids[ci]);
      }
    }
  }catch(e){}
  if(!out.length){
    try{
      var ch=clip&& (clip._children||clip.children) || [];
      for(var i=0;i<ch.length;i++) out.push(ch[i]);
    }catch(e){}
  }
  return out;
}

function getRootProjectChildren(){
  var proj=getProject();
  var out=[];
  if(!proj) return out;
  try{
    var activeFrame = null;
    if(typeof proj.getCurrentFrame === "function") activeFrame = proj.getCurrentFrame();
    if(!activeFrame && proj.activeFrame) activeFrame = proj.activeFrame;
    if(!activeFrame && proj.focus && proj.focus.timeline && proj.focus.timeline.activeFrame) activeFrame = proj.focus.timeline.activeFrame;
    if(activeFrame){
      var kids=getFrameChildren(activeFrame);
      for(var i=0;i<kids.length;i++) out.push(kids[i]);
    }
  }catch(e){}
  try{
    if(!out.length && proj.root && proj.root.timeline){
      var layers=proj.root.timeline.layers||[];
      for(var li=0;li<layers.length;li++){
        var layer=layers[li]; if(!layer) continue;
        var frames=layer.frames||[];
        for(var fi=0;fi<frames.length;fi++){
          var kids=getFrameChildren(frames[fi]);
          for(var ci=0;ci<kids.length;ci++) out.push(kids[ci]);
        }
      }
    }
  }catch(e){}
  try{
    if(!out.length && proj.focus && proj.focus.timeline){
      var layers2=proj.focus.timeline.layers||[];
      for(var li2=0;li2<layers2.length;li2++){
        var layer2=layers2[li2]; if(!layer2) continue;
        var frames2=layer2.frames||[];
        if(!frames2.length && layer2.activeFrame) frames2=[layer2.activeFrame];
        for(var fi2=0;fi2<frames2.length;fi2++){
          var kids2=getFrameChildren(frames2[fi2]);
          for(var ci2=0;ci2<kids2.length;ci2++) out.push(kids2[ci2]);
        }
      }
    }
  }catch(e){}
  return out;
}

function scanProjectTargets(){
  childNames=[];
  allTargetPaths=[];

  var rootChildren=getRootProjectChildren();
  var seenByPath={};
  var seenByUUID={};

  function addPath(path){
    if(!path || seenByPath[path]) return;
    seenByPath[path]=true;
    allTargetPaths.push(path);
    childNames.push(path);
  }

  function walkClip(clip, parentPath){
    if(!clip) return;
    var uuid=getObjectUUID(clip);
    var nm=getObjectName(clip);
    var thisPath=parentPath;
    if(nm){
      thisPath=parentPath ? (parentPath+"."+nm) : ("project."+nm); // FIX
      addPath(thisPath);
    }
    if(uuid && seenByUUID[uuid]) return;
    if(uuid) seenByUUID[uuid]=true;
    var kids=getTimelineChildrenFromClip(clip);
    for(var i=0;i<kids.length;i++){
      walkClip(kids[i], thisPath || parentPath || "");
    }
  }

  for(var i=0;i<rootChildren.length;i++) walkClip(rootChildren[i], "");

  allTargetPaths.sort(function(a,b){ return a.localeCompare(b); });
  childNames = allTargetPaths.slice();
  renderChildPills();

  if(!allTargetPaths.length){
    setStatus("Only project was found. The project scan did not see any named clips on the active timeline. Refresh with the right clip or frame active.", "warn");
  }
}

}

function parseConfig(scriptText){
  if(!scriptText) return null;
  var m = scriptText.match(/\/\/ RANDO_CONFIG: (\{.+\}|\[.+\])/);
  if(!m) return null;
  try{
    var parsed = JSON.parse(m[1]);
    if(!Array.isArray(parsed)) return null;
    return parsed.map(function(r){
      return {
        id: mkId(),
        varName: r.varName||"rng",
        min: String(r.min||"0"),
        max: String(r.max||"10"),
        isInt: !!r.isInt,
        onlyOnce: !!r.onlyOnce,
        storeVar: !!r.storeVar,
        rules: (r.rules||[]).map(function(rule){
          return { id:mkId(), op:rule.op||"==", val:String(rule.val||""), clipName:rule.clipName||"self", action:rule.action||"none", frame:String(rule.frame||"1") };
        })
      };
    });
  }catch(e){ return null; }
}

function buildScript(rngList){
  if(!rngList.length) return "";
  var L=[];
  L.push("// Rando BEGIN");
  L.push("// RANDO_CONFIG: "+JSON.stringify(rngList));
  rngList.forEach(function(r, ri){
    var vn=(r.varName||"rng").replace(/[^a-zA-Z0-9_]/g,"_")||"rng"+ri;
    var mn=parseFloat(r.min)||0;
    var mx=parseFloat(r.max)||10;
    L.push("");
    L.push("// RNG: "+vn);
    if(r.onlyOnce) L.push("if(this._rando_"+vn+"_done===undefined){");
    var ind=r.onlyOnce?"  ":"";
    if(r.isInt){
      L.push(ind+"var "+vn+" = Math.floor(Math.random()*("+(mx-mn+1)+")+"+mn+");");
    } else {
      L.push(ind+"var "+vn+" = +(Math.random()*("+mx+"-"+mn+")+"+mn+").toFixed(4);");
    }
    if(r.storeVar) L.push(ind+"this."+vn+" = "+vn+";");
    r.rules.forEach(function(rule){
      if(rule.action==="none") return;
      var hasIf=(String(rule.val||"").trim()!=="");
      var fn=rule.action;
      var fv=String(rule.frame||"1").trim();
      var fvStr=isNaN(Number(fv))?JSON.stringify(fv):fv;
      var tgt=String(rule.clipName||"self").trim();
      var tgtExpr="this";
      if(tgt && tgt!=="self") tgtExpr=tgt;
      if(hasIf) L.push(ind+"if("+vn+" "+rule.op+" "+String(rule.val).trim()+"){");
      var ii=ind+(hasIf?"  ":"");
      if(fn==="gotoAndStop"||fn==="gotoAndPlay"){
        L.push(ii+tgtExpr+"."+fn+"("+fvStr+")");
      } else if(fn==="stop"||fn==="play"||fn==="gotoNextFrame"||fn==="gotoPrevFrame"){
        L.push(ii+tgtExpr+"."+fn+"()");
      } else if(fn==="setOpacity"){
        L.push(ii+tgtExpr+".opacity="+vn);
      } else if(fn==="setX"){
        L.push(ii+tgtExpr+".x="+vn);
      } else if(fn==="setY"){
        L.push(ii+tgtExpr+".y="+vn);
      }
      if(hasIf) L.push(ind+"};");
    });
    if(r.onlyOnce){ L.push("  this._rando_"+vn+"_done=true;"); L.push("}"); }
  });
  L.push("");
  L.push("// Rando END");
  return L.join("\n");
}

document.getElementById("btnCreate").addEventListener("click", function(){
  if(!currentInfo||!currentInfo.uuid){ setStatus("No clip selected. Use Refresh.", "warn"); return; }
  if(!rngs.length){ setStatus("No RNGs added.", "warn"); return; }
  var bridge=getBridge();
  if(!bridge){ setStatus("Bridge not found.", "bad"); return; }

  var code=buildScript(rngs);
  var existing="";
  try{ if(typeof bridge.getUpdateScriptByInfo==="function") existing=bridge.getUpdateScriptByInfo(currentInfo)||""; }catch(e){}
  if(!existing){ try{ if(typeof bridge.getUpdateScriptOnSelection==="function") existing=bridge.getUpdateScriptOnSelection()||""; }catch(e){} }
  existing=existing.replace(/\/\/ Rando BEGIN[\s\S]*?\/\/ Rando END\n?/g,"").trim();
  var combined=(existing?existing+"\n\n":"")+code;

  try{
    var res=null;
    if(typeof bridge.setUpdateScriptByInfo==="function") res=bridge.setUpdateScriptByInfo(currentInfo, combined);
    if(!res||res.ok===false){
      res=bridge.setUpdateScriptOnSelection(combined);
    }
    if(res&&res.ok===false) throw new Error(res.reason||"script write failed");
    setStatus("Injected "+rngs.length+" RNG(s) into: "+(currentInfo.identifier||currentInfo.name||currentInfo.uuid), "good");
  }catch(e){ setStatus("Failed: "+e.message, "bad"); }
});
