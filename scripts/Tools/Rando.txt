// ── Bridge ────────────────────────────────────────────────────────────────────
function getBridge(){
  try{ if(window.LukeToolsBridge) return window.LukeToolsBridge; }catch(e){}
  try{ if(window.parent&&window.parent.LukeToolsBridge) return window.parent.LukeToolsBridge; }catch(e){}
  try{ if(window.top&&window.top.LukeToolsBridge) return window.top.LukeToolsBridge; }catch(e){}
  return null;
}
function getEditor(){
  try{ if(window.parent.wickEditor) return window.parent.wickEditor; }catch(e){}
  try{ var br=getBridge(); if(br&&br.getEditor) return br.getEditor(); }catch(e){}
  return null;
}
function getProject(){
  var ed=getEditor();
  try{ if(ed&&ed.project) return ed.project; }catch(e){}
  return null;
}

var currentInfo=null;
var childNames=[];

function tryLoadExisting(){
  var bridge=getBridge();
  if(!bridge){ setStatus("No bridge found.", "bad"); return; }
  if(!currentInfo||!currentInfo.uuid){ setStatus("Select a clip then click Create."); return; }

  var script="";
  var method="none";

  // Method 1: getUpdateScriptByInfo — by UUID, no selection needed
  try{
    if(typeof bridge.getUpdateScriptByInfo==="function"){
      var r1=bridge.getUpdateScriptByInfo(currentInfo);
      if(typeof r1==="string"&&r1.length){ script=r1; method="byInfo"; }
    }
  }catch(e){ setStatus("getUpdateScriptByInfo error: "+e.message,"bad"); return; }

  // Method 2: direct clip object walk
  if(!script){
    try{
      var ed=getEditor(), proj=getProject();
      var clip=null;
      var sel=proj&&proj.selection||ed&&ed.selection||null;
      if(sel&&typeof sel.getSelectedObject==="function") clip=sel.getSelectedObject();
      if(!clip&&ed&&ed.selectedObjects) clip=ed.selectedObjects[0]||null;
      if(clip){
        var target=clip.clipAsset||clip;
        // Try every known script reading method
        if(!script&&typeof target.getScript==="function"){ var gs=target.getScript("update"); if(typeof gs==="string") script=gs; else if(gs&&gs.source) script=gs.source; }
        if(!script&&target.scripts&&target.scripts.update){ var us=target.scripts.update; script=typeof us==="string"?us:(us.src||us.source||us.code||us.script||""); }
        if(!script&&target.updateScript) script=target.updateScript;
        if(script) method="directClip";
      }
    }catch(e){}
  }

  // Method 3: getUpdateScriptOnSelection
  if(!script){
    try{
      if(typeof bridge.getUpdateScriptOnSelection==="function"){
        var r3=bridge.getUpdateScriptOnSelection();
        if(typeof r3==="string"&&r3.length){ script=r3; method="onSelection"; }
      }
    }catch(e){}
  }

  if(!script){
    setStatus("Clip loaded ("+( currentInfo.identifier||currentInfo.uuid)+") — no script found. Configure and click Create.");
    return;
  }

  var hasRando=script.indexOf("Rando BEGIN")>=0;
  if(!hasRando){
    setStatus("Clip loaded — script exists ("+method+") but no Rando block. Configure and click Create.");
    return;
  }

  var parsed=parseConfig(script);
  if(parsed&&parsed.length){
    rngs=parsed;
    renderAll();
    setStatus("✎ Loaded "+parsed.length+" RNG(s) via "+method+" — edit and click Create to update.", "warn");
  } else {
    setStatus("Rando block found but config comment missing — cannot parse. Re-inject to upgrade.", "warn");
  }
}

function refreshSelection(){
  var bridge=getBridge();
  if(!bridge){ document.getElementById("selLabel").textContent="No bridge"; return; }
  try{
    if(typeof bridge.getSelectionInfo==="function"){
      var info=bridge.getSelectionInfo();
      if(info&&info.uuid){
        currentInfo=info;
        document.getElementById("selLabel").textContent=info.identifier||info.name||info.uuid;
        scanChildren();
        rebuildAllClipDropdowns();
        // Try to load existing Rando config from clip's update script
        tryLoadExisting();
        return;
      }
    }
  }catch(e){}
  currentInfo=null;
  document.getElementById("selLabel").textContent="No Clip Chosen";
  childNames=[];
  renderChildPills();
}

function scanChildren(){
  childNames=[];
  var ed=getEditor(), proj=getProject();
  if(!currentInfo||!currentInfo.uuid){ renderChildPills(); return; }

  var clip=null;
  try{ var sel=proj&&proj.selection||ed&&ed.selection||null;
    if(sel&&typeof sel.getSelectedObject==="function") clip=sel.getSelectedObject();
    if(!clip&&sel&&typeof sel.getSelectedObjects==="function"){ var a=sel.getSelectedObjects(); clip=a&&a[0]||null; }
    if(!clip&&ed&&ed.selectedObjects) clip=ed.selectedObjects[0]||null;
  }catch(e){}
  if(!clip){ renderChildPills(); return; }

  var found=[], seen={};
  function addClip(c){
    if(!c) return;
    var u=c.uuid||c._uuid||null;
    if(u&&seen[u]) return; if(u) seen[u]=true;
    var nm=String(c._identifier||c.identifier||c.name||"").trim();
    if(nm) found.push(nm);
  }
  try{
    var tl=clip.timeline||(clip.clipAsset&&clip.clipAsset.timeline)||null;
    if(tl){
      var layers=tl.layers||[];
      for(var li=0;li<layers.length;li++){
        var layer=layers[li]; if(!layer) continue;
        var frames=layer.frames||[];
        if(!frames.length){ var af=layer.activeFrame||layer.currentFrame; if(af) frames=[af]; }
        for(var fi=0;fi<frames.length;fi++){
          var frame=frames[fi]; if(!frame) continue;
          var clips=frame.clips||frame.children||frame._children||frame.objects||[];
          for(var ci=0;ci<clips.length;ci++) addClip(clips[ci]);
        }
      }
    }
  }catch(e){}
  if(!found.length){ try{ var ch=clip._children||clip.children||[]; for(var i=0;i<ch.length;i++) addClip(ch[i]); }catch(e){} }
  childNames=found;
  renderChildPills();
}

}

// ── Code generation ───────────────────────────────────────────────────────────
// Parse a saved RANDO_CONFIG comment back into rngs state
function parseConfig(scriptText){
  if(!scriptText) return null;
  var m = scriptText.match(/\/\/ RANDO_CONFIG: (\{.+\}|\[.+\])/);
  if(!m) return null;
  try{
    var parsed = JSON.parse(m[1]);
    if(!Array.isArray(parsed)) return null;
    // Re-stamp with fresh IDs so DOM elements don't clash
    return parsed.map(function(r){
      return {
        id: mkId(),
        varName: r.varName||"rng",
        min: String(r.min||"0"),
        max: String(r.max||"10"),
        isInt: !!r.isInt,
        onlyOnce: !!r.onlyOnce,
        storeVar: !!r.storeVar,
        rules: (r.rules||[]).map(function(rule){
          return { id:mkId(), op:rule.op||"==", val:String(rule.val||""), clipName:rule.clipName||"self", action:rule.action||"none", frame:String(rule.frame||"1") };
        })
      };
    });
  }catch(e){ return null; }
}

function buildScript(rngList){
  if(!rngList.length) return "";
  var L=[];
  L.push("// ── Rando BEGIN ──────────────────────────────────────────");
  L.push("// RANDO_CONFIG: "+JSON.stringify(rngList));
  rngList.forEach(function(r, ri){
    var vn=(r.varName||"rng").replace(/[^a-zA-Z0-9_]/g,"_")||"rng"+ri;
    var mn=parseFloat(r.min)||0;
    var mx=parseFloat(r.max)||10;
    L.push("");
    L.push("// RNG: "+vn);
    if(r.onlyOnce) L.push("if(this._rando_"+vn+"_done===undefined){");
    var ind=r.onlyOnce?"  ":"";

    // Roll
    if(r.isInt){
      L.push(ind+"var "+vn+" = Math.floor(Math.random()*("+(mx-mn+1)+")+"+mn+");");
    } else {
      L.push(ind+"var "+vn+" = +(Math.random()*("+mx+"-"+mn+")+"+mn+").toFixed(4);");
    }
    if(r.storeVar) L.push(ind+"this."+vn+" = "+vn+";");

    // Rules
    r.rules.forEach(function(rule){
      if(rule.action==="none") return;
      var hasIf=(String(rule.val||"").trim()!=="");
      var ci=ind+(hasIf?"  ":"");

      // Target: self = "this", child = "this.ChildName" (direct property access in Wick)
      var tgt=rule.clipName||"self";
      var tRef=(tgt==="self"||tgt==="")?  "this" : "this."+tgt;

      if(hasIf) L.push(ind+"if("+vn+" "+rule.op+" "+String(rule.val).trim()+"){");
      var ii=ci;
      var fn=rule.action, fv=String(rule.frame||"1").trim();
      if(fn==="gotoAndStop"||fn==="gotoAndPlay"){
        var fvStr=isNaN(Number(fv))?"\""+fv+"\"":fv;
        L.push(ii+"try{ "+tRef+"."+fn+"("+fvStr+"); }catch(e){}");
      } else if(fn==="stop"||fn==="play"||fn==="gotoNextFrame"||fn==="gotoPrevFrame"){
        L.push(ii+"try{ "+tRef+"."+fn+"(); }catch(e){}");
      } else if(fn==="setOpacity"){
        L.push(ii+"try{ "+tRef+".opacity="+vn+"; }catch(e){}");
      } else if(fn==="setX"){
        L.push(ii+"try{ "+tRef+".x="+vn+"; }catch(e){}");
      } else if(fn==="setY"){
        L.push(ii+"try{ "+tRef+".y="+vn+"; }catch(e){}");
      }
      if(hasIf) L.push(ind+"}");
    });

    if(r.onlyOnce){ L.push("  this._rando_"+vn+"_done=true;"); L.push("}"); }
  });
  L.push("");
  L.push("// ── Rando END ────────────────────────────────────────────");
  return L.join("\n");
}

// ── Create / Inject ───────────────────────────────────────────────────────────
document.getElementById("btnCreate").addEventListener("click", function(){
  if(!currentInfo||!currentInfo.uuid){ setStatus("No clip selected. Use Refresh.", "warn"); return; }
  if(!rngs.length){ setStatus("No RNGs added.", "warn"); return; }
  var bridge=getBridge();
  if(!bridge){ setStatus("Bridge not found.", "bad"); return; }

  var code=buildScript(rngs);
  var existing="";
  // Read by UUID — no selection needed
  try{ if(typeof bridge.getUpdateScriptByInfo==="function") existing=bridge.getUpdateScriptByInfo(currentInfo)||""; }catch(e){}
  if(!existing){ try{ if(typeof bridge.getUpdateScriptOnSelection==="function") existing=bridge.getUpdateScriptOnSelection()||""; }catch(e){} }
  existing=existing.replace(/\/\/ ── Rando BEGIN[\s\S]*?\/\/ ── Rando END\n?/g,"").trim();
  var combined=(existing?existing+"\n\n":"")+code;

  try{
    // Write by UUID — no selection needed
    var res=null;
    if(typeof bridge.setUpdateScriptByInfo==="function") res=bridge.setUpdateScriptByInfo(currentInfo, combined);
    if(!res||res.ok===false){
      // Fallback: select then write
      res=bridge.setUpdateScriptOnSelection(combined);
    }
    if(res&&res.ok===false) throw new Error(res.reason||"script write failed");
    setStatus("✓ Injected "+rngs.length+" RNG(s) into: "+(currentInfo.identifier||currentInfo.name||currentInfo.uuid), "good");
  }catch(e){ setStatus("Failed: "+e.message, "bad"); }
});
