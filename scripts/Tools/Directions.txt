<script>
(function() {
    "use strict";
    function $(id) { return document.getElementById(id); }

    var Bridge = null;
    try { Bridge = window.LukeToolsBridge || parent.LukeToolsBridge; } catch(e) {}

    function linkRangeNumber(rangeId, numId) {
        var r = $(rangeId), n = $(numId);
        if (!r || !n) return;
        function sync(src, dst) { dst.value = src.value; }
        r.addEventListener("input", function() { sync(r, n); });
        r.addEventListener("change", function() { sync(r, n); });
        n.addEventListener("input", function() { sync(n, r); });
        n.addEventListener("change", function() { sync(n, r); });
    }
    linkRangeNumber("speed", "speedVal");

    function showStatus(msg, kind) {
        var el = $("status");
        el.textContent = msg;
        el.className = "status " + (kind || "");
    }

    function getSelection() {
        if (!Bridge) return null;
        try { return Bridge.getSelectionInfo(); } catch(e) { return null; }
    }

    function refreshSelection() {
        var info = getSelection();
        $("selLabel").textContent = (info && info.ok) ? (info.name || info.identifier || "Selected") : "No Clip";
        
        if (info && info.ok && Bridge) {
            try {
                var script = Bridge.getUpdateScriptOnSelection() || "";
                var match = script.match(/var\s+LT_CFG\s*=\s*(\{[^}]+\})/);
                if (match) {
                    var cfg = JSON.parse(match[1]);
                    if (cfg.type === "LT_8DIR") {
                        if (cfg.speed !== undefined) { $("speed").value = cfg.speed; $("speedVal").value = cfg.speed; }
                        if (cfg.useWASD !== undefined) $("useWASD").checked = cfg.useWASD;
                        if (cfg.useArrows !== undefined) $("useArrows").checked = cfg.useArrows;
                        if (cfg.diagonal !== undefined) $("diagonal").checked = cfg.diagonal;
                        if (cfg.normalize !== undefined) $("normalize").checked = cfg.normalize;
                        if (cfg.solidCollision !== undefined) $("solidCollision").checked = cfg.solidCollision;
                        showStatus("Found 1 block, 6 variables", "success");
                        return info;
                    }
                }
            } catch(e) {}
        }
        return info;
    }

    function generateScript() {
        var cfg = {
            type: "LT_8DIR",
            speed: parseFloat($("speedVal").value) || 5,
            useWASD: $("useWASD").checked,
            useArrows: $("useArrows").checked,
            diagonal: $("diagonal").checked,
            normalize: $("normalize").checked,
            solidCollision: $("solidCollision").checked
        };

        var lines = [];
        lines.push('// LukeTools Type: 8 Direction');
        lines.push('var LT_CFG = ' + JSON.stringify(cfg) + ';');
        lines.push('var p = this.project || null;');
        lines.push('');
        
        // AABB helper functions
        lines.push('// AABB collision helpers');
        lines.push('function __8d_num(v, d) { v = Number(v); return isFinite(v) ? v : d; }');
        lines.push('function __8d_bounds(o) {');
        lines.push('  try {');
        lines.push('    if (o && typeof o.getBounds === "function") {');
        lines.push('      var b = o.getBounds();');
        lines.push('      if (b) {');
        lines.push('        if (b.left !== undefined) return {l:__8d_num(b.left,0), t:__8d_num(b.top,0), r:__8d_num(b.right,0), b:__8d_num(b.bottom,0)};');
        lines.push('        if (b.x !== undefined && b.width !== undefined) return {l:__8d_num(b.x,0), t:__8d_num(b.y,0), r:__8d_num(b.x,0)+__8d_num(b.width,0), b:__8d_num(b.y,0)+__8d_num(b.height,0)};');
        lines.push('      }');
        lines.push('    }');
        lines.push('  } catch(e) {}');
        lines.push('  var w = __8d_num(o && o.width, 50);');
        lines.push('  var h = __8d_num(o && o.height, 50);');
        lines.push('  var x = __8d_num(o && o.x, 0);');
        lines.push('  var y = __8d_num(o && o.y, 0);');
        lines.push('  return {l:x-w*0.5, t:y-h*0.5, r:x+w*0.5, b:y+h*0.5};');
        lines.push('}');
        lines.push('function __8d_inter(a, b) { return !(a.r <= b.l || a.l >= b.r || a.b <= b.t || a.t >= b.b); }');
        lines.push('');
        
        // Resolve X
        lines.push('function __8d_resolveX(self, dx) {');
        lines.push('  if (!dx) return;');
        lines.push('  if (!p || !p._LT_SOLIDS) { self.x += dx; return; }');
        lines.push('  self.x += dx;');
        lines.push('  var a = __8d_bounds(self);');
        lines.push('  var hw = (a.r - a.l) * 0.5;');
        lines.push('  for (var k in p._LT_SOLIDS) {');
        lines.push('    var s = p._LT_SOLIDS[k];');
        lines.push('    if (!s || s === self) continue;');
        lines.push('    var sb = __8d_bounds(s);');
        lines.push('    if (__8d_inter(a, sb)) {');
        lines.push('      if (dx > 0) self.x = sb.l - hw - 0.01;');
        lines.push('      else self.x = sb.r + hw + 0.01;');
        lines.push('      a = __8d_bounds(self);');
        lines.push('    }');
        lines.push('  }');
        lines.push('}');
        lines.push('');
        
        // Resolve Y
        lines.push('function __8d_resolveY(self, dy) {');
        lines.push('  if (!dy) return;');
        lines.push('  if (!p || !p._LT_SOLIDS) { self.y += dy; return; }');
        lines.push('  self.y += dy;');
        lines.push('  var a = __8d_bounds(self);');
        lines.push('  var hh = (a.b - a.t) * 0.5;');
        lines.push('  for (var k in p._LT_SOLIDS) {');
        lines.push('    var s = p._LT_SOLIDS[k];');
        lines.push('    if (!s || s === self) continue;');
        lines.push('    var sb = __8d_bounds(s);');
        lines.push('    if (__8d_inter(a, sb)) {');
        lines.push('      if (dy > 0) self.y = sb.t - hh - 0.01;');
        lines.push('      else self.y = sb.b + hh + 0.01;');
        lines.push('      a = __8d_bounds(self);');
        lines.push('    }');
        lines.push('  }');
        lines.push('}');
        lines.push('');
        
        // Input
        lines.push('// Input');
        lines.push('var _dx = 0, _dy = 0;');
        lines.push('if (LT_CFG.useArrows) {');
        lines.push('  if (isKeyDown("left")) _dx = -1;');
        lines.push('  else if (isKeyDown("right")) _dx = 1;');
        lines.push('  if (isKeyDown("up")) _dy = -1;');
        lines.push('  else if (isKeyDown("down")) _dy = 1;');
        lines.push('}');
        lines.push('if (LT_CFG.useWASD) {');
        lines.push('  if (isKeyDown("a")) _dx = -1;');
        lines.push('  else if (isKeyDown("d")) _dx = 1;');
        lines.push('  if (isKeyDown("w")) _dy = -1;');
        lines.push('  else if (isKeyDown("s")) _dy = 1;');
        lines.push('}');
        lines.push('');
        
        // Diagonal handling
        lines.push('// Diagonal');
        lines.push('if (!LT_CFG.diagonal && _dx !== 0 && _dy !== 0) _dy = 0;');
        lines.push('if (LT_CFG.normalize && _dx !== 0 && _dy !== 0) {');
        lines.push('  var _len = Math.sqrt(_dx*_dx + _dy*_dy);');
        lines.push('  _dx /= _len; _dy /= _len;');
        lines.push('}');
        lines.push('');
        
        // Apply movement
        lines.push('// Movement');
        lines.push('var _mx = _dx * LT_CFG.speed;');
        lines.push('var _my = _dy * LT_CFG.speed;');
        lines.push('');
        lines.push('if (LT_CFG.solidCollision) {');
        lines.push('  __8d_resolveX(this, _mx);');
        lines.push('  __8d_resolveY(this, _my);');
        lines.push('} else {');
        lines.push('  this.x += _mx;');
        lines.push('  this.y += _my;');
        lines.push('}');
        lines.push('// End');

        return lines.join('\n');
    }

    function applyScript() {
        if (!Bridge) { showStatus("Bridge not available", "error"); return; }

        var info = getSelection();
        if (!info || !info.ok) { showStatus("No clip selected", "error"); return; }

        var existing = "";
        try { existing = Bridge.getUpdateScriptOnSelection() || ""; } catch(e) {}

        var newScript = generateScript();

        // Remove old 8DIR code if present
        if (existing.indexOf('// LukeTools Type: 8 Direction') !== -1) {
            var start = existing.indexOf('// LukeTools Type: 8 Direction');
            var end = existing.indexOf('// End', start);
            if (end !== -1) {
                end = existing.indexOf('\n', end);
                if (end === -1) end = existing.length;
                else end += 1;
                existing = existing.substring(0, start) + existing.substring(end);
                existing = existing.trim();
            }
        }
        
        var finalScript = existing ? (existing + '\n\n' + newScript) : newScript;
        var result = Bridge.setUpdateScriptOnSelection(finalScript);
        
        if (result && result.ok) {
            showStatus("Applied to " + (info.name || info.identifier), "success");
        } else {
            showStatus("Error: " + (result ? result.reason : "Failed"), "error");
        }
    }

    $("btnRefresh").addEventListener("click", refreshSelection);
    $("btnApply").addEventListener("click", applyScript);

    document.addEventListener("click", function() { setTimeout(refreshSelection, 50); });

    refreshSelection();
})();
</script>
