<script>
(function() {
    var Bridge = null;
    try { Bridge = parent.LukeToolsBridge || window.LukeToolsBridge; } catch(e) {}

    var applyBtn = document.getElementById('applyBtn');
    var statusEl = document.getElementById('status');

    function showStatus(msg, isError) {
        statusEl.textContent = msg;
        statusEl.className = 'status ' + (isError ? 'error' : 'success');
    }

    function generateCode() {
        var speed = parseFloat(document.getElementById('speed').value) || 5;
        var useWASD = document.getElementById('useWASD').checked;
        var useArrows = document.getElementById('useArrows').checked;
        var diagonal = document.getElementById('diagonal').checked;
        var normalize = document.getElementById('normalize').checked;

        var code = [];
        code.push('// 8 Direction Movement - LukeTools');
        code.push('var _8DIR_SPEED = ' + speed + ';');
        code.push('var _8DIR_dx = 0;');
        code.push('var _8DIR_dy = 0;');

        var leftKeys = [], rightKeys = [], upKeys = [], downKeys = [];

        if (useArrows) {
            leftKeys.push('isKeyDown("left")');
            rightKeys.push('isKeyDown("right")');
            upKeys.push('isKeyDown("up")');
            downKeys.push('isKeyDown("down")');
        }
        if (useWASD) {
            leftKeys.push('isKeyDown("a")');
            rightKeys.push('isKeyDown("d")');
            upKeys.push('isKeyDown("w")');
            downKeys.push('isKeyDown("s")');
        }
        if (leftKeys.length === 0) {
            leftKeys.push('false'); rightKeys.push('false'); upKeys.push('false'); downKeys.push('false');
        }

        code.push('if (' + leftKeys.join(' || ') + ') { _8DIR_dx = -1; }');
        code.push('else if (' + rightKeys.join(' || ') + ') { _8DIR_dx = 1; }');
        code.push('else { _8DIR_dx = 0; }');
        code.push('if (' + upKeys.join(' || ') + ') { _8DIR_dy = -1; }');
        code.push('else if (' + downKeys.join(' || ') + ') { _8DIR_dy = 1; }');
        code.push('else { _8DIR_dy = 0; }');

        if (!diagonal) {
            code.push('if (_8DIR_dx !== 0 && _8DIR_dy !== 0) { _8DIR_dy = 0; }');
        }
        if (normalize && diagonal) {
            code.push('if (_8DIR_dx !== 0 && _8DIR_dy !== 0) {');
            code.push('  var _8DIR_len = Math.sqrt(_8DIR_dx*_8DIR_dx + _8DIR_dy*_8DIR_dy);');
            code.push('  _8DIR_dx /= _8DIR_len; _8DIR_dy /= _8DIR_len;');
            code.push('}');
        }

        code.push('this.x += _8DIR_dx * _8DIR_SPEED;');
        code.push('this.y += _8DIR_dy * _8DIR_SPEED;');

        return code.join('\n');
    }

    applyBtn.addEventListener('click', function() {
        if (!Bridge) { showStatus('Error: Bridge not available', true); return; }

        var info = Bridge.getSelectionInfo();
        if (!info || !info.ok) { showStatus('Error: No clip selected', true); return; }

        var existing = '';
        try { existing = Bridge.getUpdateScriptOnSelection() || ''; } catch(e) {}

        var newCode = generateCode();

        if (existing.indexOf('// 8 Direction Movement') !== -1) {
            var start = existing.indexOf('// 8 Direction Movement');
            var end = existing.indexOf('this.y += _8DIR_dy * _8DIR_SPEED;');
            if (start !== -1 && end !== -1) {
                end = existing.indexOf('\n', end);
                if (end === -1) end = existing.length;
                existing = existing.substring(0, start) + existing.substring(end);
                existing = existing.trim();
            }
        }

        var finalScript = existing ? (existing + '\n\n' + newCode) : newCode;
        var result = Bridge.setUpdateScriptOnSelection(finalScript);

        if (result && result.ok) {
            showStatus('Applied to "' + (info.name || info.identifier || 'clip') + '"', false);
        } else {
            showStatus('Error: ' + (result ? result.reason : 'Failed'), true);
        }
    });
})();
</script>
