function getRigScriptBlock(){
  if(!Bridge) return "";
  try{ return Bridge.getUpdateScriptOnSelection() || ""; }catch(e){ return ""; }
}
function loadRigStateFromScript(){
  var raw = getRigScriptBlock();
  state.poses = {};
  if(!raw) return false;
  try{
    var cfgMatch = raw.match(/var\s+LT_RIG_CFG\s*=\s*(\{[\s\S]*?\});/);
    if(cfgMatch){
      var cfg = JSON.parse(cfgMatch[1]);
      if(cfg.toolFrameCount !== undefined) $("frameCount").value = String(parseInt(cfg.toolFrameCount, 10) || 60);
      if(cfg.projectStartFrame !== undefined) $("startFrame").value = String(parseInt(cfg.projectStartFrame, 10) || 1);
      if(cfg.timelineSource !== undefined) $("timelineSource").value = String(cfg.timelineSource || "project");
    }
  }catch(e0){}
  try{
    var poseMatch = raw.match(/var\s+LT_RIG_POSES\s*=\s*(\{[\s\S]*?\});/);
    if(poseMatch) state.poses = JSON.parse(poseMatch[1]) || {};
  }catch(e1){ state.poses = {}; }
  return true;
}

}
function loadRotoState(){ // FIX
  try{
    var key = "LT_RotoMedia_" + (($("selLabel").textContent || "None").trim() || "None");
    var raw = localStorage.getItem(key);
    if(!raw) return;
    var data = JSON.parse(raw);
    if(data && typeof data === "object"){
      state.rotoMedia = Object.assign(state.rotoMedia, data);
      state.rotoMedia.objectUrl = ""; // FIX
    }
  }catch(e){}
}

function renderRotoMedia(){ // FIX
  clearMediaLayer();
  try{ while(rotoHtmlStage.firstChild) rotoHtmlStage.removeChild(rotoHtmlStage.firstChild); }catch(e0){}
  try{ while(rotoControlOverlay.firstChild) rotoControlOverlay.removeChild(rotoControlOverlay.firstChild); }catch(e0b){}
  state.rotoVideoEl = null;
  state.rotoVideoReady = false;

  var m = state.rotoMedia;
  if(!m || !m.visible || !m.src) return;

  // Media wrap - goes behind bones (in rotoHtmlStage, z-index 1)
  var wrap = document.createElement("div");
  wrap.className = "rotoHtmlWrap";
  wrap.style.left = m.x + "px";
  wrap.style.top = m.y + "px";
  wrap.style.width = m.w + "px";
  wrap.style.height = m.h + "px";
  wrap.style.transform = "rotate(" + m.rot + "deg)";

  var media;
  if(m.type && m.type.indexOf("video/") === 0){
    media = document.createElement("video");
    media.src = m.src;
    media.preload = "auto";
    media.muted = true;
    media.autoplay = false;
    media.loop = false;
    media.playsInline = true;
    media.className = "rotoHtmlMedia";
    state.rotoVideoEl = media;
    // Set videoReady and duration once loaded, then PAUSE and seek to current frame
    media.addEventListener("loadedmetadata", function(){
      state.rotoVideoReady = true;
      state.rotoVideoDuration = Number(media.duration) || 0;
      media.pause();
      seekRotoToFrame(state.currentFrame);
    });
    media.addEventListener("canplay", function(){ media.pause(); });
  }else{
    media = document.createElement("img");
    media.src = m.src;
    media.className = "rotoHtmlMedia";
  }
  wrap.appendChild(media);
  rotoHtmlStage.appendChild(wrap);

  // Controls wrap - goes above bones (in rotoControlOverlay, z-index 3)
  if(!m.locked){
    var ctrlWrap = document.createElement("div");
    ctrlWrap.className = "rotoHtmlWrap";
    ctrlWrap.style.left = m.x + "px";
    ctrlWrap.style.top = m.y + "px";
    ctrlWrap.style.width = m.w + "px";
    ctrlWrap.style.height = m.h + "px";
    ctrlWrap.style.transform = "rotate(" + m.rot + "deg)";
    ctrlWrap.style.pointerEvents = "auto";

    scale.setAttribute("data-roto", "scale");
    ctrlWrap.appendChild(scale);

    var rotate = document.createElement("div");
    rotate.className = "rotoHtmlRotate";
    rotate.setAttribute("data-roto", "rotate");
    ctrlWrap.appendChild(rotate);

    rotoControlOverlay.appendChild(ctrlWrap);
  }
}

function seekRotoToFrame(frame){ // FIX
  if(!state.rotoVideoEl || !state.rotoVideoReady) return;
  try{
    var fps = parseInt($("rotoFps").value, 10) || 12;
    var t = (frame - 1) / fps;
    if(t < 0) t = 0;
    if(state.rotoVideoDuration && t > state.rotoVideoDuration){
      t = state.rotoVideoDuration;
    }
    state.rotoVideoEl.currentTime = t;
  }catch(e){}
}
function revokeRotoObjectUrl(){ // FIX
  try{
    if(state.rotoMedia && state.rotoMedia.objectUrl){
      URL.revokeObjectURL(state.rotoMedia.objectUrl);
      state.rotoMedia.objectUrl = "";
    }
  }catch(e){}
}
function loadRotoFile(file){ // FIX
  if(!file){
    setStatus("No media file selected.", "warn");
    return;
  }
  try{
    revokeRotoObjectUrl();
    var url = URL.createObjectURL(file);
    state.rotoMedia.src = url;
    state.rotoMedia.objectUrl = url;
    state.rotoMedia.type = file.type || "";
    state.rotoMedia.visible = true;
    state.rotoMedia.locked = false;
    updateLockButton();
    updateMediaFileName();
    saveRotoState();
    renderRig(state.rig);
    setStatus("Rotoscope media imported and displayed behind the bones.", "good");
  }catch(e){
    setStatus("Could not load media file: " + e.message, "bad");
  }
}

function guessProjectFrameCount(){
  var project = getProject();
  if(!project) return null;
  var candidates = [];
  try{ if(project.frameCount !== undefined) candidates.push(project.frameCount); }catch(e0){}
  try{ if(project.length !== undefined) candidates.push(project.length); }catch(e1){}
  try{ if(project.duration !== undefined) candidates.push(project.duration); }catch(e2){}
  try{ if(project.activeTimeline && project.activeTimeline.length !== undefined) candidates.push(project.activeTimeline.length); }catch(e3){}
  try{ if(project.timeline && project.timeline.length !== undefined) candidates.push(project.timeline.length); }catch(e4){}
  try{ if(project.activeTimeline && project.activeTimeline.frameCount !== undefined) candidates.push(project.activeTimeline.frameCount); }catch(e5){}
  try{ if(project.timeline && project.timeline.frameCount !== undefined) candidates.push(project.timeline.frameCount); }catch(e6){}
  for(var i=0;i<candidates.length;i++){ var n = Number(candidates[i]); if(isFinite(n) && n > 0) return Math.floor(n); }
  return null;
}
function LT_guessTimelineFrameCountFromObject(obj){
  if(!obj) return null;
  var tl = null;
  try{ tl = obj.timeline || obj._timeline || (obj.clipAsset ? obj.clipAsset.timeline : null); }catch(e0){ tl = null; }
  if(!tl) return null;
  var candidates = [];
  try{ if(tl.length !== undefined) candidates.push(tl.length); }catch(e1){}
  try{ if(tl.frameCount !== undefined) candidates.push(tl.frameCount); }catch(e2){}
  try{ if(tl.playLength !== undefined) candidates.push(tl.playLength); }catch(e3){}
  try{
    var layers = tl.layers || tl._layers;
    if(layers && layers.length){
      var maxEnd = 0;
      for(var i=0;i<layers.length;i++){
        var layer = layers[i];
        if(!layer || !layer.frames) continue;
        for(var j=0;j<layer.frames.length;j++){
          var fr = layer.frames[j];
          if(!fr) continue;
          var start = Number(fr.start) || 1;
          var end = Number(fr.end) || start;
          if(end < start) end = start;
          if(end > maxEnd) maxEnd = end;
        }
      }
      if(maxEnd > 0) candidates.push(maxEnd);
    }
  }catch(e4){}
  for(var k=0;k<candidates.length;k++){ var n = Number(candidates[k]); if(isFinite(n) && n > 0) return Math.floor(n); }
  return null;
}
function getParentObject(obj){ try{ if(obj && obj.parentClip) return obj.parentClip; }catch(e0){} try{ if(obj && obj.parent) return obj.parent; }catch(e1){} return null; }
function getTimelineSourceObject(){
  var mode = $("timelineSource").value || "project";
  if(mode === "project") return null;
  var selected = getSelectedObject();
  if(mode === "selected") return selected;
  if(mode === "parent") return getParentObject(selected);
  return null;
}
function guessFrameCountFromTimelineSource(){
  var mode = $("timelineSource").value || "project";
  if(mode === "project") return guessProjectFrameCount();
  return LT_guessTimelineFrameCountFromObject(getTimelineSourceObject());
}

function getFrameChildClips(frame){
  if(!frame) return [];
  try{ if(frame.clips && frame.clips.length) return frame.clips.slice(); }catch(e){}
  try{
    if(frame.children && frame.children.length){
      var out = [];
      for(var i=0;i<frame.children.length;i++){
        var c = frame.children[i];

function getPathBoundsFromClipContent(obj){
  var tl = getTimeline(obj), layers = getLayers(tl);
  var minX=Infinity,minY=Infinity,maxX=-Infinity,maxY=-Infinity;
  if(!layers || !layers.length) return null;
  for(var i=0;i<layers.length;i++){
    var fr = getActiveFrame(layers[i],1);
    if(!fr) continue;
    var arr = null;
    try{ arr = fr.paths || []; }catch(e){}
    if(!arr) continue;
    for(var p=0;p<arr.length;p++){
      try{
        var json = arr[p].json || null;
        if(json && json[1] && json[1].segments){
          var segs = json[1].segments;
          for(var s=0;s<segs.length;s++){
            var pt = segs[s][0]; if(!pt) continue;
            var x = Number(pt[0])||0, y = Number(pt[1])||0;
            minX = Math.min(minX,x); minY = Math.min(minY,y); maxX = Math.max(maxX,x); maxY = Math.max(maxY,y);
          }
        }
      }catch(e4){}
    }
  }
  if(!isFinite(minX)) return null;
  return {x:minX,y:minY,w:maxX-minX,h:maxY-minY,cx:(minX+maxX)/2,cy:(minY+maxY)/2};
}
function makeNode(obj,parentNode){
  var t = null;
  try{ t = obj.transformation || obj._transformation || null; }catch(e){ t = null; }
  var node = {
    uid:"n"+(state.uidCounter++),
    obj:obj,
    info:{
      identifier:(function(){ try{ return obj._identifier || obj.identifier || obj.name || ""; }catch(e){ return ""; } })(),
      uuid:(function(){ try{ return obj.uuid || obj.UUID || obj.id || ""; }catch(e){ return ""; } })()
    },
    name:(function(){ try{ return obj._identifier || obj.identifier || obj.name || "Clip"; }catch(e){ return "Clip"; } })(),
    localRot:t ? (Number(t.rotation)||0) : (Number(obj.rotation) || 0),
    localX:t ? (Number(t.x)||0) : (Number(obj.x)||0),
    localY:t ? (Number(t.y)||0) : (Number(obj.y)||0),
    restVecX:0,
    restVecY:0,
    length:80,
    parent:parentNode||null,
    children:[],
    worldX:0, worldY:0, worldRot:0
  };
  state.nodeMap[node.uid] = node;
  return node;
}
function getDirectChildren(obj){
  var tl = getTimeline(obj), layers = getLayers(tl), out = [];
  if(!layers || !layers.length) return out;
  for(var i=0;i<layers.length;i++){
    var pos = 1;
    try{ pos = tl.playheadPosition || 1; }catch(e){}
    var fr = getActiveFrame(layers[i], pos);
    if(!fr) continue;
    var clips = getFrameChildClips(fr);
    for(var c=0;c<clips.length;c++) out.push(clips[c]);
  }
  return out;
}
function pickBestChild(currentObj){
  var kids = getDirectChildren(currentObj);
  if(!kids.length) return null;
  var best = null, bestD = Infinity;
  for(var i=0;i<kids.length;i++){
    var k = kids[i], x = 0, y = 0;
    try{ var t = k.transformation || k._transformation || null; x = t ? (Number(t.x)||0) : (Number(k.x)||0); y = t ? (Number(t.y)||0) : (Number(k.y)||0); }catch(e){}
    var d = Math.sqrt(x*x + y*y);
    if(d > 0.01 && d < bestD){ bestD = d; best = k; }
  }
  return best || kids[0];
}
function buildChainRig(rootObj){
  state.uidCounter = 1;
  state.nodeMap = {};
  var root = makeNode(rootObj,null);
  var currentNode = root, currentObj = rootObj, guard = 0;
  while(guard < 30){
    guard += 1;
    var childObj = pickBestChild(currentObj);
    if(!childObj) break;
    var childNode = makeNode(childObj,currentNode);
    currentNode.children = [childNode];
    currentNode = childNode;
    currentObj = childObj;
  }
  function assignRestVectors(node){
    if(node.children.length){
      var child = node.children[0];
      node.restVecX = Number(child.localX)||0;
      node.restVecY = Number(child.localY)||0;
      node.length = Math.max(12, Math.sqrt(node.restVecX*node.restVecX + node.restVecY*node.restVecY));
    }else{
      var b = getPathBoundsFromClipContent(node.obj);
      var len = b ? Math.max(18, b.h || b.w || 40) : 60;
      if(b && b.h >= b.w){ node.restVecX = 0; node.restVecY = len; }
      else { node.restVecX = len; node.restVecY = 0; }
      node.length = len;
    }
    for(var i=0;i<node.children.length;i++) assignRestVectors(node.children[i]);
  }
  assignRestVectors(root);
  return root;
}
function updateWorld(node){
  if(!node.parent){
    node.worldX = 0;
    node.worldY = 0;
    node.worldRot = node.localRot || 0;
  }else{
    var parentEnd = getBoneEnd(node.parent);
    node.worldX = parentEnd.x;
    node.worldY = parentEnd.y;
    node.worldRot = node.parent.worldRot + node.localRot;
  }
  for(var i=0;i<node.children.length;i++) updateWorld(node.children[i]);
}
function flatten(node, arr){ arr.push(node); for(var i=0;i<node.children.length;i++) flatten(node.children[i], arr); return arr; }
function getBoneEnd(node){
  var r = rad(node.worldRot), vx = node.restVecX, vy = node.restVecY;
  return { x: node.worldX + (vx * Math.cos(r) - vy * Math.sin(r)), y: node.worldY + (vx * Math.sin(r) + vy * Math.cos(r)) };
}

  return el;
}
function makeClipSnapshotData(node){ // FIX
  var obj = node.obj;
  var b = getPathBoundsFromClipContent(obj);
  if(!b) return null;

  // Try exact style exports first
  try{
    if(obj && typeof obj.exportSVG === "function"){
      var svgText = String(obj.exportSVG() || "");
      if(svgText) return {kind:"svg", data:svgText, bounds:b};
    }
  }catch(e0){}
  try{
    if(obj && typeof obj.toSVG === "function"){
      var svgText2 = String(obj.toSVG() || "");
      if(svgText2) return {kind:"svg", data:svgText2, bounds:b};
    }
  }catch(e1){}

  // Best effort bitmap snapshot from Wick style path data
  try{
    var tl = getTimeline(obj), layers = getLayers(tl);
    if(!layers || !layers.length) return null;
    var canvas = document.createElement("canvas");
    canvas.width = Math.max(8, Math.ceil(b.w + 8));
    canvas.height = Math.max(8, Math.ceil(b.h + 8));
    var ctx = canvas.getContext("2d");
    if(!ctx) return null;
    ctx.translate(-b.x + 4, -b.y + 4);

    for(var i=0;i<layers.length;i++){
      var fr = getActiveFrame(layers[i],1);
      if(!fr) continue;
      var arr = null;
      try{ arr = fr.paths || []; }catch(e2){}
      if(!arr) continue;

      for(var p=0;p<arr.length;p++){
        try{
          var pathObj = arr[p];
          var json = pathObj.json || null;
          if(!json || !json[1] || !json[1].segments) continue;
          var segs = json[1].segments;
          if(!segs.length) continue;

          ctx.beginPath();
          for(var s=0;s<segs.length;s++){
            var seg = segs[s];
            var pt = seg[0];
            var hIn = seg[1];
            var hOut = seg[2];
            if(!pt) continue;
            var x = Number(pt[0])||0, y = Number(pt[1])||0;

            if(s === 0){
              ctx.moveTo(x, y);
            }else{
              var prev = segs[s-1];
              var prevPt = prev[0];
              var prevOut = prev[2];
              var c1x = prevOut ? ((Number(prevPt[0])||0) + (Number(prevOut[0])||0)) : (Number(prevPt[0])||0);
              var c1y = prevOut ? ((Number(prevPt[1])||0) + (Number(prevOut[1])||0)) : (Number(prevPt[1])||0);
              var c2x = hIn ? (x + (Number(hIn[0])||0)) : x;
              var c2y = hIn ? (y + (Number(hIn[1])||0)) : y;
              if(prevOut || hIn){
                ctx.bezierCurveTo(c1x, c1y, c2x, c2y, x, y);
              }else{
                ctx.lineTo(x, y);
              }
            }
          }

          var closed = false;
          try{ closed = !!json[1].closed; }catch(e3){}
          if(closed && segs.length > 1){
            var first = segs[0];
            var last = segs[segs.length-1];
            var firstPt = first[0], firstIn = first[1];
            var lastPt = last[0], lastOut = last[2];
            var lx = Number(lastPt[0])||0, ly = Number(lastPt[1])||0;
            var fx = Number(firstPt[0])||0, fy = Number(firstPt[1])||0;
            var c1x2 = lastOut ? (lx + (Number(lastOut[0])||0)) : lx;
            var c1y2 = lastOut ? (ly + (Number(lastOut[1])||0)) : ly;
            var c2x2 = firstIn ? (fx + (Number(firstIn[0])||0)) : fx;
            var c2y2 = firstIn ? (fy + (Number(firstIn[1])||0)) : fy;
            if(lastOut || firstIn) ctx.bezierCurveTo(c1x2, c1y2, c2x2, c2y2, fx, fy);
            else ctx.closePath();
          }

          var fill = null, stroke = null, strokeWidth = 1;
          try{ fill = pathObj.fillColor || (json[1].fillColor || null); }catch(e4){}
          try{ stroke = pathObj.strokeColor || (json[1].strokeColor || null); }catch(e5){}
          try{ strokeWidth = Number(pathObj.strokeWidth || json[1].strokeWidth || 1) || 1; }catch(e6){}

          function toCssColor(c){
            if(!c) return null;
            if(typeof c === "string") return c;
            if(Array.isArray(c) && c.length >= 3){
              var a = (c.length > 3) ? Number(c[3]) : 1;
              return "rgba(" + Math.round((Number(c[0])||0)*255) + "," + Math.round((Number(c[1])||0)*255) + "," + Math.round((Number(c[2])||0)*255) + "," + a + ")";
            }
            if(typeof c === "object" && c.red !== undefined){
              return "rgba(" + Math.round((Number(c.red)||0)*255) + "," + Math.round((Number(c.green)||0)*255) + "," + Math.round((Number(c.blue)||0)*255) + "," + (c.alpha !== undefined ? Number(c.alpha) : 1) + ")";
            }
            return null;
          }

          var fillCss = toCssColor(fill);
          var strokeCss = toCssColor(stroke);

          if(fillCss){
            ctx.fillStyle = fillCss;
            ctx.fill();
          }else{
            ctx.fillStyle = "rgba(120,170,255,0.12)";
            ctx.fill();
          }
          if(strokeCss || strokeWidth > 0){
            ctx.strokeStyle = strokeCss || "rgba(90,140,220,0.55)";
            ctx.lineWidth = strokeWidth;
            ctx.stroke();
          }
        }catch(e7){}
      }
    }

    return {kind:"png", data:canvas.toDataURL("image/png"), bounds:b};
  }catch(e8){}

  return null;
}
function drawClipProxy(node){
  var snap = makeClipSnapshotData(node); // FIX
  if(!snap){
    var b = getPathBoundsFromClipContent(node.obj);
    if(!b) return;
    var g0 = svgEl("g",{transform:"translate(" + node.worldX + "," + node.worldY + ") rotate(" + node.worldRot + ")"});
    var rect = svgEl("rect", {x:b.x, y:b.y, width:b.w, height:b.h, class:"clipProxy"});
    g0.appendChild(rect);
    artLayer.appendChild(g0);
    return;
  }

  var g = svgEl("g",{transform:"translate(" + node.worldX + "," + node.worldY + ") rotate(" + node.worldRot + ")"});

  if(snap.kind === "svg"){
    var enc = "data:image/svg+xml;charset=utf-8," + encodeURIComponent(snap.data);
    var img = svgEl("image", {
      x: snap.bounds.x,
      y: snap.bounds.y,
      width: snap.bounds.w,
      height: snap.bounds.h,
      href: enc,
      class: "clipSnapshot"
    });
    g.appendChild(img);
  }else{
    var png = svgEl("image", {
      x: snap.bounds.x - 4,
      y: snap.bounds.y - 4,
      width: snap.bounds.w + 8,
      height: snap.bounds.h + 8,
      href: snap.data,
      class: "clipSnapshot"
    });
    g.appendChild(png);
  }

  artLayer.appendChild(g);
}
function renderRig(root){
  clearMediaLayer();
  clearArt();
  clearRig();
  renderRotoMedia();
  if(!root) return;
  updateWorld(root);
  var nodes = flatten(root,[]);
  for(var i=0;i<nodes.length;i++){
    var node = nodes[i], end = getBoneEnd(node);
    drawClipProxy(node);

    rigLayer.appendChild(svgEl("line",{x1:node.worldX,y1:node.worldY,x2:end.x,y2:end.y,class:"boneHit","data-node-id":node.uid})); // FIX
    rigLayer.appendChild(svgEl("line",{x1:node.worldX,y1:node.worldY,x2:end.x,y2:end.y,class:"boneLine"}));

    rigLayer.appendChild(svgEl("circle",{cx:node.worldX,cy:node.worldY,r:12,class:"jointHit","data-node-id":node.uid}));
    rigLayer.appendChild(svgEl("circle",{cx:end.x,cy:end.y,r:18,class:"hitHandle","data-node-id":node.uid}));
    var cls = "tipHandle" + (node.parent ? "" : " rootHandle");
    if(state.selectedNodeId === node.uid) cls += " selected";
    rigLayer.appendChild(svgEl("circle",{cx:end.x,cy:end.y,r:9,class:cls,"data-node-id":node.uid}));
    rigLayer.appendChild(svgEl("circle",{cx:node.worldX,cy:node.worldY,r:7,class:"jointDot","data-node-id":node.uid})); // FIX
    var label = svgEl("text",{x:end.x+10,y:end.y-10,class:"nodeLabel" + (state.selectedNodeId === node.uid ? " selected" : ""),"data-node-id":node.uid});
    label.textContent = node.name;
    rigLayer.appendChild(label);
  }
}

}
function findHandleElement(nodeId){ return rigLayer.querySelector('[data-node-id="'+nodeId+'"]'); }
function beginDrag(node, handle, evt){
  evt.preventDefault(); evt.stopPropagation();
  state.drag = {nodeId:node.uid,node:node,handle:handle,before:{}};
  if(state.rig){
    var dn = flatten(state.rig, []);
    for(var di=0; di<dn.length; di++) state.drag.before[dn[di].name] = round2(dn[di].localRot);
  }
  try{ handle.classList.add("dragging"); }catch(e0){}
  window.addEventListener("pointermove", onHandleMove, true);
  window.addEventListener("pointerup", onHandleUp, true);
}
function onRigPointerDown(evt){
  var handle = evt.target.closest ? evt.target.closest("[data-node-id]") : null;
  if(!handle) return false;
  var node = state.nodeMap[handle.getAttribute("data-node-id")];
  if(!node) return false;

  state.selectedNodeId = node.uid;
  renderRig(state.rig);
  refreshTimelineUi();

  // FIX:
  // label only selects
  // blue bone hit area, pivot, and end handle all drag rotation from the bone pivot
  if(handle.classList && handle.classList.contains("nodeLabel")){
    return true;
  }

  beginDrag(node, handle, evt);
  return true;
}
function setNodeRotationFromPointer(node, pt){
  var ax = pt.x - node.worldX, ay = pt.y - node.worldY;
  var restAng = Math.atan2(node.restVecY, node.restVecX) * 180 / Math.PI;
  var desiredWorld = Math.atan2(ay, ax) * 180 / Math.PI;
  var parentWorld = node.parent ? node.parent.worldRot : 0;
  node.localRot = desiredWorld - parentWorld - restAng;
}
function onHandleMove(evt){
  if(!state.drag || !state.drag.node) return;
  var pt = getSvgPoint(evt);
  setNodeRotationFromPointer(state.drag.node, pt);
  renderRig(state.rig);
  var handle = findHandleElement(state.drag.nodeId);
  if(handle){ state.drag.handle = handle; try{ handle.classList.add("dragging"); }catch(e0){} }
}
function onHandleUp(){
  if(!state.drag || !state.drag.node) return;
  try{ if(state.drag.handle) state.drag.handle.classList.remove("dragging"); }catch(e0){}
  window.removeEventListener("pointermove", onHandleMove, true);
  window.removeEventListener("pointerup", onHandleUp, true);

  if(state.rig){
    var nodes = flatten(state.rig, []);
    if(!state.poses[state.currentFrame]) state.poses[state.currentFrame] = {};
    var changed = [];
    for(var i=0;i<nodes.length;i++){
      var nm = nodes[i].name;
      var before = state.drag.before && state.drag.before[nm] !== undefined ? Number(state.drag.before[nm]) : Number(nodes[i].localRot);
      var after = round2(nodes[i].localRot);
      if(Math.abs(after - before) > 0.01){
        state.poses[state.currentFrame][nm] = after;
        changed.push(nm);
      }
    }
    if(Object.keys(state.poses[state.currentFrame]).length === 0) delete state.poses[state.currentFrame];
    if(changed.length){
      refreshTimelineUi();
      setStatus("Auto keyed changed clips at frame " + state.currentFrame + ": " + changed.join(", "), "good");
    }
  }
  state.drag = null;
}
function fitView(){
  if(!state.rig) return;
  updateWorld(state.rig);
  var nodes = flatten(state.rig, []);
  var minX=Infinity,minY=Infinity,maxX=-Infinity,maxY=-Infinity;
  for(var i=0;i<nodes.length;i++){
    var n = nodes[i], e = getBoneEnd(n);
    minX = Math.min(minX,n.worldX,e.x); minY = Math.min(minY,n.worldY,e.y);
    maxX = Math.max(maxX,n.worldX,e.x); maxY = Math.max(maxY,n.worldY,e.y);
  }
  if(!isFinite(minX)) return;
  var pad = 260;
  state.viewBox = {x:minX-pad,y:minY-pad,w:Math.max(900,(maxX-minX)+pad*2),h:Math.max(700,(maxY-minY)+pad*2)};
  applyViewBox();
}
function resetView(){ state.viewBox = {x:-900,y:-680,w:1800,h:1360}; applyViewBox(); }
function zoomAt(scale,cx,cy){
  var vb = state.viewBox, nw = vb.w * scale, nh = vb.h * scale;
  state.viewBox = {x:cx-(cx-vb.x)*(nw/vb.w),y:cy-(cy-vb.y)*(nh/vb.h),w:nw,h:nh};
  applyViewBox();
}
function capturePose(){
  if(!state.rig){ setStatus("Build rig first.", "warn"); return; }
  if(!state.selectedNodeId){ setStatus("Select a rig item first.", "warn"); return; }
  var node = state.nodeMap[state.selectedNodeId];
  if(!node){ setStatus("Selected rig item not found.", "warn"); return; }
  if(!state.poses[state.currentFrame]) state.poses[state.currentFrame] = {};
  state.poses[state.currentFrame][node.name] = round2(node.localRot);
  refreshTimelineUi();
  setStatus("Key set for " + node.name + " at tool frame " + state.currentFrame + ".", "good");
}
function deletePose(){

function getSortedPoseFrames(){ return Object.keys(state.poses).map(function(v){ return parseInt(v,10); }).filter(function(v){ return isFinite(v); }).sort(function(a,b){ return a-b; }); }
function nearestKeyFrames(frame){
  var ks = getSortedPoseFrames();
  if(!ks.length) return { prev:null, next:null };
  var prev = null, next = null;
  for(var i=0;i<ks.length;i++){
    var f = ks[i];
    if(f <= frame) prev = f;
    if(f >= frame){ next = f; break; }
  }
  if(prev === null) prev = ks[0];
  if(next === null) next = ks[ks.length-1];
  return { prev: prev, next: next };
}
function applyPoseForFrame(frame){
  if(!state.rig) return;
  var pair = nearestKeyFrames(frame);
  var nodes = flatten(state.rig, []);
  if(pair.prev === null || pair.next === null){ renderRig(state.rig); return; }
  var A = state.poses[pair.prev], B = state.poses[pair.next];
  if(!A && !B){ renderRig(state.rig); return; }
  if(A && !B) B = A;
  if(B && !A) A = B;
  if(pair.prev === pair.next){
    for(var i=0;i<nodes.length;i++) if(A[nodes[i].name] !== undefined) nodes[i].localRot = Number(A[nodes[i].name]) || 0;
    renderRig(state.rig); return;
  }
  var t = clamp((frame - pair.prev) / (pair.next - pair.prev), 0, 1);
  t = easeInOutTween(t);
  for(var k=0;k<nodes.length;k++){
    var nm = nodes[k].name;
    var a = (A[nm] !== undefined) ? Number(A[nm]) : nodes[k].localRot;
    var b = (B[nm] !== undefined) ? Number(B[nm]) : a;
    nodes[k].localRot = lerpAngleDeg(a, b, t);
  }
  renderRig(state.rig);
}

function buildRigScript(){
  if(!state.rig) throw new Error("Build rig first");
  var nodes = flatten(state.rig, []);
  var poseMap = {};
  var frames = getSortedPoseFrames();
  for(var i=0;i<frames.length;i++) poseMap[frames[i]] = state.poses[frames[i]];
  var names = nodes.map(function(n){ return n.name; });
  var cfg = {
    toolFrameCount: parseInt($("frameCount").value,10) || 60,
    projectStartFrame: parseInt($("startFrame").value,10) || 1,
    timelineSource: $("timelineSource").value || "project",
    chainNames: names
  };

  var lines = [];
  lines.push('// LukeTools Rig Timeline');
  lines.push('var LT_RIG_CFG = ' + JSON.stringify(cfg) + ';');
  lines.push('var LT_RIG_POSES = ' + JSON.stringify(poseMap) + ';');
  lines.push('function LT_findByName(root, wanted){');
  lines.push('  if(!root || !wanted) return null;');
  lines.push('  var seen = [];');
  lines.push('  function walk(obj){');
  lines.push('    if(!obj) return null;');
  lines.push('    if(seen.indexOf(obj) !== -1) return null;');
  lines.push('    seen.push(obj);');
  lines.push('    try{ if(obj._identifier === wanted || obj.identifier === wanted || obj.name === wanted) return obj; }catch(e0){}');
  lines.push('    var tl = null;');
  lines.push('    try{ tl = obj.timeline || obj._timeline || (obj.clipAsset ? obj.clipAsset.timeline : null); }catch(e1){ tl = null; }');
  lines.push('    if(!tl) return null;');
  lines.push('    var layers = null;');
  lines.push('    try{ layers = tl.layers || tl._layers; }catch(e2){ layers = null; }');
  lines.push('    if(!layers || !layers.length) return null;');
  lines.push('    for(var i=0;i<layers.length;i++){');
  lines.push('      var layer = layers[i];');
  lines.push('      if(!layer || !layer.frames) continue;');
  lines.push('      for(var j=0;j<layer.frames.length;j++){');
  lines.push('        var fr = layer.frames[j];');
  lines.push('        if(!fr) continue;');
  lines.push('        var clips = [];');
  lines.push('        try{ if(fr.clips) clips = fr.clips; else if(fr.children) clips = fr.children; }catch(e3){}');
  lines.push('        for(var k=0;k<clips.length;k++){ var f = walk(clips[k]); if(f) return f; }');
  lines.push('      }');
  lines.push('    }');
  lines.push('    return null;');
  lines.push('  }');
  lines.push('  return walk(root);');
  lines.push('}');
  lines.push('function LT_lerpAngleDeg(a,b,t){ var d = ((b - a + 540) % 360) - 180; return a + d*t; }');
  lines.push('function LT_easeInOutTween(t){ if(t < 0) t = 0; if(t > 1) t = 1; return t * t * (3 - 2 * t); }');
  lines.push('function LT_getTimelineCursorObject(ctx, mode){ if(mode === "selected") return ctx; if(mode === "parent"){ try{ if(ctx && ctx.parentClip) return ctx.parentClip; }catch(e0){} try{ if(ctx && ctx.parent) return ctx.parent; }catch(e1){} } return null; }');
  lines.push('function LT_getFrameFromObjectTimeline(obj){ if(!obj) return null; try{ var tl = obj.timeline || obj._timeline || (obj.clipAsset ? obj.clipAsset.timeline : null); if(!tl) return null; if(tl.playheadPosition !== undefined) return Number(tl.playheadPosition) || 1; if(tl.currentFrame !== undefined) return Number(tl.currentFrame) || 1; }catch(e2){} return null; }');
  lines.push('function LT_getProjectFrame(ctx){');
  lines.push('  var mode = LT_RIG_CFG.timelineSource || "project";');
  lines.push('  if(mode !== "project"){ var src = LT_getTimelineCursorObject(ctx, mode); var f = LT_getFrameFromObjectTimeline(src); if(f !== null) return f; }');
  lines.push('  try{ if(ctx && ctx.project && ctx.project.currentFrameNumber !== undefined) return Number(ctx.project.currentFrameNumber) || 1; }catch(e3){}');
  lines.push('  try{ if(typeof project !== "undefined" && project && project.currentFrameNumber !== undefined) return Number(project.currentFrameNumber) || 1; }catch(e4){}');
  lines.push('  try{ if(ctx && ctx.parentClip && ctx.parentClip.project && ctx.parentClip.project.currentFrameNumber !== undefined) return Number(ctx.parentClip.project.currentFrameNumber) || 1; }catch(e5){}');
  lines.push('  try{ if(ctx && ctx.root && ctx.root.project && ctx.root.project.currentFrameNumber !== undefined) return Number(ctx.root.project.currentFrameNumber) || 1; }catch(e6){}');
  lines.push('  return 1;');
  lines.push('}');
  lines.push('function LT_interpPose(frame){');
  lines.push('  if(LT_RIG_POSES[frame]) return LT_RIG_POSES[frame];');
  lines.push('  var keys = Object.keys(LT_RIG_POSES).map(function(v){ return parseInt(v,10); }).filter(function(v){ return isFinite(v); }).sort(function(a,b){ return a-b; });');
  lines.push('  if(!keys.length) return null;');
  lines.push('  var prev = null, next = null;');
  lines.push('  for(var i=0;i<keys.length;i++){ if(keys[i] <= frame) prev = keys[i]; if(keys[i] >= frame){ next = keys[i]; break; } }');
  lines.push('  if(prev === null) prev = keys[0];');
  lines.push('  if(next === null) next = keys[keys.length-1];');
  lines.push('  var A = LT_RIG_POSES[prev], B = LT_RIG_POSES[next];');
  lines.push('  if(!A && !B) return null; if(A && !B) B = A; if(B && !A) A = B;');
  lines.push('  if(prev === next) return A;');
  lines.push('  var t = (frame - prev) / (next - prev); if(t < 0) t = 0; if(t > 1) t = 1;');
  lines.push('  t = LT_easeInOutTween(t);');
  lines.push('  var out = {};');
  lines.push('  for(var nm in A){ var a = Number(A[nm]); var b = (B[nm] !== undefined) ? Number(B[nm]) : a; out[nm] = LT_lerpAngleDeg(a, b, t); }');
  lines.push('  return out;');
  lines.push('}');
  lines.push('function LT_setRotation(target, value){ var v = Number(value) || 0; try{ target.rotation = v; }catch(e4){} try{ if(target.transformation) target.transformation.rotation = v; }catch(e5){} try{ if(target._transformation) target._transformation.rotation = v; }catch(e6){} }');
  lines.push('function LT_markDirty(ctx){ try{ if(ctx && ctx.project && ctx.project.markAsModified) ctx.project.markAsModified(); }catch(e7){} }');
  lines.push('var __lt_projFrame = LT_getProjectFrame(this);');
  lines.push('var __lt_toolFrame = (__lt_projFrame - (Number(LT_RIG_CFG.projectStartFrame)||1)) + 1;');
  lines.push('if(__lt_toolFrame < 1) __lt_toolFrame = 1;');
  lines.push('if(__lt_toolFrame > (Number(LT_RIG_CFG.toolFrameCount)||1)) __lt_toolFrame = Number(LT_RIG_CFG.toolFrameCount)||1;');
  lines.push('var __lt_pose = LT_interpPose(__lt_toolFrame);');
  lines.push('if(__lt_pose){');
  lines.push('  for(var i=0;i<LT_RIG_CFG.chainNames.length;i++){ var nm = LT_RIG_CFG.chainNames[i]; var target = LT_findByName(this, nm); if(!target) continue; if(__lt_pose[nm] === undefined) continue; LT_setRotation(target, __lt_pose[nm]); }');
  lines.push('  LT_markDirty(this);');
  lines.push('}');
  lines.push('// End');
  return lines.join("\n");
}
function writeRigScript(){
  if(!Bridge){ setStatus("Bridge not available.", "bad"); return; }
  try{
    var scriptText = buildRigScript();
    var existing = "";
    try{ existing = Bridge.getUpdateScriptOnSelection() || ""; }catch(e0){}
    if(existing.indexOf('// LukeTools Rig Timeline') !== -1){
      var start = existing.indexOf('// LukeTools Rig Timeline');
      var end = existing.indexOf('// End', start);
      if(end !== -1){
        end = existing.indexOf('\n', end);
        if(end === -1) end = existing.length; else end++;
        existing = (existing.substring(0, start) + existing.substring(end)).trim();
      }
    }
    var finalText = existing ? existing + '\n\n' + scriptText : scriptText;
    var r = Bridge.setUpdateScriptOnSelection(finalText);
    if(r && r.ok === false) throw new Error(r.reason || "update script failed");
    loadRigStateFromScript(); // FIX
    refreshTimelineUi(); // FIX
    setStatus("Rig script written to selected clip update.", "good");
  }catch(err){
    setStatus("Write script failed: " + err.message, "bad");
  }
}
function build(){
  if(!Bridge){ setStatus("LukeToolsBridge not available.", "bad"); return false; }
  var info = selInfo();
  if(!info || !info.ok){ setStatus("No clip selected.", "warn"); return false; }
  var currentKey = (info.uuid || info.identifier || info.name || "Selected");
  if(state._lastBuiltKey !== currentKey) state.selectedNodeId = null;
  var obj = getSelectedObject();
  if(!obj){ setStatus("Could not read selected Wick object.", "bad"); return false; }
  var autoFrameCount = guessFrameCountFromTimelineSource();
  if(autoFrameCount && autoFrameCount > 0) $("frameCount").value = String(autoFrameCount);
  state.rig = buildChainRig(obj);
  loadRigStateFromScript();
  if(autoFrameCount && autoFrameCount > 0) $("frameCount").value = String(autoFrameCount);
  if(!state.selectedNodeId){
    var nodes = flatten(state.rig, []);
    if(nodes.length) state.selectedNodeId = nodes[0].uid;
  }
  state._lastBuiltKey = currentKey;
  renderRig(state.rig);
  setCurrentFrame(state.currentFrame || 1);
  fitView();
  setStatus("Rig built automatically and frames set from the chosen timeline source.", "good");
  return true;
}
