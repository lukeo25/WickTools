  function applyRigScale(rig) {
    var sc = Math.max(0.1, Number(rig.rigScale) || 1);
    rig.shoulderHalfWidth = rig.shoulderHalfWidthBase * sc;
    rig.hipHalfWidth = rig.hipHalfWidthBase * sc;

    var keys = Object.keys(rig.bones);
    for (var i=0;i<keys.length;i+=1){
      var b = rig.bones[keys[i]];
      b.len = (Number(b.baseLen) || b.len) * sc;
    }
  }

  // Default pose preset: T pose
  // NOTE: bone angles in this tool are RELATIVE to their parent (computeRigWorld sums angles up the chain).
  // These angles produce a classic T pose in the "Front" view:
  // - Spine points up
  // - Arms extend left/right
  // - Legs point down
  function setTPoseAngles(rig){
    if(!rig || !rig.bones) return;

    // Spine up
    rig.bones.Spine1.ang = -90;
    rig.bones.Spine2.ang = 0;
    rig.bones.Head.ang   = 0;

    // Arms: world angles should be 180 (left) and 0 (right).
    // Parent (Spine2) world is -90, so relatives are:
    //   Left: 180 - (-90) = 270 => -90
    //   Right: 0 - (-90) = 90
    rig.bones.UpperArmL.ang = -90;
    rig.bones.LowerArmL.ang = 0;
    rig.bones.HandL.ang     = 0;

    rig.bones.UpperArmR.ang = 90;
    rig.bones.LowerArmR.ang = 0;
    rig.bones.HandR.ang     = 0;

    // FIX: Legs are parented to Pelvis. Pelvis world angle is 0 (pointing right).
    // FIX: Down is +90 in this coordinate system.
    rig.bones.UpperLegL.ang = 90; // FIX
    rig.bones.LowerLegL.ang = 0;
    rig.bones.FootL.ang     = 0;

    rig.bones.UpperLegR.ang = 90; // FIX
    rig.bones.LowerLegR.ang = 0;
    rig.bones.FootR.ang     = 0;
  }

  function computeRigWorld(rig) {
    rig.world = {};
    rig.joints = {};

    var pelvis = { x: rig.pelvis.x, y: rig.pelvis.y };
    rig.joints.Pelvis = pelvis;

    function worldAngleOf(boneKey){
      var b = rig.bones[boneKey];
      if(!b) return 0;
      if(!b.parent) return b.ang;
      return worldAngleOf(b.parent) + b.ang;
    }

    function getBaseJoint(boneKey){
      var b = rig.bones[boneKey];
      if(boneKey === "UpperLegL"){
        var hxL = pelvis.x - (Number(rig.hipHalfWidth)||0);
        var hyL = pelvis.y;
        if(rig.unclipHips && rig.jointOffsets && rig.jointOffsets.HipL){ hxL += Number(rig.jointOffsets.HipL.x)||0; hyL += Number(rig.jointOffsets.HipL.y)||0; }
        return { x:hxL, y:hyL }; // FIX Start left leg at HipL while keeping hips rigid
      }
      if(boneKey === "UpperLegR"){
        var hxR = pelvis.x + (Number(rig.hipHalfWidth)||0);
        var hyR = pelvis.y;
        if(rig.unclipHips && rig.jointOffsets && rig.jointOffsets.HipR){ hxR += Number(rig.jointOffsets.HipR.x)||0; hyR += Number(rig.jointOffsets.HipR.y)||0; }
        return { x:hxR, y:hyR }; // FIX Start right leg at HipR while keeping hips rigid
      }
      if(!b) return pelvis;
      if(!b.parent) return pelvis;
      var pw = rig.world[b.parent];
      return pw ? pw.end : pelvis;
    }

    function ensureWorld(k){
      if(rig.world[k]) return;
      var b = rig.bones[k];
      if(!b) return;
      if(b.parent) ensureWorld(b.parent);

      var base = getBaseJoint(k);
      var ang = worldAngleOf(k);
      var r = degToRad(ang);
      var end = { x: base.x + Math.cos(r) * b.len, y: base.y + Math.sin(r) * b.len };
      rig.world[k] = { base:{x:base.x,y:base.y}, end:end, ang:ang, len:b.len };
    }

    var keys = Object.keys(rig.bones);
    for(var i=0;i<keys.length;i+=1) ensureWorld(keys[i]);

    var spine1 = rig.world.Spine1 ? rig.world.Spine1 : null;
    var spine2 = rig.world.Spine2 ? rig.world.Spine2 : null;

    var u1 = { x:0, y:-1 };
    if(spine1){
      var dx1 = spine1.end.x - spine1.base.x;
      var dy1 = spine1.end.y - spine1.base.y;
      var l1 = Math.sqrt(dx1*dx1 + dy1*dy1) || 1;
      u1.x = dx1/l1; u1.y = dy1/l1;
    }
    var p1 = { x:1, y:0 }; // FIX Keep hips rigid to pelvis in world axis

    rig.joints.HipL = { x: pelvis.x - p1.x * rig.hipHalfWidth, y: pelvis.y - p1.y * rig.hipHalfWidth };
    rig.joints.HipR = { x: pelvis.x + p1.x * rig.hipHalfWidth, y: pelvis.y + p1.y * rig.hipHalfWidth };

    
    if(rig.unclipHips && rig.jointOffsets){
      rig.joints.HipL.x += Number(rig.jointOffsets.HipL.x)||0;
      rig.joints.HipL.y += Number(rig.jointOffsets.HipL.y)||0;
      rig.joints.HipR.x += Number(rig.jointOffsets.HipR.x)||0;
      rig.joints.HipR.y += Number(rig.jointOffsets.HipR.y)||0;
    } else if(rig.jointOffsets){
      rig.jointOffsets.HipL.x = 0; rig.jointOffsets.HipL.y = 0;
      rig.jointOffsets.HipR.x = 0; rig.jointOffsets.HipR.y = 0;
    }

var neck = spine2 ? spine2.end : (spine1 ? spine1.end : pelvis);
    rig.joints.Neck = { x: neck.x, y: neck.y };

    var u2 = { x:0, y:-1 };
    if(spine2){
      var dx2 = spine2.end.x - spine2.base.x;
      var dy2 = spine2.end.y - spine2.base.y;
      var l2 = Math.sqrt(dx2*dx2 + dy2*dy2) || 1;
      u2.x = dx2/l2; u2.y = dy2/l2;
    }
    var p2 = { x:-u2.y, y:u2.x };

    rig.joints.ShoulderL = { x: neck.x - p2.x * rig.shoulderHalfWidth, y: neck.y - p2.y * rig.shoulderHalfWidth };
    rig.joints.ShoulderR = { x: neck.x + p2.x * rig.shoulderHalfWidth, y: neck.y + p2.y * rig.shoulderHalfWidth };

    
    if(rig.unclipShoulders && rig.jointOffsets){
      rig.joints.ShoulderL.x += Number(rig.jointOffsets.ShoulderL.x)||0;
      rig.joints.ShoulderL.y += Number(rig.jointOffsets.ShoulderL.y)||0;
      rig.joints.ShoulderR.x += Number(rig.jointOffsets.ShoulderR.x)||0;
      rig.joints.ShoulderR.y += Number(rig.jointOffsets.ShoulderR.y)||0;
    } else if(rig.jointOffsets){
      rig.jointOffsets.ShoulderL.x = 0; rig.jointOffsets.ShoulderL.y = 0;
      rig.jointOffsets.ShoulderR.x = 0; rig.jointOffsets.ShoulderR.y = 0;
    }

function overrideBase(boneKey, jointObj){
      var bw = rig.world[boneKey];
      if(!bw || !jointObj) return;
      var ang = bw.ang;
      var len = bw.len;
      var r = degToRad(ang);
      rig.world[boneKey] = {
        base:{x:jointObj.x,y:jointObj.y},
        end:{x:jointObj.x + Math.cos(r)*len, y:jointObj.y + Math.sin(r)*len},
        ang:ang, len:len
      };
    }

    overrideBase("UpperArmL", rig.joints.ShoulderL);
    overrideBase("UpperArmR", rig.joints.ShoulderR);
    overrideBase("UpperLegL", rig.joints.HipL);
    overrideBase("UpperLegR", rig.joints.HipR);

  // FIX: Fit rig to the visible stage on start so all bones are seen
  function fitRigToStage(state, view){
    try{
      if(!state || !view || !view.canvas) return;
      if(state.__fitDone) return;

      var rig = state.rig;
      if(!rig) return;

      computeRigWorld(rig);

      var minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;

      function addPt(p){
        if(!p) return;
        if(p.x < minX) minX = p.x;
        if(p.y < minY) minY = p.y;
        if(p.x > maxX) maxX = p.x;
        if(p.y > maxY) maxY = p.y;
      }

      if(rig.joints){
        var jk = Object.keys(rig.joints);
        for(var i=0;i<jk.length;i+=1){
          addPt(rig.joints[jk[i]]);
        }
      }

      if(rig.world){
        var wk = Object.keys(rig.world);
        for(var j=0;j<wk.length;j+=1){
          addPt(rig.world[wk[j]].base);
          addPt(rig.world[wk[j]].end);
        }
      }

      if(!isFinite(minX) || !isFinite(minY) || !isFinite(maxX) || !isFinite(maxY)) return;

      var bw = Math.max(1, maxX - minX);
      var bh = Math.max(1, maxY - minY);

      var cw = view.canvas.width;
      var ch = view.canvas.height;

      var z = Math.min((cw * 0.9) / bw, (ch * 0.9) / bh);
      z = clamp(z, 0.2, 5);

      // Center rig around origin so stage centering shows everything
      var cx = (minX + maxX) * 0.5;
      var cy = (minY + maxY) * 0.5;

      rig.pelvis.x -= cx;
      rig.pelvis.y -= cy;

      state.zoom = z;
      state.__fitDone = true;

      computeRigWorld(rig);
    }catch(e){}
  }


    function chain(childKey, parentKey){
      if(!rig.world[parentKey] || !rig.world[childKey]) return;
      var base = rig.world[parentKey].end;
      var ang = rig.world[childKey].ang;
      var len = rig.world[childKey].len;
      var r = degToRad(ang);
      rig.world[childKey] = {
        base:{x:base.x,y:base.y},
        end:{x:base.x + Math.cos(r)*len, y:base.y + Math.sin(r)*len},
        ang:ang, len:len
      };
    }

    chain("LowerArmL","UpperArmL"); chain("HandL","LowerArmL");
    chain("LowerArmR","UpperArmR"); chain("HandR","LowerArmR");
    chain("LowerLegL","UpperLegL"); chain("FootL","LowerLegL");
    chain("LowerLegR","UpperLegR"); chain("FootR","LowerLegR");
    chain("Spine2","Spine1");
    chain("Head","Spine2");
  }

    function isRightBoneKey(boneKey){
    return !!(boneKey && typeof boneKey === "string" && boneKey.slice(-1) === "R");
  }

  function isLeftBoneKey(boneKey){
    return !!(boneKey && typeof boneKey === "string" && boneKey.slice(-1) === "L");
  }

  function boneColor(boneKey){
    if(boneKey === "Spine1" || boneKey === "Spine2" || boneKey === "Head") return "#bfbfbf";
    if(isRightBoneKey(boneKey)) return "#34e06b";
    if(isLeftBoneKey(boneKey)) return "#ff4a4a";

  }

  function sideDimAlpha(viewPreset, boneKey){
    if(viewPreset === "Side Left"){
      if(isRightBoneKey(boneKey)) return 0.28;
      return 0.95;
    }
    if(viewPreset === "Side Right"){
      if(isLeftBoneKey(boneKey)) return 0.28;
      return 0.95;
    }
    if(viewPreset === "Three Quarter Left"){
      if(isRightBoneKey(boneKey)) return 0.55;
      return 0.95;
    }
    if(viewPreset === "Three Quarter Right"){
      if(isLeftBoneKey(boneKey)) return 0.55;
      return 0.95;
    }
    return 0.95;
  }


  function drawBoneWedge(ctx, base, end, color, baseW, endW){
    var dx=end.x-base.x, dy=end.y-base.y;
    var len=Math.sqrt(dx*dx+dy*dy)||1;
    dx/=len; dy/=len;
    var px=-dy, py=dx;

    var bx1 = base.x + px*(baseW*0.5);
    var by1 = base.y + py*(baseW*0.5);
    var bx2 = base.x - px*(baseW*0.5);
    var by2 = base.y - py*(baseW*0.5);

    var ex1 = end.x + px*(endW*0.5);
    var ey1 = end.y + py*(endW*0.5);
    var ex2 = end.x - px*(endW*0.5);
    var ey2 = end.y - py*(endW*0.5);

    ctx.beginPath();
    ctx.moveTo(bx1,by1);
    ctx.lineTo(ex1,ey1);
    ctx.lineTo(ex2,ey2);
    ctx.lineTo(bx2,by2);
    ctx.closePath();
    ctx.fillStyle=color;
    ctx.fill();

    ctx.globalAlpha = 0.85;
    ctx.lineWidth = 1;
    ctx.strokeStyle = "rgba(0,0,0,0.65)";
    ctx.stroke();
    ctx.globalAlpha = 1;
  }

  function drawJointGlobe(ctx,x,y,r,isSelected){
    ctx.save();
    ctx.beginPath(); ctx.arc(x,y,r,0,Math.PI*2);
    ctx.fillStyle="#0e0e0e"; ctx.fill();
    ctx.lineWidth = isSelected ? 3 : 2;
    ctx.strokeStyle = isSelected ? "#ffffff" : "rgba(230,230,230,0.92)";
    ctx.stroke();

    ctx.globalAlpha = 0.8;
    ctx.lineWidth = 1;
    ctx.beginPath(); ctx.arc(x,y,r*0.65,0,Math.PI*2); ctx.stroke();
    ctx.beginPath(); ctx.moveTo(x-r*0.8,y); ctx.lineTo(x+r*0.8,y); ctx.stroke();
    ctx.beginPath(); ctx.moveTo(x,y-r*0.8); ctx.lineTo(x,y+r*0.8); ctx.stroke();
    ctx.restore();
  }

  function capturePose(state){
    var rig = state.rig;
    var angs = {};
    var keys = Object.keys(rig.bones);
    for(var i=0;i<keys.length;i+=1){
      angs[keys[i]] = Number(rig.bones[keys[i]].ang) || 0;
    }

    var jOff = {
      ShoulderL:{ x: (rig.jointOffsets && rig.jointOffsets.ShoulderL) ? (Number(rig.jointOffsets.ShoulderL.x)||0) : 0,
                 y: (rig.jointOffsets && rig.jointOffsets.ShoulderL) ? (Number(rig.jointOffsets.ShoulderL.y)||0) : 0 },
      ShoulderR:{ x: (rig.jointOffsets && rig.jointOffsets.ShoulderR) ? (Number(rig.jointOffsets.ShoulderR.x)||0) : 0,
                 y: (rig.jointOffsets && rig.jointOffsets.ShoulderR) ? (Number(rig.jointOffsets.ShoulderR.y)||0) : 0 },
      HipL:{ x: (rig.jointOffsets && rig.jointOffsets.HipL) ? (Number(rig.jointOffsets.HipL.x)||0) : 0,
             y: (rig.jointOffsets && rig.jointOffsets.HipL) ? (Number(rig.jointOffsets.HipL.y)||0) : 0 },
      HipR:{ x: (rig.jointOffsets && rig.jointOffsets.HipR) ? (Number(rig.jointOffsets.HipR.x)||0) : 0,
             y: (rig.jointOffsets && rig.jointOffsets.HipR) ? (Number(rig.jointOffsets.HipR.y)||0) : 0 }
    };

    var pose = {
      pelvis:{ x: rig.pelvis.x, y: rig.pelvis.y },
      angs: angs,
      jOff: jOff
    };

    if(state.keyScale){
      var scales = {};
      for(var p=0;p<PARTS.length;p+=1){
        var pk = PARTS[p].key;
        var pp = state.partParams[pk];
        scales[pk] = { sx: pp ? (Number(pp.sx)||1) : 1, sy: pp ? (Number(pp.sy)||1) : 1 };
      }
      pose.scales = scales;
    }

    return pose;
  }
  function applyPose(state, pose){
    if(!pose) return;    // FIX: Removed ApplyPose debug log (logs must be on stage bone creation only)
    var rig = state.rig;

    if(pose.pelvis){
      rig.pelvis.x = Number(pose.pelvis.x) || rig.pelvis.x;
      rig.pelvis.y = Number(pose.pelvis.y) || rig.pelvis.y;
    }

    if(pose.angs){
      var keys = Object.keys(pose.angs);
      for(var i=0;i<keys.length;i+=1){
        var k = keys[i];
        if(rig.bones[k]){
          rig.bones[k].ang = Number(pose.angs[k]) || rig.bones[k].ang;
          try { __LT_log("ApplyPose", "bone", k, "ang", rig.bones[k].ang, "frame", (state && state.currentFrame)); } catch(eApLog){} // FIX: LOG inside applyPose
        }
      }
    }

    if(pose.jOff && rig.jointOffsets){
      rig.jointOffsets.ShoulderL.x = Number(pose.jOff.ShoulderL && pose.jOff.ShoulderL.x) || 0;
      rig.jointOffsets.ShoulderL.y = Number(pose.jOff.ShoulderL && pose.jOff.ShoulderL.y) || 0;
      rig.jointOffsets.ShoulderR.x = Number(pose.jOff.ShoulderR && pose.jOff.ShoulderR.x) || 0;

      rig.jointOffsets.HipL.x = Number(pose.jOff.HipL && pose.jOff.HipL.x) || 0;

      rig.jointOffsets.HipR.y = Number(pose.jOff.HipR && pose.jOff.HipR.y) || 0;
    }

    if(pose.scales){
      var pks = Object.keys(pose.scales);
      for(var j=0;j<pks.length;j+=1){
        var pk = pks[j];
        if(state.partParams[pk]){
          state.partParams[pk].sx = Number(pose.scales[pk].sx) || state.partParams[pk].sx || 1;
          state.partParams[pk].sy = Number(pose.scales[pk].sy) || state.partParams[pk].sy || 1;

  function nearestKeyFrames(map, frame){
    var ks = Object.keys(map || {}).map(function(k){ return Number(k); }).filter(function(n){ return isFinite(n); });
    if(!ks.length) return { prev:null, next:null };
    ks.sort(function(a,b){ return a-b; });

    var prev=null, next=null;
    for(var i=0;i<ks.length;i+=1){
      var f = ks[i];
      if(f <= frame) prev = f;
      if(f >= frame){ next = f; break; }
    }
    if(prev === null) prev = ks[0];
    if(next === null) next = ks[ks.length-1];
    return { prev: prev, next: next };
  }

  function evalPoseAtFrame(state, frame){
    var map = state.poseKeys || {};
    var pair = nearestKeyFrames(map, frame);
    if(pair.prev === null || pair.next === null) return capturePose(state);

    var A = map[String(pair.prev)];
    var B = map[String(pair.next)];
    if(!A && !B) return capturePose(state);
    if(A && !B) return A;
    if(!A && B) return B;
    if(pair.prev === pair.next) return A;

    var t = (frame - pair.prev) / (pair.next - pair.prev);
    t = clamp(t, 0, 1);

    var out = { pelvis:{x:0,y:0}, angs:{}, jOff:{}, scales:null };
    out.pelvis.x = lerp(A.pelvis.x, B.pelvis.x, t);
    out.pelvis.y = lerp(A.pelvis.y, B.pelvis.y, t);

    var rigKeys = Object.keys(state.rig.bones);
    for(var i=0;i<rigKeys.length;i+=1){
      var k = rigKeys[i];
      var a = (A.angs && isFinite(A.angs[k])) ? A.angs[k] : (state.rig.bones[k].ang || 0);
      var b = (B.angs && isFinite(B.angs[k])) ? B.angs[k] : (state.rig.bones[k].ang || 0);
      out.angs[k] = lerpAngleDeg(a, b, t);
    }

    function jv(pose, key, axis){
      if(!pose || !pose.jOff || !pose.jOff[key]) return 0;
      return Number(pose.jOff[key][axis]) || 0;
    }

    out.jOff.ShoulderL = { x: lerp(jv(A,"ShoulderL","x"), jv(B,"ShoulderL","x"), t), y: lerp(jv(A,"ShoulderL","y"), jv(B,"ShoulderL","y"), t) };
    out.jOff.ShoulderR = { x: lerp(jv(A,"ShoulderR","x"), jv(B,"ShoulderR","x"), t), y: lerp(jv(A,"ShoulderR","y"), jv(B,"ShoulderR","y"), t) };
    out.jOff.HipL      = { x: lerp(jv(A,"HipL","x"),      jv(B,"HipL","x"),      t), y: lerp(jv(A,"HipL","y"),      jv(B,"HipL","y"),      t) };
    out.jOff.HipR      = { x: lerp(jv(A,"HipR","x"),      jv(B,"HipR","x"),      t), y: lerp(jv(A,"HipR","y"),      jv(B,"HipR","y"),      t) };

    if(A.scales || B.scales){
      out.scales = {};
      for(var p=0;p<PARTS.length;p+=1){
        var pk = PARTS[p].key;
        var aSx = (A.scales && A.scales[pk]) ? (Number(A.scales[pk].sx)||1) : (state.partParams[pk] ? (Number(state.partParams[pk].sx)||1) : 1);
        var aSy = (A.scales && A.scales[pk]) ? (Number(A.scales[pk].sy)||1) : (state.partParams[pk] ? (Number(state.partParams[pk].sy)||1) : 1);
        var bSx = (B.scales && B.scales[pk]) ? (Number(B.scales[pk].sx)||1) : (state.partParams[pk] ? (Number(state.partParams[pk].sx)||1) : 1);
        var bSy = (B.scales && B.scales[pk]) ? (Number(B.scales[pk].sy)||1) : (state.partParams[pk] ? (Number(state.partParams[pk].sy)||1) : 1);
        out.scales[pk] = { sx: lerp(aSx, bSx, t), sy: lerp(aSy, bSy, t) };
      }
    }

    return out;
  }

  function drawBones(state, ctx){
    if(!state.showBones) return;

    var rig = state.rig;
    computeRigWorld(rig);

    ctx.save();

    var keys = Object.keys(rig.world);

    function z(k){
      if(k==="Spine1") return 0;
      if(k==="Spine2") return 1;
      if(k==="Head") return 2;
      if(isLeftBoneKey(k)) return 3;
      if(isRightBoneKey(k)) return 4;
      return 10;
    }
    keys.sort(function(a,b){ return z(a)-z(b); });

    for(var i=0;i<keys.length;i+=1){
      var k = keys[i];
      var bw = rig.world[k];
      if(!bw) continue;

      var col = boneColor(k);
      var bwid = (k==="Spine1"||k==="Spine2") ? 26 : 22;
      var ewid = (k==="Spine1"||k==="Spine2") ? 10 : 7;

      ctx.globalAlpha = sideDimAlpha(rig.viewPreset, k);
      drawBoneWedge(ctx, bw.base, bw.end, col, bwid, ewid);
    }

    ctx.globalAlpha = 1;

    var jointPts = [];
    jointPts.push({ name:"Pelvis", pt: rig.joints.Pelvis });
    jointPts.push({ name:"HipL", pt: rig.joints.HipL });
    jointPts.push({ name:"HipR", pt: rig.joints.HipR });
    jointPts.push({ name:"Neck", pt: rig.joints.Neck });
    jointPts.push({ name:"ShoulderL", pt: rig.joints.ShoulderL });
    jointPts.push({ name:"ShoulderR", pt: rig.joints.ShoulderR });

    function pushEnd(k){ if(rig.world[k]) jointPts.push({ name:k, pt: rig.world[k].end }); }
    pushEnd("Head");
    pushEnd("UpperArmL"); pushEnd("LowerArmL"); pushEnd("HandL");
    pushEnd("UpperArmR"); pushEnd("LowerArmR"); pushEnd("HandR");
    pushEnd("UpperLegL"); pushEnd("LowerLegL"); pushEnd("FootL");
    pushEnd("UpperLegR"); pushEnd("LowerLegR"); pushEnd("FootR");

    for(var j=0;j<jointPts.length;j+=1){
      var jp = jointPts[j];
      if(!jp || !jp.pt) continue;
      var isSel = (state.selectedBone === "Pelvis" && jp.name === "Pelvis");
      drawJointGlobe(ctx, jp.pt.x, jp.pt.y, 8, isSel);
    }

    if(state.selectedBone && state.selectedBone !== "Pelvis" && rig.world[state.selectedBone]){
      var se = rig.world[state.selectedBone].end;
      drawJointGlobe(ctx, se.x, se.y, 9, true);
    }

    ctx.restore();
  }

  function getPartTransformFromBone(state, partKey){
    var rig = state.rig;
    computeRigWorld(rig);

    var meta = null;
    for(var i=0;i<PARTS.length;i+=1) if(PARTS[i].key === partKey) meta = PARTS[i];
    if(!meta) return null;

    var bw = rig.world[meta.bone];
    if(!bw) return null;

    var mid = { x:(bw.base.x+bw.end.x)*0.5, y:(bw.base.y+bw.end.y)*0.5 };
    var p = state.partParams[partKey] || { ox:0, oy:0, rotOff:0, sx:1, sy:1, px:0.5, py:0.5 };

    var ang = bw.ang;
    var ar = degToRad(ang);
    var rx = p.ox * Math.cos(ar) - p.oy * Math.sin(ar);
    var ry = p.ox * Math.sin(ar) + p.oy * Math.cos(ar);

    return {
      x: mid.x + rx,
      y: mid.y + ry,
      r: ang + (p.rotOff || 0),
      sx: p.sx || 1,
      sy: p.sy || 1,
      px: clamp(p.px, 0, 1),
      py: clamp(p.py, 0, 1)
    };
  }

  function drawPinnedImages(state, ctx){
    for(var i=0;i<PARTS.length;i+=1){
      var pk = PARTS[i].key;
      var img = state.partImages[pk];
      if(!img) continue;

      var tfm = getPartTransformFromBone(state, pk);
      if(!tfm) continue;

      var iw = img.naturalWidth || img.width || 1;
      var ih = img.naturalHeight || img.height || 1;

      ctx.save();
      ctx.translate(tfm.x, tfm.y);
      ctx.rotate((tfm.r||0) * Math.PI/180);
      ctx.scale(tfm.sx || 1, tfm.sy || 1);

      var ox = -iw * (tfm.px || 0.5);
      var oy = -ih * (tfm.py || 0.5);
      ctx.globalAlpha = 0.96;
      ctx.drawImage(img, ox, oy);
      ctx.restore();
    }
  }

  function drawTimeline(state, view){
    var c = view.tlCanvas;
    var ctx = c.getContext("2d");
    if(!ctx) return;

    var w = c.width;
    var h = c.height;

    ctx.clearRect(0,0,w,h);
    ctx.fillStyle="#0f0f0f";
    ctx.fillRect(0,0,w,h);

    var fc = Math.max(1, state.frameCount);
    var pxPerFrame = w / fc;

    for(var f=1; f<=fc; f+=1){
      var x = Math.floor((f-1)*pxPerFrame);
      var bw = Math.ceil(pxPerFrame);
      if(bw < 2) bw = 2;

      var alt = (f % 2 === 0);
      ctx.fillStyle = alt ? "rgba(255,255,255,0.05)" : "rgba(255,255,255,0.02)";
      ctx.fillRect(x, 0, bw, h);

      ctx.fillStyle = "rgba(0,0,0,0.35)";
      ctx.fillRect(x, 0, 1, h);
    }

    var ks = Object.keys(state.poseKeys || {}).map(function(k){ return Number(k); }).filter(function(n){ return isFinite(n); });
    ks.sort(function(a,b){ return a-b; });

    for(var i=0;i<ks.length;i+=1){
      var kf = ks[i];
      var xk = frameToX(state, view, kf) + pxPerFrame*0.5;

      var size = 10;
      var y = h - 14;

      var isSel = (state.selectedKeyFrame === kf);
      ctx.fillStyle = isSel ? "#ffffff" : "#7bd88f";
      ctx.fillRect(xk - size/2, y - size/2, size, size);

      ctx.strokeStyle = "rgba(0,0,0,0.55)";
      ctx.lineWidth = 1;
      ctx.strokeRect(xk - size/2, y - size/2, size, size);
    }

    var xcur = frameToX(state, view, state.currentFrame) + pxPerFrame*0.5;
    drawPlayhead(ctx, xcur, h);
  }

      var bw=rig.world[k];
      if(!bw) continue;
      var d = dist2(mx,my,bw.end.x,bw.end.y);
      if(d < bestD){ bestD=d; best=k; }
    }
    if(bestD <= 20*20) return best;
    return null;
  }

  function pickPelvis(state, mx, my){
    computeRigWorld(state.rig);
    var p = state.rig.joints.Pelvis;
    if(!p) return false;
    return dist2(mx,my,p.x,p.y) <= 22*22;
  }

  function pickSpecialJoint(state, mx, my){
    computeRigWorld(state.rig);
    var rig = state.rig;
    var best = null;

    function consider(key, radius){
      var p = rig.joints[key];
      if(!p) return;
      var d2 = dist2(mx,my,p.x,p.y);
      if(d2 <= radius*radius){
        best = key;
      }
    }

    if(rig.unclipShoulders){
      consider("ShoulderL", 16);
      consider("ShoulderR", 16);
    }

    if(rig.unclipHips){
      var pel = rig.joints.Pelvis;
      var pelD2 = pel ? dist2(mx,my,pel.x,pel.y) : 1e9;
      if(pelD2 > 12*12){
        consider("HipL", 16);
        consider("HipR", 16);
      }
    }

    return best;
  }


  function applyMirrorLengths(state, changedKey){
    var rig = state.rig;

    function mirrorPair(a,b){
      if(!rig.bones[a] || !rig.bones[b]) return;
      rig.bones[b].len = rig.bones[a].len;
    }

    if(state.mirrorArms){
      if(changedKey === "UpperArmL") mirrorPair("UpperArmL","UpperArmR");
      if(changedKey === "UpperArmR") mirrorPair("UpperArmR","UpperArmL");
      if(changedKey === "LowerArmL") mirrorPair("LowerArmL","LowerArmR");
      if(changedKey === "LowerArmR") mirrorPair("LowerArmR","LowerArmL");
      if(changedKey === "HandL") mirrorPair("HandL","HandR");
      if(changedKey === "HandR") mirrorPair("HandR","HandL");
    }

    if(state.mirrorLegs){
      if(changedKey === "UpperLegL") mirrorPair("UpperLegL","UpperLegR");
      if(changedKey === "UpperLegR") mirrorPair("UpperLegR","UpperLegL");
      if(changedKey === "LowerLegL") mirrorPair("LowerLegL","LowerLegR");
      if(changedKey === "LowerLegR") mirrorPair("LowerLegR","LowerLegL");
      if(changedKey === "FootL") mirrorPair("FootL","FootR");
      if(changedKey === "FootR") mirrorPair("FootR","FootL");
    }
  }

  // Wait until Wick's render objectCache knows about an imported asset uuid.
  // This helps avoid failures when placing newly imported image assets into clips.
  async function waitObjectCacheAsset(assetUUID, label){
    var uuid = String(assetUUID || "");
    if(!uuid) return true;

    // This is a best effort wait. If it never appears, we continue anyway.
    var start = Date.now();
    var timeoutMs = 12000; // allow time for Wick ObjectCache to build after asset import

    while(Date.now() - start < timeoutMs){
      try{
        var p = window.parent || window;

        var caches = [];
        try { if(p && p.wickEditor && p.wickEditor.canvas && p.wickEditor.canvas.objectCache) caches.push(p.wickEditor.canvas.objectCache); } catch(e0){}
        try { if(p && p.WickEditor && p.WickEditor.editor && p.WickEditor.editor.canvas && p.WickEditor.editor.canvas.objectCache) caches.push(p.WickEditor.editor.canvas.objectCache); } catch(e1){}
        try { if(p && p.wickEditor && p.wickEditor.project && p.wickEditor.project.objectCache) caches.push(p.wickEditor.project.objectCache); } catch(e2){}
        try { if(p && p.wickEditor && p.wickEditor.objectCache) caches.push(p.wickEditor.objectCache); } catch(e3){}

        for(var i=0;i<caches.length;i++){
          var cache = caches[i];
          try{
            if(cache){
              if(typeof cache.getAsset === "function"){
                var a = cache.getAsset(uuid);
                if(a) return true;
              }
              if(cache.assets && typeof cache.assets === "object"){
                if(cache.assets[uuid]) return true;
              }
            }
          }catch(e4){}
        }

        // Also check Wick.ObjectCache, which is the most reliable cache for createImagePathFromAsset
        try{
          var W = __LT_getWickGlobalAny();
          if(W && W.ObjectCache && typeof W.ObjectCache.getObjectByUUID === "function"){
            if(W.ObjectCache.getObjectByUUID(uuid)) return true;
          }
        }catch(eW){}
      }catch(e5){}

      await new Promise(function(res){ setTimeout(res, 60); });
    }

    __LT_warn("[LukeTools] objectCache asset not ready (continuing)", label, uuid);
    return false;
  }

function __LT_pngWedgeDataUrl(fillHex){
    // Bitmap wedge used for Wick assets (base joint is at the IMAGE CENTER).
    // This makes it easy to place/rotate/scale the resulting clip around its base.
    try{
      var c = document.createElement("canvas");
      c.width = 220;
      c.height = 90;
      var ctx = c.getContext("2d");
      ctx.clearRect(0,0,c.width,c.height);

      // Base joint at canvas center
      var cx = c.width/2;
      var cy = c.height/2;

      // Bone length in pixels from base to tip inside the bitmap
      // (poseClipToBone scales this to match rig lengths)
      var L = 110;

      ctx.translate(cx, cy);

      ctx.lineJoin = "round";
      ctx.lineWidth = 4;
      ctx.strokeStyle = "#000000";
      ctx.fillStyle = String(fillHex || "#8a8a8a");

      // Main wedge (base at 0,0 pointing to +X)
      ctx.beginPath();
      ctx.moveTo(0,-12);
      ctx.lineTo(L,0);
      ctx.lineTo(0,12);
      ctx.closePath();
      ctx.fill();
      ctx.stroke();

      // Base joint circle
      ctx.fillStyle = "#000000";
      ctx.beginPath();
      ctx.arc(0,0,9,0,Math.PI*2);
      ctx.fill();

      ctx.fillStyle = "#ffffff";
      ctx.beginPath();
      ctx.arc(0,0,6,0,Math.PI*2);
      ctx.fill();

      // Tip joint circle (smaller)
      ctx.fillStyle = "#000000";
      ctx.beginPath();
      ctx.arc(L,0,6,0,Math.PI*2);
      ctx.fill();

      ctx.fillStyle = "#ffffff";
      ctx.beginPath();
      ctx.arc(L,0,4,0,Math.PI*2);
      ctx.fill();

      return c.toDataURL("image/png");
    }catch(e1){
      return "";
    }
  }

    function dataUrlToFile(dataUrl, filename){
      try{
        var parts = String(dataUrl || "").split(",");
        if(parts.length < 2) return null;
        var header = parts[0];
        var b64 = parts[1];
        var mimeMatch = header.match(/data:([^;]+);base64/i);
        var mime = mimeMatch ? mimeMatch[1] : "application/octet-stream";
        var bin = atob(b64);
        var len = bin.length;
        var bytes = new Uint8Array(len);
        for(var i=0;i<len;i++) bytes[i] = bin.charCodeAt(i);
        var blob = new Blob([bytes], { type: mime });
        try { return new File([blob], filename, { type: mime }); } catch(e2){
          // File may be blocked, fallback to Blob with name
          blob.name = filename;
          return blob;
        }
      }catch(e1){}
      return null;
    }

  function bindStageInteractions(state, view){
    function mousePos(ev){
      var r = view.canvas.getBoundingClientRect();
      var z = (state && state.zoom) ? state.zoom : 1; // FIX
      return { x: ((ev.clientX - r.left) - (r.width * 0.5)) / z, y: ((ev.clientY - r.top) - (r.height * 0.5)) / z }; // FIX
    }

    view.canvas.addEventListener("mousedown", function(ev){
      var m = mousePos(ev);

      if(state.showBones){
        var jHit = pickSpecialJoint(state, m.x, m.y);
        if(jHit){
          state.selectedBone = jHit;
          var off = (state.rig.jointOffsets && state.rig.jointOffsets[jHit]) ? state.rig.jointOffsets[jHit] : {x:0,y:0};
          state.drag = { mode:"joint", joint:jHit, sx:m.x, sy:m.y, ox:Number(off.x)||0, oy:Number(off.y)||0 };
          return;
        }

        if(pickPelvis(state, m.x, m.y)){
          state.selectedBone = "Pelvis";
          if(view && view.__LT_tlScrollToLayer){ view.__LT_tlScrollToLayer(state.selectedBone); } // FIX: Jump timeline to selected bone layer
          state.drag = { mode:"pelvis", sx:m.x, sy:m.y, px: state.rig.pelvis.x, py: state.rig.pelvis.y };
          return;
        }

        var hit = pickBoneEnd(state, m.x, m.y);
        if(hit){
          state.selectedBone = hit;
          computeRigWorld(state.rig);
          var bw = state.rig.world[hit];
          state.drag = { mode:"bone", bone: hit, base:{x:bw.base.x,y:bw.base.y} };
          return;
        }
      }
      state.drag = null;
    });

    window.addEventListener("mouseup", function(){ state.drag = null; });

    window.addEventListener("mousemove", function(ev){
      if(!state.drag) return;
      var m = mousePos(ev);

      if(state.drag.mode === "joint"){
        var dxJ = m.x - state.drag.sx;
        var dyJ = m.y - state.drag.sy;
        if(state.rig.jointOffsets && state.rig.jointOffsets[state.drag.joint]){
          state.rig.jointOffsets[state.drag.joint].x = state.drag.ox + dxJ;
          state.rig.jointOffsets[state.drag.joint].y = state.drag.oy + dyJ;
        }
        if(state.autoKeyPose) setPoseKey(state, state.currentFrame);
        return;
      }

      if(state.drag.mode === "pelvis"){
        var dx = m.x - state.drag.sx;
        var dy = m.y - state.drag.sy;
        state.rig.pelvis.x = state.drag.px + dx;
        state.rig.pelvis.y = state.drag.py + dy;

        if(state.autoKeyPose) setPoseKey(state, state.currentFrame);
        return;
      }

      if(state.drag.mode === "bone"){
        var bone = state.drag.bone;
        computeRigWorld(state.rig);
        var bw = state.rig.world[bone];
        if(!bw) return;

        var bx = bw.base.x, by = bw.base.y;
        var ang = radToDeg(Math.atan2(m.y - by, m.x - bx));

        var b = state.rig.bones[bone];
        if(!b) return;

        if(!b.parent) b.ang = ang;
        else {
          var pw = state.rig.world[b.parent];
          var pa = pw ? pw.ang : 0;
          b.ang = ang - pa;
        }

        if(state.autoKeyPose) setPoseKey(state, state.currentFrame);
        return;
      }
    });
  }


    // FIX: Mouse wheel zoom on stage
    try{
      view.canvas.addEventListener("wheel", function(ev){
        ev.preventDefault();
        var z = (state && state.zoom) ? state.zoom : 1;
        if(ev.deltaY < 0) z *= 1.1;
        else z *= 0.9;

  function bindTimelineInteractions(state, view){

    function tlMouse(ev){

    function keyHitTest(x, y){
      var fc = Math.max(1, state.frameCount);
      var w = view.tlCanvas.width;
      var pxPerFrame = w / fc;

      var ks = Object.keys(state.poseKeys || {}).map(function(k){ return Number(k); }).filter(function(n){ return isFinite(n); });
      for(var i=0;i<ks.length;i+=1){
        var kf = ks[i];
        var xk = frameToX(state, view, kf) + pxPerFrame*0.5;
        var dx = Math.abs(x - xk);
        var dy = Math.abs(y - (view.tlCanvas.height - 14));
        if(dx <= 9 && dy <= 12) return kf;
      }

      var y = m.y * (view.tlCanvas.height / Math.max(1, m.h));

      var hitKey = keyHitTest(x, y);
      if(hitKey !== null){
        state.selectedKeyFrame = hitKey;
        if(hitKey === 1){
          toast("Frame 1 key locked", true);
          return;
        }
        state.keyDrag = { frame: hitKey };
        return;

      var pxPerFrame = view.tlCanvas.width / Math.max(1, state.frameCount|0);
      var phX = frameToX(state, view, state.currentFrame) + pxPerFrame * 0.5;

      state.tlDrag = { active:true, offsetX: inHandle ? (phX - x) : 0 }; // FIX

      if(target === 1){ toast("Frame 1 key locked", true); return; }
      if(state.poseKeys[String(target)]){
        deletePoseKey(state, target);
        toast("Key deleted", false);
      } else {
        toast("No key to delete", true);
      }
    });

    view.btnPlay.addEventListener("click", function(){
      // FIX: Allow Play even when no video is loaded (advance the frame counter at FPS)
      state.playing = !state.playing;
      state.lastPlayMs = nowMs();
      updatePlayButton(state, view);
    });

    function loop(){
      if(state.playing){
        // FIX: If a video is loaded, advance video time. Otherwise advance the frame counter at FPS.
        if(state.videoReady){
          try{
            var dt = (nowMs() - state.lastPlayMs) / 1000;
            state.lastPlayMs = nowMs();

            var cur = Number(view.video.currentTime) || 0;
            cur += dt;
            if(cur > state.duration) cur = state.duration;
            view.video.currentTime = cur;

            var f = getFrameFromTime(cur, state.fps);
            if(f !== state.currentFrame) setFrame(state, view, f);
          }catch(e){}
        }else{
          try{
            var dt2 = (nowMs() - state.lastPlayMs) / 1000;
            state.lastPlayMs = nowMs();
            var ff = state.currentFrame + (dt2 * state.fps);
            var f2 = Math.floor(ff + 0.00001);
            if(f2 < 1) f2 = 1;
            if(f2 > state.frameCount){
              f2 = state.frameCount;
              state.playing = false;
              updatePlayButton(state, view);
            }
            if(f2 !== state.currentFrame) setFrame(state, view, f2);
          }catch(e){}
        }
      }

      applyPose(state, evalPoseAtFrame(state, state.currentFrame));
      drawStage(state, view);
      drawTimeline(state, view);

      requestAnimationFrame(loop);
    }

    syncTimelineControls(state, view);
    updatePlayButton(state, view);
    setFrame(state, view, 1);
    loop();
  }

  (function(){
    var root  = document.querySelector(".root");
    var btn   = document.getElementById("ltFsBtnToggle");
    var icon  = document.getElementById("ltFsIcon");
    var hud   = document.getElementById("ltFsHud");
    var fsMode = false;

    var SVG_EXPAND = '<path d="M8 3H5a2 2 0 0 0-2 2v3"/><path d="M21 8V5a2 2 0 0 0-2-2h-3"/><path d="M3 16v3a2 2 0 0 0 2 2h3"/><path d="M16 21h3a2 2 0 0 0 2-2v-3"/>';
    var SVG_SHRINK = '<path d="M4 14h6v6"/><path d="M20 10h-6V4"/><path d="M14 10l7-7"/><path d="M3 21l7-7"/>';

    function setFsMode(on) {
      fsMode = on;
      root.classList.toggle("ltFsMode", on);
      icon.innerHTML = on ? SVG_SHRINK : SVG_EXPAND;
      btn.title = on ? "Exit fullscreen panel" : "Toggle fullscreen panel";
      // Trigger resize so stage/canvas recalculates
      try { window.dispatchEvent(new Event("resize")); } catch(e) {}
    }

    if (btn) btn.addEventListener("click", function() { setFsMode(!fsMode); });

    // Escape key exits FS
    window.addEventListener("keydown", function(ev) {
      if ((ev.key === "Escape" || ev.key === "F11") && fsMode) {
        ev.preventDefault();
        setFsMode(false);
      }
      // F key toggles FS
      if (ev.key === "f" || ev.key === "F") {
        if (document.activeElement && (document.activeElement.tagName === "INPUT" || document.activeElement.tagName === "TEXTAREA" || document.activeElement.tagName === "SELECT")) return;
        setFsMode(!fsMode);
      }
    });

    // ── Wire HUD buttons to mirror the real player buttons ──
    function mirrorBtn(hudId, realId) {
      var hudEl  = document.getElementById(hudId);
      var realEl = document.getElementById(realId);
      if (hudEl && realEl) {
        hudEl.addEventListener("click", function() { realEl.click(); });
      }
    }

    // The real buttons live inside .tlTopLeft — query after a tick so buildUI has run
    setTimeout(function() {
      // Find player buttons by text content since they don't have IDs
      var allBtns = document.querySelectorAll(".btnSmall");
      var found = { prev: null, play: null, next: null, key: null, del: null };
      for (var i = 0; i < allBtns.length; i++) {
        var t = (allBtns[i].textContent || "").trim();
        if (t === "Prev"    && !found.prev) found.prev = allBtns[i];
        if ((t === "Play" || t === "Pause") && !found.play) found.play = allBtns[i];
        if (t === "Next"    && !found.next) found.next = allBtns[i];
        if (t === "Key"     && !found.key)  found.key  = allBtns[i];
        if (t === "Del Key" && !found.del)  found.del  = allBtns[i];
      }

      var hudPrev = document.getElementById("ltFsHudPrev");
      var hudPlay = document.getElementById("ltFsHudPlay");
      var hudNext = document.getElementById("ltFsHudNext");
      var hudKey  = document.getElementById("ltFsHudKey");
      var hudDel  = document.getElementById("ltFsHudDel");
      var hudFR   = document.getElementById("ltFsFrameRead");
      var hudBone = document.getElementById("ltFsHudBone");

      if (hudPrev && found.prev) hudPrev.onclick = function() { found.prev.click(); };
      if (hudNext && found.next) hudNext.onclick = function() { found.next.click(); };
      if (hudKey  && found.key)  hudKey.onclick  = function() { found.key.click(); };
      if (hudDel  && found.del)  hudDel.onclick  = function() { found.del.click(); };

      // Play button — text changes between Play/Pause, mirror it
      if (hudPlay && found.play) {
        hudPlay.onclick = function() { found.play.click(); };
        // Sync label by watching real button text
        setInterval(function() {
          if (!found.play || !hudPlay) return;
          var t2 = (found.play.textContent || "").trim();
          hudPlay.textContent = (t2 === "Pause") ? "⏸ Pause" : "▶ Play";
        }, 200);
      }

      // Frame read — mirror .tlRead
      var realRead = document.querySelector(".tlRead");
      if (hudFR && realRead) {
        setInterval(function() { hudFR.textContent = realRead.textContent; }, 100);
      }

      // Bone picker — populate from PARTS list and wire to click the bone label in left panel
      var BONE_KEYS = [
        "Head","Torso","Midsection",
        "UpperArmL","LowerArmL","HandL",
        "UpperArmR","LowerArmR","HandR",
        "UpperLegL","LowerLegL","FootL",
        "UpperLegR","LowerLegR","FootR"
      ];
      var BONE_LABELS = {
        Head:"Head", Torso:"Torso", Midsection:"Midsection",
        UpperArmL:"Upper Arm L", LowerArmL:"Lower Arm L", HandL:"Hand L",
        UpperArmR:"Upper Arm R", LowerArmR:"Lower Arm R", HandR:"Hand R",
        UpperLegL:"Upper Leg L", LowerLegL:"Lower Leg L", FootL:"Foot L",
        UpperLegR:"Upper Leg R", LowerLegR:"Lower Leg R", FootR:"Foot R"
      };
      if (hudBone) {
        var defOpt = document.createElement("option");
        defOpt.value = ""; defOpt.textContent = "— select bone —";
        hudBone.appendChild(defOpt);
        for (var bi = 0; bi < BONE_KEYS.length; bi++) {
          var opt = document.createElement("option");
          opt.value = BONE_KEYS[bi];
          opt.textContent = BONE_LABELS[BONE_KEYS[bi]] || BONE_KEYS[bi];
          hudBone.appendChild(opt);
        }
        hudBone.onchange = function() {
          var bk = hudBone.value;
          if (!bk) return;
          // Find the bone slider pill in the left panel and scroll to / click it
          var pills = document.querySelectorAll(".pill");
          for (var pi2 = 0; pi2 < pills.length; pi2++) {
            if ((pills[pi2].textContent || "").indexOf(BONE_LABELS[bk] || bk) !== -1) {
              pills[pi2].scrollIntoView({ behavior: "smooth", block: "nearest" });
              pills[pi2].click();
              break;
            }
          }
          // Also highlight the selected state label if present
          hudBone.value = "";
        };
      }
    }, 800);

  })();
