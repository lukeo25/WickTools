<script>
(function(){
"use strict";
var $ = function(id){ return document.getElementById(id); };
var Bridge = window.LukeToolsBridge || null;

function status(msg, kind){ var el=$("status"); el.textContent=msg; el.className="status"+(kind?" "+kind:""); }
function sel(){ if(!Bridge) return null; try{ return Bridge.getSelectionInfo(); }catch(e){ return null; } }

// Open Google Fonts in new tab
$("btnGoogleFonts").onclick = function(){
  window.open("https://fonts.google.com/", "_blank");
};

// When custom font is typed, clear dropdown selection visual indicator
$("customFont").oninput = function(){
  if($("customFont").value.trim()){
    $("fontFamily").style.opacity = "0.5";
  } else {
    $("fontFamily").style.opacity = "1";
  }
};

// When dropdown changes, clear custom font
$("fontFamily").onchange = function(){
  $("customFont").value = "";
  $("fontFamily").style.opacity = "1";
};

function getSelectedFont(){
  var custom = $("customFont").value.trim();
  if(custom) return custom;
  return $("fontFamily").value || "Poppins";
}

function refresh(){
  var info = sel();
  $("selLabel").textContent = (info && info.ok) ? (info.name || info.identifier || "Selected") : "None";
  
  if(info && info.ok && Bridge){
    try{
      var script = Bridge.getUpdateScriptOnSelection() || "";
      var m = script.match(/var\s+LT_TIMER_CFG\s*=\s*(\{[\s\S]*?\});/);
      if(m){
        var cfg = JSON.parse(m[1]);
        if(cfg.mode) $("timerMode").value = cfg.mode;
        if(cfg.startSeconds !== undefined) $("startSeconds").value = cfg.startSeconds;
        if(cfg.endSeconds !== undefined) $("endSeconds").value = cfg.endSeconds;
        if(cfg.fps) $("fps").value = cfg.fps;
        if(cfg.varName) $("varName").value = cfg.varName;
        if(cfg.textName) $("textName").value = cfg.textName;
        if(cfg.displayMode){
          if(cfg.displayMode === "seconds") $("dispSecs").checked = true;
          if(cfg.displayMode === "tenths") $("dispTenths").checked = true;
          if(cfg.displayMode === "hundredths") $("dispHundredths").checked = true;
        }
        if(cfg.fontFamily){
          // Check if font is in dropdown
          var found = false;
          for(var i=0; i<$("fontFamily").options.length; i++){
            if($("fontFamily").options[i].value === cfg.fontFamily){
              $("fontFamily").value = cfg.fontFamily;
              $("customFont").value = "";
              $("fontFamily").style.opacity = "1";
              found = true;
              break;
            }
          }
          if(!found){
            $("customFont").value = cfg.fontFamily;
            $("fontFamily").style.opacity = "0.5";
          }
        }
        if(cfg.gotoFrameOnComplete) $("gotoFrame").value = cfg.gotoFrameOnComplete;
        if(cfg.loop !== undefined) $("loop").checked = cfg.loop;
        if(cfg.stopOnComplete !== undefined) $("stopOnComplete").checked = cfg.stopOnComplete;
        if(cfg.showConsole !== undefined) $("showConsole").checked = cfg.showConsole;
        status("Found config", "good");
      }
    }catch(e){}
  }
  return info;
}

function getDisplayMode(){
  if($("dispHundredths").checked) return "hundredths";
  if($("dispTenths").checked) return "tenths";
  return "seconds";
}

function buildScript(){
  var gotoVal = String($("gotoFrame").value || "").trim();
  var textName = $("textName").value || "TimerText";
  var fontFamily = getSelectedFont();
  
  var cfg = {
    type: "LT_GAME_SPRITE_TIMER",
    mode: $("timerMode").value,
    startSeconds: parseInt($("startSeconds").value) || 30,
    endSeconds: parseInt($("endSeconds").value) || 0,
    fps: parseInt($("fps").value) || 30,
    loop: $("loop").checked,
    stopOnComplete: $("stopOnComplete").checked,
    gotoFrameOnComplete: gotoVal === "" ? null : parseInt(gotoVal),
    showConsole: $("showConsole").checked,
    varName: $("varName").value || "LT_Time",
    textName: textName,
    displayMode: getDisplayMode(),
    fontFamily: fontFamily,
    fontWeights: "400;700"
  };
  
  var L = [];
  L.push('// LukeTools Type: Timer');
  L.push('var LT_TIMER_CFG = ' + JSON.stringify(cfg) + ';');
  L.push('');
  
  // State init
  L.push('if(!this._tmr) this._tmr = {frames:0, value:LT_TIMER_CFG.startSeconds, done:false, fontOk:false};');
  L.push('var _t = this._tmr;');
  L.push('var _fps = LT_TIMER_CFG.fps || 30;');
  L.push('');
  
  // Format function
  L.push('function _pad2(n){ return (n<10?"0":"")+n; }');
  L.push('function _fmt(v){');
  L.push('  v = Math.max(0, Number(v) || 0);');
  L.push('  if(LT_TIMER_CFG.displayMode === "hundredths"){ v = Math.round(v * 100) / 100; }');
  L.push('  else if(LT_TIMER_CFG.displayMode === "tenths"){ v = Math.round(v * 10) / 10; }');
  L.push('  else { v = Math.floor(v); }');
  L.push('  var m = Math.floor(v / 60);');
  L.push('  var rem = v - (m * 60);');
  L.push('  var s = Math.floor(rem);');
  L.push('  var frac = rem - s;');
  L.push('  if(LT_TIMER_CFG.displayMode === "hundredths"){');
  L.push('    var h = Math.round(frac * 100);');
  L.push('    if(h >= 100){ h = 0; s += 1; }');
  L.push('    if(s >= 60){ s = 0; m += 1; }');
  L.push('    return _pad2(m) + ":" + _pad2(s) + "." + (h<10?"0":"") + h;');
  L.push('  }');
  L.push('  if(LT_TIMER_CFG.displayMode === "tenths"){');
  L.push('    var t = Math.round(frac * 10);');
  L.push('    if(t >= 10){ t = 0; s += 1; }');
  L.push('    if(s >= 60){ s = 0; m += 1; }');
  L.push('    return _pad2(m) + ":" + _pad2(s) + "." + t;');
  L.push('  }');
  L.push('  if(s >= 60){ s = 0; m += 1; }');
  L.push('  return _pad2(m) + ":" + _pad2(s);');
  L.push('}');
  L.push('');
  
  // Font loading
  L.push('if(!_t.fontOk && LT_TIMER_CFG.fontFamily){');
  L.push('  _t.fontOk = true;');
  L.push('  try{');
  L.push('    var _fkey = "__LT_F_"+LT_TIMER_CFG.fontFamily;');
  L.push('    if(!window[_fkey]){');
  L.push('      window[_fkey] = true;');
  L.push('      var _link = document.createElement("link");');
  L.push('      _link.rel = "stylesheet";');
  L.push('      _link.href = "https://fonts.googleapis.com/css2?family="+LT_TIMER_CFG.fontFamily.replace(/ /g,"+")+":wght@"+LT_TIMER_CFG.fontWeights+"&display=swap";');
  L.push('      document.head.appendChild(_link);');
  L.push('    }');
  L.push('  }catch(e){}');
  L.push('}');
  L.push('');
  
  // Timer logic
  L.push('if(!_t.done){');
  L.push('  _t.frames++;');
  L.push('  var elapsed = _t.frames / _fps;');  // Don't floor - keep fractional seconds
  L.push('  if(LT_TIMER_CFG.mode === "countup"){');
  L.push('    _t.value = LT_TIMER_CFG.startSeconds + elapsed;');
  L.push('    if(LT_TIMER_CFG.endSeconds > 0 && _t.value >= LT_TIMER_CFG.endSeconds){ _t.value = LT_TIMER_CFG.endSeconds; _t.done = true; }');
  L.push('  } else {');
  L.push('    _t.value = LT_TIMER_CFG.startSeconds - elapsed;');
  L.push('    if(_t.value <= LT_TIMER_CFG.endSeconds){ _t.value = LT_TIMER_CFG.endSeconds; _t.done = true; }');
  L.push('  }');
  L.push('}');
  L.push('');
  
  // Export variables
  L.push('var _timeStr = _fmt(_t.value);');
  L.push('this.timerSeconds = _t.value;');
  L.push('this.timerText = _timeStr;');
  L.push('this[LT_TIMER_CFG.varName] = _t.value;');
  L.push('window[LT_TIMER_CFG.varName] = _t.value;');
  L.push('');
  
  // Update text - use this.TextName.setText() and fontFamily property
  L.push('// Update text');
  L.push('var _txt = this.' + textName + ';');
  L.push('if(_txt){');
  L.push('  try{ _txt.setText(_timeStr); }catch(e){}');
  L.push('  try{ _txt.fontFamily = LT_TIMER_CFG.fontFamily; }catch(e){}');
  L.push('}');
  L.push('');
  
  // Console log
  L.push('if(LT_TIMER_CFG.showConsole && _t.frames % _fps === 0){');
  L.push('  console.log("Timer:", _timeStr, LT_TIMER_CFG.varName+":", _t.value);');
  L.push('}');
  L.push('');
  
  // Completion
  L.push('if(_t.done){');
  L.push('  if(LT_TIMER_CFG.loop){');
  L.push('    this._tmr = {frames:0, value:LT_TIMER_CFG.startSeconds, done:false, fontOk:true};');
  L.push('  } else {');
  L.push('    if(LT_TIMER_CFG.gotoFrameOnComplete) try{ this.gotoAndStop(LT_TIMER_CFG.gotoFrameOnComplete); }catch(e){}');
  L.push('    if(LT_TIMER_CFG.stopOnComplete) try{ this.stop(); }catch(e){}');
  L.push('  }');
  L.push('}');
  L.push('// End');
  
  return L.join('\n');
}

function apply(){
  if(!Bridge){ status("Bridge not available", "bad"); return; }
  var info = sel();
  if(!info || !info.ok){ status("No clip selected", "warn"); return; }
  
  var existing = "";
  try{ existing = Bridge.getUpdateScriptOnSelection() || ""; }catch(e){}
  
  var script = buildScript();
  
  // Remove old timer code
  if(existing.indexOf('// LukeTools Type: Timer') !== -1){
    var start = existing.indexOf('// LukeTools Type: Timer');
    var end = existing.indexOf('// End', start);
    if(end !== -1){
      end = existing.indexOf('\n', end);
      if(end === -1) end = existing.length; else end++;
      existing = existing.substring(0, start) + existing.substring(end);
      existing = existing.trim();
    }
  }
  if(existing.indexOf('// LukeTools Timer Type') !== -1){
    var start = existing.indexOf('// LukeTools Timer Type');
    var end = existing.lastIndexOf('}');
    if(end !== -1){ existing = existing.substring(0, start).trim(); }
  }
  
  var final = existing ? existing + '\n\n' + script : script;
  var r = Bridge.setUpdateScriptOnSelection(final);
  
  if(r && r.ok) status("Applied to " + (info.name || info.identifier), "good");
  else status("Error: " + (r ? r.reason : "Failed"), "bad");
}

$("btnRefresh").onclick = refresh;
$("btnCreate").onclick = apply;
document.onclick = function(){ setTimeout(refresh, 50); };
refresh();
})();
