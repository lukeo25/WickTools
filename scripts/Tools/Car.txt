<script>
(function() {
    "use strict";
    function $(id) { return document.getElementById(id); }

    var Bridge = null;
    try { Bridge = window.LukeToolsBridge || parent.LukeToolsBridge; } catch(e) {}

    function linkRangeNumber(rangeId, numId) {
        var r = $(rangeId), n = $(numId);
        if (!r || !n) return;
        function sync(src, dst) { dst.value = src.value; }
        r.addEventListener("input", function() { sync(r, n); });
        r.addEventListener("change", function() { sync(r, n); });
        n.addEventListener("input", function() { sync(n, r); });
        n.addEventListener("change", function() { sync(n, r); });
    }
    
    linkRangeNumber("maxSpeed", "maxSpeedVal");
    linkRangeNumber("accel", "accelVal");
    linkRangeNumber("brake", "brakeVal");
    linkRangeNumber("friction", "frictionVal");
    linkRangeNumber("turnSpeed", "turnSpeedVal");
    linkRangeNumber("reverseMax", "reverseMaxVal");
    linkRangeNumber("shake", "shakeVal");

    function showStatus(msg, kind) {
        var el = $("status");
        el.textContent = msg;
        el.className = "status " + (kind || "");
    }

    function getSelection() {
        if (!Bridge) return null;
        try { return Bridge.getSelectionInfo(); } catch(e) { return null; }
    }

    function refreshSelection() {
        var info = getSelection();
        $("selLabel").textContent = (info && info.ok) ? (info.name || info.identifier || "Selected") : "No Clip";
        
        if (info && info.ok && Bridge) {
            try {
                var script = Bridge.getUpdateScriptOnSelection() || "";
                var match = script.match(/var\s+LT_CFG\s*=\s*(\{[^}]+\})/);
                if (match) {
                    var cfg = JSON.parse(match[1]);
                    if (cfg.type === "LT_CAR") {
                        if (cfg.maxSpeed !== undefined) { $("maxSpeed").value = cfg.maxSpeed; $("maxSpeedVal").value = cfg.maxSpeed; }
                        if (cfg.accel !== undefined) { $("accel").value = cfg.accel; $("accelVal").value = cfg.accel; }
                        if (cfg.brake !== undefined) { $("brake").value = cfg.brake; $("brakeVal").value = cfg.brake; }
                        if (cfg.friction !== undefined) { $("friction").value = cfg.friction; $("frictionVal").value = cfg.friction; }
                        if (cfg.turnSpeed !== undefined) { $("turnSpeed").value = cfg.turnSpeed; $("turnSpeedVal").value = cfg.turnSpeed; }
                        if (cfg.reverseMax !== undefined) { $("reverseMax").value = cfg.reverseMax; $("reverseMaxVal").value = cfg.reverseMax; }
                        if (cfg.shake !== undefined) { $("shake").value = cfg.shake; $("shakeVal").value = cfg.shake; }
                        if (cfg.enableRough !== undefined) $("enableRough").checked = cfg.enableRough;
                        if (cfg.useArrows !== undefined) $("useArrows").checked = cfg.useArrows;
                        if (cfg.useWASD !== undefined) $("useWASD").checked = cfg.useWASD;
                        if (cfg.useMouse !== undefined) $("useMouse").checked = cfg.useMouse;
                        if (cfg.useClips !== undefined) $("useClips").checked = cfg.useClips;
                        if (cfg.solidCollision !== undefined) $("solidCollision").checked = cfg.solidCollision;
                        if (cfg.bounceOnHit !== undefined) $("bounceOnHit").checked = cfg.bounceOnHit;
                        showStatus("Found 1 block, 14 variables", "success");
                        return info;
                    }
                }
            } catch(e) {}
        }
        return info;
    }

    function generateScript() {
        var cfg = {
            type: "LT_CAR",
            maxSpeed: parseFloat($("maxSpeedVal").value) || 12,
            accel: parseFloat($("accelVal").value) || 0.8,
            brake: parseFloat($("brakeVal").value) || 0.6,
            friction: parseFloat($("frictionVal").value) || 0.985,
            turnSpeed: parseFloat($("turnSpeedVal").value) || 5,
            reverseMax: parseFloat($("reverseMaxVal").value) || 6,
            shake: parseFloat($("shakeVal").value) || 2,
            enableRough: $("enableRough").checked,
            useArrows: $("useArrows").checked,
            useWASD: $("useWASD").checked,
            useMouse: $("useMouse").checked,
            useClips: $("useClips").checked,
            solidCollision: $("solidCollision").checked,
            bounceOnHit: $("bounceOnHit").checked
        };

        // Build the script as array of lines
        var lines = [];
        
        lines.push('// LukeTools Type: Car');
        lines.push('var LT_CFG = ' + JSON.stringify(cfg) + ';');
        lines.push('var p = this.project || null;');
        lines.push('');
        lines.push('// Init speed on this clip');
        lines.push('if (this._carSpd === undefined) this._carSpd = 0;');
        lines.push('');
        
        // AABB helper functions (same pattern as SideRunner)
        lines.push('// AABB collision helpers');
        lines.push('function __car_num(v, d) { v = Number(v); return isFinite(v) ? v : d; }');
        lines.push('function __car_bounds(o) {');
        lines.push('  try {');
        lines.push('    if (o && typeof o.getBounds === "function") {');
        lines.push('      var b = o.getBounds();');
        lines.push('      if (b) {');
        lines.push('        if (b.left !== undefined) return {l:__car_num(b.left,0), t:__car_num(b.top,0), r:__car_num(b.right,0), b:__car_num(b.bottom,0)};');
        lines.push('        if (b.x !== undefined && b.width !== undefined) return {l:__car_num(b.x,0), t:__car_num(b.y,0), r:__car_num(b.x,0)+__car_num(b.width,0), b:__car_num(b.y,0)+__car_num(b.height,0)};');
        lines.push('      }');
        lines.push('    }');
        lines.push('  } catch(e) {}');
        lines.push('  var w = __car_num(o && o.width, 50);');
        lines.push('  var h = __car_num(o && o.height, 50);');
        lines.push('  var x = __car_num(o && o.x, 0);');
        lines.push('  var y = __car_num(o && o.y, 0);');
        lines.push('  return {l:x-w*0.5, t:y-h*0.5, r:x+w*0.5, b:y+h*0.5};');
        lines.push('}');
        lines.push('function __car_inter(a, b) { return !(a.r <= b.l || a.l >= b.r || a.b <= b.t || a.t >= b.b); }');
        lines.push('');
        
        // Resolve X movement with AABB collision
        lines.push('function __car_resolveX(self, dx) {');
        lines.push('  if (!dx) return false;');
        lines.push('  if (!p || !p._LT_SOLIDS) { self.x += dx; return false; }');
        lines.push('  self.x += dx;');
        lines.push('  var a = __car_bounds(self);');
        lines.push('  var hw = (a.r - a.l) * 0.5;');
        lines.push('  for (var k in p._LT_SOLIDS) {');
        lines.push('    var s = p._LT_SOLIDS[k];');
        lines.push('    if (!s || s === self) continue;');
        lines.push('    var sb = __car_bounds(s);');
        lines.push('    if (__car_inter(a, sb)) {');
        lines.push('      if (dx > 0) self.x = sb.l - hw - 0.01;');
        lines.push('      else self.x = sb.r + hw + 0.01;');
        lines.push('      return true;');
        lines.push('    }');
        lines.push('  }');
        lines.push('  return false;');
        lines.push('}');
        lines.push('');
        
        // Resolve Y movement with AABB collision
        lines.push('function __car_resolveY(self, dy) {');
        lines.push('  if (!dy) return false;');
        lines.push('  if (!p || !p._LT_SOLIDS) { self.y += dy; return false; }');
        lines.push('  self.y += dy;');
        lines.push('  var a = __car_bounds(self);');
        lines.push('  var hh = (a.b - a.t) * 0.5;');
        lines.push('  for (var k in p._LT_SOLIDS) {');
        lines.push('    var s = p._LT_SOLIDS[k];');
        lines.push('    if (!s || s === self) continue;');
        lines.push('    var sb = __car_bounds(s);');
        lines.push('    if (__car_inter(a, sb)) {');
        lines.push('      if (dy > 0) self.y = sb.t - hh - 0.01;');
        lines.push('      else self.y = sb.b + hh + 0.01;');
        lines.push('      return true;');
        lines.push('    }');
        lines.push('  }');
        lines.push('  return false;');
        lines.push('}');
        lines.push('');
        
        // Input
        lines.push('// Input');
        lines.push('var _gas = false, _brk = false, _left = false, _right = false;');
        lines.push('if (LT_CFG.useArrows) {');
        lines.push('  if (isKeyDown("up")) _gas = true;');
        lines.push('  if (isKeyDown("down")) _brk = true;');
        lines.push('  if (isKeyDown("left")) _left = true;');
        lines.push('  if (isKeyDown("right")) _right = true;');
        lines.push('}');
        lines.push('if (LT_CFG.useWASD) {');
        lines.push('  if (isKeyDown("w")) _gas = true;');
        lines.push('  if (isKeyDown("s")) _brk = true;');
        lines.push('  if (isKeyDown("a")) _left = true;');
        lines.push('  if (isKeyDown("d")) _right = true;');
        lines.push('}');
        lines.push('if (LT_CFG.useMouse && mouseDown) {');
        lines.push('  _gas = true;');
        lines.push('  var _tdx = mouseX - this.x, _tdy = mouseY - this.y;');
        lines.push('  var _tAng = Math.atan2(_tdy, _tdx) * 180 / Math.PI + 90;');
        lines.push('  var _aDif = _tAng - this.rotation;');
        lines.push('  while (_aDif > 180) _aDif -= 360;');
        lines.push('  while (_aDif < -180) _aDif += 360;');
        lines.push('  if (Math.abs(this._carSpd) > 0.1) {');
        lines.push('    if (_aDif > 2) _right = true;');
        lines.push('    else if (_aDif < -2) _left = true;');
        lines.push('  }');
        lines.push('}');
        lines.push('if (LT_CFG.useClips) {');
        lines.push('  try { var _bg = project.findClipByName("btn_gas"); if (_bg && _bg.isHoveredOver && mouseDown) _gas = true; } catch(e) {}');
        lines.push('  try { var _bb = project.findClipByName("btn_brake"); if (_bb && _bb.isHoveredOver && mouseDown) _brk = true; } catch(e) {}');
        lines.push('  try { var _bl = project.findClipByName("btn_left"); if (_bl && _bl.isHoveredOver && mouseDown) _left = true; } catch(e) {}');
        lines.push('  try { var _br = project.findClipByName("btn_right"); if (_br && _br.isHoveredOver && mouseDown) _right = true; } catch(e) {}');
        lines.push('}');
        lines.push('');
        
        // Speed - use LT_CFG values directly
        lines.push('// Speed');
        lines.push('if (_gas) {');
        lines.push('  this._carSpd += LT_CFG.accel;');
        lines.push('  if (this._carSpd > LT_CFG.maxSpeed) this._carSpd = LT_CFG.maxSpeed;');
        lines.push('} else if (_brk) {');
        lines.push('  this._carSpd -= LT_CFG.brake;');
        lines.push('  if (this._carSpd < -LT_CFG.reverseMax) this._carSpd = -LT_CFG.reverseMax;');
        lines.push('} else {');
        lines.push('  this._carSpd *= LT_CFG.friction;');
        lines.push('  if (Math.abs(this._carSpd) < 0.08) this._carSpd = 0;');
        lines.push('}');
        lines.push('');
        
        // Steering
        lines.push('// Steering');
        lines.push('if (Math.abs(this._carSpd) > 0.2) {');
        lines.push('  var _dir = this._carSpd > 0 ? 1 : -1;');
        lines.push('  if (_left) this.rotation -= LT_CFG.turnSpeed * _dir;');
        lines.push('  if (_right) this.rotation += LT_CFG.turnSpeed * _dir;');
        lines.push('}');
        lines.push('');
        
        // Movement with AABB collision
        lines.push('// Movement');
        lines.push('var _rad = (this.rotation - 90) * Math.PI / 180;');
        lines.push('var _mx = Math.cos(_rad) * this._carSpd;');
        lines.push('var _my = Math.sin(_rad) * this._carSpd;');
        lines.push('');
        lines.push('var _hitX = false, _hitY = false;');
        lines.push('if (LT_CFG.solidCollision) {');
        lines.push('  _hitX = __car_resolveX(this, _mx);');
        lines.push('  _hitY = __car_resolveY(this, _my);');
        lines.push('  if (_hitX || _hitY) {');
        lines.push('    if (LT_CFG.bounceOnHit) this._carSpd = -this._carSpd * 0.3;');
        lines.push('    else this._carSpd *= 0.5;');
        lines.push('  }');
        lines.push('} else {');
        lines.push('  this.x += _mx;');
        lines.push('  this.y += _my;');
        lines.push('}');
        lines.push('');
        
        // Rough terrain
        lines.push('// Rough terrain');
        lines.push('if (LT_CFG.enableRough && this._carRough && Math.abs(this._carSpd) > 0.5) {');
        lines.push('  this.x += (Math.random() - 0.5) * LT_CFG.shake;');
        lines.push('  this.y += (Math.random() - 0.5) * LT_CFG.shake;');
        lines.push('}');
        lines.push('// End');

        return lines.join('\n');
    }

    function applyScript() {
        if (!Bridge) { showStatus("Bridge not available", "error"); return; }

        var info = getSelection();
        if (!info || !info.ok) { showStatus("No clip selected", "error"); return; }

        // Always replace the full script
        var existing = "";
        try { existing = Bridge.getUpdateScriptOnSelection() || ""; } catch(e) {}

        var newScript = generateScript();

        // Remove old car code if present
        if (existing.indexOf('// LukeTools Type: Car') !== -1) {
            var start = existing.indexOf('// LukeTools Type: Car');
            var end = existing.indexOf('// End', start);
            if (end !== -1) {
                end = existing.indexOf('\n', end);
                if (end === -1) end = existing.length;
                else end += 1;
                existing = existing.substring(0, start) + existing.substring(end);
                existing = existing.trim();
            }
        }
        
        var finalScript = existing ? (existing + '\n\n' + newScript) : newScript;
        var result = Bridge.setUpdateScriptOnSelection(finalScript);
        
        if (result && result.ok) {
            showStatus("Applied to " + (info.name || info.identifier), "success");
        } else {
            showStatus("Error: " + (result ? result.reason : "Failed"), "error");
        }
    }

    $("btnRefresh").addEventListener("click", refreshSelection);
    $("btnApply").addEventListener("click", applyScript);

    document.addEventListener("click", function() { setTimeout(refreshSelection, 50); });

    refreshSelection();
})();
</script>
