<script>
(function(){
"use strict";

var Bridge = window.LukeToolsBridge || null;

function $(id){ return document.getElementById(id); }
function status(msg, kind){
  var el = $("status");
  el.textContent = msg;
  el.className = "status" + (kind ? (" " + kind) : "");
}
function currentFontFamily(){
  var custom = String($("customFont").value || "").trim();
  if(custom) return custom;
  return String($("fontSelect").value || "Roboto").trim();
}
function googleFontCssUrl(family){
  var familyTrim = String(family || "").trim();
  var familyQuery = familyTrim.replace(/\s+/g, "+");
  return "https://fonts.googleapis.com/css2?family=" + familyQuery + ":wght@400;700&display=swap";
}
function updatePreview(){
  var family = currentFontFamily();
  var txt = String($("previewText").value || "").trim() || "The quick brown fox jumps over the lazy dog";
  var preview = $("preview");
  preview.textContent = txt;

  var old = document.getElementById("lt_google_font_preview_link");
  if(old) old.remove();

  var link = document.createElement("link");
  link.id = "lt_google_font_preview_link";
  link.rel = "stylesheet";
  link.href = googleFontCssUrl(family);
  document.head.appendChild(link);

  preview.style.fontFamily = "'" + family.replace(/'/g, "\\'") + "', sans-serif";
}
function refreshSelection(){
  if(!Bridge){
    $("selInfo").textContent = "LukeToolsBridge not available";
    status("Bridge not available.", "bad");
    return null;
  }
  var info = null;
  try{
    info = Bridge.getSelectionInfo ? Bridge.getSelectionInfo() : null;
  }catch(e){}
  if(!info || !info.ok){
    $("selInfo").textContent = "No clip selected";
    status("No clip selected.", "warn");
    return null;
  }
  var label = (info.name || info.identifier || info.uuid || "Selected clip");
  $("selInfo").textContent = label;
  status("Selection found.", "good");
  return info;
}
function buildScript(){
  var textName = String($("textName").value || "").trim();
  var fontFamily = currentFontFamily();

  if(!textName) throw new Error("Text item name is required");
  if(!fontFamily) throw new Error("Font family is required");

  var cfg = {
    textName: textName,
    fontFamily: fontFamily,
    fontCssUrl: googleFontCssUrl(fontFamily)
  };

  var L = [];
  L.push("// LukeTools Type: Google Font Text");
  L.push("// Loads a Google font and applies it to a named text item inside this clip.");
  L.push("var LT_GF_CFG = " + JSON.stringify(cfg) + ";");
  L.push("");
  L.push("if(!this._LT_GF_STATE){");
  L.push("  this._LT_GF_STATE = { loaded:false, tried:false };");
  L.push("}");
  L.push("");
  L.push("function LT_GF_ensureLink(href){");
  L.push("  try{");
  L.push("    if(!href) return;");
  L.push("    if(typeof document === 'undefined' || !document.head) return;");
  L.push("    var id = 'lt_google_font_' + href.replace(/[^a-z0-9]+/gi, '_');");
  L.push("    var ex = document.getElementById(id);");
  L.push("    if(ex) return;");
  L.push("    var link = document.createElement('link');");
  L.push("    link.id = id;");
  L.push("    link.rel = 'stylesheet';");
  L.push("    link.href = href;");
  L.push("    document.head.appendChild(link);");
  L.push("  }catch(e0){}");
  L.push("}");
  L.push("");
  L.push("function LT_GF_findText(root, wanted){");
  L.push("  if(!root || !wanted) return null;");
  L.push("  try{ if(root[wanted]) return root[wanted]; }catch(e1){}");
  L.push("  var seen = [];");
  L.push("  function walk(obj){");
  L.push("    if(!obj) return null;");
  L.push("    if(seen.indexOf(obj) !== -1) return null;");
  L.push("    seen.push(obj);");
  L.push("    try{ if(obj._identifier === wanted || obj.identifier === wanted || obj.name === wanted) return obj; }catch(e2){}");
  L.push("    var tl = null;");
  L.push("    try{ tl = obj.timeline || obj._timeline || (obj.clipAsset ? obj.clipAsset.timeline : null); }catch(e3){ tl = null; }");
  L.push("    if(!tl) return null;");
  L.push("    var layers = null;");
  L.push("    try{ layers = tl.layers || tl._layers; }catch(e4){ layers = null; }");
  L.push("    if(!layers || !layers.length) return null;");
  L.push("    for(var i=0;i<layers.length;i++){");
  L.push("      var layer = layers[i];");
  L.push("      if(!layer || !layer.frames) continue;");
  L.push("      for(var j=0;j<layer.frames.length;j++){");
  L.push("        var fr = layer.frames[j];");
  L.push("        if(!fr) continue;");
  L.push("        var kids = [];");
  L.push("        try{");
  L.push("          if(fr.clips) kids = kids.concat(fr.clips);");
  L.push("          if(fr.children) kids = kids.concat(fr.children);");
  L.push("          if(fr.paths) kids = kids.concat(fr.paths);");
  L.push("          if(fr.texts) kids = kids.concat(fr.texts);");
  L.push("        }catch(e5){}");
  L.push("        for(var k=0;k<kids.length;k++){");
  L.push("          var got = walk(kids[k]);");
  L.push("          if(got) return got;");
  L.push("        }");
  L.push("      }");
  L.push("    }");
  L.push("    return null;");
  L.push("  }");
  L.push("  return walk(root);");
  L.push("}");
  L.push("");
  L.push("LT_GF_ensureLink(LT_GF_CFG.fontCssUrl);");
  L.push("var _txt = LT_GF_findText(this, LT_GF_CFG.textName);");
  L.push("if(_txt){");
  L.push("  try{ _txt.fontFamily = LT_GF_CFG.fontFamily; }catch(e6){}");
  L.push("  try{ if(_txt.style) _txt.style.fontFamily = LT_GF_CFG.fontFamily; }catch(e7){}");
  L.push("  try{ if(_txt.textData) _txt.textData.fontFamily = LT_GF_CFG.fontFamily; }catch(e8){}");
  L.push("  try{ if(_txt._textData) _txt._textData.fontFamily = LT_GF_CFG.fontFamily; }catch(e9){}");
  L.push("}");
  L.push("// End");
  return L.join("\n");
}
function removeOldBlock(existing){
  existing = String(existing || "");
  var start = existing.indexOf("// LukeTools Type: Google Font Text");
  if(start !== -1){
    var end = existing.indexOf("// End", start);
    if(end !== -1){
      end = existing.indexOf("\n", end);
      if(end === -1) end = existing.length;
      else end += 1;
      existing = (existing.substring(0, start) + existing.substring(end)).trim();
    }
  }
  return existing;
}
function apply(){
  if(!Bridge){
    status("Bridge not available.", "bad");
    return;
  }
  var info = refreshSelection();
  if(!info || !info.ok){
    status("No clip selected.", "warn");
    return;
  }

  var existing = "";
  try{ existing = Bridge.getUpdateScriptOnSelection() || ""; }catch(e0){}
  var script = buildScript();
  existing = removeOldBlock(existing);
  var finalText = existing ? (existing + "\n\n" + script) : script;

  var r = Bridge.setUpdateScriptOnSelection(finalText);
  if(r && r.ok === false){
    status("Error: " + (r.reason || "Failed"), "bad");
    return;
  }
  status("Applied Google font text type to " + (info.name || info.identifier || "selected clip") + ".", "good");
}

$("btnRefresh").onclick = refreshSelection;
$("btnApply").onclick = apply;
$("fontSelect").onchange = updatePreview;
$("customFont").oninput = updatePreview;
$("previewText").oninput = updatePreview;

updatePreview();
refreshSelection();

})();
</script>
