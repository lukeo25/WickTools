<script>
(function(){
"use strict";
var $ = function(id){ return document.getElementById(id); };
var Bridge = window.LukeToolsBridge || null;

function status(msg, kind){ var el=$("status"); el.textContent=msg; el.className="status"+(kind?" "+kind:""); }
function sel(){ if(!Bridge) return null; try{ return Bridge.getSelectionInfo(); }catch(e){ return null; } }

$("btnGoogleFonts").onclick = function(){
  window.open("https://fonts.google.com/", "_blank");
};

$("customFont").oninput = function(){
  if($("customFont").value.trim()){ $("fontFamily").style.opacity = "0.5"; }
  else { $("fontFamily").style.opacity = "1"; }
};

$("fontFamily").onchange = function(){
  $("customFont").value = "";
  $("fontFamily").style.opacity = "1";
};

function getSelectedFont(){
  var custom = $("customFont").value.trim();
  if(custom) return custom;
  return $("fontFamily").value || "Poppins";
}

function refresh(){
  var info = sel();
  $("selLabel").textContent = (info && info.ok) ? (info.name || info.identifier || "Selected") : "None";

  if(info && info.ok && Bridge){
    try{
      var script = Bridge.getUpdateScriptOnSelection() || "";
      var m = script.match(/var\s+LT_LIFE_CFG\s*=\s*(\{[\s\S]*?\});/);
      if(m){
        var cfg = JSON.parse(m[1]);
        if(cfg.varName) $("varName").value = cfg.varName;
        if(cfg.textName) $("textName").value = cfg.textName;
        if(cfg.defaultValue !== undefined) $("defaultValue").value = cfg.defaultValue;
        if(cfg.prefix !== undefined) $("prefix").value = cfg.prefix;
        if(cfg.suffix !== undefined) $("suffix").value = cfg.suffix;
        if(cfg.fontFamily){
          var found = false;
          for(var i=0; i<$("fontFamily").options.length; i++){
            if($("fontFamily").options[i].value === cfg.fontFamily){
              $("fontFamily").value = cfg.fontFamily;
              $("customFont").value = "";
              $("fontFamily").style.opacity = "1";
              found = true;
              break;
            }
          }
          if(!found){
            $("customFont").value = cfg.fontFamily;
            $("fontFamily").style.opacity = "0.5";
          }
        }
        if(cfg.showConsole !== undefined) $("showConsole").checked = cfg.showConsole;
        status("Found config", "good");
      }
    }catch(e){}
  }
  return info;
}

function buildScript(){
  var textName = $("textName").value || "LifeText";
  var fontFamily = getSelectedFont();

  var cfg = {
    type: "LT_LIFE_DISPLAY",
    varName: $("varName").value || "LT_Life",
    textName: textName,
    defaultValue: $("defaultValue").value || "0",
    prefix: $("prefix").value || "",
    suffix: $("suffix").value || "",
    fontFamily: fontFamily,
    fontWeights: "400;700",
    showConsole: $("showConsole").checked
  };

  var L = [];
  L.push('// LukeTools Type: Life');
  L.push('var LT_LIFE_CFG = ' + JSON.stringify(cfg) + ';');
  L.push('');
  L.push('if(!this._lifeDisp) this._lifeDisp = {fontOk:false,lastText:null};');
  L.push('var _ld = this._lifeDisp;');
  L.push('');
  L.push('if(!_ld.fontOk && LT_LIFE_CFG.fontFamily){');
  L.push('  _ld.fontOk = true;');
  L.push('  try{');
  L.push('    var _fkey = "__LT_F_"+LT_LIFE_CFG.fontFamily;');
  L.push('    if(!window[_fkey]){');
  L.push('      window[_fkey] = true;');
  L.push('      var _link = document.createElement("link");');
  L.push('      _link.rel = "stylesheet";');
  L.push('      _link.href = "https://fonts.googleapis.com/css2?family="+LT_LIFE_CFG.fontFamily.replace(/ /g,"+")+":wght@"+LT_LIFE_CFG.fontWeights+"&display=swap";');
  L.push('      document.head.appendChild(_link);');
  L.push('    }');
  L.push('  }catch(e){}');
  L.push('}');
  L.push('');
  L.push('var _lifeVal = LT_LIFE_CFG.defaultValue;');
  L.push('try{ if(window[LT_LIFE_CFG.varName] !== undefined && window[LT_LIFE_CFG.varName] !== null) _lifeVal = window[LT_LIFE_CFG.varName]; }catch(e){}');
  L.push('try{ if(this[LT_LIFE_CFG.varName] !== undefined && this[LT_LIFE_CFG.varName] !== null) _lifeVal = this[LT_LIFE_CFG.varName]; }catch(e){}');
  L.push('');
  L.push('this.lifeValue = _lifeVal;');
  L.push('this[LT_LIFE_CFG.varName] = _lifeVal;');
  L.push('');
  L.push('var _lifeText = String(LT_LIFE_CFG.prefix) + String(_lifeVal) + String(LT_LIFE_CFG.suffix);');
  L.push('');
  L.push('var _txt = this.' + textName + ';');
  L.push('if(_txt){');
  L.push('  try{ _txt.setText(_lifeText); }catch(e){}');
  L.push('  try{ _txt.text = _lifeText; }catch(e){}');
  L.push('  try{ _txt.fontFamily = LT_LIFE_CFG.fontFamily; }catch(e){}');
  L.push('  try{ _txt.font = LT_LIFE_CFG.fontFamily; }catch(e){}');
  L.push('}');
  L.push('');
  L.push('if(LT_LIFE_CFG.showConsole && _ld.lastText !== _lifeText){');
  L.push('  _ld.lastText = _lifeText;');
  L.push('  console.log("Life:", LT_LIFE_CFG.varName + ":", _lifeVal, "Text:", _lifeText);');
  L.push('}');
  L.push('// End');
  return L.join("\n");
}

function apply(){
  if(!Bridge){ status("Bridge not available", "bad"); return; }
  var info = sel();
  if(!info || !info.ok){ status("No clip selected", "warn"); return; }

  var existing = "";
  try{ existing = Bridge.getUpdateScriptOnSelection() || ""; }catch(e){}

  var script = buildScript();

  if(existing.indexOf('// LukeTools Type: Life') !== -1){
    var start = existing.indexOf('// LukeTools Type: Life');
    var end = existing.indexOf('// End', start);
    if(end !== -1){
      end = existing.indexOf('\n', end);
      if(end === -1) end = existing.length; else end++;
      existing = existing.substring(0, start) + existing.substring(end);
      existing = existing.trim();
    }
  }

  var final = existing ? existing + '\n\n' + script : script;
  var r = Bridge.setUpdateScriptOnSelection(final);

  if(r && r.ok) status("Applied to " + (info.name || info.identifier), "good");
  else status("Error: " + (r ? r.reason : "Failed"), "bad");
}

$("btnRefresh").onclick = refresh;
$("btnCreate").onclick = apply;
document.onclick = function(){ setTimeout(refresh, 50); };
refresh();
})();
</script>
