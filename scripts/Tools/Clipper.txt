<script>
(function () {
  'use strict';

  var TOOL_ID = 'lt-stage-recorder-clipper-v4';
  if (window[TOOL_ID]) {
    try {
      window[TOOL_ID].focus();
    } catch (e) {}
    return;
  }

  var state = {
    panel: null,
    statusEl: null,
    recordBtn: null,
    findBtn: null,
    filenameInput: null,
    timerEl: null,
    canvas: null,
    canvasOwnerWindow: null,
    recorder: null,
    chunks: [],
    lastBlob: null,
    lastUrl: '',
    isRecording: false,
    autoDownload: true,
    startTimer: null,
    captureCanvas: null,
    captureCtx: null,
    captureStream: null,
    drawLoopId: 0,
    recordStartedAt: 0,
    timerInterval: 0
  };

  window[TOOL_ID] = {
    focus: function () {
      if (state.panel) {
        state.panel.style.display = 'block';
        state.panel.style.opacity = '1';
      }
    }
  };

  function setStatus(text) {
    if (state.statusEl) {
      state.statusEl.textContent = text;
    }
    console.log('[LukeTools Recorder]', text);
  }

  function makeEl(tag, styles, text) {
    var el = document.createElement(tag);
    if (styles) {
      Object.keys(styles).forEach(function (key) {
        el.style[key] = styles[key];
      });
    }
    if (typeof text === 'string') {
      el.textContent = text;
    }
    return el;
  }

  function formatMs(ms) {
    var totalSeconds = Math.max(0, Math.floor(ms / 1000));
    var hours = Math.floor(totalSeconds / 3600);
    var minutes = Math.floor((totalSeconds % 3600) / 60);
    var seconds = totalSeconds % 60;

    function pad(n) {
      return String(n).padStart(2, '0');
    }

    return pad(hours) + ':' + pad(minutes) + ':' + pad(seconds);
  }

  function updateTimerDisplay() {
    if (!state.timerEl) return;
    if (!state.isRecording || !state.recordStartedAt) {
      state.timerEl.textContent = '00:00:00';
      return;
    }
    state.timerEl.textContent = formatMs(Date.now() - state.recordStartedAt);
  }

  function startTimerDisplay() {
    stopTimerDisplay();
    state.recordStartedAt = Date.now();
    updateTimerDisplay();
    state.timerInterval = setInterval(updateTimerDisplay, 250);
  }

  function stopTimerDisplay() {
    if (state.timerInterval) {
      clearInterval(state.timerInterval);
      state.timerInterval = 0;
    }
  }

  function createPanel() {
    var panel = makeEl('div', {
      position: 'fixed',
      top: '12px',
      right: '12px',
      width: '260px',
      background: '#1e1e1e',
      color: '#f2f2f2',
      border: '1px solid #444',
      borderRadius: '12px',
      boxShadow: '0 10px 30px rgba(0,0,0,0.35)',
      zIndex: '2147483647',
      fontFamily: 'Arial, Helvetica, sans-serif',
      fontSize: '12px',
      overflow: 'hidden'
    });

    var header = makeEl('div', {
      padding: '9px 10px',
      background: '#2a2a2a',
      fontWeight: '700',
      letterSpacing: '0.2px'
    }, 'Luke Tools Stage Recorder');

    var body = makeEl('div', {
      padding: '10px'
    });

    var subtitle = makeEl('div', {
      marginBottom: '8px',
      color: '#cccccc',
      lineHeight: '1.4'
    }, 'Records the visible Wick canvas exactly as shown on stage.');

    var timerWrap = makeEl('div', {
      marginBottom: '10px',
      padding: '10px',
      borderRadius: '8px',
      background: '#111',
      border: '1px solid #444',
      textAlign: 'center'
    });

    var timerLabel = makeEl('div', {
      color: '#bdbdbd',
      fontSize: '11px',
      marginBottom: '4px'
    }, 'Recorded time');

    state.timerEl = makeEl('div', {
      fontSize: '22px',
      fontWeight: '700',
      letterSpacing: '1px',
      color: '#ffffff'
    }, '00:00:00');

    timerWrap.appendChild(timerLabel);
    timerWrap.appendChild(state.timerEl);

    var grid = makeEl('div', {
      display: 'grid',
      gridTemplateColumns: '1fr',
      gap: '8px',
      alignItems: 'stretch'
    });

    function addRow(labelText, inputEl) {
      var wrap = makeEl('div', {
        display: 'block'
      });
      var label = makeEl('label', {
        color: '#d8d8d8',
        display: 'block',
        marginBottom: '4px',
        fontSize: '12px'
      }, labelText);
      wrap.appendChild(label);
      wrap.appendChild(inputEl);
      grid.appendChild(wrap);
    }

    function makeInput(value) {
      return makeEl('input', {
        width: '100%',
        boxSizing: 'border-box',
        padding: '7px 9px',
        borderRadius: '7px',
        border: '1px solid #555',
        background: '#111',
        color: '#fff',
        outline: 'none'
      });
    }

    state.filenameInput = makeInput('wick-stage-recording');
    state.filenameInput.type = 'text';

    addRow('File name', state.filenameInput);

    var buttonRow = makeEl('div', {
      display: 'grid',
      gridTemplateColumns: '1fr',
      gap: '6px',
      marginTop: '10px'
    });

    function makeButton(text, bg) {
      return makeEl('button', {
        padding: '9px 10px',
        borderRadius: '7px',
        border: '1px solid #555',
        background: bg,
        color: '#fff',
        cursor: 'pointer',
        fontWeight: '700'
      }, text);
    }

    state.recordBtn = makeButton('Record', '#0b5ed7');
    state.findBtn = makeButton('Find Canvas', '#444');

    buttonRow.appendChild(state.recordBtn);
    buttonRow.appendChild(state.findBtn);

    state.statusEl = makeEl('div', {
      marginTop: '10px',
      padding: '10px',
      minHeight: '34px',
      borderRadius: '7px',
      background: '#111',
      color: '#cfcfcf',
      lineHeight: '1.35'
    }, 'Ready.');

    body.appendChild(subtitle);
    body.appendChild(timerWrap);
    body.appendChild(grid);
    body.appendChild(buttonRow);
    body.appendChild(state.statusEl);

    panel.appendChild(header);
    panel.appendChild(body);
    document.body.appendChild(panel);

    state.panel = panel;
  }

  function getDocumentsToSearch() {
    var docs = [];
    var seen = [];

    function addDoc(doc) {
      if (!doc) return;
      if (seen.indexOf(doc) !== -1) return;
      seen.push(doc);
      docs.push(doc);
    }

    addDoc(document);

    try {
      if (window.parent && window.parent.document) addDoc(window.parent.document);
    } catch (e) {}

    try {
      if (window.top && window.top.document) addDoc(window.top.document);
    } catch (e) {}

    return docs;
  }

  function getVisibleCanvasScore(canvas) {
    if (!canvas || !canvas.getBoundingClientRect) return -1;
    var rect = canvas.getBoundingClientRect();
    if (!rect) return -1;
    if (rect.width < 20 || rect.height < 20) return -1;

    var style;
    try {
      style = (canvas.ownerDocument.defaultView || window).getComputedStyle(canvas);
    } catch (e) {
      style = null;
    }

    if (style) {
      if (style.display === 'none' || style.visibility === 'hidden' || parseFloat(style.opacity || '1') === 0) return -1;
    }

    var score = rect.width * rect.height;
    if (canvas.width && canvas.height) score += canvas.width * canvas.height * 0.25;
    return score;
  }

  function isToolCanvas(canvas) {
    if (!canvas) return true;
    if (state.panel && state.panel.contains(canvas)) return true;
    return false;
  }

  function findBestCanvas() {
    var docs = getDocumentsToSearch();
    var best = null;
    var bestScore = -1;

    docs.forEach(function (doc) {
      var canvases = [];
      try {
        canvases = Array.prototype.slice.call(doc.querySelectorAll('canvas'));
      } catch (e) {
        canvases = [];
      }

      canvases.forEach(function (canvas) {
        if (isToolCanvas(canvas)) return;
        var score = getVisibleCanvasScore(canvas);
        if (score > bestScore) {
          bestScore = score;
          best = canvas;
        }
      });
    });

    return best;
  }

  function describeCanvas(canvas) {
    if (!canvas) return 'No canvas';
    var rect = canvas.getBoundingClientRect();
    return 'Canvas ' + Math.round(rect.width) + ' x ' + Math.round(rect.height) + ' internal ' + (canvas.width || 0) + ' x ' + (canvas.height || 0);
  }

  function updateRecordButton() {
    if (!state.recordBtn) return;
    if (state.isRecording) {
      state.recordBtn.textContent = 'Stop Recording';
      state.recordBtn.style.background = '#b02a37';
    } else {
      state.recordBtn.textContent = 'Record';
      state.recordBtn.style.background = '#0b5ed7';
    }
  }

  function ensureCanvas() {
    var canvas = findBestCanvas();
    if (!canvas) {
      state.canvas = null;
      state.canvasOwnerWindow = null;
      setStatus('Could not find the Wick stage canvas.');
      return false;
    }
    state.canvas = canvas;
    state.canvasOwnerWindow = canvas.ownerDocument.defaultView || window;
    setStatus('Canvas found. ' + describeCanvas(canvas));
    return true;
  }

  function releaseLastUrl() {
    if (state.lastUrl) {
      try {
        URL.revokeObjectURL(state.lastUrl);
      } catch (e) {}
      state.lastUrl = '';
    }
  }

  function stopDrawLoop() {
    if (state.drawLoopId) {
      try {
        cancelAnimationFrame(state.drawLoopId);
      } catch (e) {}
      state.drawLoopId = 0;
    }
  }

  function cleanupCaptureSurfaces() {
    stopDrawLoop();
    if (state.captureStream) {
      try {
        state.captureStream.getTracks().forEach(function (track) { track.stop(); });
      } catch (e) {}
    }
    state.captureCanvas = null;
    state.captureCtx = null;
    state.captureStream = null;
  }

  function drawCanvasLoop() {
    if (!state.captureCtx || !state.captureCanvas || !state.canvas || !state.isRecording) return;

    try {
      if (state.captureCanvas.width !== state.canvas.width || state.captureCanvas.height !== state.canvas.height) {
        state.captureCanvas.width = state.canvas.width || 1280;
        state.captureCanvas.height = state.canvas.height || 720;
      }

      state.captureCtx.clearRect(0, 0, state.captureCanvas.width, state.captureCanvas.height);
      state.captureCtx.drawImage(state.canvas, 0, 0, state.captureCanvas.width, state.captureCanvas.height);
    } catch (e) {}

    state.drawLoopId = requestAnimationFrame(drawCanvasLoop);
  }

  function chooseMimeType() {
    var mimeTypes = [
      'video/webm;codecs=vp9,opus',
      'video/webm;codecs=vp8,opus',
      'video/webm;codecs=vp9',
      'video/webm;codecs=vp8',
      'video/webm'
    ];

    for (var i = 0; i < mimeTypes.length; i++) {
      var type = mimeTypes[i];
      try {
        if (window.MediaRecorder && MediaRecorder.isTypeSupported && MediaRecorder.isTypeSupported(type)) {
          return type;
        }
      } catch (e) {}
    }
    return '';
  }

  function sanitizeFileName(name) {
    var s = String(name || 'wick-stage-recording').trim();
    if (!s) s = 'wick-stage-recording';
    s = s.replace(/[\\/:*?"<>|]+/g, '_');
    if (!/\.webm$/i.test(s)) s += '.webm';
    return s;
  }

  function buildRecorder() {
    if (!state.canvas) return null;

    state.captureCanvas = document.createElement('canvas');
    state.captureCanvas.width = state.canvas.width || 1280;
    state.captureCanvas.height = state.canvas.height || 720;
    state.captureCtx = state.captureCanvas.getContext('2d', { alpha: false });

    state.captureStream = state.captureCanvas.captureStream(30);

    var mimeType = chooseMimeType();
    var options = {};
    if (mimeType) options.mimeType = mimeType;
    options.videoBitsPerSecond = 8000000;

    var recorder;
    try {
      recorder = new MediaRecorder(state.captureStream, options);
    } catch (e) {
      setStatus('Could not create MediaRecorder. ' + (e && e.message ? e.message : e));
      cleanupCaptureSurfaces();
      return null;
    }

    state.chunks = [];

    recorder.ondataavailable = function (event) {
      if (event.data && event.data.size > 0) {
        state.chunks.push(event.data);
      }
    };

    recorder.onerror = function (event) {
      var err = event && event.error ? event.error.message || String(event.error) : 'Unknown recorder error';
      setStatus('Recorder error. ' + err);
    };

    recorder.onstop = function () {
      state.isRecording = false;
      updateRecordButton();
      stopTimerDisplay();
      cleanupCaptureSurfaces();

      if (!state.chunks.length) {
        setStatus('Recording stopped, but no video data was captured.');
        return;
      }

      releaseLastUrl();
      state.lastBlob = new Blob(state.chunks, { type: recorder.mimeType || 'video/webm' });
      state.lastUrl = URL.createObjectURL(state.lastBlob);
      setStatus('Recording saved. ' + formatMs(Date.now() - state.recordStartedAt) + ' captured.');

      if (state.autoDownload) {
        downloadLastRecording();
      }
    };

    return recorder;
  }

  function downloadLastRecording() {
    if (!state.lastBlob || !state.lastUrl) {
      setStatus('No recording to download yet.');
      return;
    }

    var a = document.createElement('a');
    a.href = state.lastUrl;
    a.download = sanitizeFileName(state.filenameInput.value);
    document.body.appendChild(a);
    a.click();
    a.remove();
  }

  function startRecording() {
    if (state.isRecording) return;

    if (!ensureCanvas()) return;

    setStatus('Canvas found. ' + describeCanvas(state.canvas));

    state.recorder = buildRecorder();
    if (!state.recorder) return;

    state.isRecording = true;
    updateRecordButton();
    startTimerDisplay();

    drawCanvasLoop();
    try {
      state.recorder.start(250);
      setStatus('Recording started. Recording the visible Wick canvas.');
    } catch (e) {
      state.isRecording = false;
      updateRecordButton();
      stopTimerDisplay();
      cleanupCaptureSurfaces();
      setStatus('Could not start recording. ' + (e && e.message ? e.message : e));
    }
  }

  function stopRecording() {
    if (state.startTimer) {
      clearTimeout(state.startTimer);
      state.startTimer = null;
    }

    if (!state.isRecording) {
      setStatus('Recorder is not running.');
      return;
    }

    if (state.recorder && state.recorder.state !== 'inactive') {
      try {
        state.recorder.stop();
      } catch (e) {
        state.isRecording = false;
        updateRecordButton();
        stopTimerDisplay();
        cleanupCaptureSurfaces();
        setStatus('Stop failed. ' + (e && e.message ? e.message : e));
      }
    } else {
      state.isRecording = false;
      updateRecordButton();
      stopTimerDisplay();
      cleanupCaptureSurfaces();
      setStatus('Recording stopped.');
    }
  }

  function wireEvents() {
    state.findBtn.addEventListener('click', function () {
      ensureCanvas();
    });

    state.recordBtn.addEventListener('click', function () {
      if (state.isRecording) {
        stopRecording();
      } else {
        startRecording();
      }
    });
  }

  function init() {
    createPanel();
    wireEvents();
    updateRecordButton();
    updateTimerDisplay();
    ensureCanvas();
  }

  init();
})();

</script>
