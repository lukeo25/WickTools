  <script>
    (function () {
      var KEY_BIPED_ACTIVE = "LukeToolsBipedActive";

      function byId(id) { return document.getElementById(id); }

      function getBridge() {
        try {
          if (window.LukeToolsBridge) return window.LukeToolsBridge;
        } catch (e1) {}
        try {
          if (parent && parent.LukeToolsBridge) return parent.LukeToolsBridge;
        } catch (e2) {}
        return null;
      }

      function setStatus(t) {
        byId("status").textContent = String(t || "");
      }

      function setBipedActive(flag) {
        try { parent.localStorage.setItem(KEY_BIPED_ACTIVE, flag ? "1" : "0"); } catch (e1) {}
        try { localStorage.setItem(KEY_BIPED_ACTIVE, flag ? "1" : "0"); } catch (e2) {}
      }

      // Mark that biped is open so the addon can show the pose panel after Close
      setBipedActive(true);

      byId("btnArrange").addEventListener("click", function () {
        var bridge = getBridge();
        if (!bridge) {
          setStatus("Bridge not found");
          return;
        }

        if (typeof bridge.arrangeArmRigOnSelection !== "function") {
          setStatus("Arrange error TypeError: bridge.arrangeArmRigOnSelection is not a function");
          return;
        }

        var sh = byId("shoulderId").value || "Shoulder_R";
        var el = byId("elbowId").value || "Elbo_R";

        try {
          var ok = bridge.arrangeArmRigOnSelection(sh, el);
          setStatus(ok ? "Arrange ok" : "Arrange failed");
        } catch (e) {
          setStatus("Arrange error " + e);
        }
      });

      byId("btnPose").addEventListener("click", function () {
        try {
          if (parent && typeof parent.LukeToolsPosePanelShow === "function") {
            parent.LukeToolsPosePanelShow();
            setStatus("Pose panel shown");
            return;
          }
        } catch (e1) {}

        var bridge = getBridge();
        if (bridge && typeof bridge.showPosePanel === "function") {
          try {
            bridge.showPosePanel();
            setStatus("Pose panel shown");
            return;
          } catch (e2) {}
        }

        setStatus("Pose panel not available");
      });
    })();
  </script>
