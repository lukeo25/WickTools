  <script>
    (function () {
      function byId(id) { return document.getElementById(id); }
      var status = byId("status");

      function setStatus(t) {
        status.textContent = String(t || "");
      }

      function getBridge() {
        try { return window.LukeToolsBridge || parent.LukeToolsBridge; } catch (e) { return null; }
      }

      byId("btnArrange").addEventListener("click", function () {
        var bridge = getBridge();
        if (!bridge) {
          setStatus("Bridge not found. This tool must be opened from the Luke Tools dropdown.");
          return;
        }

        if (typeof bridge.arrangeArmRigOnSelection !== "function") {
          setStatus("Arrange failed. bridge.arrangeArmRigOnSelection is not a function.");
          return;
        }

        var sh = byId("shoulderId").value || "Shoulder_R";
        var el = byId("elbowId").value || "Elbo_R";

        try {
          var ok = bridge.arrangeArmRigOnSelection(sh, el);
          setStatus(ok ? "Arrange ok" : "Arrange failed");
        } catch (e) {
          setStatus("Arrange error: " + e);
        }
      });

      byId("btnShowPose").addEventListener("click", function () {
        var bridge = getBridge();
        if (!bridge || typeof bridge.showPosePanel !== "function") {
          setStatus("Pose Panel call not available. Update the userscript.");
          return;
        }

        try {
          bridge.showPosePanel();
          setStatus("Pose Panel shown in Wick");
        } catch (e) {
          setStatus("Pose Panel error: " + e);
        }
      });
    })();
  </script>
